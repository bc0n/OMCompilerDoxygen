52 static fmiBoolean invalidNumber(ModelInstance* comp, const char* f, const char* arg, int n, int nExpected)
53 {
54  if (n != nExpected)
55  {
56  comp->state = modelError;
57  comp->functions.logger(comp, comp->instanceName, fmiError, "error",
58  "%s: Invalid argument %s = %d. Expected %d.", f, arg, n, nExpected);
59  return fmiTrue;
60  }
61  return fmiFalse;
62 }

64 static fmiBoolean invalidState(ModelInstance* comp, const char* f, int statesExpected)
65 {
66  if (!comp)
67  return fmiTrue;
68  if (!(comp->state & statesExpected))
69  {
70  comp->state = modelError;
71  comp->functions.logger(comp, comp->instanceName, fmiError, "error",
72  "%s: Illegal call sequence. Expected State: %d.", f, statesExpected);
73  return fmiTrue;
74  }
75  return fmiFalse;
76 }

126 fmiComponent fmiInstantiateModel(fmiString instanceName, fmiString GUID, fmiCallbackFunctions functions, fmiBoolean loggingOn)
127 {
128  ModelInstance* comp;
129  if (!functions.logger)
130  return NULL;
131  if (!functions.allocateMemory || !functions.freeMemory){
132  functions.logger(NULL, instanceName, fmiError, "error",
133  "fmiInstantiateModel: Missing callback function.");
134  return NULL;
135  }
136  if (!instanceName || strlen(instanceName)==0) {
137  functions.logger(NULL, instanceName, fmiWarning, "Warning",
138  "fmiInstantiateModel: Missing instance name.");
139  }
140  if (strcmp(GUID, MODEL_GUID) != 0) {
141  functions.logger(NULL, instanceName, fmiError, "error",
142  "fmiInstantiateModel: Wrong GUID %s. Expected %s.", GUID, MODEL_GUID);
143  return NULL;
144  }
145  comp = (ModelInstance *)functions.allocateMemory(1, sizeof(ModelInstance));
146  if (comp) {
147  DATA* fmudata = NULL;
148  MODEL_DATA* modelData = NULL;
149  SIMULATION_INFO* simInfo = NULL;
150  threadData_t *threadData = NULL;
151 
152  comp->functions = functions;
153  comp->loggingOn = loggingOn;
154  comp->state = modelInstantiated;
155  comp->instanceName = functions.allocateMemory(1 + strlen(instanceName), sizeof(char));
156  comp->GUID = functions.allocateMemory(1 + strlen(GUID), sizeof(char));
157  /* Cannot use functions.allocateMemory since the pointer might not be stored on the stack of the parent */
158  fmudata = (DATA *)functions.allocateMemory(1, sizeof(DATA));
159  modelData = (MODEL_DATA *)functions.allocateMemory(1, sizeof(MODEL_DATA));
160  simInfo = (SIMULATION_INFO *)functions.allocateMemory(1, sizeof(SIMULATION_INFO));
161  fmudata->modelData = modelData;
162  fmudata->simulationInfo = simInfo;
163 
164 
165  threadData = (threadData_t *)functions.allocateMemory(1, sizeof(threadData_t));
166  memset(threadData, 0, sizeof(threadData_t));
167  /*
168  pthread_key_create(&fmu1_thread_data_key,NULL);
169  pthread_setspecific(fmu1_thread_data_key, threadData);
170  */
171 
172  comp->threadData = threadData;
173  comp->fmuData = fmudata;
174 
175  if (!comp->fmuData) {
176  functions.logger(NULL, instanceName, fmiError, "error",
177  "fmiInstantiateModel: Error: Could not initialize the global data structure file.");
178  return NULL;
179  }
180  }
181  if (!comp || !comp->instanceName || !comp->GUID) {
182  functions.logger(NULL, instanceName, fmiError, "error", "fmiInstantiateModel: Out of memory.");
183  return NULL;
184  }
185  /* intialize modelData */
186  fmu1_model_interface_setupDataStruc(comp->fmuData, comp->threadData);
187  useStream[LOG_STDOUT] = 1;
188  useStream[LOG_ASSERT] = 1;
189  initializeDataStruc(comp->fmuData, comp->threadData);
190  /* setup model data with default start data */
191  setDefaultStartValues(comp);
192  setAllVarsToStart(comp->fmuData);
193  setAllParamsToStart(comp->fmuData);
194  comp->fmuData->callback->read_input_fmu(comp->fmuData->modelData, comp->fmuData->simulationInfo);
195  modelInfoInit(&(comp->fmuData->modelData->modelDataXml));
196 
197  strcpy((char*)comp->instanceName, (const char*)instanceName);
198  strcpy((char*)comp->GUID, (const char*)GUID);
199 
200  /* read input vars */
201  /* input_function(comp->fmuData);*/
202  /* allocate memory for non-linear system solvers */
203 #if !defined(OMC_NUM_NONLINEAR_SYSTEMS) || OMC_NUM_NONLINEAR_SYSTEMS>0
204  initializeNonlinearSystems(comp->fmuData, comp->threadData);
205 #endif
206  /* allocate memory for non-linear system solvers */
207 #if !defined(OMC_NUM_LINEAR_SYSTEMS) || OMC_NUM_LINEAR_SYSTEMS>0
208  initializeLinearSystems(comp->fmuData, comp->threadData);
209 #endif
210  /* allocate memory for mixed system solvers */
211 #if !defined(OMC_NUM_MIXED_SYSTEMS) || OMC_NUM_MIXED_SYSTEMS>0
212  initializeMixedSystems(comp->fmuData, comp->threadData);
213 #endif
214  /* allocate memory for state selection */
215 #if !defined(OMC_NO_STATESELECTION)
216  initializeStateSetJacobians(comp->fmuData, comp->threadData);
217 #endif
218  return comp;
219 }

251 fmiStatus fmiSetReal(fmiComponent c, const fmiValueReference vr[], size_t nvr, const fmiReal value[])
252 {
253  unsigned int i=0;
254  ModelInstance* comp = (ModelInstance *)c;
255  if (invalidState(comp, "fmiSetReal", modelInstantiated|modelInitialized))
256  return fmiError;
257  if (nvr>0 && nullPointer(comp, "fmiSetReal", "vr[]", vr))
258  return fmiError;
259  if (nvr>0 && nullPointer(comp, "fmiSetReal", "value[]", value))
260  return fmiError;
261  if (comp->loggingOn) comp->functions.logger(c, comp->instanceName, fmiOK, "log",
262  "fmiSetReal: nvr = %d", nvr);
263  // no check wether setting the value is allowed in the current state
264  for (i=0; i<nvr; i++) {
265  if (vrOutOfRange(comp, "fmiSetReal", vr[i], NUMBER_OF_REALS+NUMBER_OF_STATES))
266  return fmiError;
267  if (comp->loggingOn) comp->functions.logger(c, comp->instanceName, fmiOK, "log",
268  "fmiSetReal: #r%d# = %.16g", vr[i], value[i]);
269  if (setReal(comp, vr[i],value[i]) != fmiOK) // to be implemented by the includer of this file
270  return fmiError;
271  }
272 
273  return fmiOK;
274 }

685 fmiStatus fmiEventUpdate(fmiComponent c, fmiBoolean intermediateResults, fmiEventInfo* eventInfo)
686 {
687  int i;
688  ModelInstance* comp = (ModelInstance *)c;
689  threadData_t *threadData = comp->threadData;
690  if (invalidState(comp, "fmiEventUpdate", modelInitialized))
691  return fmiError;
692  if (nullPointer(comp, "fmiEventUpdate", "eventInfo", eventInfo))
693  return fmiError;
694  eventInfo->stateValuesChanged = fmiFalse;
695 
696  if (comp->loggingOn) comp->functions.logger(c, comp->instanceName, fmiOK, "log",
697  "fmiEventUpdate: Start Event Update! Next Sample Event %g", eventInfo->nextEventTime);
698 
699  /* try */
700  MMC_TRY_INTERNAL(simulationJumpBuffer)
701 
702 #if !defined(OMC_NO_STATESELECTION)
703  if (stateSelection(comp->fmuData, threadData, 1, 1))
704  {
705  if (comp->loggingOn) comp->functions.logger(c, comp->instanceName, fmiOK, "log",
706  "fmiEventUpdate: Need to iterate state values changed!");
707  /* if new set is calculated reinit the solver */
708  eventInfo->stateValuesChanged = fmiTrue;
709  }
710 #endif
711 
712  storePreValues(comp->fmuData);
713 
714  /* activate sample event */
715  for(i=0; i<comp->fmuData->modelData->nSamples; ++i)
716  {
717  if(comp->fmuData->simulationInfo->nextSampleTimes[i] <= comp->fmuData->localData[0]->timeValue)
718  {
719  comp->fmuData->simulationInfo->samples[i] = 1;
720  infoStreamPrint(LOG_EVENTS, 0, "[%ld] sample(%g, %g)", comp->fmuData->modelData->samplesInfo[i].index, comp->fmuData->modelData->samplesInfo[i].start, comp->fmuData->modelData->samplesInfo[i].interval);
721  }
722  }
723 
724  comp->fmuData->callback->functionDAE(comp->fmuData, threadData);
725 
726  /* deactivate sample events */
727  for(i=0; i<comp->fmuData->modelData->nSamples; ++i)
728  {
729  if(comp->fmuData->simulationInfo->samples[i])
730  {
731  comp->fmuData->simulationInfo->samples[i] = 0;
732  comp->fmuData->simulationInfo->nextSampleTimes[i] += comp->fmuData->modelData->samplesInfo[i].interval;
733  }
734  }
735 
736  for(i=0; i<comp->fmuData->modelData->nSamples; ++i)
737  if((i == 0) || (comp->fmuData->simulationInfo->nextSampleTimes[i] < comp->fmuData->simulationInfo->nextSampleEvent))
738  comp->fmuData->simulationInfo->nextSampleEvent = comp->fmuData->simulationInfo->nextSampleTimes[i];
739 
740  if (checkForDiscreteChanges(comp->fmuData, threadData) || comp->fmuData->simulationInfo->needToIterate || checkRelations(comp->fmuData) || eventInfo->stateValuesChanged)
741  {
742  intermediateResults = fmiTrue;
743  if (comp->loggingOn)
744  comp->functions.logger(c, comp->instanceName, fmiOK, "log", "fmiEventUpdate: Need to iterate(discrete changes)!");
745  eventInfo->iterationConverged = fmiTrue;
746  eventInfo->stateValueReferencesChanged = fmiFalse;
747  eventInfo->stateValuesChanged = fmiTrue;
748  eventInfo->terminateSimulation = fmiFalse;
749  }
750  else
751  {
752  intermediateResults = fmiFalse;
753  eventInfo->iterationConverged = fmiTrue;
754  eventInfo->stateValueReferencesChanged = fmiFalse;
755  eventInfo->terminateSimulation = fmiFalse;
756  }
757 
758  /* due to an event overwrite old values */
759  overwriteOldSimulationData(comp->fmuData);
760 
761  /* TODO: check the event iteration for relation
762  * in fmi import and export. This is an workaround,
763  * since the iteration seem not starting.
764  */
765  storePreValues(comp->fmuData);
766  updateRelationsPre(comp->fmuData);
767 
768  if (comp->loggingOn)
769  comp->functions.logger(c, comp->instanceName, fmiOK, "log", "fmiEventUpdate: intermediateResults = %d", intermediateResults);
770 
771  //Get Next Event Time
772  double nextSampleEvent=0;
773  nextSampleEvent = getNextSampleTimeFMU(comp->fmuData);
774  if (nextSampleEvent == -1)
775  {
776  eventInfo->upcomingTimeEvent = fmiFalse;
777  }
778  else
779  {
780  eventInfo->upcomingTimeEvent = fmiTrue;
781  eventInfo->nextEventTime = nextSampleEvent;
782  }
783  if (comp->loggingOn)
784  comp->functions.logger(c, comp->instanceName, fmiOK, "log", "fmiEventUpdate: Checked for Sample Events! Next Sample Event %g",eventInfo->nextEventTime);
785 
786  return fmiOK;
787 
788  /* catch */
789  MMC_CATCH_INTERNAL(simulationJumpBuffer)
790 
791  comp->functions.logger(c, comp->instanceName, fmiError, "error", "fmiEventUpdate: terminated by an assertion.");
