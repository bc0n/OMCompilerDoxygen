116 /*
117  * -----------------------------------------------------------------
118  * Function : IDASpbcg
119  * -----------------------------------------------------------------
120  * This routine initializes the memory record and sets various function
121  * fields specific to the IDASPBCG linear solver module.
122  *
123  * IDASpbcg first calls the existing lfree routine if this is not NULL.
124  * It then sets the ida_linit, ida_lsetup, ida_lsolve, ida_lperf, and
125  * ida_lfree fields in (*IDA_mem) to be IDASpbcgInit, IDASpbcgSetup,
126  * IDASpbcgSolve, IDASpbcgPerf, and IDASpbcgFree, respectively.
127  * It allocates memory for a structure of type IDASpilsMemRec and sets
128  * the ida_lmem field in (*IDA_mem) to the address of this structure.
129  * It sets setupNonNull in (*IDA_mem). It then sets various fields
130  * in the IDASpilsMemRec structure. Finally, IDASpbcg allocates memory
131  * for ytemp, yptemp, and xx, and calls SpbcgMalloc to allocate memory
132  * for the Spbcg solver.
133  *
134  * The return value of IDASpbcg is:
135  * IDASPILS_SUCCESS = 0 if successful
136  * IDASPILS_MEM_FAIL = -1 if IDA_mem is NULL or a memory
137  * allocation failed
138  * IDASPILS_ILL_INPUT = -2 if a required vector operation is not
139  * implemented.
140  * -----------------------------------------------------------------
141  */
142 
143 int IDASpbcg(void *ida_mem, int maxl)
144 {
145  IDAMem IDA_mem;
146  IDASpilsMem idaspils_mem;
147  SpbcgMem spbcg_mem;
148  int flag, maxl1;
149 
150  /* Return immediately if ida_mem is NULL */
151  if (ida_mem == NULL) {
152  IDAProcessError(NULL, IDASPILS_MEM_NULL, "IDASPBCG", "IDASpbcg", MSGS_IDAMEM_NULL);
153  return(IDASPILS_MEM_NULL);
154  }
155  IDA_mem = (IDAMem) ida_mem;
156 
157  /* Check if N_VDotProd is present */
158  if (vec_tmpl->ops->nvdotprod == NULL) {
159  IDAProcessError(NULL, IDASPILS_ILL_INPUT, "IDASPBCG", "IDASpbcg", MSGS_BAD_NVECTOR);
160  return(IDASPILS_ILL_INPUT);
161  }
162 
163  if (lfree != NULL) flag = lfree((IDAMem) ida_mem);
164 
165  /* Set five main function fields in ida_mem */
166  linit = IDASpbcgInit;
167  lsetup = IDASpbcgSetup;
168  lsolve = IDASpbcgSolve;
169  lperf = IDASpbcgPerf;
170  lfree = IDASpbcgFree;
171 
172  /* Get memory for IDASpilsMemRec */
173  idaspils_mem = NULL;
174  idaspils_mem = (IDASpilsMem) malloc(sizeof(struct IDASpilsMemRec));
175  if (idaspils_mem == NULL) {
176  IDAProcessError(NULL, IDASPILS_MEM_FAIL, "IDASPBCG", "IDASpbcg", MSGS_MEM_FAIL);
177  return(IDASPILS_MEM_FAIL);
178  }
179 
180  /* Set ILS type */
181  idaspils_mem->s_type = SPILS_SPBCG;
182 
183  /* Set SPBCG parameters that were passed in call sequence */
184  maxl1 = (maxl <= 0) ? IDA_SPILS_MAXL : maxl;
185  idaspils_mem->s_maxl = maxl1;
186 
187  /* Set defaults for Jacobian-related fileds */
188  jtimesDQ = TRUE;
189  jtimes = NULL;
190  jdata = NULL;
191 
192  /* Set defaults for preconditioner-related fields */
193  idaspils_mem->s_pset = NULL;
194  idaspils_mem->s_psolve = NULL;
195  idaspils_mem->s_pfree = NULL;
196  idaspils_mem->s_pdata = IDA_mem->ida_user_data;
197 
198  /* Set default values for the rest of the Spbcg parameters */
199  idaspils_mem->s_eplifac = PT05;
200  idaspils_mem->s_dqincfac = ONE;
201 
202  idaspils_mem->s_last_flag = IDASPILS_SUCCESS;
203 
204  /* Set setupNonNull to FALSE */
205  setupNonNull = FALSE;
206 
207  /* Allocate memory for ytemp, yptemp, and xx */
208 
209  ytemp = N_VClone(vec_tmpl);
210  if (ytemp == NULL) {
211  IDAProcessError(NULL, IDASPILS_MEM_FAIL, "IDASPBCG", "IDASpbcg", MSGS_MEM_FAIL);
212  free(idaspils_mem); idaspils_mem = NULL;
213  return(IDASPILS_MEM_FAIL);
214  }
215 
216  yptemp = N_VClone(vec_tmpl);
217  if (yptemp == NULL) {
218  IDAProcessError(NULL, IDASPILS_MEM_FAIL, "IDASPBCG", "IDASpbcg", MSGS_MEM_FAIL);
219  N_VDestroy(ytemp);
220  free(idaspils_mem); idaspils_mem = NULL;
221  return(IDASPILS_MEM_FAIL);
222  }
223 
224  xx = N_VClone(vec_tmpl);
225  if (xx == NULL) {
226  IDAProcessError(NULL, IDASPILS_MEM_FAIL, "IDASPBCG", "IDASpbcg", MSGS_MEM_FAIL);
227  N_VDestroy(ytemp);
228  N_VDestroy(yptemp);
229  free(idaspils_mem); idaspils_mem = NULL;
230  return(IDASPILS_MEM_FAIL);
231  }
232 
233  /* Compute sqrtN from a dot product */
234  N_VConst(ONE, ytemp);
235  sqrtN = SUNRsqrt(N_VDotProd(ytemp, ytemp));
236 
237  /* Call SpbcgMalloc to allocate workspace for Spbcg */
238  spbcg_mem = NULL;
239  spbcg_mem = SpbcgMalloc(maxl1, vec_tmpl);
240  if (spbcg_mem == NULL) {
241  IDAProcessError(NULL, IDASPILS_MEM_FAIL, "IDASPBCG", "IDASpbcg", MSGS_MEM_FAIL);
242  N_VDestroy(ytemp);
243  N_VDestroy(yptemp);
244  N_VDestroy(xx);
245  free(idaspils_mem); idaspils_mem = NULL;
246  return(IDASPILS_MEM_FAIL);
247  }
248 
249  /* Attach SPBCG memory to spils memory structure */
250  spils_mem = (void *)spbcg_mem;
251 
252  /* Attach linear solver memory to the integrator memory */
253  lmem = idaspils_mem;
254 
255  return(IDASPILS_SUCCESS);
256 }

332 /*
333  * -----------------------------------------------------------------
334  * Function : IDASpbcgSolve
335  * -----------------------------------------------------------------
336  * Note: The x-scaling and b-scaling arrays are both equal to weight.
337  *
338  * We set the initial guess, x = 0, then call SpbcgSolve.
339  * We copy the solution x into b, and update the counters nli, nps,
340  * and ncfl. If SpbcgSolve returned nli_inc = 0 (hence x = 0), we
341  * take the SPBCG vtemp vector (= P_inverse F) as the correction
342  * vector instead. Finally, we set the return value according to the
343  * success of SpbcgSolve.
344  * -----------------------------------------------------------------
345  */
346 
347 static int IDASpbcgSolve(IDAMem IDA_mem, N_Vector bb, N_Vector weight,
348  N_Vector yy_now, N_Vector yp_now, N_Vector rr_now)
349 {
350  IDASpilsMem idaspils_mem;
351  SpbcgMem spbcg_mem;
352  int pretype, nli_inc, nps_inc, retval;
353  realtype res_norm;
354 
355  idaspils_mem = (IDASpilsMem) lmem;
356 
357  spbcg_mem = (SpbcgMem)spils_mem;
358 
359  /* Set SpbcgSolve convergence test constant epslin, in terms of the
360  Newton convergence test constant epsNewt and safety factors. The factor
361  sqrt(Neq) assures that the Bi-CGSTAB convergence test is applied to the
362  WRMS norm of the residual vector, rather than the weighted L2 norm. */
363  epslin = sqrtN*eplifac*epsNewt;
364 
365  /* Set vectors ycur, ypcur, and rcur for use by the Atimes and Psolve */
366  ycur = yy_now;
367  ypcur = yp_now;
368  rcur = rr_now;
369 
370  /* Set SpbcgSolve inputs pretype and initial guess xx = 0 */
371  pretype = (psolve == NULL) ? PREC_NONE : PREC_LEFT;
372  N_VConst(ZERO, xx);
373 
374  /* Call SpbcgSolve and copy xx to bb */
375  retval = SpbcgSolve(spbcg_mem, IDA_mem, xx, bb, pretype, epslin,
376  IDA_mem, weight, weight, IDASpilsAtimes,
377  IDASpilsPSolve, &res_norm, &nli_inc, &nps_inc);
378  last_flag = retval;
379  if (nli_inc == 0) N_VScale(ONE, SPBCG_VTEMP(spbcg_mem), bb);
380  else N_VScale(ONE, xx, bb);
381 
382  /* Increment counters nli, nps, and return if successful */
383  nli += nli_inc;
384  nps += nps_inc;
385  if (retval != SPBCG_SUCCESS) ncfl++;
386 
387  /* Interpret return value from SpbcgSolve */
388 
389  last_flag = retval;
390 
391  switch(retval) {
392 
393  case SPBCG_SUCCESS:
394  return(0);
395  break;
396  case SPBCG_RES_REDUCED:
397  return(1);
398  break;
399  case SPBCG_CONV_FAIL:
400  return(1);
401  break;
402  case SPBCG_PSOLVE_FAIL_REC:
403  return(1);
404  break;
405  case SPBCG_ATIMES_FAIL_REC:
406  return(1);
407  break;
408  case SPBCG_MEM_NULL:
409  return(-1);
410  break;
411  case SPBCG_ATIMES_FAIL_UNREC:
412  IDAProcessError(IDA_mem, SPBCG_ATIMES_FAIL_UNREC, "IDaSPBCG", "IDASpbcgSolve", MSGS_JTIMES_FAILED);
413  return(-1);
414  break;
415  case SPBCG_PSOLVE_FAIL_UNREC:
416  IDAProcessError(IDA_mem, SPBCG_PSOLVE_FAIL_UNREC, "IDASPBCG", "IDASpbcgSolve", MSGS_PSOLVE_FAILED);
417  return(-1);
418  break;
419  }
420 
421  return(0);
422 }

424 /*
425  * -----------------------------------------------------------------
426  * Function : IDASpbcgPerf
427  * -----------------------------------------------------------------
428  * This routine handles performance monitoring specific to the
429  * IDASPBCG linear solver. When perftask = 0, it saves values of
430  * various counters. When perftask = 1, it examines difference
431  * quotients in these counters, and depending on their values, it
432  * prints up to three warning messages. Messages are printed up to
433  * a maximum of 10 times.
434  * -----------------------------------------------------------------
435  */
436 
437 static int IDASpbcgPerf(IDAMem IDA_mem, int perftask)
438 {
439  IDASpilsMem idaspils_mem;
440  realtype avdim, rcfn, rcfl;
441  long int nstd, nnid;
442  booleantype lavd, lcfn, lcfl;
443 
444  idaspils_mem = (IDASpilsMem) lmem;
445 
446  if (perftask == 0) {
447  nst0 = nst; nni0 = nni; nli0 = nli;
448  ncfn0 = ncfn; ncfl0 = ncfl;
449  nwarn = 0;
450  return(0);
451  }
452 
453  nstd = nst - nst0; nnid = nni - nni0;
454  if (nstd == 0 || nnid == 0) return(0);
455  avdim = (realtype) ((nli - nli0)/((realtype) nnid));
456  rcfn = (realtype) ((ncfn - ncfn0)/((realtype) nstd));
457  rcfl = (realtype) ((ncfl - ncfl0)/((realtype) nnid));
458  lavd = (avdim > ((realtype) maxl));
459  lcfn = (rcfn > PT9);
460  lcfl = (rcfl > PT9);
461  if (!(lavd || lcfn || lcfl)) return(0);
462  nwarn++;
463  if (nwarn > 10) return(1);
464  if (lavd)
465  IDAProcessError(IDA_mem, IDA_WARNING, "IDASPBCG", "IDASpbcgPerf", MSGS_AVD_WARN, tn, avdim);
466  if (lcfn)
467  IDAProcessError(IDA_mem, IDA_WARNING, "IDASPBCG", "IDASpbcgPerf", MSGS_CFN_WARN, tn, rcfn);
468  if (lcfl)
469  IDAProcessError(IDA_mem, IDA_WARNING, "IDASPBCG", "IDASpbcgPerf", MSGS_CFL_WARN, tn, rcfl);
470 
471  return(0);
472 }

509  * IDASpbcgB
510  *
511  * Wrapper for the backward phase

514 
515 int IDASpbcgB(void *ida_mem, int which, int maxlB)
516 {
517  IDAMem IDA_mem;
518  IDAadjMem IDAADJ_mem;
519  IDABMem IDAB_mem;
520  IDASpilsMemB idaspilsB_mem;
521  void *ida_memB;
522  int flag;
523 
524  /* Check if ida_mem is allright. */
525  if (ida_mem == NULL) {
526  IDAProcessError(NULL, IDASPILS_MEM_NULL, "IDASPBCG", "IDASpbcgB", MSGS_IDAMEM_NULL);
527  return(IDASPILS_MEM_NULL);
528  }
529  IDA_mem = (IDAMem) ida_mem;
530 
531  /* Is ASA initialized? */
532  if (IDA_mem->ida_adjMallocDone == FALSE) {
533  IDAProcessError(IDA_mem, IDASPILS_NO_ADJ, "IDASPBCG", "IDASpbcgB", MSGS_NO_ADJ);
534  return(IDASPILS_NO_ADJ);
535  }
536  IDAADJ_mem = IDA_mem->ida_adj_mem;
537 
538  /* Check the value of which */
539  if ( which >= IDAADJ_mem->ia_nbckpbs ) {
540  IDAProcessError(IDA_mem, IDASPILS_ILL_INPUT, "IDASPBCG", "IDASpbcgB", MSGS_BAD_WHICH);
541  return(IDASPILS_ILL_INPUT);
542  }
543 
544  /* Find the IDABMem entry in the linked list corresponding to 'which'. */
545  IDAB_mem = IDAADJ_mem->IDAB_mem;
546  while (IDAB_mem != NULL) {
547  if( which == IDAB_mem->ida_index ) break;
548  /* advance */
549  IDAB_mem = IDAB_mem->ida_next;
550  }
551  /* ida_mem corresponding to 'which' problem. */
552  ida_memB = (void *) IDAB_mem->IDA_mem;
553 
554 
555  /* Get memory for IDASpilsMemRecB */
556  idaspilsB_mem = NULL;
557  idaspilsB_mem = (IDASpilsMemB) malloc(sizeof(struct IDASpilsMemRecB));
558  if (idaspilsB_mem == NULL) {
559  IDAProcessError(IDA_mem, IDASPILS_MEM_FAIL, "IDASPBCG", "IDASpbcgB", MSGS_MEM_FAIL);
560  return(IDASPILS_MEM_FAIL);
561  }
562 
563  idaspilsB_mem->s_psetB = NULL;
564  idaspilsB_mem->s_psolveB = NULL;
565  idaspilsB_mem->s_P_dataB = NULL;
566 
567  /* initialize Jacobian function */
568  idaspilsB_mem->s_jtimesB = NULL;
569 
570  /* attach lmem and lfree */
571  IDAB_mem->ida_lmem = idaspilsB_mem;
572  IDAB_mem->ida_lfree = IDASpbcgFreeB;
573 
574  flag = IDASpbcg(IDAB_mem->IDA_mem, maxlB);
575 
576  if (flag != IDASPILS_SUCCESS) {
577  free(idaspilsB_mem);
578  idaspilsB_mem = NULL;
579  }
580 
581  return(flag);
582 }
