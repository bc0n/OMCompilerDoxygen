166 void SolverDefaultImplementation::setZeroState()
167 {
168  // Reset Zero-State
169  _zeroStatus = ISolver::UNCHANGED_SIGN;;
170 
171  // Alle Elemente im ZeroFunction-Array durchgehen
172  for (int i=0; i<_dimZeroFunc; ++i)
173  {
174  // Überprüfung auf Vorzeichenwechsel
175  if ((_zeroVal[i] < 0 && _zeroValLastSuccess[i] > 0) || (_zeroVal[i] > 0 && _zeroValLastSuccess[i] < 0))
176  {
177  // Vorzeichenwechsel, aber Eintrag ist größer (oder kleiner) als Toleranzbereich
178  _zeroStatus = ISolver::EQUAL_ZERO;
179 
180  // Rest ZeroSign
181  _events[i] = true;
182 
183  // Zeitpunkt des letzten verworfenen Schrittes abspeichern
184  _tLastUnsucess = _tCurrent;
185  break;
186  }
187  else
188  _events[i] = false;
189  }
