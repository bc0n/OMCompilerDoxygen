19 c performs the matrix sum C = A+B.
20 c-----------------------------------------------------------------------
21 c on entry:
22 c ---------
23 c nrow = integer. The row dimension of A and B
24 c ncol = integer. The column dimension of A and B.
25 c job = integer. Job indicator. When job = 0, only the structure
26 c (i.e. the arrays jc, ic) is computed and the
27 c real values are ignored.
28 c
29 c a,
30 c ja,
31 c ia = Matrix A in compressed sparse row format.
32 c
33 c b,
34 c jb,
35 c ib = Matrix B in compressed sparse row format.
36 c
37 c nzmax = integer. The length of the arrays c and jc.

39 c of elements that exceeds exceeds nzmax. See ierr.
40 c
41 c on return:
42 c----------
43 c c,

46 c

61  ierr = 0

63  ic(1) = 1
64  do 1 j=1, ncol
65  iw(j) = 0
66  1 continue
67 c
68  do 500 ii=1, nrow
69 c row i
70  do 200 ka=ia(ii), ia(ii+1)-1

77  200 continue
78 c

83  len = len+1

86  if (values) c(len) = b(kb)
87  iw(jcol)= len
88  else
89  if (values) c(jpos) = c(jpos) + b(kb)
90  endif
91  300 continue
92  do 301 k=ic(ii), len
93  iw(jc(k)) = 0
94  301 continue
95  ic(ii+1) = len+1
96  500 continue
97  return
98  999 ierr = ii
99  return

318 c on entry:
319 c ---------

321 c
322 c job = integer. job indicator. Job=0 means get array b only

330 c
331 c on return:

333 c

422  subroutine aplsca (nrow, a, ja, ia, scal,iw)

424  integer ja(*), ia(nrow+1),iw(*)
425 c-----------------------------------------------------------------------
426 c Adds a scalar to the diagonal entries of a sparse matrix A :=A + s I
427 c-----------------------------------------------------------------------
428 c on entry:
429 c ---------
430 c nrow = integer. The row dimension of A
431 c
432 c a,
433 c ja,
434 c ia = Matrix A in compressed sparse row format.
435 c
436 c scal = real. scalar to add to the diagonal entries.
437 c
438 c on return:
439 c----------
440 c
441 c a,
442 c ja,
443 c ia = matrix A with diagonal elements shifted (or created).
444 c
445 c iw = integer work array of length n. On return iw will
446 c contain the positions of the diagonal entries in the
447 c output matrix. (i.e., a(iw(k)), ja(iw(k)), k=1,...n,
448 c are the values/column indices of the diagonal elements
449 c of the output matrix. ).
450 c
451 c Notes:
452 c-------
453 c The column dimension of A is not needed.
454 c important: the matrix a may be expanded slightly to allow for
455 c additions of nonzero elements to previously nonexisting diagonals.
456 c The is no checking as to whether there is enough space appended
457 c to the arrays a and ja. if not sure allow for n additional
458 c elemnts.
459 c coded by Y. Saad. Latest version July, 19, 1990
460 c-----------------------------------------------------------------------
461  logical test
462 c
463  call diapos (nrow,ja,ia,iw)
464  icount = 0
465  do 1 j=1, nrow
466  if (iw(j) .eq. 0) then
467  icount = icount+1
468  else
469  a(iw(j)) = a(iw(j)) + scal
470  endif
471  1 continue
472 c
473 c if no diagonal elements to insert in data structure return.
474 c
475  if (icount .eq. 0) return
476 c
477 c shift the nonzero elements if needed, to allow for created
478 c diagonal elements.
479 c
480  ko = ia(nrow+1)+icount
481 c
482 c copy rows backward
483 c
484  do 5 ii=nrow, 1, -1
485 c
486 c go through row ii
487 c
488  k1 = ia(ii)
489  k2 = ia(ii+1)-1
490  ia(ii+1) = ko
491  test = (iw(ii) .eq. 0)
492  do 4 k = k2,k1,-1
493  j = ja(k)
494  if (test .and. (j .lt. ii)) then
495  test = .false.
496  ko = ko - 1
497  a(ko) = scal
498  ja(ko) = ii
499  iw(ii) = ko
500  endif
501  ko = ko-1
502  a(ko) = a(k)
503  ja(ko) = j
504  4 continue
505 c diagonal element has not been added yet.
506  if (test) then
507  ko = ko-1
508  a(ko) = scal
509  ja(ko) = ii
510  iw(ii) = ko
511  endif
512  5 continue
513  ia(1) = ko
514  return
515 c-----------------------------------------------------------------------
516 c----------end-of-aplsca------------------------------------------------
517  end

579 c-----------------------------------------------------------------------
580 c writes a matrix in Harwell-Boeing format into a file.
581 c assumes that the matrix is stored in COMPRESSED SPARSE COLUMN FORMAT.
582 c some limited functionality for right hand sides.
583 c Author: Youcef Saad - Date: Sept., 1989 - updated Oct. 31, 1989 to
584 c cope with new format.
585 c-----------------------------------------------------------------------
586 c on entry:
587 c---------
588 c nrow = number of rows in matrix
589 c ncol = number of columns in matrix
590 c a = real*8 array containing the values of the matrix stored
591 c columnwise
592 c ja = integer array of the same length as a containing the column
593 c indices of the corresponding matrix elements of array a.
594 c ia = integer array of containing the pointers to the beginning of
595 c the row in arrays a and ja.
596 c rhs = real array containing the right-hand-side (s) and optionally
597 c the associated initial guesses and/or exact solutions
598 c in this order. See also guesol for details. the vector rhs will
599 c be used only if job .gt. 2 (see below). Only full storage for
600 c the right hand sides is supported.
601 c
602 c guesol = a 2-character string indicating whether an initial guess
603 c (1-st character) and / or the exact solution (2-nd)
604 c character) is provided with the right hand side.
605 c if the first character of guesol is 'G' it means that an
606 c an intial guess is provided for each right-hand sides.
607 c These are assumed to be appended to the right hand-sides in
608 c the array rhs.
609 c if the second character of guesol is 'X' it means that an
610 c exact solution is provided for each right-hand side.
611 c These are assumed to be appended to the right hand-sides
612 c and the initial guesses (if any) in the array rhs.
613 c
614 c title = character*72 = title of matrix test ( character a*72 ).
615 c key = character*8 = key of matrix
616 c type = charatcer*3 = type of matrix.
617 c
618 c ifmt = integer specifying the format chosen for the real values
619 c to be output (i.e., for a, and for rhs-guess-sol if
620 c applicable). The meaning of ifmt is as follows.
621 c * if (ifmt .lt. 100) then the D descriptor is used,
622 c format Dd.m, in which the length (m) of the mantissa is
623 c precisely the integer ifmt (and d = ifmt+6)
624 c * if (ifmt .gt. 100) then prtmt will use the
625 c F- descriptor (format Fd.m) in which the length of the
626 c mantissa (m) is the integer mod(ifmt,100) and the length
627 c of the integer part is k=ifmt/100 (and d = k+m+2)
628 c Thus ifmt= 4 means D10.4 +.xxxxD+ee while
629 c ifmt=104 means F7.4 +x.xxxx
630 c ifmt=205 means F9.5 +xx.xxxxx
631 c Note: formats for ja, and ia are internally computed.
632 c
633 c job = integer to indicate whether matrix values and
634 c a right-hand-side is available to be written
635 c job = 1 write srtucture only, i.e., the arrays ja and ia.
636 c job = 2 write matrix including values, i.e., a, ja, ia
637 c job = 3 write matrix and one right hand side: a,ja,ia,rhs.
638 c job = nrhs+2 write matrix and nrhs successive right hand sides
639 c Note that there cannot be any right-hand-side if the matrix
640 c has no values. Also the initial guess and exact solutions when
641 c provided are for each right hand side. For example if nrhs=2
642 c and guesol='GX' there are 6 vectors to write.
643 c
644 c
645 c iounit = logical unit number where to write the matrix into.
646 c
647 c on return:
648 c----------
649 c the matrix a, ja, ia will be written in output unit iounit
650 c in the Harwell-Boeing format. None of the inputs is modofied.
651 c
652 c Notes: 1) This code attempts to pack as many elements as possible per
653 c 80-character line.
654 c 2) this code attempts to avoid as much as possible to put
655 c blanks in the formats that are written in the 4-line header
656 c (This is done for purely esthetical reasons since blanks
657 c are ignored in format descriptors.)
658 c 3) sparse formats for right hand sides and guesses are not
659 c supported.
660 c-----------------------------------------------------------------------
661  character title*72,key*8,type*3,ptrfmt*16,indfmt*16,valfmt*20,
662  * guesol*2, rhstyp*3
663  integer totcrd, ptrcrd, indcrd, valcrd, rhscrd, nrow, ncol,
664  1 nnz, nrhs, len, nperli, nrwindx
665  integer ja(*), ia(*)
666  real*8 a(*),rhs(*)

668 c compute pointer format
669 c--------------
670  nnz = ia(ncol+1) -1
671  len = int( alog10(0.1+real(nnz+1))) + 1
672  nperli = 80/len
673  ptrcrd = ncol/nperli + 1
674  if (len .gt. 9) then
675  assign 101 to ix
676  else
677  assign 100 to ix

936  subroutine coicsr (n,nnz,job,a,ja,ia,iwk)
937  integer ia(nnz),ja(nnz),iwk(n+1)
938  real*8 a(*)
939 c------------------------------------------------------------------------
940 c IN-PLACE coo-csr conversion routine.
941 c------------------------------------------------------------------------
942 c this subroutine converts a matrix stored in coordinate format into
943 c the csr format. The conversion is done in place in that the arrays
944 c a,ja,ia of the result are overwritten onto the original arrays.
945 c------------------------------------------------------------------------
946 c on entry:
947 c---------
948 c n = integer. row dimension of A.
949 c nnz = integer. number of nonzero elements in A.
950 c job = integer. Job indicator. when job=1, the real values in a are
951 c filled. Otherwise a is not touched and the structure of the
952 c array only (i.e. ja, ia) is obtained.
953 c a = real array of size nnz (number of nonzero elements in A)
954 c containing the nonzero elements
955 c ja = integer array of length nnz containing the column positions
956 c of the corresponding elements in a.
957 c ia = integer array of length nnz containing the row positions
958 c of the corresponding elements in a.
959 c iwk = integer work array of length n+1
960 c on return:
961 c----------
962 c a
963 c ja
964 c ia = contains the compressed sparse row data structure for the
965 c resulting matrix.
966 c Note:
967 c-------
968 c the entries of the output matrix are not sorted (the column
969 c indices in each are not in increasing order) use coocsr
970 c if you want them sorted.
971 c----------------------------------------------------------------------c
972 c Coded by Y. Saad, Sep. 26 1989 c
973 c----------------------------------------------------------------------c
974  real*8 t,tnext
975  logical values
976 c-----------------------------------------------------------------------
977  values = (job .eq. 1)
978 c find pointer array for resulting matrix.
979  do 35 i=1,n+1
980  iwk(i) = 0
981  35 continue
982  do 4 k=1,nnz
983  i = ia(k)
984  iwk(i+1) = iwk(i+1)+1
985  4 continue
986 c------------------------------------------------------------------------
987  iwk(1) = 1
988  do 44 i=2,n
989  iwk(i) = iwk(i-1) + iwk(i)
990  44 continue
991 c
992 c loop for a cycle in chasing process.
993 c
994  init = 1
995  k = 0
996  5 if (values) t = a(init)
997  i = ia(init)
998  j = ja(init)
999  ia(init) = -1
1000 c------------------------------------------------------------------------
1001  6 k = k+1
1002 c current row number is i. determine where to go.
1003  ipos = iwk(i)
1004 c save the chased element.
1005  if (values) tnext = a(ipos)
1006  inext = ia(ipos)
1007  jnext = ja(ipos)
1008 c then occupy its location.
1009  if (values) a(ipos) = t
1010  ja(ipos) = j
1011 c update pointer information for next element to come in row i.
1012  iwk(i) = ipos+1
1013 c determine next element to be chased,
1014  if (ia(ipos) .lt. 0) goto 65
1015  t = tnext
1016  i = inext
1017  j = jnext
1018  ia(ipos) = -1
1019  if (k .lt. nnz) goto 6
1020  goto 70
1021  65 init = init+1
1022  if (init .gt. nnz) goto 70
1023  if (ia(init) .lt. 0) goto 65
1024 c restart chasing --
1025  goto 5
1026  70 do 80 i=1,n
1027  ia(i+1) = iwk(i)
1028  80 continue
1029  ia(1) = 1
1030  return
1031 c----------------- end of coicsr ----------------------------------------
1032 c------------------------------------------------------------------------
1033  end

1038 c-----------------------------------------------------------------------

1042 c a, ir, jc into a row general sparse ao, jao, iao format.
1043 c

1058 c

1060 c ja,

1065 c

1069 c
1070 c nnz = number of nonzero elements in matrix.
1071 c ierr = integer error indicator.

1073 c ierr .eq. 1 means that the the code stopped

1076 c
1077 c NOTES: 1)This routine is PARTIALLY in place: csrcoo can be called with
1078 c ao being the same array as as a, and jc the same array as ja.
1079 c but ir CANNOT be the same as ia.
1080 c 2) note the order in the output arrays,
1081 c------------------------------------------------------------------------
1082  ierr = 0
1083  nnz = ia(nrow+1)-1
1084  if (nnz .gt. nzmax) then
1085  ierr = 1
1086  return
1087  endif

1089  goto (3,2,1) job
1090  1 do 10 k=1,nnz

1095  11 continue

1097 c copy backward to allow for in-place processing.
1098 c
1099  3 do 13 i=nrow,1,-1
1100  k1 = ia(i+1)-1
1101  k2 = ia(i)
1102  do 12 k=k1,k2,-1
1103  ir(k) = i
1104  12 continue
1105  13 continue
1106  return

1108 c-----------------------------------------------------------------------
1109  end

1227  subroutine csrdia (n,idiag,job,a,ja,ia,ndiag,
1228  * diag,ioff,ao,jao,iao,ind)
1229  real*8 diag(ndiag,idiag), a(*), ao(*)
1230  integer ia(*), ind(*), ja(*), jao(*), iao(*), ioff(*)
1231 c-----------------------------------------------------------------------
1232 c Compressed sparse row to diagonal format
1233 c-----------------------------------------------------------------------
1234 c this subroutine extracts idiag diagonals from the input matrix a,
1235 c a, ia, and puts the rest of the matrix in the output matrix ao,
1236 c jao, iao. The diagonals to be extracted depend on the value of job
1237 c (see below for details.) In the first case, the diagonals to be
1238 c extracted are simply identified by their offsets provided in ioff
1239 c by the caller. In the second case, the code internally determines
1240 c the idiag most significant diagonals, i.e., those diagonals of the
1241 c matrix which have the largest number of nonzero elements, and
1242 c extracts them.
1243 c-----------------------------------------------------------------------
1244 c on entry:
1245 c----------
1246 c n = dimension of the matrix a.
1247 c idiag = integer equal to the number of diagonals to be extracted.
1248 c Note: on return idiag may be modified.
1249 c a, ja,
1250 c ia = matrix stored in a, ja, ia, format
1251 c job = integer. serves as a job indicator. Job is better thought
1252 c of as a two-digit number job=xy. If the first (x) digit
1253 c is one on entry then the diagonals to be extracted are
1254 c internally determined. In this case csrdia exctracts the
1255 c idiag most important diagonals, i.e. those having the largest
1256 c number on nonzero elements. If the first digit is zero
1257 c then csrdia assumes that ioff(*) contains the offsets
1258 c of the diagonals to be extracted. there is no verification
1259 c that ioff(*) contains valid entries.
1260 c The second (y) digit of job determines whether or not
1261 c the remainder of the matrix is to be written on ao,jao,iao.
1262 c If it is zero then ao, jao, iao is not filled, i.e.,
1263 c the diagonals are found and put in array diag and the rest is
1264 c is discarded. if it is one, ao, jao, iao contains matrix
1265 c of the remaining elements.
1266 c Thus:

1268 c defined by ioff) and do not fill ao,jao,iao

1270 c and fill ao,jao,iao

1272 c and do not fill ao,jao,iao

1275 c

1277 c

1280 c

1282 c on entry.

1284 c of A on return

1293 c needed only when job.ge.10 i.e., in case the diagonals are to
1294 c be selected internally.
1295 c
1296 c Notes:
1297 c-------
1298 c 1) The algorithm is in place: ao, jao, iao can be overwritten on
1299 c a, ja, ia if desired
1300 c 2) When the code is required to select the diagonals (job .ge. 10)
1301 c the selection of the diagonals is done from left to right
1302 c as a result if several diagonals have the same weight (number
1303 c of nonzero elemnts) the leftmost one is selected first.
1304 c-----------------------------------------------------------------------
1305  job1 = job/10
1306  job2 = job-job1*10
1307  if (job1 .eq. 0) goto 50
1308  n2 = n+n-1
1309  call infdia(n,ja,ia,ind,idum)

1311 c-----------------------------------------------------------------------

1313  4 ii=ii+1
1314  jmax = 0

1316  j = ind(k)
1317  if (j .le. jmax) goto 41
1318  i = k
1319  jmax = j
1320  41 continue
1321  if (jmax .le. 0) then
1322  ii = ii-1
1323  goto 42
1324  endif

1326  ind(i) = - jmax

1328  42 idiag = ii

1333  diag(i,j) = 0.0d0
1334  54 continue
1335  55 continue

1338 c-----------------------------------------------------------------------
1339 c extract diagonals and accumulate remaining matrix.
1340 c-----------------------------------------------------------------------
1341  do 6 i=1, n
1342  do 51 k=ia(i),ia(i+1)-1
1343  j = ja(k)
1344  do 52 l=1,idiag
1345  if (j-i .ne. ioff(l)) goto 52
1346  diag(i,l) = a(k)
1347  goto 51
1348  52 continue

1350  if (job2 .eq. 0) goto 51
1351  ao(ko) = a(k)
1352  jao(ko) = j
1353  ko = ko+1
1354  51 continue
1355  if (job2 .ne. 0 ) ind(i+1) = ko
1356  6 continue
1357  if (job2 .eq. 0) return
1358 c finish with iao
1359  iao(1) = 1
1360  do 7 i=2,n+1
1361  iao(i) = ind(i)
1362  7 continue
1363  return

1367  subroutine csrbnd (n,a,ja,ia,job,abd,nabd,lowd,ml,mu,ierr)
1368  real*8 a(*),abd(nabd,n)
1369  integer ia(n+1),ja(*)
1370 c-----------------------------------------------------------------------
1371 c Compressed Sparse Row to Banded (Linpack ) format.
1372 c-----------------------------------------------------------------------
1373 c this subroutine converts a general sparse matrix stored in
1374 c compressed sparse row format into the banded format. for the
1375 c banded format,the Linpack conventions are assumed (see below).
1376 c-----------------------------------------------------------------------
1377 c on entry:
1378 c----------
1379 c n = integer,the actual row dimension of the matrix.
1380 c
1381 c a,
1382 c ja,
1383 c ia = input matrix stored in compressed sparse row format.
1384 c
1385 c job = integer. if job=1 then the values of the lower bandwith ml
1386 c and the upper bandwidth mu are determined internally.
1387 c otherwise it is assumed that the values of ml and mu
1388 c are the correct bandwidths on input. See ml and mu below.
1389 c
1390 c nabd = integer. first dimension of array abd.
1391 c
1392 c lowd = integer. this should be set to the row number in abd where
1393 c the lowest diagonal (leftmost) of A is located.
1394 c lowd should be ( 1 .le. lowd .le. nabd).
1395 c if it is not known in advance what lowd should be
1396 c enter lowd = 0 and the default value lowd = ml+mu+1
1397 c will be chosen. Alternative: call routine getbwd from unary
1398 c first to detrermione ml and mu then define lowd accordingly.
1399 c (Note: the banded solvers in linpack use lowd=2*ml+mu+1. )
1400 c
1401 c ml = integer. equal to the bandwidth of the strict lower part of A
1402 c mu = integer. equal to the bandwidth of the strict upper part of A
1403 c thus the total bandwidth of A is ml+mu+1.
1404 c if ml+mu+1 is found to be larger than lowd then an error
1405 c flag is raised (unless lowd = 0). see ierr.
1406 c
1407 c note: ml and mu are assumed to have the correct bandwidth values
1408 c as defined above if job is set to zero on entry.
1409 c
1410 c on return:
1411 c-----------
1412 c
1413 c abd = real array of dimension abd(nabd,n).
1414 c on return contains the values of the matrix stored in
1415 c banded form. The j-th column of abd contains the elements
1416 c of the j-th column of the original matrix comprised in the
1417 c band ( i in (j-ml,j+mu) ) with the lowest diagonal at
1418 c the bottom row (row lowd). See details below for this format.
1419 c
1420 c ml = integer. equal to the bandwidth of the strict lower part of A
1421 c mu = integer. equal to the bandwidth of the strict upper part of A
1422 c if job=1 on entry then these two values are internally computed.
1423 c
1424 c lowd = integer. row number in abd where the lowest diagonal
1425 c (leftmost) of A is located on return. In case lowd = 0
1426 c on return, then it is defined to ml+mu+1 on return and the
1427 c lowd will contain this value on return. `
1428 c
1429 c ierr = integer. used for error messages. On return:
1430 c ierr .eq. 0 :means normal return
1431 c ierr .eq. -1 : means invalid value for lowd. (either .lt. 0
1432 c or larger than nabd).
1433 c ierr .eq. -2 : means that lowd is not large enough and as
1434 c result the matrix cannot be stored in array abd.
1435 c lowd should be at least ml+mu+1, where ml and mu are as
1436 c provided on output.
1437 c
1438 c----------------------------------------------------------------------*
1439 c Additional details on banded format. (this closely follows the *
1440 c format used in linpack. may be useful for converting a matrix into *
1441 c this storage format in order to use the linpack banded solvers). *
1442 c----------------------------------------------------------------------*
1443 c --- band storage format for matrix abd --- *
1444 c uses ml+mu+1 rows of abd(nabd,*) to store the diagonals of *
1445 c a in rows of abd starting from the lowest (sub)-diagonal which is *
1446 c stored in row number lowd of abd. the minimum number of rows needed *
1447 c in abd is ml+mu+1, i.e., the minimum value for lowd is ml+mu+1. the *
1448 c j-th column of abd contains the elements of the j-th column of a, *
1449 c from bottom to top: the element a(j+ml,j) is stored in position *
1450 c abd(lowd,j), then a(j+ml-1,j) in position abd(lowd-1,j) and so on. *
1451 c Generally, the element a(j+k,j) of original matrix a is stored in *
1452 c position abd(lowd+k-ml,j), for k=ml,ml-1,..,0,-1, -mu. *
1453 c The first dimension nabd of abd must be .ge. lowd *
1454 c *
1455 c example [from linpack ]: if the original matrix is *
1456 c *
1457 c 11 12 13 0 0 0 *
1458 c 21 22 23 24 0 0 *
1459 c 0 32 33 34 35 0 original banded matrix *
1460 c 0 0 43 44 45 46 *
1461 c 0 0 0 54 55 56 *
1462 c 0 0 0 0 65 66 *
1463 c *
1464 c then n = 6, ml = 1, mu = 2. lowd should be .ge. 4 (=ml+mu+1) and *
1465 c if lowd = 5 for example, abd should be: *
1466 c *
1467 c untouched --> x x x x x x *
1468 c * * 13 24 35 46 *
1469 c * 12 23 34 45 56 resulting abd matrix in banded *
1470 c 11 22 33 44 55 66 format *
1471 c row lowd--> 21 32 43 54 65 * *
1472 c *
1473 c * = not used *
1474 c
1475 *
1476 c----------------------------------------------------------------------*
1477 c first determine ml and mu.
1478 c-----------------------------------------------------------------------
1479  ierr = 0
1480 c-----------
1481  if (job .eq. 1) call getbwd(n,a,ja,ia,ml,mu)
1482  m = ml+mu+1
1483  if (lowd .eq. 0) lowd = m
1484  if (m .gt. lowd) ierr = -2
1485  if (lowd .gt. nabd .or. lowd .lt. 0) ierr = -1
1486  if (ierr .lt. 0) return
1487 c------------
1488  do 15 i=1,m
1489  ii = lowd -i+1
1490  do 10 j=1,n
1491  abd(ii,j) = 0.0d0
1492  10 continue
1493  15 continue
1494 c---------------------------------------------------------------------
1495  mdiag = lowd-ml
1496  do 30 i=1,n
1497  do 20 k=ia(i),ia(i+1)-1
1498  j = ja(k)
1499  abd(i-j+mdiag,j) = a(k)
1500  20 continue
1501  30 continue
1502  return
1503 c------------- end of csrbnd -------------------------------------------
1504 c-----------------------------------------------------------------------
1505  end

2127 c-----------------------------------------------------------------------
2128  implicit none
2129  integer n
2130  real*8 a(*),alu(*),w(n),droptol
2131  integer ja(*),ia(n+1),jlu(*),ju(n),jw(2*n),lfil,iwk,ierr
2132 c----------------------------------------------------------------------*
2133 c *** ILUT preconditioner *** *
2134 c incomplete LU factorization with dual truncation mechanism *
2135 c----------------------------------------------------------------------*
2136 c Author: Yousef Saad *May, 5, 1990, Latest revision, August 1996 *
2137 c----------------------------------------------------------------------*
2138 c PARAMETERS
2139 c-----------
2140 c
2141 c on entry:
2142 c==========
2143 c n = integer. The row dimension of the matrix A. The matrix
2144 c
2145 c a,ja,ia = matrix stored in Compressed Sparse Row format.
2146 c
2147 c lfil = integer. The fill-in parameter. Each row of L and each row
2148 c of U will have a maximum of lfil elements (excluding the
2149 c diagonal element). lfil must be .ge. 0.
2150 c ** WARNING: THE MEANING OF LFIL HAS CHANGED WITH RESPECT TO
2151 c EARLIER VERSIONS.
2152 c
2153 c droptol = real*8. Sets the threshold for dropping small terms in the
2154 c factorization. See below for details on dropping strategy.
2155 c
2156 c
2157 c iwk = integer. The lengths of arrays alu and jlu. If the arrays
2158 c are not big enough to store the ILU factorizations, ilut
2159 c will stop with an error message.
2160 c
2161 c On return:
2162 c===========
2163 c
2164 c alu,jlu = matrix stored in Modified Sparse Row (MSR) format containing
2165 c the L and U factors together. The diagonal (stored in
2166 c alu(1:n) ) is inverted. Each i-th row of the alu,jlu matrix
2167 c contains the i-th row of L (excluding the diagonal entry=1)
2168 c followed by the i-th row of U.
2169 c
2170 c ju = integer array of length n containing the pointers to
2171 c the beginning of each row of U in the matrix alu,jlu.
2172 c
2173 c ierr = integer. Error message with the following meaning.
2174 c ierr = 0 --> successful return.
2175 c ierr .gt. 0 --> zero pivot encountered at step number ierr.
2176 c ierr = -1 --> Error. input matrix may be wrong.
2177 c (The elimination process has generated a
2178 c row in L or U whose length is .gt. n.)
2179 c ierr = -2 --> The matrix L overflows the array al.
2180 c ierr = -3 --> The matrix U overflows the array alu.
2181 c ierr = -4 --> Illegal value for lfil.
2182 c ierr = -5 --> zero row encountered.
2183 c
2184 c work arrays:
2185 c=============
2186 c jw = integer work array of length 2*n.
2187 c w = real work array of length n
2188 c
2189 c----------------------------------------------------------------------
2190 c w, ju (1:n) store the working array [1:ii-1 = L-part, ii:n = u]
2191 c jw(n+1:2n) stores nonzero indicators
2192 c
2193 c Notes:
2194 c ------
2195 c The diagonal elements of the input matrix must be nonzero (at least
2196 c 'structurally').
2197 c
2198 c----------------------------------------------------------------------*
2199 c---- Dual drop strategy works as follows. *
2200 c *
2201 c 1) Theresholding in L and U as set by droptol. Any element whose *
2202 c magnitude is less than some tolerance (relative to the abs *
2203 c value of diagonal element in u) is dropped. *
2204 c *
2205 c 2) Keeping only the largest lfil elements in the i-th row of L *
2206 c and the largest lfil elements in the i-th row of U (excluding *
2207 c diagonal elements). *
2208 c *
2209 c Flexibility: one can use droptol=0 to get a strategy based on *
2210 c keeping the largest elements in each row of L and U. Taking *
2211 c droptol .ne. 0 but lfil=n will give the usual threshold strategy *
2212 c (however, fill-in is then mpredictible). *
2213 c----------------------------------------------------------------------*
2214 c locals
2215  integer ju0,k,j1,j2,j,ii,i,lenl,lenu,jj,jrow,jpos,len
2216  real*8 tnorm, t, abs, s, fact
2217  if (lfil .lt. 0) goto 998
2218 c-----------------------------------------------------------------------
2219 c initialize ju0 (points to next element to be added to alu,jlu)
2220 c and pointer array.
2221 c-----------------------------------------------------------------------
2222  ju0 = n+2
2223  jlu(1) = ju0
2224 c
2225 c initialize nonzero indicator array.
2226 c
2227  do 1 j=1,n
2228  jw(n+j) = 0
2229  1 continue
2230 c-----------------------------------------------------------------------
2231 c beginning of main loop.
2232 c-----------------------------------------------------------------------
2233  do 500 ii = 1, n
2234  j1 = ia(ii)
2235  j2 = ia(ii+1) - 1
2236  tnorm = 0.0d0
2237  do 501 k=j1,j2
2238  tnorm = tnorm+abs(a(k))
2239  501 continue
2240  if (tnorm .eq. 0.0d0) goto 999
2241  tnorm = tnorm/real(j2-j1+1)

2243 c unpack L-part and U-part of row of A in arrays w
2244 c
2245  lenu = 1
2246  lenl = 0
2247  jw(ii) = ii
2248  w(ii) = 0.0
2249  jw(n+ii) = ii
2250 c
2251  do 170 j = j1, j2
2252  k = ja(j)
2253  t = a(j)
2254  if (k .lt. ii) then
2255  lenl = lenl+1
2256  jw(lenl) = k
2257  w(lenl) = t
2258  jw(n+k) = lenl
2259  else if (k .eq. ii) then
2260  w(ii) = t
2261  else

2263  jpos = ii+lenu-1
2264  jw(jpos) = k
2265  w(jpos) = t
2266  jw(n+k) = jpos
2267  endif
2268  170 continue

2273 c

2275  if (jj .gt. lenl) goto 160

2277 c in order to do the elimination in the correct order we must select
2278 c the smallest column index among jw(k), k=jj+1, ..., lenl.
2279 c-----------------------------------------------------------------------
2280  jrow = jw(jj)
2281  k = jj
2282 c
2283 c determine smallest column index
2284 c
2285  do 151 j=jj+1,lenl
2286  if (jw(j) .lt. jrow) then
2287  jrow = jw(j)
2288  k = j
2289  endif
2290  151 continue
2291 c
2292  if (k .ne. jj) then
2293 c exchange in jw
2294  j = jw(jj)
2295  jw(jj) = jw(k)
2296  jw(k) = j
2297 c exchange in jr
2298  jw(n+jrow) = jj
2299  jw(n+j) = k
2300 c exchange in w
2301  s = w(jj)
2302  w(jj) = w(k)
2303  w(k) = s
2304  endif
2305 c
2306 c zero out element in row by setting jw(n+jrow) to zero.
2307 c
2308  jw(n+jrow) = 0
2309 c
2310 c get the multiplier for row to be eliminated (jrow).
2311 c
2312  fact = w(jj)*alu(jrow)
2313  if (abs(fact) .le. droptol) goto 150
2314 c
2315 c combine current row and row jrow
2316 c
2317  do 203 k = ju(jrow), jlu(jrow+1)-1
2318  s = fact*alu(k)
2319  j = jlu(k)
2320  jpos = jw(n+j)
2321  if (j .ge. ii) then
2322 c
2323 c dealing with upper part.
2324 c
2325  if (jpos .eq. 0) then
2326 c
2327 c this is a fill-in element
2328 c
2329  lenu = lenu+1
2330  if (lenu .gt. n) goto 995
2331  i = ii+lenu-1
2332  jw(i) = j
2333  jw(n+j) = i
2334  w(i) = - s
2335  else
2336 c
2337 c this is not a fill-in element
2338 c
2339  w(jpos) = w(jpos) - s
2340  endif
2341  else
2342 c
2343 c dealing with lower part.
2344 c
2345  if (jpos .eq. 0) then
2346 c
2347 c this is a fill-in element
2348 c

2350  if (lenl .gt. n) goto 995

2352  jw(n+j) = lenl
2353  w(lenl) = - s
2354  else
2355 c
2356 c this is not a fill-in element
2357 c

2359  endif
2360  endif

2364 c overlap with the working elements in L (pivots).
2365 c

2371 c

2373 c
2374  do 308 k=1, lenu
2375  jw(n+jw(ii+k-1)) = 0
2376  308 continue
2377 c

2379 c

2382 c
2383 c sort by quick-split
2384 c
2385  call qsplit (w,jw,lenl,len)
2386 c
2387 c store L-part
2388 c
2389  do 204 k=1, len
2390  if (ju0 .gt. iwk) goto 996
2391  alu(ju0) = w(k)
2392  jlu(ju0) = jw(k)
2393  ju0 = ju0+1
2394  204 continue
2395 c

2397 c
2398  ju(ii) = ju0
2399 c
2400 c update U-matrix -- first apply dropping strategy
2401 c
2402  len = 0
2403  do k=1, lenu-1
2404  if (abs(w(ii+k)) .gt. droptol*tnorm) then
2405  len = len+1
2406  w(ii+len) = w(ii+k)
2407  jw(ii+len) = jw(ii+k)
2408  endif
2409  enddo
2410  lenu = len+1
2411  len = min0(lenu,lfil)
2412 c
2413  if (lenu .gt. 1) call qsplit (w(ii+1), jw(ii+1), lenu-1,len)
2414 c
2415 c copy
2416 c

2418  if (len + ju0 .gt. iwk) goto 997
2419  do 302 k=ii+1,ii+len-1
2420  jlu(ju0) = jw(k)
2421  alu(ju0) = w(k)
2422  t = t + abs(w(k) )
2423  ju0 = ju0+1
2424  302 continue
2425 c
2426 c store inverse of diagonal element of u
2427 c
2428  if (w(ii) .eq. 0.0d0) w(ii) = (0.0001 + droptol)*tnorm
2429 c
2430  alu(ii) = 1.0d0/ w(ii)
2431 c
2432 c update pointer to beginning of next row of U.
2433 c
2434  jlu(ii+1) = ju0
2435 c-----------------------------------------------------------------------
2436 c end main loop
2437 c-----------------------------------------------------------------------
2438  500 continue
2439  ierr = 0
2440  return
2441 c
2442 c incomprehensible error. Matrix must be wrong.
2443 c
2444  995 ierr = -1
2445  return
2446 c
2447 c insufficient storage in L.
2448 c
2449  996 ierr = -2
2450  return
2451 c
2452 c insufficient storage in U.
2453 c
2454  997 ierr = -3
2455  return
2456 c
2457 c illegal lfil entered.
2458 c
2459  998 ierr = -4
2460  return
2461 c
2462 c zero row encountered
2463 c
2464  999 ierr = -5
2465  return
2466 c----------------end-of-ilut--------------------------------------------
2467 c-----------------------------------------------------------------------
2468  end
