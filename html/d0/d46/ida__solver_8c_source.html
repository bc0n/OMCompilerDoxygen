135 int
136 ida_solver_initial(DATA* data, threadData_t *threadData, SOLVER_INFO* solverInfo, IDA_SOLVER *idaData){
137 
138  TRACE_PUSH
139 
140  int flag;
141  long int i;
142  double* tmp;
143 
144  /* default max order */
145  int maxOrder = 5;
146 
147  /* sim data */
148  idaData->simData = (IDA_USERDATA*)malloc(sizeof(IDA_USERDATA));
149  idaData->simData->data = data;
150  idaData->simData->threadData = threadData;
151 
152  /* initialize constants */
153  idaData->setInitialSolution = 0;
154 
155  /* start initialization routines of sundials */
156  idaData->ida_mem = IDACreate();
157  if (idaData->ida_mem == NULL){
158  throwStreamPrint(threadData, "##IDA## Initialization of IDA solver failed!");
159  }
160 
161  idaData->daeMode = 0;
162  idaData->residualFunction = residualFunctionIDA;
163  idaData->N = (long int)data->modelData->nStates;
164 
165 
166  /* change parameter for DAE mode */
167  if (compiledInDAEMode)
168  {
169  idaData->daeMode = 1;
170  idaData->N = (long int) data->modelData->nStates + data->simulationInfo->daeModeData->nAlgebraicDAEVars;
171  }
172  infoStreamPrint(LOG_SOLVER, 1, "## IDA ## Initializing solver of size %ld %s.", idaData->N, idaData->daeMode?"in DAE mode":"");
173 
174  /* initialize states and der(states) */
175  if (idaData->daeMode)
176  {
177  idaData->states = (double*) malloc(idaData->N*sizeof(double));
178  idaData->statesDer = (double*) calloc(idaData->N,sizeof(double));
179 
180  memcpy(idaData->states, data->localData[0]->realVars, sizeof(double)*data->modelData->nStates);
181  // and also algebraic vars
182  getAlgebraicDAEVars(data, idaData->states + data->modelData->nStates);
183  memcpy(idaData->statesDer, data->localData[0]->realVars + data->modelData->nStates, sizeof(double)*data->modelData->nStates);
184 
185  idaData->y = N_VMake_Serial(idaData->N, idaData->states);
186  idaData->yp = N_VMake_Serial(idaData->N, idaData->statesDer);
187  }
188  else
189  {
190  idaData->y = N_VMake_Serial(idaData->N, data->localData[0]->realVars);
191  idaData->yp = N_VMake_Serial(idaData->N, data->localData[0]->realVars + data->modelData->nStates);
192  }
193 
194 
195  flag = IDAInit(idaData->ida_mem,
196  idaData->residualFunction,
197  data->simulationInfo->startTime,
198  idaData->y,
199  idaData->yp);
200 
201  /* allocate memory for jacobians calculation */
202  idaData->ysave = (double*) malloc(idaData->N*sizeof(double));
203  idaData->ypsave = (double*) malloc(idaData->N*sizeof(double));
204  idaData->delta_hh = (double*) malloc(idaData->N*sizeof(double));
205  idaData->errwgt = N_VNew_Serial(idaData->N);
206  idaData->newdelta = N_VNew_Serial(idaData->N);
207 
208  /* allocate memory for initialization process */
209  tmp = (double*) malloc(idaData->N*sizeof(double));
210 
211  if (checkIDAflag(flag)){
212  throwStreamPrint(threadData, "##IDA## Something goes wrong while initialize IDA solver!");
213  }
214 
215  flag = IDASetUserData(idaData->ida_mem, idaData);
216  if (checkIDAflag(flag)){
217  throwStreamPrint(threadData, "##IDA## Something goes wrong while initialize IDA solver!");
218  }
219 
220  flag = IDASetErrHandlerFn(idaData->ida_mem, errOutputIDA, idaData);
221  if (checkIDAflag(flag)){
222  throwStreamPrint(threadData, "##IDA## Something goes wrong while set error handler!");
223  }
224 
225 
226  /* set nominal values of the states for absolute tolerances */
227  infoStreamPrint(LOG_SOLVER, 1, "The relative tolerance is %g. Following absolute tolerances are used for the states: ", data->simulationInfo->tolerance);
228 
229  for(i=0; i < data->modelData->nStates; ++i)
230  {
231  tmp[i] = fmax(fabs(data->modelData->realVarsData[i].attribute.nominal), 1e-32);
232  infoStreamPrint(LOG_SOLVER_V, 0, "%ld. %s -> %g", i+1, data->modelData->realVarsData[i].info.name, tmp[i]);
233  }
234 
235  /* daeMode: set nominal values for algebraic variables */
236  if (idaData->daeMode)
237  {
238  getAlgebraicDAEVarNominals(data, tmp + data->modelData->nStates);
239  }
240  /* multiply by tolerance to obtain a relative tolerace */
241  for(i=0; i < idaData->N; ++i)
242  {
243  tmp[i] *= data->simulationInfo->tolerance;
244  }
245  messageClose(LOG_SOLVER);
246  flag = IDASVtolerances(idaData->ida_mem,
247  data->simulationInfo->tolerance,
248  N_VMake_Serial(idaData->N,tmp));
249  if (checkIDAflag(flag)){
250  throwStreamPrint(threadData, "##IDA## Setting tolerances fails while initialize IDA solver!");
251  }
252 
253 
254  if (omc_flag[FLAG_IDA_SCALING]) // idaNoScaling
255  {
256  /* allocate memory for scaling */
257  idaData->yScale = (double*) malloc(idaData->N*sizeof(double));
258  idaData->ypScale = (double*) malloc(idaData->N*sizeof(double));
259  idaData->resScale = (double*) malloc(idaData->N*sizeof(double));
260 
261  /* set yScale from nominal values */
262  for(i=0; i < data->modelData->nStates; ++i)
263  {
264  idaData->yScale[i] = fabs(data->modelData->realVarsData[i].attribute.nominal);
265  idaData->ypScale[i] = 1.0;
266  }
267  /* daeMode: set nominal values for algebraic variables */
268  if (idaData->daeMode)
269  {
270  getAlgebraicDAEVarNominals(data, idaData->yScale + data->modelData->nStates);
271  for(i=data->modelData->nStates; i < idaData->N; ++i)
272  {
273  idaData->ypScale[i] = 1.0;
274  }
275  }
276  infoStreamPrint(LOG_SOLVER_V, 1, "The scale factors for all ida states: ");
277  for(i=0; i < idaData->N; ++i)
278  {
279  infoStreamPrint(LOG_SOLVER_V, 0, "%ld. scaleFactor: %g", i+1, idaData->yScale[i]);
280  }
281  messageClose(LOG_SOLVER_V);
282  }
283  /* initialize */
284  idaData->disableScaling = 0;
285 
286  /* set root function */
287  /* if FLAG_NO_ROOTFINDING is set, solve perform without internal root finding*/
288  if(!omc_flag[FLAG_NO_ROOTFINDING])
289  {
290  solverInfo->solverRootFinding = 1;
291  flag = IDARootInit(idaData->ida_mem, data->modelData->nZeroCrossings, rootsFunctionIDA);
292  if (checkIDAflag(flag)){
293  throwStreamPrint(threadData, "##IDA## Setting root function fails while initialize IDA solver!");
294  }
295  }
296  infoStreamPrint(LOG_SOLVER, 0, "ida uses internal root finding method %s", solverInfo->solverRootFinding?"YES":"NO");
297 
298  /* define maximum integration order of dassl */
299  if (omc_flag[FLAG_MAX_ORDER])
300  {
301  maxOrder = atoi(omc_flagValue[FLAG_MAX_ORDER]);
302 
303  flag = IDASetMaxOrd(idaData->ida_mem, maxOrder);
304  if (checkIDAflag(flag)){
305  throwStreamPrint(threadData, "##IDA## Failed to set max integration order!");
306  }
307  }
308  infoStreamPrint(LOG_SOLVER, 0, "maximum integration order %d", maxOrder);
309 
310 
311  /* if FLAG_NOEQUIDISTANT_GRID is set, choose ida step method */
312  if (omc_flag[FLAG_NOEQUIDISTANT_GRID])
313  {
314  idaData->internalSteps = 1; /* TRUE */
315  solverInfo->solverNoEquidistantGrid = 1;
316  }
317  else
318  {
319  idaData->internalSteps = 0; /* FALSE */
320  }
321  infoStreamPrint(LOG_SOLVER, 0, "use equidistant time grid %s", idaData->internalSteps?"NO":"YES");
322 
323  /* check if Flags FLAG_NOEQUIDISTANT_OUT_FREQ or FLAG_NOEQUIDISTANT_OUT_TIME are set */
324  if (idaData->internalSteps){
325  if (omc_flag[FLAG_NOEQUIDISTANT_OUT_FREQ])
326  {
327  idaData->stepsFreq = atoi(omc_flagValue[FLAG_NOEQUIDISTANT_OUT_FREQ]);
328  }
329  else if (omc_flag[FLAG_NOEQUIDISTANT_OUT_TIME])
330  {
331  idaData->stepsTime = atof(omc_flagValue[FLAG_NOEQUIDISTANT_OUT_TIME]);
332  flag = IDASetMaxStep(idaData->ida_mem, idaData->stepsTime);
333  if (checkIDAflag(flag)){
334  throwStreamPrint(threadData, "##IDA## Setting max steps of the IDA solver!");
335  }
336  infoStreamPrint(LOG_SOLVER, 0, "maximum step size %g", idaData->stepsTime);
337  } else {
338  idaData->stepsFreq = 1;
339  idaData->stepsTime = 0.0;
340  }
341 
342  if (omc_flag[FLAG_NOEQUIDISTANT_OUT_FREQ] && omc_flag[FLAG_NOEQUIDISTANT_OUT_TIME]){
343  warningStreamPrint(LOG_STDOUT, 0, "The flags are \"noEquidistantOutputFrequency\" "
344  "and \"noEquidistantOutputTime\" are in opposition "
345  "to each other. The flag \"noEquidistantOutputFrequency\" superiors.");
346  }
347  infoStreamPrint(LOG_SOLVER, 0, "as the output frequency control is used: %d", idaData->stepsFreq);
348  infoStreamPrint(LOG_SOLVER, 0, "as the output frequency time step control is used: %f", idaData->stepsTime);
349  }
350 
351  /* if FLAG_IDA_LS is set, choose ida linear solver method */
352  if (omc_flag[FLAG_IDA_LS])
353  {
354  for(i=1; i< IDA_LS_MAX;i++)
355  {
356  if(!strcmp((const char*)omc_flagValue[FLAG_IDA_LS], IDA_LS_METHOD[i])){
357  idaData->linearSolverMethod = (int)i;
358  break;
359  }
360  }
361  if(idaData->linearSolverMethod == IDA_LS_UNKNOWN)
362  {
363  if (ACTIVE_WARNING_STREAM(LOG_SOLVER))
364  {
365  warningStreamPrint(LOG_SOLVER, 1, "unrecognized ida linear solver method %s, current options are:", (const char*)omc_flagValue[FLAG_IDA_LS]);
366  for(i=1; i < IDA_LS_MAX; ++i)
367  {
368  warningStreamPrint(LOG_SOLVER, 0, "%-15s [%s]", IDA_LS_METHOD[i], IDA_LS_METHOD_DESC[i]);
369  }
370  messageClose(LOG_SOLVER);
371  }
372  throwStreamPrint(threadData,"unrecognized ida linear solver method %s", (const char*)omc_flagValue[FLAG_IDA_LS]);
373  }
374  }
375  else
376  {
377  idaData->linearSolverMethod = IDA_LS_KLU;
378  }
379 
380  switch (idaData->linearSolverMethod){
381  case IDA_LS_SPGMR:
382  flag = IDASpgmr(idaData->ida_mem, 0);
383  break;
384  case IDA_LS_SPBCG:
385  flag = IDASpbcg(idaData->ida_mem, 0);
386  break;
387  case IDA_LS_SPTFQMR:
388  flag = IDASptfqmr(idaData->ida_mem, 0);
389  break;
390  case IDA_LS_DENSE:
391  flag = IDADense(idaData->ida_mem, idaData->N);
392  break;
393  case IDA_LS_KLU:
394  /* Set KLU after initialized sparse pattern of the jacobian for nnz */
395  break;
396  default:
397  throwStreamPrint(threadData,"unrecognized linear solver method %s", (const char*)omc_flagValue[FLAG_IDA_LS]);
398  break;
399  }
400  if (checkIDAflag(flag)){
401  throwStreamPrint(threadData, "##IDA## Setting linear solver method fails while initialize IDA solver!");
402  } else {
403  infoStreamPrint(LOG_SOLVER, 0, "IDA linear solver method selected %s", IDA_LS_METHOD_DESC[idaData->linearSolverMethod]);
404  }
405 
406 
407  /* if FLAG_JACOBIAN is set, choose jacobian calculation method */
408  if (omc_flag[FLAG_JACOBIAN])
409  {
410  for(i=1; i< JAC_MAX;i++)
411  {
412  if(!strcmp((const char*)omc_flagValue[FLAG_JACOBIAN], JACOBIAN_METHOD[i])){
413  idaData->jacobianMethod = (int)i;
414  break;
415  }
416  }
417  if(idaData->daeMode && ( idaData->jacobianMethod == COLOREDSYMJAC || idaData->jacobianMethod == SYMJAC ))
418  {
419  warningStreamPrint(LOG_STDOUT, 1, "Symbolic Jacobians are currently not supported by DAE Mode. Switch back to numerical Jacobian!");
420  idaData->jacobianMethod = COLOREDNUMJAC;
421  }
422  if(idaData->jacobianMethod == JAC_UNKNOWN)
423  {
424  if (ACTIVE_WARNING_STREAM(LOG_SOLVER))
425  {
426  warningStreamPrint(LOG_SOLVER, 1, "unrecognized jacobian calculation method %s, current options are:", (const char*)omc_flagValue[FLAG_JACOBIAN]);
427  for(i=1; i < JAC_MAX; ++i)
428  {
429  warningStreamPrint(LOG_SOLVER, 0, "%-15s [%s]", JACOBIAN_METHOD[i], JACOBIAN_METHOD_DESC[i]);
430  }
431  messageClose(LOG_SOLVER);
432  }
433  throwStreamPrint(threadData,"unrecognized jacobian calculation method %s", (const char*)omc_flagValue[FLAG_JACOBIAN]);
434  }
435  /* default case colored numerical jacobian */
436  }
437  else
438  {
439  idaData->jacobianMethod = COLOREDNUMJAC;
440  }
441 
442  /* selects the calculation method of the jacobian */
443  /* in daeMode sparse pattern is already initialized in DAEMODE_DATA */
444  if(!idaData->daeMode && (idaData->jacobianMethod == COLOREDNUMJAC ||
445  idaData->jacobianMethod == COLOREDSYMJAC ||
446  idaData->jacobianMethod == SYMJAC))
447  {
448  ANALYTIC_JACOBIAN* jacobian = &(data->simulationInfo->analyticJacobians[data->callback->INDEX_JAC_A]);
449  if (data->callback->initialAnalyticJacobianA(data, threadData, jacobian))
450  {
451  infoStreamPrint(LOG_STDOUT, 0, "Jacobian or SparsePattern is not generated or failed to initialize! Switch back to normal.");
452  idaData->jacobianMethod = INTERNALNUMJAC;
453  if (idaData->linearSolverMethod == IDA_LS_KLU)
454  {
455  idaData->linearSolverMethod = IDA_LS_DENSE;
456  flag = IDADense(idaData->ida_mem, idaData->N);
457  warningStreamPrint(LOG_STDOUT, 0, "IDA linear solver method also switched back to %s", IDA_LS_METHOD_DESC[idaData->linearSolverMethod]);
458  }
459  } else {
460  ANALYTIC_JACOBIAN* jac = &data->simulationInfo->analyticJacobians[data->callback->INDEX_JAC_A];
461  infoStreamPrint(LOG_SIMULATION, 1, "Initialized colored Jacobian:");
462  infoStreamPrint(LOG_SIMULATION, 0, "columns: %d rows: %d", jac->sizeCols, jac->sizeRows);
463  infoStreamPrint(LOG_SIMULATION, 0, "NNZ: %d colors: %d", jac->sparsePattern.numberOfNoneZeros, jac->sparsePattern.maxColors);
464  messageClose(LOG_SIMULATION);
465  }
466 
467  }
468  /* set up the appropriate function pointer */
469  if (idaData->linearSolverMethod == IDA_LS_KLU)
470  {
471  if (idaData->daeMode)
472  {
473  idaData->NNZ = data->simulationInfo->daeModeData->sparsePattern->numberOfNoneZeros;
474  flag = IDAKLU(idaData->ida_mem, idaData->N, idaData->NNZ);
475  idaData->jacobianMethod = COLOREDNUMJAC; /* In DAE mode only numerical matrix is supported */
476  }
477  else
478  {
479  idaData->NNZ = data->simulationInfo->analyticJacobians[data->callback->INDEX_JAC_A].sparsePattern.numberOfNoneZeros;
480  flag = IDAKLU(idaData->ida_mem, idaData->N, idaData->NNZ);
481  /* to add a cj identety matrix */
482  idaData->tmpJac = NewSparseMat(idaData->N, idaData->N, idaData->N);
483  }
484  if (checkIDAflag(flag)){
485  throwStreamPrint(threadData, "##IDA## Setup of linear solver KLU failed!");
486  }
487 
488  switch (idaData->jacobianMethod){
489  case SYMJAC:
490  case NUMJAC:
491  case COLOREDSYMJAC:
492  case COLOREDNUMJAC:
493  flag = IDASlsSetSparseJacFn(idaData->ida_mem, callSparseJacobian);
494  break;
495  default:
496  throwStreamPrint(threadData,"For the klu solver jacobian calculation method has to be %s or %s", JACOBIAN_METHOD[COLOREDSYMJAC], JACOBIAN_METHOD[COLOREDNUMJAC]);
497  break;
498  }
499  }
500  else
501  {
502  switch (idaData->jacobianMethod){
503  case SYMJAC:
504  case COLOREDSYMJAC:
505  case COLOREDNUMJAC:
506  case NUMJAC:
507  /* set jacobian function */
508  flag = IDADlsSetDenseJacFn(idaData->ida_mem, callDenseJacobian);
509  break;
510  case INTERNALNUMJAC:
511  break;
512  default:
513  throwStreamPrint(threadData,"unrecognized jacobian calculation method %s", (const char*)omc_flagValue[FLAG_JACOBIAN]);
514  break;
515  }
516  }
517 
518  if (checkIDAflag(flag)){
519  throwStreamPrint(threadData, "##IDA## Setting jacobian function fails while initialize IDA solver!");
520  } else {
521  infoStreamPrint(LOG_SOLVER, 0, "jacobian is calculated by %s", JACOBIAN_METHOD_DESC[idaData->jacobianMethod]);
522  }
523 
524  /* set max error test fails */
525  if (omc_flag[FLAG_IDA_MAXERRORTESTFAIL])
526  {
527  int maxErrorTestFails = atoi(omc_flagValue[FLAG_IDA_MAXERRORTESTFAIL]);
528  flag = IDASetMaxErrTestFails(idaData->ida_mem, maxErrorTestFails);
529  if (checkIDAflag(flag)){
530  throwStreamPrint(threadData, "##IDA## set IDASetMaxErrTestFails failed!");
531  }
532  }
533 
534  /* set maximum number of nonlinear solver iterations at one step */
535  if (omc_flag[FLAG_IDA_MAXNONLINITERS])
536  {
537  int maxNonlinIters = atoi(omc_flagValue[FLAG_IDA_MAXNONLINITERS]);
538  flag = IDASetMaxNonlinIters(idaData->ida_mem, maxNonlinIters);
539  if (checkIDAflag(flag)){
540  throwStreamPrint(threadData, "##IDA## set IDASetMaxNonlinIters failed!");
541  }
542  }
543 
544  /* maximum number of nonlinear solver convergence failures at one step */
545  if (omc_flag[FLAG_IDA_MAXCONVFAILS])
546  {
547  int maxConvFails = atoi(omc_flagValue[FLAG_IDA_MAXCONVFAILS]);
548  flag = IDASetMaxConvFails(idaData->ida_mem, maxConvFails);
549  if (checkIDAflag(flag)){

808  }

812  idaData->setInitialSolution = 0;
813  }
814

816  if (!idaData->setInitialSolution)
817  {
818  debugStreamPrint(LOG_SOLVER, 0, "Re-initialized IDA Solver");

821  if (idaData->daeMode)
822  {
823  memcpy(idaData->states, data->localData[0]->realVars, sizeof(double)*data->modelData->nStates);

836  idaScaleData(idaData);

838  }

841  solverInfo->currentTime,

847  }

851  {

864  NV_Ith_S(idaData->ySp[i],j) = 0;

866  }

868  if (checkIDAflag(flag)){
869  throwStreamPrint(threadData, "##IDA## set IDASensInit failed!");
870  }
871  }
872  idaData->setInitialSolution = 1;
873  }
874 
875  saveJumpState = threadData->currentErrorStage;
876  threadData->currentErrorStage = ERROR_INTEGRATOR;

879 #if !defined(OMC_EMCC)
880  MMC_TRY_INTERNAL(simulationJumpBuffer)
881 #endif

883 
884  /* Check that tout is not less than timeValue otherwise the solver
885  * will come in trouble.
886  * If that is the case we skip the current step. */
887  if (solverInfo->currentStepSize < DASSL_STEP_EPS)
888  {
889  infoStreamPrint(LOG_SOLVER, 0, "Desired step to small try next one");
890  infoStreamPrint(LOG_SOLVER, 0, "Interpolate linear");
891

893  for(i = 0; i < idaData->N; i++)

896  }

898  data->callback->functionODE(data, threadData);
899  solverInfo->currentTime = sData->timeValue;
900 
901  TRACE_POP
902  return 0;
903  }

907  if (idaData->internalSteps)
908  {
909  /* If internalSteps are selected, let IDA run to stopTime or next sample event */
910  if (data->simulationInfo->nextSampleEvent < data->simulationInfo->stopTime)
911  {
912  tout = data->simulationInfo->nextSampleEvent;
913  }
914  else
915  {
916  tout = data->simulationInfo->stopTime;
917  }
918  stepsMode = IDA_ONE_STEP;
919  flag = IDASetStopTime(idaData->ida_mem, tout);
920  if (checkIDAflag(flag)){
921  throwStreamPrint(threadData, "##IDA## Something goes wrong while set stopTime!");
922  }
923  }
924  else
925  {
926  tout = solverInfo->currentTime + solverInfo->currentStepSize;
927  stepsMode = IDA_NORMAL;
928  }

930

933  infoStreamPrint(LOG_SOLVER, 1, "##IDA## new step from %.15g to %.15g", solverInfo->currentTime, tout);
934 
935  /* read input vars */
936  externalInputUpdate(data);
937  data->callback->input_function(data, threadData);
938 
939  if (omc_flag[FLAG_IDA_SCALING])
940  {
941  /* scale idaData->y and idaData->yp */
942  idaScaleData(idaData);
943  }
944 
945  flag = IDASolve(idaData->ida_mem, tout, &solverInfo->currentTime, idaData->y, idaData->yp, stepsMode);
946

948  {
949  /* rescale idaData->y and idaData->yp */
950  idaReScaleData(idaData);
951  }
952 
953  /* set time to current time */
954  sData->timeValue = solverInfo->currentTime;
955 
956  /* error handling */
957  if ( !checkIDAflag(flag) && solverInfo->currentTime >= tout)
958  {
959  infoStreamPrint(LOG_SOLVER, 0, "##IDA## step done to time = %.15g", solverInfo->currentTime);
960  finished = TRUE;
961  }
962  else
963  {
964  if (!checkIDAflag(flag))
965  {
966  infoStreamPrint(LOG_SOLVER, 0, "##IDA## continue integration time = %.15g", solverInfo->currentTime);
967  }
968  else if (flag == IDA_ROOT_RETURN)
969  {
970  infoStreamPrint(LOG_SOLVER, 0, "##IDA## root found at time = %.15g", solverInfo->currentTime);
971  finished = TRUE;
972  }
973  else if (flag == IDA_TOO_MUCH_WORK)
974  {
975  warningStreamPrint(LOG_SOLVER, 0, "##IDA## has done too much work with small steps at time = %.15g", solverInfo->currentTime);
976  }

978  {
979  flag = IDAReInit(idaData->ida_mem,
980  solverInfo->currentTime,
981  idaData->y,
982  idaData->yp);
983  restartAfterLSFail = 1;
984  warningStreamPrint(LOG_SOLVER, 0, "##IDA## linear solver failed try once again = %.15g", solverInfo->currentTime);
985  }
986  else
987  {
988  infoStreamPrint(LOG_STDOUT, 0, "##IDA## %d error occurred at time = %.15g", flag, solverInfo->currentTime);
989  finished = TRUE;
990  retVal = flag;
991  }
992  }
993 
994  /* closing new step message */
995  messageClose(LOG_SOLVER);
996 
997  /* emit step, if step mode is selected */
998  if (idaData->internalSteps)
999  {
1000  infoStreamPrint(LOG_SOLVER, 0, "##IDA## noEquadistant stepsOutputCounter %d by freq %d at time = %.15g", stepsOutputCounter, idaData->stepsFreq, solverInfo->currentTime);
1001  if (omc_flag[FLAG_NOEQUIDISTANT_OUT_FREQ]){
1002  /* output every n-th time step */
1003  if (stepsOutputCounter >= idaData->stepsFreq){
1004  stepsOutputCounter = 1; /* next line set it to one */
1005  infoStreamPrint(LOG_SOLVER, 0, "##IDA## noEquadistant output %d by freq at time = %.15g", stepsOutputCounter, solverInfo->currentTime);
1006  break;
1007  }
1008  stepsOutputCounter++;
1009  } else if (omc_flag[FLAG_NOEQUIDISTANT_OUT_TIME]){
1010  /* output when time>=k*timeValue */
1011  if (solverInfo->currentTime > stepsOutputCounter * idaData->stepsTime){
1012  stepsOutputCounter++;
1013  infoStreamPrint(LOG_SOLVER, 0, "##IDA## noEquadistant output %d by time freq at time = %.15g", stepsOutputCounter, solverInfo->currentTime);
1014  break;
1015  }
1016  } else {
1017  break;
1018  }
1019  }
1020 
1021  } while(!finished);
1022 
1023 #if !defined(OMC_EMCC)
1024  MMC_CATCH_INTERNAL(simulationJumpBuffer)
1025 #endif
1026  threadData->currentErrorStage = saveJumpState;
1027 
1028  /* if a state event occurs than no sample event does need to be activated */
1029  if (data->simulationInfo->sampleActivated && solverInfo->currentTime < data->simulationInfo->nextSampleEvent)
1030  {
1031  data->simulationInfo->sampleActivated = 0;
1032  }
1033 
1034  /* initialize states and der(states) */
1035  if (idaData->daeMode)
1036  {
1037  memcpy(data->localData[0]->realVars, idaData->states, sizeof(double)*data->modelData->nStates);
1038  // and also algebraic vars
1039  setAlgebraicDAEVars(data, idaData->states + data->modelData->nStates);
1040  memcpy(data->localData[0]->realVars + data->modelData->nStates, idaData->statesDer, sizeof(double)*data->modelData->nStates);
1041  sData->timeValue = solverInfo->currentTime;
1042  }
1043 
1044  /* sensitivity mode */
1045  if (idaData->idaSmode)
1046  {
1047  flag = IDAGetSens(idaData->ida_mem, &solverInfo->currentTime, idaData->ySResult);
1048  if (checkIDAflag(flag)){
1049  throwStreamPrint(threadData, "##IDA## Something goes wrong while obtain results for parameter sensitivities!");
1050  }
1051  }
1052 
1053  /* save stats */
1054  /* steps */
1055  tmp = 0;
1056  flag = IDAGetNumSteps(idaData->ida_mem, &tmp);
1057  if (flag == IDA_SUCCESS)
1058  {
1059  solverInfo->solverStatsTmp[0] = tmp;
1060  }
1061 
1062  /* functionODE evaluations */
1063  tmp = 0;
1064  flag = IDAGetNumResEvals(idaData->ida_mem, &tmp);
1065  if (flag == IDA_SUCCESS)
1066  {
1067  solverInfo->solverStatsTmp[1] = tmp;
1068  }
1069 
1070  /* Jacobians evaluations */
1071  tmp = 0;
1072  if (idaData->linearSolverMethod == IDA_LS_KLU)
1073  {
1074  flag = IDASlsGetNumJacEvals(idaData->ida_mem, &tmp);
1075  }
1076  else
1077  {
1078  flag = IDADlsGetNumJacEvals(idaData->ida_mem, &tmp);
1079  }
1080 
1081  if (flag == IDA_SUCCESS)
1082  {
1083  solverInfo->solverStatsTmp[2] = tmp;
1084  }
1085 
1086  /* local error test failures */
1087  tmp = 0;
1088  flag = IDAGetNumErrTestFails(idaData->ida_mem, &tmp);
1089  if (flag == IDA_SUCCESS)
1090  {
1091  solverInfo->solverStatsTmp[3] = tmp;
1092  }
1093 
1094  /* local error test failures */
1095  tmp = 0;
1096  flag = IDAGetNumNonlinSolvConvFails(idaData->ida_mem, &tmp);
1097  if (flag == IDA_SUCCESS)
1098  {
1099  solverInfo->solverStatsTmp[4] = tmp;
1100  }
1101 
1102  /* get more statistics */
1103  if (useStream[LOG_SOLVER_V])
1104  {
1105  long int tmp1,tmp2;
1106  double dtmp;
1107 
1108  infoStreamPrint(LOG_SOLVER_V, 1, "### IDAStats ###");

1121 
1122  messageClose(LOG_SOLVER_V);
1123  }
1124 
1125  infoStreamPrint(LOG_SOLVER, 0, "##IDA## Finished Integrator step.");
1126
