128 /*
129  * -----------------------------------------------------------------
130  * IDALapackDense
131  * -----------------------------------------------------------------
132  * This routine initializes the memory record and sets various function
133  * fields specific to the linear solver module. IDALapackDense first
134  * calls the existing lfree routine if this is not NULL. Then it sets
135  * the ida_linit, ida_lsetup, ida_lsolve, ida_lfree fields in (*ida_mem)
136  * to be idaLapackDenseInit, idaLapackDenseSetup, idaLapackDenseSolve,
137  * and idaLapackDenseFree, respectively. It allocates memory for a
138  * structure of type IDADlsMemRec and sets the ida_lmem field in
139  * (*ida_mem) to the address of this structure. It sets setupNonNull
140  * in (*ida_mem) to TRUE, and the d_jac field to the default
141  * idaLapackDenseDQJac. Finally, it allocates memory for M, pivots.
142  *
143  * The return value is SUCCESS = 0, or LMEM_FAIL = -1.
144  *
145  * NOTE: The dense linear solver assumes a serial implementation
146  * of the NVECTOR package. Therefore, IDALapackDense will first
147  * test for a compatible N_Vector internal representation
148  * by checking that N_VGetArrayPointer and N_VSetArrayPointer
149  * exist.
150  * -----------------------------------------------------------------
151  */
152 int IDALapackDense(void *ida_mem, int N)
153 {
154  IDAMem IDA_mem;
155  IDADlsMem idadls_mem;
156 
157  /* Return immediately if ida_mem is NULL */
158  if (ida_mem == NULL) {
159  IDAProcessError(NULL, IDADLS_MEM_NULL, "IDASLAPACK", "IDALapackDense", MSGD_IDAMEM_NULL);
160  return(IDADLS_MEM_NULL);
161  }
162  IDA_mem = (IDAMem) ida_mem;
163 
164  /* Test if the NVECTOR package is compatible with the LAPACK solver */
165  if (tempv->ops->nvgetarraypointer == NULL ||
166  tempv->ops->nvsetarraypointer == NULL) {
167  IDAProcessError(IDA_mem, IDADLS_ILL_INPUT, "IDASLAPACK", "IDALapackDense", MSGD_BAD_NVECTOR);
168  return(IDADLS_ILL_INPUT);
169  }
170 
171  if (lfree !=NULL) lfree(IDA_mem);
172 
173  /* Set four main function fields in IDA_mem */
174  linit = idaLapackDenseInit;
175  lsetup = idaLapackDenseSetup;
176  lsolve = idaLapackDenseSolve;
177  lperf = NULL;
178  lfree = idaLapackDenseFree;
179 
180  /* Get memory for IDADlsMemRec */
181  idadls_mem = NULL;
182  idadls_mem = (IDADlsMem) malloc(sizeof(struct IDADlsMemRec));
183  if (idadls_mem == NULL) {
184  IDAProcessError(IDA_mem, IDADLS_MEM_FAIL, "IDASLAPACK", "IDALapackDense", MSGD_MEM_FAIL);
185  return(IDADLS_MEM_FAIL);
186  }
187 
188  /* Set matrix type */
189  mtype = SUNDIALS_DENSE;
190 
191  /* Set default Jacobian routine and Jacobian data */
192  jacDQ = TRUE;
193  djac = NULL;
194  J_data = NULL;
195 
196  last_flag = IDADLS_SUCCESS;
197  setupNonNull = TRUE;
198 
199  /* Set problem dimension */
200  n = (long int) N;
201 
202  /* Allocate memory for JJ and pivot array */
203  JJ = NULL;
204  pivots = NULL;
205 
206  JJ = NewDenseMat(n, n);
207  if (JJ == NULL) {
208  IDAProcessError(IDA_mem, IDADLS_MEM_FAIL, "IDASLAPACK", "IDALapackDense", MSGD_MEM_FAIL);
209  free(idadls_mem); idadls_mem = NULL;
210  return(IDADLS_MEM_FAIL);
211  }
212  pivots = NewIntArray(N);
213  if (pivots == NULL) {
214  IDAProcessError(IDA_mem, IDADLS_MEM_FAIL, "IDASLAPACK", "IDALapackDense", MSGD_MEM_FAIL);
215  DestroyMat(JJ);
216  free(idadls_mem); idadls_mem = NULL;
217  return(IDADLS_MEM_FAIL);
218  }
219 
220  /* Attach linear solver memory to integrator memory */
221  lmem = idadls_mem;
222 
223  return(IDADLS_SUCCESS);
224 }

226 /*
227  * -----------------------------------------------------------------
228  * IDALapackBand
229  * -----------------------------------------------------------------
230  * This routine initializes the memory record and sets various function
231  * fields specific to the band linear solver module. It first calls
232  * the existing lfree routine if this is not NULL. It then sets the
233  * ida_linit, ida_lsetup, ida_lsolve, and ida_lfree fields in (*ida_mem)
234  * to be idaLapackBandInit, idaLapackBandSetup, idaLapackBandSolve,
235  * and idaLapackBandFree, respectively. It allocates memory for a
236  * structure of type IDALapackBandMemRec and sets the ida_lmem field in
237  * (*ida_mem) to the address of this structure. It sets setupNonNull
238  * in (*ida_mem) to be TRUE, mu to be mupper, ml to be mlower, and
239  * the jacE and jacI field to NULL.
240  * Finally, it allocates memory for M and pivots.
241  * The IDALapackBand return value is IDADLS_SUCCESS = 0,
242  * IDADLS_MEM_FAIL = -1, or IDADLS_ILL_INPUT = -2.
243  *
244  * NOTE: The IDALAPACK linear solver assumes a serial implementation
245  * of the NVECTOR package. Therefore, IDALapackBand will first
246  * test for compatible a compatible N_Vector internal
247  * representation by checking that the function
248  * N_VGetArrayPointer exists.
249  * -----------------------------------------------------------------
250  */
251 int IDALapackBand(void *ida_mem, int N, int mupper, int mlower)
252 {
253  IDAMem IDA_mem;
254  IDADlsMem idadls_mem;
255 
256  /* Return immediately if ida_mem is NULL */
257  if (ida_mem == NULL) {
258  IDAProcessError(NULL, IDADLS_MEM_NULL, "IDASLAPACK", "IDALapackBand", MSGD_IDAMEM_NULL);
259  return(IDADLS_MEM_NULL);
260  }
261  IDA_mem = (IDAMem) ida_mem;
262 
263  /* Test if the NVECTOR package is compatible with the BAND solver */
264  if (tempv->ops->nvgetarraypointer == NULL) {
265  IDAProcessError(IDA_mem, IDADLS_ILL_INPUT, "IDASLAPACK", "IDALapackBand", MSGD_BAD_NVECTOR);
266  return(IDADLS_ILL_INPUT);
267  }
268 
269  if (lfree != NULL) lfree(IDA_mem);
270 
271  /* Set four main function fields in IDA_mem */
272  linit = idaLapackBandInit;
273  lsetup = idaLapackBandSetup;
274  lsolve = idaLapackBandSolve;
275  lperf = NULL;
276  lfree = idaLapackBandFree;
277 
278  /* Get memory for IDADlsMemRec */
279  idadls_mem = NULL;
280  idadls_mem = (IDADlsMem) malloc(sizeof(struct IDADlsMemRec));
281  if (idadls_mem == NULL) {
282  IDAProcessError(IDA_mem, IDADLS_MEM_FAIL, "IDASLAPACK", "IDALapackBand", MSGD_MEM_FAIL);
283  return(IDADLS_MEM_FAIL);
284  }
285 
286  /* Set matrix type */
287  mtype = SUNDIALS_BAND;
288 
289  /* Set default Jacobian routine and Jacobian data */
290  jacDQ = TRUE;
291  bjac = NULL;
292  J_data = NULL;
293 
294  last_flag = IDADLS_SUCCESS;
295  setupNonNull = TRUE;
296 
297  /* Load problem dimension */
298  n = (long int) N;
299 
300  /* Load half-bandwiths in idadls_mem */
301  ml = (long int) mlower;
302  mu = (long int) mupper;
303 
304  /* Test ml and mu for legality */
305  if ((ml < 0) || (mu < 0) || (ml >= n) || (mu >= n)) {
306  IDAProcessError(IDA_mem, IDADLS_ILL_INPUT, "IDASLAPACK", "IDALapackBand", MSGD_BAD_SIZES);
307  free(idadls_mem); idadls_mem = NULL;
308  return(IDADLS_ILL_INPUT);
309  }
310 
311  /* Set extended upper half-bandwith for M (required for pivoting) */
312  smu = mu + ml;
313 
314  /* Allocate memory for JJ and pivot arrays */
315  JJ = NULL;
316  pivots = NULL;
317 
318  JJ = NewBandMat(n, mu, ml, smu);
319  if (JJ == NULL) {
320  IDAProcessError(IDA_mem, IDADLS_MEM_FAIL, "IDASLAPACK", "IDALapackBand", MSGD_MEM_FAIL);
321  free(idadls_mem); idadls_mem = NULL;
322  return(IDADLS_MEM_FAIL);
323  }
324  pivots = NewIntArray(N);
325  if (pivots == NULL) {
326  IDAProcessError(IDA_mem, IDADLS_MEM_FAIL, "IDASLAPACK", "IDALapackBand", MSGD_MEM_FAIL);
327  DestroyMat(JJ);
328  free(idadls_mem); idadls_mem = NULL;
329  return(IDADLS_MEM_FAIL);
330  }
331 
332  /* Attach linear solver memory to integrator memory */
333  lmem = idadls_mem;
334 
335  return(IDADLS_SUCCESS);
336 }

368 /*
369  * idaLapackDenseSetup does the setup operations for the dense linear solver.
370  * It calls the Jacobian function to obtain the Newton matrix M = F_y + c_j*F_y',
371  * updates counters, and calls the dense LU factorization routine.
372  */
373 static int idaLapackDenseSetup(IDAMem IDA_mem,
374  N_Vector yP, N_Vector ypP, N_Vector fctP,
375  N_Vector tmp1, N_Vector tmp2, N_Vector tmp3)
376 {
377  IDADlsMem idadls_mem;
378  int ier, retval;
379  int intn;
380 
381  idadls_mem = (IDADlsMem) lmem;
382 
383  intn = (int) n;
384 
385  /* Call Jacobian function */
386  nje++;
387  SetToZero(JJ);
388  retval = djac(n, tn, cj, yP, ypP, fctP, JJ, J_data, tmp1, tmp2, tmp3);
389  if (retval < 0) {
390  IDAProcessError(IDA_mem, IDADLS_JACFUNC_UNRECVR, "IDASLAPACK", "idaLapackDenseSetup", MSGD_JACFUNC_FAILED);
391  last_flag = IDADLS_JACFUNC_UNRECVR;
392  return(-1);
393  } else if (retval > 0) {
394  last_flag = IDADLS_JACFUNC_RECVR;
395  return(1);
396  }
397 
398  /* Do LU factorization of M */
399  dgetrf_f77(&intn, &intn, JJ->data, &intn, pivots, &ier);
400 
401  /* Return 0 if the LU was complete; otherwise return 1 */
402  last_flag = (long int) ier;
403  if (ier > 0) return(1);
404  return(0);
405 }

407 /*
408  * idaLapackDenseSolve handles the solve operation for the dense linear solver
409  * by calling the dense backsolve routine.
410  */
411 static int idaLapackDenseSolve(IDAMem IDA_mem, N_Vector b, N_Vector weight,
412  N_Vector yC, N_Vector ypC, N_Vector fctC)
413 {
414  IDADlsMem idadls_mem;
415  realtype *bd, fact;
416  int ier, one = 1;
417  int intn;
418 
419  idadls_mem = (IDADlsMem) lmem;
420 
421  intn = (int) n;
422 
423  bd = N_VGetArrayPointer(b);
424 
425  dgetrs_f77("N", &intn, &one, JJ->data, &intn, pivots, bd, &intn, &ier, 1);
426  if (ier > 0) return(1);
427 
428  /* Scale the correction to account for change in cj. */
429  if (cjratio != ONE) {
430  fact = TWO/(ONE + cjratio);
431  dscal_f77(&intn, &fact, bd, &one);
432  }
433 
434  last_flag = IDADLS_SUCCESS;
435  return(0);
436 }

488  * updates counters, and calls the band LU factorization routine.

490 static int idaLapackBandSetup(IDAMem IDA_mem,
491  N_Vector yP, N_Vector ypP, N_Vector fctP,
492  N_Vector tmp1, N_Vector tmp2, N_Vector tmp3)

503  ldmat = JJ->ldim;
504 
505  /* Call Jacobian function */
506  nje++;

515  return(+1);

590 
591 int IDALapackDenseB(void *ida_mem, int which, int NeqB)
592 {
593  IDAMem IDA_mem;
594  IDAadjMem IDAADJ_mem;
595  IDABMem IDAB_mem;
596  IDADlsMemB idadlsB_mem;
597  void *ida_memB;
598  int flag;
599 
600  /* Is ida_mem allright? */
601  if (ida_mem == NULL) {
602  IDAProcessError(NULL, IDADLS_MEM_NULL, "IDASLAPACK", "IDALapackDenseB", MSGD_CAMEM_NULL);
603  return(IDADLS_MEM_NULL);
604  }
605  IDA_mem = (IDAMem) ida_mem;
606 
607  /* Is ASA initialized? */
608  if (IDA_mem->ida_adjMallocDone == FALSE) {
609  IDAProcessError(IDA_mem, IDADLS_NO_ADJ, "IDASLAPACK", "IDALapackDenseB", MSGD_NO_ADJ);
610  return(IDADLS_NO_ADJ);
611  }
612  IDAADJ_mem = IDA_mem->ida_adj_mem;
613 
614  /* Check the value of which */
615  if ( which >= IDAADJ_mem->ia_nbckpbs ) {
616  IDAProcessError(IDA_mem, IDADLS_ILL_INPUT, "IDASLAPACK", "IDALapackDenseB", MSGD_BAD_WHICH);
617  return(IDADLS_ILL_INPUT);
618  }
619 
620  /* Find the IDABMem entry in the linked list corresponding to 'which'. */
621  IDAB_mem = IDAADJ_mem->IDAB_mem;
622  while (IDAB_mem != NULL) {
623  if( which == IDAB_mem->ida_index ) break;
624  /* advance */
625  IDAB_mem = IDAB_mem->ida_next;
626  }
