72  * Addresses are passed to FKDJAC, using the macro DENSE_COL

79  DlsMat J, void *user_data, N_Vector vtemp1, N_Vector vtemp2)
80 {
81  realtype *uu_data, *fval_data, *jacdata, *v1_data, *v2_data;
82  int ier;
83 
84  /* Initialize all pointers to NULL */
85  uu_data = fval_data = jacdata = v1_data = v2_data = NULL;
86 
87  /* NOTE: The user-supplied routine should set ier to an
88  appropriate value, but we preset the value to zero
89  (meaning SUCCESS) so the user need only reset the
90  value if an error occurred */
91  ier = 0;
92
