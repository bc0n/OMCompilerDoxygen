190 int solveSystemWithTotalPivotSearchLS(int n, double* x, double* A, int* indRow, int* indCol, int *rank)
191 {
192  int i, k, j, l, m=n+1, nrsh=1, singular=0;
193  int pCol, pRow;
194  double hValue;
195  double hInt;
196  double absMax;
197  int r,s;
198  int permutation = 1;
199 
200  /* assume full rank of matrix [n x (n+1)] */
201  *rank = n;
202 
203  for (i=0; i<n; i++) {
204  indRow[i] = i;
205  }
206  for (i=0; i<m; i++) {
207  indCol[i] = i;
208  }
209 
210  for (i = 0; i < n; i++) {
211  getIndicesOfPivotElementLS(&n, &n, &i, A, indRow, indCol, &pRow, &pCol, &absMax);
212  /* this criteria should be evaluated and may be improved in future */
213  if (absMax<DBL_EPSILON) {
214  *rank = i;
215  warningStreamPrint(LOG_LS, 0, "Matrix singular!");
216  debugIntLS(LOG_LS,"rank = ", *rank);
217  break;
218  }
219  /* swap row indices */
220  if (pRow!=i) {
221  hInt = indRow[i];
222  indRow[i] = indRow[pRow];
223  indRow[pRow] = hInt;
224  }
225  /* swap column indices */
226  if (pCol!=i) {
227  hInt = indCol[i];
228  indCol[i] = indCol[pCol];
229  indCol[pCol] = hInt;
230  }
231 
232  /* Gauss elimination of row indRow[i] */
233  for (k=i+1; k<n; k++) {
234  hValue = -A[indRow[k] + indCol[i]*n]/A[indRow[i] + indCol[i]*n];
235  for (j=i+1; j<m; j++) {
236  A[indRow[k] + indCol[j]*n] = A[indRow[k] + indCol[j]*n] + hValue*A[indRow[i] + indCol[j]*n];
237  }
238  A[indRow[k] + indCol[i]*n] = 0;
239  }
240  }
241 
242  debugMatrixDoubleLS(LOG_LS_V,"LGS: matrix Ab manipulated",A, n, n+1);
243  /* solve even singular matrix !!! */
244  for (i=n-1;i>=0; i--) {
245  if (i>=*rank) {
246  /* this criteria should be evaluated and may be improved in future */
247  if (fabs(A[indRow[i] + n*n])>1e-12) {
248  warningStreamPrint(LOG_LS, 0, "under-determined linear system not solvable!");
249  return -1;
250  } else {
251  x[indCol[i]] = 0.0;
252  }
253  } else {
254  x[indCol[i]] = -A[indRow[i] + n*n];
255  for (j=n-1; j>i; j--) {
256  x[indCol[i]] = x[indCol[i]] - A[indRow[i] + indCol[j]*n]*x[indCol[j]];
257  }
258  x[indCol[i]]=x[indCol[i]]/A[indRow[i] + indCol[i]*n];
259  }
260  }
261  x[n]=1.0;
262  debugVectorDoubleLS(LOG_LS_V,"LGS: solution vector x",x, n+1);
263 
264  return 0;
265 }

321 int getAnalyticalJacobianTotalPivot(DATA* data, threadData_t *threadData, double* jac, int sysNumber)
322 {
323  int i,j,k,l,ii,currentSys = sysNumber;
324  LINEAR_SYSTEM_DATA* systemData = &(((DATA*)data)->simulationInfo->linearSystemData[currentSys]);
325 
326  const int index = systemData->jacobianIndex;
327  ANALYTIC_JACOBIAN* jacobian = &(data->simulationInfo->analyticJacobians[systemData->jacobianIndex]);
328 
329  memset(jac, 0, (systemData->size)*(systemData->size)*sizeof(double));
330 
331  for(i=0; i < jacobian->sparsePattern.maxColors; i++)
332  {
333  /* activate seed variable for the corresponding color */
334  for(ii=0; ii < jacobian->sizeCols; ii++)
335  if(jacobian->sparsePattern.colorCols[ii]-1 == i)
336  jacobian->seedVars[ii] = 1;
337 
338  ((systemData->analyticalJacobianColumn))(data, threadData, jacobian, systemData->parentJacobian);
339 
340  for(j = 0; j < jacobian->sizeCols; j++)
341  {
342  if(jacobian->seedVars[j] == 1)
343  {
344  ii = jacobian->sparsePattern.leadindex[j];
345  while(ii < jacobian->sparsePattern.leadindex[j+1]) {
346  l = jacobian->sparsePattern.index[ii];
347  k = j*jacobian->sizeRows + l;
348  jac[k] = jacobian->resultVars[l];
349  ii++;
350  }
351  }
352  /* de-activate seed variable for the corresponding color */
353  if(jacobian->sparsePattern.colorCols[j]-1 == i) {
354  jacobian->seedVars[j] = 0;
355  }
356  }
357 
358  }
359  return 0;
360 }

385 int solveTotalPivot(DATA *data, threadData_t *threadData, int sysNumber, double* aux_x)
386 {
387  void *dataAndThreadData[2] = {data, threadData};
388  int i, j;
389  LINEAR_SYSTEM_DATA* systemData = &(data->simulationInfo->linearSystemData[sysNumber]);
390  DATA_TOTALPIVOT* solverData = (DATA_TOTALPIVOT*) systemData->solverData[1];
391  int n = systemData->size, status;
392  double fdeps = 1e-8;
393  double xTol = 1e-8;
394  int eqSystemNumber = systemData->equationIndex;
395  int indexes[2] = {1,eqSystemNumber};
396  int rank;
397 
398  /* We are given the number of the linear system.
399  * We want to look it up among all equations. */
400  /* int eqSystemNumber = systemData->equationIndex; */
401  int success = 1;
402  double tmpJacEvalTime;
403 
404  infoStreamPrintWithEquationIndexes(LOG_LS, 0, indexes, "Start solving Linear System %d (size %d) at time %g with Total Pivot Solver",
405  eqSystemNumber, (int) systemData->size,
406  data->localData[0]->timeValue);
407 
408  debugVectorDoubleLS(LOG_LS_V,"SCALING",systemData->nominal,n);
409  debugVectorDoubleLS(LOG_LS_V,"Old VALUES",aux_x,n);
410 
411  rt_ext_tp_tick(&(solverData->timeClock));
412  if (0 == systemData->method){
413 
414  /* reset matrix A */
415  vecConstLS(n*n, 0.0, systemData->A);
416  /* update matrix A -> first n columns of matrix Ab*/
417  systemData->setA(data, threadData, systemData);
418  vecCopyLS(n*n, systemData->A, solverData->Ab);
419 
420  /* update vector b (rhs) -> -b is last column of matrix Ab*/
421  rt_ext_tp_tick(&(solverData->timeClock));
422  systemData->setb(data, threadData, systemData);
423  vecScalarMultLS(n, systemData->b, -1.0, solverData->Ab + n*n);
424 
425  } else {
426 
427  /* calculate jacobian -> first n columns of matrix Ab*/
428  if(systemData->jacobianIndex != -1){
429  getAnalyticalJacobianTotalPivot(data, threadData, solverData->Ab, sysNumber);
430  } else {
431  assertStreamPrint(threadData, 1, "jacobian function pointer is invalid" );
432  }
433  /* calculate vector b (rhs) -> -b is last column of matrix Ab */
434  wrapper_fvec_totalpivot(aux_x, solverData->Ab + n*n, dataAndThreadData, sysNumber);
435  }
436  tmpJacEvalTime = rt_ext_tp_tock(&(solverData->timeClock));
437  systemData->jacobianTime += tmpJacEvalTime;
438  infoStreamPrint(LOG_LS_V, 0, "### %f time to set Matrix A and vector b.", tmpJacEvalTime);
439  debugMatrixDoubleLS(LOG_LS_V,"LGS: matrix Ab",solverData->Ab, n, n+1);
440 
441  rt_ext_tp_tick(&(solverData->timeClock));
442  status = solveSystemWithTotalPivotSearchLS(n, solverData->x, solverData->Ab, solverData->indRow, solverData->indCol, &rank);
443  infoStreamPrint(LOG_LS_V, 0, "Solve System: %f", rt_ext_tp_tock(&(solverData->timeClock)));
444 
445  if (status != 0)
446  {
447  warningStreamPrint(LOG_STDOUT, 0, "Error solving linear system of equations (no. %d) at time %f.", (int)systemData->equationIndex, data->localData[0]->timeValue);
448  success = 0;
449  } else {
450 
451  debugVectorDoubleLS(LOG_LS_V, "SOLUTION:", solverData->x, n+1);
452  if (1 == systemData->method){
453  /* add the solution to old solution vector*/
454  vecAddLS(n, aux_x, solverData->x, aux_x);
455  wrapper_fvec_totalpivot(aux_x, solverData->b, dataAndThreadData, sysNumber);
456  } else {
457  /* take the solution */
458  vecCopyLS(n, solverData->x, aux_x);
459  }
460 
461  if (ACTIVE_STREAM(LOG_LS_V)){
462  infoStreamPrint(LOG_LS_V, 1, "Solution x:");
463  infoStreamPrint(LOG_LS_V, 0, "System %d numVars %d.", eqSystemNumber, modelInfoGetEquation(&data->modelData->modelDataXml,eqSystemNumber).numVar);
464  for(i=0; i<systemData->size; ++i)
465  {
466  infoStreamPrint(LOG_LS_V, 0, "[%d] %s = %g", i+1, modelInfoGetEquation(&data->modelData->modelDataXml,eqSystemNumber).vars[i], aux_x[i]);
467  }
468  messageClose(LOG_LS_V);
469  }
470  }
471  return success;
472 }
