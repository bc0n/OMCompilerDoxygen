86 int IDABBDPrecInit(void *ida_mem, long int Nlocal,
87  long int mudq, long int mldq,
88  long int mukeep, long int mlkeep,
89  realtype dq_rel_yy,
90  IDABBDLocalFn Gres, IDABBDCommFn Gcomm)
91 {
92  IDAMem IDA_mem;
93  IDASpilsMem idaspils_mem;
94  IBBDPrecData pdata;
95  N_Vector tempv4;
96  long int muk, mlk, storage_mu;
97  int flag;
98 
99  if (ida_mem == NULL) {
100  IDAProcessError(NULL, IDASPILS_MEM_NULL, "IDABBDPRE", "IDABBDPrecInit", MSGBBD_MEM_NULL);
101  return(IDASPILS_MEM_NULL);
102  }
103  IDA_mem = (IDAMem) ida_mem;
104 
105  /* Test if one of the SPILS linear solvers has been attached */
106  if (IDA_mem->ida_lmem == NULL) {
107  IDAProcessError(IDA_mem, IDASPILS_LMEM_NULL, "IDABBDPRE", "IDABBDPrecInit", MSGBBD_LMEM_NULL);
108  return(IDASPILS_LMEM_NULL);
109  }
110  idaspils_mem = (IDASpilsMem) IDA_mem->ida_lmem;
111 
112  /* Test if the NVECTOR package is compatible with BLOCK BAND preconditioner */
113  if(vec_tmpl->ops->nvgetarraypointer == NULL) {
114  IDAProcessError(IDA_mem, IDASPILS_ILL_INPUT, "IDABBDPRE", "IDABBDPrecInit", MSGBBD_BAD_NVECTOR);
115  return(IDASPILS_ILL_INPUT);
116  }
117 
118  /* Allocate data memory. */
119  pdata = NULL;
120  pdata = (IBBDPrecData) malloc(sizeof *pdata);
121  if (pdata == NULL) {
122  IDAProcessError(IDA_mem, IDASPILS_MEM_FAIL, "IDABBDPRE", "IDABBDPrecInit", MSGBBD_MEM_FAIL);
123  return(IDASPILS_MEM_FAIL);
124  }
125 
126  /* Set pointers to glocal and gcomm; load half-bandwidths. */
127  pdata->ida_mem = IDA_mem;
128  pdata->glocal = Gres;
129  pdata->gcomm = Gcomm;
130  pdata->mudq = SUNMIN(Nlocal-1, SUNMAX(0, mudq));
131  pdata->mldq = SUNMIN(Nlocal-1, SUNMAX(0, mldq));
132  muk = SUNMIN(Nlocal-1, SUNMAX(0, mukeep));
133  mlk = SUNMIN(Nlocal-1, SUNMAX(0, mlkeep));
134  pdata->mukeep = muk;
135  pdata->mlkeep = mlk;
136 
137  /* Set extended upper half-bandwidth for PP (required for pivoting). */
138  storage_mu = SUNMIN(Nlocal-1, muk+mlk);
139 
140  /* Allocate memory for preconditioner matrix. */
141  pdata->PP = NULL;
142  pdata->PP = NewBandMat(Nlocal, muk, mlk, storage_mu);
143  if (pdata->PP == NULL) {
144  free(pdata); pdata = NULL;
145  IDAProcessError(IDA_mem, IDASPILS_MEM_FAIL, "IDABBDPRE", "IDABBDPrecInit", MSGBBD_MEM_FAIL);
146  return(IDASPILS_MEM_FAIL);
147  }
148 
149  /* Allocate memory for lpivots. */
150  pdata->lpivots = NULL;
151  pdata->lpivots = NewLintArray(Nlocal);
152  if (pdata->lpivots == NULL) {
153  DestroyMat(pdata->PP);
154  free(pdata); pdata = NULL;
155  IDAProcessError(IDA_mem, IDASPILS_MEM_FAIL, "IDABBDPRE", "IDABBDPrecInit", MSGBBD_MEM_FAIL);
156  return(IDASPILS_MEM_FAIL);
157  }
158 
159  /* Allocate tempv4 for use by IBBDDQJac */
160  tempv4 = NULL;
161  tempv4 = N_VClone(vec_tmpl);
162  if (tempv4 == NULL){
163  DestroyMat(pdata->PP);
164  DestroyArray(pdata->lpivots);
165  free(pdata); pdata = NULL;
166  IDAProcessError(IDA_mem, IDASPILS_MEM_FAIL, "IDABBDPRE", "IDABBDPrecInit", MSGBBD_MEM_FAIL);
167  return(IDASPILS_MEM_FAIL);
168  }
169  pdata->tempv4 = tempv4;
170 
171  /* Set rel_yy based on input value dq_rel_yy (0 implies default). */
172  pdata->rel_yy = (dq_rel_yy > ZERO) ? dq_rel_yy : SUNRsqrt(uround);
173 
174  /* Store Nlocal to be used in IDABBDPrecSetup */
175  pdata->n_local = Nlocal;
176 
177  /* Set work space sizes and initialize nge. */
178  pdata->rpwsize = Nlocal*(mlk + storage_mu + 1);
179  pdata->ipwsize = Nlocal;
180  pdata->nge = 0;
181 
182  /* Overwrite the pdata field in the SPILS memory */
183  idaspils_mem->s_pdata = pdata;
184 
185  /* Attach the pfree function */
186  idaspils_mem->s_pfree = IDABBDPrecFree;
187 
188  /* Attach preconditioner solve and setup functions */
189  flag = IDASpilsSetPreconditioner(ida_mem, IDABBDPrecSetup, IDABBDPrecSolve);
190 
191  return(flag);
192 }

313 /*
314  * -----------------------------------------------------------------
315  * Function : IDABBDPrecSetup
316  * -----------------------------------------------------------------
317  * IDABBDPrecSetup generates a band-block-diagonal preconditioner
318  * matrix, where the local block (on this processor) is a band
319  * matrix. Each local block is computed by a difference quotient
320  * scheme via calls to the user-supplied routines glocal, gcomm.
321  * After generating the block in the band matrix PP, this routine
322  * does an LU factorization in place in PP.
323  *
324  * The IDABBDPrecSetup parameters used here are as follows:
325  *
326  * tt is the current value of the independent variable t.
327  *
328  * yy is the current value of the dependent variable vector,
329  * namely the predicted value of y(t).
330  *
331  * yp is the current value of the derivative vector y',
332  * namely the predicted value of y'(t).
333  *
334  * c_j is the scalar in the system Jacobian, proportional to 1/hh.
335  *
336  * bbd_data is the pointer to BBD memory set by IDABBDInit
337  *
338  * tmp1, tmp2, tmp3 are pointers to vectors of type
339  * N_Vector, used for temporary storage or
340  * work space.
341  *
342  * The arguments Neq, rr, res, uround, and nrePtr are not used.
343  *
344  * Return value:
345  * The value returned by this IDABBDPrecSetup function is a int
346  * flag indicating whether it was successful. This value is
347  * 0 if successful,
348  * > 0 for a recoverable error (step will be retried), or
349  * < 0 for a nonrecoverable error (step fails).
350  * -----------------------------------------------------------------
351  */
352 
353 static int IDABBDPrecSetup(realtype tt,
354  N_Vector yy, N_Vector yp, N_Vector rr,
355  realtype c_j, void *bbd_data,
356  N_Vector tempv1, N_Vector tempv2, N_Vector tempv3)
357 {
358  int retval;
359  long int ier;
360  IBBDPrecData pdata;
361  IDAMem IDA_mem;
362 
363  pdata =(IBBDPrecData) bbd_data;
364 
365  IDA_mem = (IDAMem) pdata->ida_mem;
366 
367  /* Call IBBDDQJac for a new Jacobian calculation and store in PP. */
368  SetToZero(PP);
369  retval = IBBDDQJac(pdata, tt, c_j, yy, yp,
370  tempv1, tempv2, tempv3, tempv4);
371  if (retval < 0) {
372  IDAProcessError(IDA_mem, -1, "IDABBDPRE", "IDABBDPrecSetup", MSGBBD_FUNC_FAILED);
373  return(-1);
374  }
375  if (retval > 0) {
376  return(+1);
377  }
378 
379  /* Do LU factorization of preconditioner block in place (in PP). */
380  ier = BandGBTRF(PP, lpivots);
381 
382  /* Return 0 if the LU was complete, or +1 otherwise. */
383  if (ier > 0) return(+1);
384  return(0);
385 }

457 /*
458  * -----------------------------------------------------------------
459  * IBBDDQJac
460  * -----------------------------------------------------------------
461  * This routine generates a banded difference quotient approximation
462  * to the local block of the Jacobian of G(t,y,y'). It assumes that
463  * a band matrix of type BandMat is stored column-wise, and that
464  * elements within each column are contiguous.
465  *
466  * All matrix elements are generated as difference quotients, by way
467  * of calls to the user routine glocal. By virtue of the band
468  * structure, the number of these calls is bandwidth + 1, where
469  * bandwidth = mldq + mudq + 1. But the band matrix kept has
470  * bandwidth = mlkeep + mukeep + 1. This routine also assumes that
471  * the local elements of a vector are stored contiguously.
472  *
473  * Return values are: 0 (success), > 0 (recoverable error),
474  * or < 0 (nonrecoverable error).
475  * -----------------------------------------------------------------
476  */
477 
478 static int IBBDDQJac(IBBDPrecData pdata, realtype tt, realtype cj,
479  N_Vector yy, N_Vector yp, N_Vector gref,
480  N_Vector ytemp, N_Vector yptemp, N_Vector gtemp)
481 {
482  IDAMem IDA_mem;
483  realtype inc, inc_inv;
484  int retval;
485  long int group, i, j, width, ngroups, i1, i2;
486  realtype *ydata, *ypdata, *ytempdata, *yptempdata, *grefdata, *gtempdata;
487  realtype *cnsdata = NULL, *ewtdata;
488  realtype *col_j, conj, yj, ypj, ewtj;
489 
490  IDA_mem = (IDAMem) pdata->ida_mem;
491 
492  /* Initialize ytemp and yptemp. */
493 
494  N_VScale(ONE, yy, ytemp);
495  N_VScale(ONE, yp, yptemp);
496 
497  /* Obtain pointers as required to the data array of vectors. */
498 
499  ydata = N_VGetArrayPointer(yy);
500  ypdata = N_VGetArrayPointer(yp);
501  gtempdata = N_VGetArrayPointer(gtemp);
502  ewtdata = N_VGetArrayPointer(ewt);
503  if (constraints != NULL)
504  cnsdata = N_VGetArrayPointer(constraints);
505  ytempdata = N_VGetArrayPointer(ytemp);
506  yptempdata= N_VGetArrayPointer(yptemp);
507  grefdata = N_VGetArrayPointer(gref);
508 
509  /* Call gcomm and glocal to get base value of G(t,y,y'). */
510 
511  if (gcomm != NULL) {
512  retval = gcomm(Nlocal, tt, yy, yp, user_data);
513  if (retval != 0) return(retval);
514  }
515 
516  retval = glocal(Nlocal, tt, yy, yp, gref, user_data);
517  nge++;
518  if (retval != 0) return(retval);
519 
520 
521  /* Set bandwidth and number of column groups for band differencing. */
522 
523  width = mldq + mudq + 1;
524  ngroups = SUNMIN(width, Nlocal);
525 
526  /* Loop over groups. */
527  for(group = 1; group <= ngroups; group++) {
528 
529  /* Loop over the components in this group. */
530  for(j = group-1; j < Nlocal; j += width) {
531  yj = ydata[j];
532  ypj = ypdata[j];
533  ewtj = ewtdata[j];
534 
535  /* Set increment inc to yj based on rel_yy*abs(yj), with
536  adjustments using ypj and ewtj if this is small, and a further
537  adjustment to give it the same sign as hh*ypj. */
538  inc = rel_yy*SUNMAX(SUNRabs(yj), SUNMAX( SUNRabs(hh*ypj), ONE/ewtj));
539  if (hh*ypj < ZERO) inc = -inc;
540  inc = (yj + inc) - yj;
541 
542  /* Adjust sign(inc) again if yj has an inequality constraint. */
543  if (constraints != NULL) {
544  conj = cnsdata[j];
545  if (SUNRabs(conj) == ONE) {if ((yj+inc)*conj < ZERO) inc = -inc;}
546  else if (SUNRabs(conj) == TWO) {if ((yj+inc)*conj <= ZERO) inc = -inc;}
547  }
548 
549  /* Increment yj and ypj. */
550  ytempdata[j] += inc;
551  yptempdata[j] += cj*inc;
552 
553  }
554 
555  /* Evaluate G with incremented y and yp arguments. */
556 
557  retval = glocal(Nlocal, tt, ytemp, yptemp, gtemp, user_data);
558  nge++;
559  if (retval != 0) return(retval);
560 
561  /* Loop over components of the group again; restore ytemp and yptemp. */
562  for(j = group-1; j < Nlocal; j += width) {
563  yj = ytempdata[j] = ydata[j];
564  ypj = yptempdata[j] = ypdata[j];
565  ewtj = ewtdata[j];
566 
567  /* Set increment inc as before .*/
568  inc = rel_yy*SUNMAX(SUNRabs(yj), SUNMAX( SUNRabs(hh*ypj), ONE/ewtj));
569  if (hh*ypj < ZERO) inc = -inc;
570  inc = (yj + inc) - yj;
571  if (constraints != NULL) {
572  conj = cnsdata[j];
573  if (SUNRabs(conj) == ONE) {if ((yj+inc)*conj < ZERO) inc = -inc;}
574  else if (SUNRabs(conj) == TWO) {if ((yj+inc)*conj <= ZERO) inc = -inc;}
575  }
576 
577  /* Form difference quotients and load into PP. */
578  inc_inv = ONE/inc;
579  col_j = BAND_COL(PP,j);
580  i1 = SUNMAX(0, j-mukeep);
581  i2 = SUNMIN(j+mlkeep, Nlocal-1);
582  for(i = i1; i <= i2; i++) BAND_COL_ELEM(col_j,i,j) =
583  inc_inv * (gtempdata[i] - grefdata[i]);
584  }
585  }
586 
587  return(0);
588 }
