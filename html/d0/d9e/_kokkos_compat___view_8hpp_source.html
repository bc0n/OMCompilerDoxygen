135  // Custom deallocator for Teuchos::ArrayRCP. It doesn't actually
136  // deallocate the ArrayRCP's data (which belongs to the View passed
137  // into Deallocator's constructor, not to the ArrayRCP). Instead,
138  // it just keeps a reference to the View. That way, the data won't
139  // go away until the View's reference count goes to zero.
140  template<class ViewType>
141  class Deallocator {
142  public:
143  typedef ViewType view_type;
144  typedef typename ViewType::value_type ptr_t;
145  typedef typename view_type::value_type value_type;
146 
147  // Constructor takes the View that owns the memory.
148  Deallocator (const ViewType& view__) : view_ (view__) {}
149 
150  // "Deallocation function" doesn't actually deallocate its input
151  // pointer; the View is responsible for deallocation of its
152  // memory.
153  void free (value_type*) {}
154  view_type view() { return view_;}
155  private:
156  view_type view_; // View that owns the memory
157 
158  };

166  // Create a "persisting view" from a Kokkos::View
167  template <typename ViewType>
168  Teuchos::ArrayRCP<typename ViewType::value_type>
169  persistingView(const ViewType& view) {
170  // mfh (05 Sep 2013) We use a custom deallocator that keeps the
171  // view, but otherwise does nothing. It doesn't deallocate the
172  // view; it just maintains the reference count for the lifetime
173  // of the deallocator object (and therefore, for the lifetime of
174  // the ArrayRCP). It seems weird that this is a nonowning
175  // ArrayRCP (the last argument is false). However, that's OK,
176  // because (a) the ArrayRCP (actually, the underlying
177  // RCPNodeTmpl; see Teuchos_RCPNode.hpp) keeps the deallocator
178  // object regardless of whether the ArrayRCP is owning, and (b)
179  // if we make this an owning ArrayRCP, the RCPNode tracking
180  // system (in a debug build: Teuchos_ENABLE_DEBUG:BOOL=ON) will
181  // report problems, because multiple owning ArrayRCPs have the
182  // same raw pointer but not the same underlying RCPNode. That
183  // would be a problem if those ArrayRCPs shared memory that was
184  // allocated (e.g.,) using new or malloc, but it's not a problem
185  // here, because the custom deallocator does not free anything.
186  // Nevertheless, it's better not to trouble the tracking system.
187  return Teuchos::arcp(view.ptr_on_device(), 0, view.capacity(),
188  deallocator(view), false);
189  }
