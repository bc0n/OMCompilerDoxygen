110 /******************** String HASH Functions ********************/
111 /*
112  * adrpo 2008-12-02
113  * http://www.cse.yorku.ca/~oz/hash.html
114  * hash functions which could be useful to replace System__hash:
115  */
116 /*** djb2 hash ***/
117 static inline unsigned long djb2_hash(const unsigned char *str)
118 {
119  unsigned long hash = 5381;
120  int c;
121  while (0 != (c = *str++)) hash = ((hash << 5) + hash) + c; /* hash * 33 + c */
122  return hash;
123 }
124 
125 /*** sdbm hash ***/
126 static inline unsigned long sdbm_hash(const unsigned char* str)
127 {
128  unsigned long hash = 0;
129  int c;
130  while (0 != (c = *str++)) hash = c + (hash << 6) + (hash << 16) - hash;
131  return hash;
132 }

184 modelica_metatype boxptr_substring(threadData_t *threadData, metamodelica_string_const str, modelica_metatype boxstart, modelica_metatype boxstop)
185 {
186  unsigned header = 0, nwords;
187  long start = MMC_UNTAGFIXNUM(boxstart) - 1;
188  long stop = MMC_UNTAGFIXNUM(boxstop) - 1;
189  long totalLen = MMC_STRLEN(str), len = stop-start+1;
190  struct mmc_string *res;
191  char *tmp;
192  modelica_metatype p;
193  /* Bad indexes */
194  if (start < 0 || start >= totalLen || stop < start || stop >= totalLen) {
195  MMC_THROW_INTERNAL();
196  }
197  header = MMC_STRINGHDR(len);
198  nwords = MMC_HDRSLOTS(header) + 1;
199  res = (struct mmc_string *) mmc_alloc_words_atomic(nwords);
200  res->header = header;
201  tmp = (char*) res->data;
202  memcpy(tmp, MMC_STRINGDATA(str) + start, len);
203  tmp[len] = '\0';
204  p = MMC_TAGPTR(res);
205  MMC_CHECK_STRING(p);
206  return p;

225 metamodelica_string stringAppendList(modelica_metatype lst)
226 {
227  /* fprintf(stderr, "stringAppendList(%s)\n", anyString(lst)); */
228  modelica_integer lstLen = 0, len = 0;
229  unsigned nbytes = 0, header = 0, nwords = 0;
230  modelica_metatype car = NULL, lstHead = NULL, lstTmp = NULL;
231  char *tmp = NULL;
232  struct mmc_string *res = NULL;
233  void *p = NULL;
234 
235  lstLen = 0;
236  nbytes = 0;
237  lstHead = lst;
238  lstTmp = lst;
239  while (!listEmpty(lstTmp)) {
240  MMC_CHECK_STRING(MMC_CAR(lstTmp));
241  nbytes += MMC_STRLEN(MMC_CAR(lstTmp));
242  /* fprintf(stderr, "stringAppendList: Has success reading input %d: %s\n", lstLen, MMC_STRINGDATA(MMC_CAR(lst))); */
243  lstTmp = MMC_CDR(lstTmp);
244  lstLen++;
245  }
246  if (nbytes == 0) return mmc_emptystring;
247  if (lstLen == 1) return MMC_CAR(lstHead);
248 
249  header = MMC_STRINGHDR(nbytes);
250  nwords = MMC_HDRSLOTS(header) + 1;
251  res = (struct mmc_string *) mmc_alloc_words_atomic(nwords);
252  res->header = header;
253  tmp = (char*) res->data;
254  nbytes = 0;
255  lstTmp = lstHead;
256  while (!listEmpty(lstTmp)) {
257  car = MMC_CAR(lstTmp);
258  len = MMC_STRLEN(car);
259  /* fprintf(stderr, "stringAppendList: %s %d %d\n", MMC_STRINGDATA(car), len, strlen(MMC_STRINGDATA(car))); */
260  /* Might be useful to check this when debugging. String literals are often done wrong :) */
261  MMC_DEBUG_ASSERT(len == strlen(MMC_STRINGDATA(car)));
262  memcpy(tmp+nbytes,MMC_STRINGDATA(car),len);
263  nbytes += len;
264  lstTmp = MMC_CDR(lstTmp);
265  }
266  tmp[nbytes] = '\0';
267  /* fprintf(stderr, "stringAppendList(%s)=>%s\n", anyString(lstHead), anyString(MMC_TAGPTR(res))); */
268  p = MMC_TAGPTR(res);
269  MMC_CHECK_STRING(p);
270  return p;
271 }

273 modelica_metatype stringDelimitList(modelica_metatype lst, metamodelica_string_const delimiter)
274 {
275  /* fprintf(stderr, "stringDelimitList(%s)\n", anyString(lst)); */
276  modelica_integer lstLen = 0, len = 0, lenDelimiter = 0;
277  unsigned nbytes = 0, header = 0, nwords = 0;
278  modelica_metatype car = NULL, lstHead = NULL, lstTmp = NULL;
279  char *tmp = 0, *delimiter_cstr = 0;
280  struct mmc_string *res = NULL;
281  void *p = NULL;
282 
283  lstLen = 0;
284  nbytes = 0;
285  lstHead = lst;
286  lstTmp = lst;
287  while (!listEmpty(lstTmp)) {
288  MMC_CHECK_STRING(MMC_CAR(lstTmp));
289  nbytes += MMC_STRLEN(MMC_CAR(lstTmp));
290  /* fprintf(stderr, "stringDelimitList: Has success reading input %d: %s\n", lstLen, MMC_STRINGDATA(MMC_CAR(lst))); */
291  lstTmp = MMC_CDR(lstTmp);
292  lstLen++;
293  }
294  if (nbytes == 0) return mmc_emptystring;
295  if (lstLen == 1) return MMC_CAR(lstHead);
296  lenDelimiter = MMC_STRLEN(delimiter);
297  nbytes += (lstLen-1)*lenDelimiter;
298  delimiter_cstr = MMC_STRINGDATA(delimiter);
299  MMC_DEBUG_ASSERT(lenDelimiter == strlen(delimiter_cstr));
300 
301  header = MMC_STRINGHDR(nbytes);
302  nwords = MMC_HDRSLOTS(header) + 1;
303  res = (struct mmc_string *) mmc_alloc_words_atomic(nwords);
304  res->header = header;
305  tmp = (char*) res->data;
306  nbytes = 0;
307  lstTmp = lstHead;
308  { /* Unrolled first element (not delimiter in front) */
309  car = MMC_CAR(lstTmp);
310  len = MMC_STRLEN(car);
311  MMC_DEBUG_ASSERT(len == strlen(MMC_STRINGDATA(car)));
312  memcpy(tmp+nbytes,MMC_STRINGDATA(car),len);
313  nbytes += len;
314  lstTmp = MMC_CDR(lstTmp);
315  }
316  while (!listEmpty(lstTmp)) {
317  memcpy(tmp+nbytes,delimiter_cstr,lenDelimiter);
318  nbytes += lenDelimiter;
319  car = MMC_CAR(lstTmp);
320  len = MMC_STRLEN(car);
321  /* fprintf(stderr, "stringDelimitList: %s %d %d\n", MMC_STRINGDATA(car), len, strlen(MMC_STRINGDATA(car))); */
322  /* Might be useful to check this when debugging. String literals are often done wrong :) */
323  MMC_DEBUG_ASSERT(len == strlen(MMC_STRINGDATA(car)));
324  memcpy(tmp+nbytes,MMC_STRINGDATA(car),len);
325  nbytes += len;
326  lstTmp = MMC_CDR(lstTmp);
327  }
328  tmp[nbytes] = '\0';
329  /* fprintf(stderr, "stringDelimitList(%s)=>%s\n", anyString(lstHead), anyString(MMC_TAGPTR(res))); */
330  p = MMC_TAGPTR(res);
331  MMC_CHECK_STRING(p);
332  return p;
333 }

425 modelica_metatype listAppend(modelica_metatype l1,modelica_metatype l2)
426 {
427  int length = 0, i = 0;
428  struct mmc_cons_struct *res = NULL;
429  struct mmc_cons_struct *p = NULL;
430  if (MMC_NILTEST(l2)) /* If l2 is empty, simply return l1; huge performance gain for some uses of listAppend */
431  return l1;
432  length = listLength(l1);
433  if (length == 0) /* We need to check for empty l1 */
434  return l2;
435  res = (struct mmc_cons_struct*)mmc_alloc_words( length * 3 /*(sizeof(struct mmc_cons_struct)/sizeof(void*))*/ ); /* Do one single big alloc. It's cheaper */
436  for (i=0; i<length-1; i++) { /* Write all except the last element... */
437  struct mmc_cons_struct *p = res+i;
438  p->header = MMC_STRUCTHDR(2, MMC_CONS_CTOR);
439  p->data[0] = MMC_CAR(l1);
440  p->data[1] = MMC_TAGPTR(res+i+1);
441  l1 = MMC_CDR(l1);
442  }
443  /* The last element is a bit special. It points to l2. */
444  p = res+length-1;
445  p->header = MMC_STRUCTHDR(2, MMC_CONS_CTOR);
446  p->data[0] = MMC_CAR(l1);
447  p->data[1] = l2;
448  return MMC_TAGPTR(res);
449 }
