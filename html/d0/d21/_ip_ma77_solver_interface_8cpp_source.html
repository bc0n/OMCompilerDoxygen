122  const std::string& prefix)
123  {
124  ma77_default_control(&control_);
125  control_.f_arrays = 1; // Use Fortran numbering (faster)
126  control_.bits=32;
127  // FIXME: HSL_MA77 should be updated to allow a matrix with new
128  // values to be refactorized after a -11 (singular) error.
129  //control_.action = 0; // false, should exit with error on singularity
130 
131  options.GetIntegerValue("ma77_print_level", control_.print_level, prefix);
132  options.GetIntegerValue("ma77_buffer_lpage", control_.buffer_lpage[0], prefix);
133  options.GetIntegerValue("ma77_buffer_lpage", control_.buffer_lpage[1], prefix);
134  options.GetIntegerValue("ma77_buffer_npage", control_.buffer_npage[0], prefix);
135  options.GetIntegerValue("ma77_buffer_npage", control_.buffer_npage[1], prefix);
136  int temp;
137  options.GetIntegerValue("ma77_file_size", temp, prefix);
138  control_.file_size = temp;
139  options.GetIntegerValue("ma77_maxstore", temp, prefix);
140  control_.maxstore = temp;
141  options.GetIntegerValue("ma77_nemin", control_.nemin, prefix);
142  options.GetNumericValue("ma77_small", control_.small, prefix);
143  options.GetNumericValue("ma77_static", control_.static_, prefix);
144  options.GetNumericValue("ma77_u", control_.u, prefix);
145  options.GetNumericValue("ma77_umax", umax_, prefix);
146  std::string order_method;
147  options.GetStringValue("ma77_order", order_method, prefix);
148  if (order_method == "metis") {
149  ordering_ = ORDER_METIS;
150  } else {
151  ordering_ = ORDER_AMD;
152  }
153  return true; // All is well
154  }
155 
156  /* Method for initializing internal structures. Here, ndim gives
157  * the number of rows and columns of the matrix, nonzeros give
158  * the number of nonzero elements, and ia and ja give the
159  * positions of the nonzero elements, given in the matrix format
160  * determined by MatrixFormat.
161  */
162  ESymSolverStatus Ma77SolverInterface::InitializeStructure(Index dim,
163  Index nonzeros, const Index* ia, const Index* ja)
164  {
165  struct ma77_info info;
166  struct mc68_control control68;
167  struct mc68_info info68;
168 
169  // Store size for later use
170  ndim_ = dim;
171 
172  if (HaveIpData()) {
173  IpData().TimingStats().LinearSystemSymbolicFactorization().Start();
174  }
175 
176  // mc68 requires a half matrix. A future version will support full
177  // matrix entry, and this code should be removed when it is available.
178  Index *ia_half = new Index[dim+1];
179  Index *ja_half = new Index[ia[dim]-1];
180  {
181  int k = 0;
182  for(int i=0; i<dim; i++) {
183  ia_half[i] = k+1;
184  for(int j=ia[i]-1; j<ia[i+1]-1; j++)
185  if(ja[j]-1 >= i)
186  ja_half[k++] = ja[j];
187  }
188  ia_half[dim] = k+1;
189  }
190 
191  // Determine an ordering
192  mc68_default_control(&control68);
193  control68.f_array_in = 1; // Use Fortran numbering (faster)
194  control68.f_array_out = 1; // Use Fortran numbering (faster)
195  Index *perm = new Index[dim];
196  if(ordering_ == ORDER_METIS) {
197  mc68_order(3, dim, ia_half, ja_half, perm, &control68, &info68); /* MeTiS */
198  if(info68.flag == -5) {
199  // MeTiS not available
200  ordering_ = ORDER_AMD;
201  } else if(info68.flag<0) {
202  delete[] ia_half;
203  delete[] ja_half;
204  return SYMSOLVER_FATAL_ERROR;
205  }
206  }
207  if(ordering_ == ORDER_AMD) {
208  mc68_order(1, dim, ia_half, ja_half, perm, &control68, &info68); /* AMD */
209  if(info68.flag<0) {
210  delete[] ia_half;
211  delete[] ja_half;
212  return SYMSOLVER_FATAL_ERROR;
213  }
214  }
215  delete[] ia_half;
216  delete[] ja_half;
217 
218  // Open files
219  ma77_open(ndim_, "ma77_int", "ma77_real", "ma77_work", "ma77_delay", &keep_,
220  &control_, &info);
221  if (info.flag < 0) return SYMSOLVER_FATAL_ERROR;
222 
223  // Store data into files
224  for(int i=0; i<dim; i++) {
225  ma77_input_vars(i+1, ia[i+1]-ia[i], &(ja[ia[i]-1]), &keep_,
226  &control_, &info);
227  if (info.flag < 0) return SYMSOLVER_FATAL_ERROR;
228  }
229 
230  // Perform analyse
231  ma77_analyse(perm, &keep_, &control_, &info);
232  delete[] perm; // Done with order
233 
234  if (HaveIpData())
235  IpData().TimingStats().LinearSystemSymbolicFactorization().End();
236 
237  // Setup memory for values
238  if (val_!=NULL) delete[] val_;
239  val_ = new double[nonzeros];
240 
241  if (info.flag>=0) {
242  return SYMSOLVER_SUCCESS;
243  }
244  else {
245  return SYMSOLVER_FATAL_ERROR;
246  }
247  }
248 
249  /* Solve operation for multiple right hand sides. Solves the
250  * linear system A * x = b with multiple right hand sides, where
251  * A is the symmtric indefinite matrix. Here, ia and ja give the
252  * positions of the values (in the required matrix data format).
253  * The actual values of the matrix will have been given to this
254  * object by copying them into the array provided by
255  * GetValuesArrayPtr. ia and ja are identical to the ones given
256  * to InitializeStructure. The flag new_matrix is set to true,
257  * if the values of the matrix has changed, and a refactorzation
258  * is required.
259  *
260  * The return code is SYMSOLV_SUCCESS if the factorization and
261  * solves were successful, SYMSOLV_SINGULAR if the linear system
262  * is singular, and SYMSOLV_WRONG_INERTIA if check_NegEVals is
263  * true and the number of negative eigenvalues in the matrix does
264  * not match numberOfNegEVals. If SYMSOLV_CALL_AGAIN is
265  * returned, then the calling function will request the pointer
266  * for the array for storing a again (with GetValuesPtr), write
267  * the values of the nonzero elements into it, and call this
268  * MultiSolve method again with the same right-hand sides. (This
269  * can be done, for example, if the linear solver realized it
270  * does not have sufficient memory and needs to redo the
271  * factorization; e.g., for MA27.)
272  *
273  * The number of right-hand sides is given by nrhs, the values of
274  * the right-hand sides are given in rhs_vals (one full right-hand
275  * side stored immediately after the other), and solutions are
276  * to be returned in the same array.
277  *
278  * check_NegEVals will not be chosen true, if ProvidesInertia()
279  * returns false.
280  */
281  ESymSolverStatus Ma77SolverInterface::MultiSolve(bool new_matrix,
282  const Index* ia, const Index* ja, Index nrhs, double* rhs_vals,
283  bool check_NegEVals, Index numberOfNegEVals)
284  {
285  struct ma77_info info;
286 
287  if (new_matrix || pivtol_changed_)
288  {
289  for(int i=0; i<ndim_; i++) {
290  ma77_input_reals(i+1, ia[i+1]-ia[i], &(val_[ia[i]-1]), &keep_,
291  &control_, &info);
292  if (info.flag < 0) return SYMSOLVER_FATAL_ERROR;
293  }
294 
295  if (HaveIpData()) {
296  IpData().TimingStats().LinearSystemFactorization().Start();
297  }
298  //ma77_factor(0, &keep_, &control_, &info, NULL);
299  ma77_factor_solve(0, &keep_, &control_, &info, NULL,
300  nrhs, ndim_, rhs_vals);
301  if (HaveIpData()) {
302  IpData().TimingStats().LinearSystemFactorization().End();
303  }
304  if (info.flag==4 || info.flag==-11) return SYMSOLVER_SINGULAR;
305  if (info.flag<0) return SYMSOLVER_FATAL_ERROR;
306  if (check_NegEVals && info.num_neg!=numberOfNegEVals)
307  return SYMSOLVER_WRONG_INERTIA;
308 
309  numneg_ = info.num_neg;
310  pivtol_changed_ = false;
311  }
312  else {
313  if (HaveIpData()) {
314  IpData().TimingStats().LinearSystemBackSolve().Start();
315  }
316  ma77_solve(0, nrhs, ndim_, rhs_vals, &keep_, &control_, &info, NULL);
317  if (HaveIpData()) {
318  IpData().TimingStats().LinearSystemBackSolve().End();
319  }
320  }
321 
322  return SYMSOLVER_SUCCESS;
323  }

325  bool Ma77SolverInterface::IncreaseQuality()
326  {
327  if (control_.u >= umax_) {
328  return false;
329  }
330  pivtol_changed_ = true;
331 
332  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
333  "Increasing pivot tolerance for HSL_MA77 from %7.2e ",
334  control_.u);
335  control_.u = Min(umax_, pow(control_.u,0.75));
336  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
337  "to %7.2e.\n",
338  control_.u);
339  return true;
340  }
