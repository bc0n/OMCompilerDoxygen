59 int zmq::tipc_address_t::resolve (const char *name)
60 {
61  unsigned int type = 0;
62  unsigned int lower = 0;
63  unsigned int upper = 0;
64  unsigned int z = 1, c = 0, n = 0;
65  char eof;
66  const char *domain;
67  const int res = sscanf (name, "{%u,%u,%u}", &type, &lower, &upper);
68 
69  /* Fetch optional domain suffix. */
70  if ((domain = strchr(name, '@'))) {
71  if (sscanf(domain, "@%u.%u.%u%c", &z, &c, &n, &eof) != 3)
72  return EINVAL;
73  }
74  if (res == 3)
75  goto nameseq;
76  else
77  if (res == 2 && type > TIPC_RESERVED_TYPES) {
78  address.family = AF_TIPC;
79  address.addrtype = TIPC_ADDR_NAME;
80  address.addr.name.name.type = type;
81  address.addr.name.name.instance = lower;
82  address.addr.name.domain = tipc_addr (z, c, n);
83  address.scope = 0;
84  return 0;
85  }
86  else
87  return EINVAL;
88 nameseq:
89  if (type < TIPC_RESERVED_TYPES || upper < lower)
90  return EINVAL;
91  address.family = AF_TIPC;
92  address.addrtype = TIPC_ADDR_NAMESEQ;
93  address.addr.nameseq.type = type;
94  address.addr.nameseq.lower = lower;
95  address.addr.nameseq.upper = upper;
96  address.scope = TIPC_ZONE_SCOPE;
97  return 0;
98 }
