39 // ================================================================== //
40 // This class is an example of matrix free format. //
41 // It represents a 3D Laplacian on a structured Cartesian grid; the //
42 // problem is discretized using a classical 7-point formula. //
43 // The computational domain is the unitary cube; Dirichlet boundary //
44 // conditions are applied on \partial \Omega. The Dirichlet boundary //
45 // nodes have been eliminated from the matrix. //
46 // //
47 // The nonzero elements are queried through a suitable getrow() //
48 // function, here given by ExtractMyRowCopy(). To specify the //
49 // communication pattern, we proceed as follows: //
50 // - an Epetra_CrsGraph is created, filled, the maps and import //
51 // objects are copied in the class; //
52 // - the Import() of Epetra_MultiVector is used in method Multiply(). //
53 // - recall that Aztec and ML require the elements returned by the //
54 // getrow() to be ordered so that external elements come after //
55 // local elements. //
56 // ================================================================== //

482  void CreateCoordinates()
483  {
484  x_coord.resize(NumMyRows());
485  y_coord.resize(NumMyRows());
486  z_coord.resize(NumMyRows());
487 
488  double lx = 1.0, ly = 1.0, lz = 1.0;
489 
490  double delta_x = lx / (nx() - 1);
491  double delta_y = ly / (ny() - 1);
492  double delta_z = lz / (nz() - 1);
493 
494  for (int i = 0 ; i < NumMyRows() ; i++)
495  {
496  int GlobalRow = RowMatrixRowMap().GID(i);
497  int ixy = GlobalRow % (nx() * ny());
498  int iz = (GlobalRow - ixy) / (nx() * ny());
499 
500  int ix = ixy % nx();
501  int iy = (ixy - ix) / ny();
502 
503  x_coord[i] = delta_x * ix;
504  y_coord[i] = delta_y * iy;
505  z_coord[i] = delta_z * iz;
506  }
507  }

531  void GetNeighboursCartesian2d(const int i, int& left, int& right,
532  int& lower, int& upper)
533  {
534  int ix, iy;
535  ix = i % nx();
536  iy = (i - ix) / nx();
537 
538  if (ix == 0)
539  left = -1;
540  else
541  left = i-1;
542  if (ix == nx() - 1)
543  right = -1;
544  else
545  right = i + 1;
546  if (iy == 0)
547  lower = -1;
548  else
549  lower = i - nx();
550  if (iy == ny() - 1)
551  upper = -1;
552  else
553  upper = i + nx();
554 
555  return;
556  }

559  void GetNeighboursCartesian3d(const int i, int& left, int& right,
560  int& lower, int& upper,
561  int& below, int& above )
562  {
563  int ixy, iz;
564  ixy = i % (nx() * ny());
565 
566  iz = (i - ixy)/(nx() * ny());
567 
568  if (iz == 0)
569  below = -1;
570  else
571  below = i - nx() * ny();
572  if (iz == nz() - 1)
573  above = -1;
574  else
575  above = i + nx() * ny();
576 
577  GetNeighboursCartesian2d(ixy, left, right, lower, upper);
578 
579  if (left != -1) left += iz * (nx() * ny());
580  if (right != -1) right += iz * (nx() * ny());
581  if (lower != -1) lower += iz * (nx() * ny());
582  if (upper != -1) upper += iz * (nx() * ny());
583 
584  return;
585  }

587  void CreateMap()
588  {
589  int modx = (nx() + (nx() % mx())) / mx();
590  int mody = (ny() + (ny() % my())) / my();
591  int modz = (nz() + (nz() % mz())) / mz();
592 
593  int MyPID = Comm().MyPID(), startx, starty, startz, endx, endy, endz;
594  int mxy = mx() * my();
595  int zpid = MyPID / mxy;
596  int xpid = (MyPID % mxy) % mx();
597  int ypid = (MyPID % mxy) / mx();
598 
599  startx = xpid * modx;
600  if ((xpid+1)*modx < nx()) endx = (xpid + 1) * modx;
601  else endx = nx();
602  starty = ypid*mody;
603  if ((ypid+1)*mody < ny()) endy = (ypid + 1) * mody;
604  else endy = ny();
605  startz = zpid*modz;
606  if ((zpid+1)*modz < nz()) endz = (zpid + 1) * modz;
607  else endz = nz();
608 
609  int NumMyElements = (endx - startx) * (endy - starty) * (endz - startz);
610  std::vector<int> MyGlobalElements(NumMyElements);
611  int count = 0;
612 
613  for (int i = startx ; i < endx ; ++i) {
614  for (int j = starty ; j < endy ; ++j) {
615  for (int k = startz ; k < endz ; ++k) {
616  MyGlobalElements[count++] = i + j * nx() + k * (nx() * ny());
617  }
618  }
619  }
620 
621  RowMap_ = new Epetra_Map (-1,NumMyElements,&MyGlobalElements[0],0,Comm());
622  }
