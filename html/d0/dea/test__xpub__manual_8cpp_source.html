36 
37  // Create a publisher

39  assert (pub);

41  assert (rc == 0);
42 
43  // set pub socket options
44  int manual = 1;
45  rc = zmq_setsockopt(pub, ZMQ_XPUB_MANUAL, &manual, 4);
46  assert (rc == 0);
47 
48  // Create a subscriber
49  void *sub = zmq_socket (ctx, ZMQ_XSUB);
50  assert (sub);
51  rc = zmq_connect (sub, "inproc://soname");
52  assert (rc == 0);
53

56  rc = zmq_send_const(sub, subscription, 2, 0);
57  assert (rc == 2);
58 
59  char buffer[2];
60

66 
67  // Subscribe socket for B instead

220

222  void *xpub_proxy = zmq_socket (ctx, ZMQ_XPUB);
223  assert (xpub_proxy);
224  assert (zmq_setsockopt (xpub_proxy, ZMQ_XPUB_MANUAL, &manual, 4) == 0);
225  assert (zmq_bind (xpub_proxy, "tcp://127.0.0.1:*") == 0);
226  len = MAX_SOCKET_STRING;
227  rc = zmq_getsockopt (xpub_proxy, ZMQ_LAST_ENDPOINT, my_endpoint_backend,
228  &len);
229  assert (rc == 0);
230 
231  // publisher
232  void *pub = zmq_socket (ctx, ZMQ_PUB);
233  assert (zmq_connect (pub, my_endpoint_frontend) == 0);
234 
235  // first subscriber subscribes
236  void *sub1 = zmq_socket (ctx, ZMQ_SUB);
237  assert (sub1);
238  assert (zmq_connect (sub1, my_endpoint_backend) == 0);
239  assert (zmq_setsockopt (sub1, ZMQ_SUBSCRIBE, topic, 1) == 0);
240

242  msleep (SETTLE_TIME);
243 
244  // proxy reroutes and confirms subscriptions
245  char sub_buff[2];

247  assert (sub_buff [0] == 1);
248  assert (sub_buff [1] == *topic);
249  assert (zmq_setsockopt (xpub_proxy, ZMQ_SUBSCRIBE, topic, 1) == 0);
250  assert (zmq_send (xsub_proxy, sub_buff, 2, 0) == 2);
251 
252  // second subscriber subscribes
253  void *sub2 = zmq_socket (ctx, ZMQ_SUB);
254  assert (sub2);
255  assert (zmq_connect (sub2, my_endpoint_backend) == 0);
256  assert (zmq_setsockopt (sub2, ZMQ_SUBSCRIBE, topic, 1) == 0);
257 
258  // wait

260

263  assert (sub_buff [0] == 1);

277

279  char topic_buff[1];
280  char data_buff[1];

282  assert (topic_buff [0] == *topic);

284  assert (data_buff [0] == *payload);

288  // wait

290 
291  // each subscriber should now get a message
292  assert (zmq_recv (sub2, topic_buff, 1, ZMQ_DONTWAIT) == 1);
293  assert (topic_buff [0] == *topic);
294  assert (zmq_recv (sub2, data_buff, 1, ZMQ_DONTWAIT) == 1);
295  assert (data_buff [0] == *payload);
296 
297  assert (zmq_recv (sub1, topic_buff, 1, ZMQ_DONTWAIT) == 1);
298  assert (topic_buff [0] == *topic);
299  assert (zmq_recv (sub1, data_buff, 1, ZMQ_DONTWAIT) == 1);
300  assert (data_buff [0] == *payload);
301 
302  // Disconnect both subscribers
303  assert (zmq_close (sub1) == 0);
304  assert (zmq_close (sub2) == 0);
305 
306  // wait
307  msleep (SETTLE_TIME);
308 
309  // unsubscribe messages are passed from proxy to publisher
310  assert (zmq_recv (xpub_proxy, sub_buff, 2, 0) == 2);
311  assert (sub_buff [0] == 0);
312  assert (sub_buff [1] == *topic);

314  assert (zmq_send (xsub_proxy, sub_buff, 2, 0) == 2);

319  assert (sub_buff [0] == 0);
320  assert (sub_buff [1] == *topic);
321  assert (zmq_setsockopt (xpub_proxy, ZMQ_UNSUBSCRIBE, topic, 1) == 0);
322  assert (zmq_send (xsub_proxy, sub_buff, 2, 0) == 2);
323 
324  // wait

326

329  assert (zmq_send (pub, payload, 1, 0) == 1);
330 
331  // wait

333 
334  // nothing should come to the proxy
335  assert (zmq_recv (xsub_proxy, topic_buff, 1, ZMQ_DONTWAIT) == -1);
336  assert (errno == EAGAIN);
337 
338  assert (zmq_close (pub) == 0);
339  assert (zmq_close (xpub_proxy) == 0);
340  assert (zmq_close (xsub_proxy) == 0);

346 int test_missing_subscriptions(void)
347 {
348  const char* topic1 = "1";
349  const char* topic2 = "2";
350  const char* payload = "X";
351 
352  size_t len = MAX_SOCKET_STRING;
353  char my_endpoint_backend[MAX_SOCKET_STRING];
354  char my_endpoint_frontend[MAX_SOCKET_STRING];

356  int manual = 1;
357 
358  void *ctx = zmq_ctx_new ();
359  assert (ctx);
360 
361  // proxy frontend
362  void *xsub_proxy = zmq_socket (ctx, ZMQ_XSUB);
363  assert (xsub_proxy);
364  assert (zmq_bind (xsub_proxy, "tcp://127.0.0.1:*") == 0);
365  int rc = zmq_getsockopt (xsub_proxy, ZMQ_LAST_ENDPOINT, my_endpoint_frontend,
366  &len);
367  assert (rc == 0);
368 
369  // proxy backend
370  void *xpub_proxy = zmq_socket (ctx, ZMQ_XPUB);
371  assert (xpub_proxy);

380  void *pub = zmq_socket (ctx, ZMQ_PUB);
381  assert (zmq_connect (pub, my_endpoint_frontend) == 0);

401

405  assert (buffer [0] == 1);
406  assert (buffer [1] == *topic1);
407  assert (zmq_setsockopt (xpub_proxy, ZMQ_SUBSCRIBE, topic1, 1) == 0);
408  assert (zmq_send (xsub_proxy, buffer, 2, 0) == 2);
409 
410  assert (zmq_recv (xpub_proxy, buffer, 2, ZMQ_DONTWAIT) == 2);
411  assert (buffer [0] == 1);
412  assert (buffer [1] == *topic2);
413  assert (zmq_setsockopt (xpub_proxy, ZMQ_SUBSCRIBE, topic2, 1) == 0);
414  assert (zmq_send (xsub_proxy, buffer, 2, 0) == 2);
415 
416  // wait
417  msleep (SETTLE_TIME);
418 
419  // let publisher send 2 msgs, each with its own topic
420  assert (zmq_send (pub, topic1, 1, ZMQ_SNDMORE) == 1);
421  assert (zmq_send (pub, payload, 1, 0) == 1);
422  assert (zmq_send (pub, topic2, 1, ZMQ_SNDMORE) == 1);
423  assert (zmq_send (pub, payload, 1, 0) == 1);
424

431  assert (zmq_recv (xsub_proxy, topic_buff, 1, ZMQ_DONTWAIT) == 1);
432  assert (topic_buff [0] == *topic1);
433  assert (zmq_recv (xsub_proxy, data_buff, 1, ZMQ_DONTWAIT) == 1);
434  assert (data_buff [0] == *payload);

439  assert (topic_buff [0] == *topic2);
