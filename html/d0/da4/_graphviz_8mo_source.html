104 protected function dumpNode "Dumps a node to a string."
105  input Node inNode;
106  output Ident outIdent;
107 algorithm
108  outIdent := match (inNode)
109  local
110  Label nm,typlbl,out,typ,lblstr;
111  Attributes newattr,attr;
112  Children children;
113  list<Label> lbl_1,lbl;
114 
115  case (NODE(type_ = typ,attributes = attr,children = children))
116  equation
117  nm = nodename(typ);
118  typlbl = makeLabel({typ});
119  newattr = ATTR("label",typlbl)::attr;
120  out = makeNode(nm, newattr);
121  print(out);
122  dumpChildren(nm, children);
123  then
124  nm;
125 
126  case (LNODE(type_ = typ,labelLst = lbl,attributes = attr,children = children))
127  equation
128  nm = nodename(typ);
129  lbl_1 = typ::lbl;
130  lblstr = makeLabel(lbl_1);
131  newattr = ATTR("label",lblstr)::attr;
132  out = makeNode(nm, newattr);
133  print(out);
134  dumpChildren(nm, children);
135  then
136  nm;
137  end match;
138 end dumpNode;
