59 int zmq::v1_decoder_t::one_byte_size_ready (unsigned char const*)
60 {
61  // First byte of size is read. If it is 0xff read 8-byte size.
62  // Otherwise allocate the buffer for message data and read the
63  // message data into it.
64  if (*tmpbuf == 0xff)
65  next_step (tmpbuf, 8, &v1_decoder_t::eight_byte_size_ready);
66  else {
67 
68  // There has to be at least one byte (the flags) in the message).
69  if (!*tmpbuf) {
70  errno = EPROTO;
71  return -1;
72  }
73 
74  if (maxmsgsize >= 0 && (int64_t) (*tmpbuf - 1) > maxmsgsize) {
75  errno = EMSGSIZE;
76  return -1;
77  }
78 
79  int rc = in_progress.close();
80  assert(rc == 0);
81  rc = in_progress.init_size (*tmpbuf - 1);
82  if (rc != 0) {
83  errno_assert (errno == ENOMEM);
84  rc = in_progress.init ();
85  errno_assert (rc == 0);
86  errno = ENOMEM;
87  return -1;
88  }
89 
90  next_step (tmpbuf, 1, &v1_decoder_t::flags_ready);
91  }
92  return 0;
93 }

95 int zmq::v1_decoder_t::eight_byte_size_ready (unsigned char const*)
96 {
97  // 8-byte payload length is read. Allocate the buffer
98  // for message body and read the message data into it.
99  const uint64_t payload_length = get_uint64 (tmpbuf);
100 
101  // There has to be at least one byte (the flags) in the message).
102  if (payload_length == 0) {
103  errno = EPROTO;
104  return -1;
105  }
106 
107  // Message size must not exceed the maximum allowed size.
108  if (maxmsgsize >= 0 && payload_length - 1 > (uint64_t) maxmsgsize) {
109  errno = EMSGSIZE;
110  return -1;
111  }
112 
113  // Message size must fit within range of size_t data type.
114  if (payload_length - 1 > std::numeric_limits <size_t>::max ()) {
115  errno = EMSGSIZE;
116  return -1;
117  }
118 
119  const size_t msg_size = static_cast <size_t> (payload_length - 1);
120 
121  int rc = in_progress.close();
122  assert(rc == 0);
123  rc = in_progress.init_size (msg_size);
124  if (rc != 0) {
125  errno_assert (errno == ENOMEM);
126  rc = in_progress.init ();
127  errno_assert (rc == 0);
128  errno = ENOMEM;
129  return -1;
130  }
131 
132  next_step (tmpbuf, 1, &v1_decoder_t::flags_ready);
133  return 0;
134 }
