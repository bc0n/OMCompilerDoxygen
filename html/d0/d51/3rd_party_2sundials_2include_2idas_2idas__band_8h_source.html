40 /*
41  * -----------------------------------------------------------------
42  * Function : IDABand
43  * -----------------------------------------------------------------
44  * A call to the IDABand function links the main integrator
45  * with the IDABAND linear solver module.
46  *
47  * ida_mem is the pointer to the integrator memory returned by
48  * IDACreate.
49  *
50  * mupper is the upper bandwidth of the banded Jacobian matrix.
51  *
52  * mlower is the lower bandwidth of the banded Jacobian matrix.
53  *
54  * The return values of IDABand are:
55  * IDADLS_SUCCESS = 0 if successful
56  * IDADLS_LMEM_FAIL = -1 if there was a memory allocation failure
57  * IDADLS_ILL_INPUT = -2 if the input was illegal or NVECTOR bad.
58  *
59  * NOTE: The band linear solver assumes a serial implementation
60  * of the NVECTOR package. Therefore, IDABand will first
61  * test for a compatible N_Vector internal representation
62  * by checking that the N_VGetArrayPointer function exists.
63  * -----------------------------------------------------------------
64  */
65 
66 SUNDIALS_EXPORT int IDABand(void *ida_mem,
67  long int Neq, long int mupper, long int mlower);
