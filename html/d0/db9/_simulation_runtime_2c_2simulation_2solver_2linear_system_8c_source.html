66 int initializeLinearSystems(DATA *data, threadData_t *threadData)
67 {
68  TRACE_PUSH
69  int i, nnz;
70  int size;
71  LINEAR_SYSTEM_DATA *linsys = data->simulationInfo->linearSystemData;
72 
73  infoStreamPrint(LOG_LS, 1, "initialize linear system solvers");
74  infoStreamPrint(LOG_LS, 0, "%ld linear systems", data->modelData->nLinearSystems);
75 
76  if (LSS_DEFAULT == data->simulationInfo->lssMethod) {
77 #ifdef WITH_UMFPACK
78  data->simulationInfo->lssMethod = LSS_KLU;
79 #elif !defined(OMC_MINIMAL_RUNTIME)
80  data->simulationInfo->lssMethod = LSS_LIS;
81 #endif
82  }
83 
84  for(i=0; i<data->modelData->nLinearSystems; ++i)
85  {
86  size = linsys[i].size;
87  nnz = linsys[i].nnz;
88 
89  linsys[i].totalTime = 0;
90  linsys[i].failed = 0;
91 
92  /* allocate system data */
93  linsys[i].b = (double*) malloc(size*sizeof(double));
94 
95  /* check if analytical jacobian is created */
96  if (1 == linsys[i].method)
97  {
98  ANALYTIC_JACOBIAN* jacobian = &(data->simulationInfo->analyticJacobians[linsys[i].jacobianIndex]);
99  if(linsys[i].jacobianIndex != -1)
100  {
101  assertStreamPrint(threadData, 0 != linsys[i].analyticalJacobianColumn, "jacobian function pointer is invalid" );
102  }
103  if(linsys[i].initialAnalyticalJacobian(data, threadData, jacobian))
104  {
105  linsys[i].jacobianIndex = -1;
106  throwStreamPrint(threadData, "Failed to initialize the jacobian for torn linear system %d.", (int)linsys[i].equationIndex);
107  }
108  nnz = jacobian->sparsePattern.numberOfNoneZeros;
109  linsys[i].nnz = nnz;
110  }
111 
112  if(nnz/(double)(size*size)<=linearSparseSolverMaxDensity && size>=linearSparseSolverMinSize)
113  {
114  linsys[i].useSparseSolver = 1;
115  infoStreamPrint(LOG_STDOUT, 0, "Using sparse solver for linear system %d,\nbecause density of %.3f remains under threshold of %.3f and size of %d exceeds threshold of %d.\nThe maximum density and the minimal system size for using sparse solvers can be specified\nusing the runtime flags '<-lssMaxDensity=value>' and '<-lssMinSize=value>'.", i, nnz/(double)(size*size), linearSparseSolverMaxDensity, size, linearSparseSolverMinSize);
116  }
117 
118  /* allocate more system data */
119  linsys[i].nominal = (double*) malloc(size*sizeof(double));
120  linsys[i].min = (double*) malloc(size*sizeof(double));
121  linsys[i].max = (double*) malloc(size*sizeof(double));
122 
123  linsys[i].initializeStaticLSData(data, threadData, &linsys[i]);
124 
125  /* allocate solver data */
126  /* the implementation of matrix A is solver-specific */
127  if(linsys[i].useSparseSolver == 1)
128  {
129  switch(data->simulationInfo->lssMethod)
130  {
131  #ifdef WITH_UMFPACK
132  case LSS_UMFPACK:
133  linsys[i].setAElement = setAElementUmfpack;
134  linsys[i].setBElement = setBElement;
135  allocateUmfPackData(size, size, nnz, linsys[i].solverData);
136  break;
137  case LSS_KLU:
138  linsys[i].setAElement = setAElementKlu;
139  linsys[i].setBElement = setBElement;
140  allocateKluData(size, size, nnz, linsys[i].solverData);
141  break;
142  #else
143  case LSS_KLU:
144  case LSS_UMFPACK:
145  throwStreamPrint(threadData, "OMC is compiled without UMFPACK, if you want use klu or umfpack please compile OMC with UMFPACK.");
146  break;
147  #endif
148  #if !defined(OMC_MINIMAL_RUNTIME)
149  case LSS_LIS:
150  linsys[i].setAElement = setAElementLis;
151  linsys[i].setBElement = setBElementLis;
152  allocateLisData(size, size, nnz, linsys[i].solverData);
153  break;
154  #else
155  case LSS_LIS_NOT_AVAILABLE:
156  throwStreamPrint(threadData, "OMC is compiled without sparse linear solver Lis.");
157  break;
158  #endif
159  #if defined(OMC_MINIMAL_RUNTIME) && !defined(WITH_UMFPACK)
160  case LSS_DEFAULT:
161  {
162  int indexes[2] = {1, linsys[i].equationIndex};
163  infoStreamPrintWithEquationIndexes(LOG_STDOUT, 0, indexes, "The simulation runtime does not have access to sparse solvers. Defaulting to a dense linear system solver instead.");
164  linsys[i].useSparseSolver = 0;
165  break;
166  }
167  #endif
168  default:
169  throwStreamPrint(threadData, "unrecognized sparse linear solver (%d)", data->simulationInfo->lssMethod);
170  }
171  }
172  if(linsys[i].useSparseSolver == 0) { /* Not an else-statement because there might not be a sparse linear solver available */
173  switch(data->simulationInfo->lsMethod)
174  {
175  case LS_LAPACK:
176  linsys[i].A = (double*) malloc(size*size*sizeof(double));
177  linsys[i].setAElement = setAElement;
178  linsys[i].setBElement = setBElement;
179  allocateLapackData(size, linsys[i].solverData);
180  break;
181 
182  #if !defined(OMC_MINIMAL_RUNTIME)
183  case LS_LIS:
184  linsys[i].setAElement = setAElementLis;
185  linsys[i].setBElement = setBElementLis;
186  allocateLisData(size, size, nnz, linsys[i].solverData);
187  break;
188  #endif
189  #ifdef WITH_UMFPACK
190  case LS_UMFPACK:
191  linsys[i].setAElement = setAElementUmfpack;
192  linsys[i].setBElement = setBElement;
193  allocateUmfPackData(size, size, nnz, linsys[i].solverData);
194  break;
195  case LS_KLU:
196  linsys[i].setAElement = setAElementKlu;
197  linsys[i].setBElement = setBElement;
198  allocateKluData(size, size, nnz, linsys[i].solverData);
199  break;
200  #else
201  case LS_UMFPACK:
202  throwStreamPrint(threadData, "OMC is compiled without UMFPACK, if you want use umfpack please compile OMC with UMFPACK.");
203  break;
204  #endif
205 
206  case LS_TOTALPIVOT:
207  linsys[i].A = (double*) malloc(size*size*sizeof(double));
208  linsys[i].setAElement = setAElement;
209  linsys[i].setBElement = setBElement;
210  allocateTotalPivotData(size, linsys[i].solverData);
211  break;
212 
213  case LS_DEFAULT:
214  linsys[i].A = (double*) malloc(size*size*sizeof(double));
215  linsys[i].setAElement = setAElement;
216  linsys[i].setBElement = setBElement;
217 
218  allocateLapackData(size, linsys[i].solverData);
219  allocateTotalPivotData(size, linsys[i].solverData);
220 
221  break;
222 
223  default:
224  throwStreamPrint(threadData, "unrecognized dense linear solver (%d)", data->simulationInfo->lsMethod);
225  }
226  }
227  }
228 
229  messageClose(LOG_LS);
230 
231  TRACE_POP
232  return 0;
233 }

396 int solve_linear_system(DATA *data, threadData_t *threadData, int sysNumber, double* aux_x)
397 {
398  TRACE_PUSH
399  int success;
400  int retVal;
401  int logLevel;
402  LINEAR_SYSTEM_DATA* linsys = &(data->simulationInfo->linearSystemData[sysNumber]);
403 
404  rt_ext_tp_tick(&(linsys->totalTimeClock));
405 
406  /* enable to avoid division by zero */
407  data->simulationInfo->noThrowDivZero = 1;
408 
409  if(linsys->useSparseSolver == 1)
410  {
411  switch(data->simulationInfo->lssMethod)
412  {
413  #if !defined(OMC_MINIMAL_RUNTIME)
414  case LSS_LIS:
415  success = solveLis(data, threadData, sysNumber, aux_x);
416  break;
417  #else
418  case LSS_LIS_NOT_AVAILABLE:
419  throwStreamPrint(threadData, "OMC is compiled without UMFPACK, if you want use umfpack please compile OMC with UMFPACK.");
420  break;
421  #endif
422  #ifdef WITH_UMFPACK
423  case LSS_KLU:
424  success = solveKlu(data, threadData, sysNumber, aux_x);
425  break;
426  case LSS_UMFPACK:
427  success = solveUmfPack(data, threadData, sysNumber, aux_x);
428  if (!success && linsys->strictTearingFunctionCall != NULL){
429  debugString(LOG_DT, "Solving the casual tearing set failed! Now the strict tearing set is used.");
430  success = linsys->strictTearingFunctionCall(data, threadData);
431  if (success) success=2;
432  }
433  break;
434  #else
435  case LSS_KLU:
436  case LSS_UMFPACK:
437  throwStreamPrint(threadData, "OMC is compiled without UMFPACK, if you want use umfpack please compile OMC with UMFPACK.");
438  break;
439  #endif
440  default:
441  throwStreamPrint(threadData, "unrecognized sparse linear solver (%d)", data->simulationInfo->lssMethod);
442  }
443  }
444 
445  else{
446  switch(data->simulationInfo->lsMethod)
447  {
448  case LS_LAPACK:
449  success = solveLapack(data, threadData, sysNumber, aux_x);
450  break;
451 
452  #if !defined(OMC_MINIMAL_RUNTIME)
453  case LS_LIS:
454  success = solveLis(data, threadData, sysNumber, aux_x);
455  break;
456  #endif
457  #ifdef WITH_UMFPACK
458  case LS_KLU:
459  success = solveKlu(data, threadData, sysNumber, aux_x);
460  break;
461  case LS_UMFPACK:
462  success = solveUmfPack(data, threadData, sysNumber, aux_x);
463  if (!success && linsys->strictTearingFunctionCall != NULL){
464  debugString(LOG_DT, "Solving the casual tearing set failed! Now the strict tearing set is used.");
465  success = linsys->strictTearingFunctionCall(data, threadData);
466  if (success) success=2;
467  }
468  break;
469  #else
470  case LS_UMFPACK:
471  throwStreamPrint(threadData, "OMC is compiled without UMFPACK, if you want use umfpack please compile OMC with UMFPACK.");
472  break;
473  #endif
474 
475  case LS_TOTALPIVOT:
476  success = solveTotalPivot(data, threadData, sysNumber, aux_x);
477  break;
478 
479  case LS_DEFAULT:
480  success = solveLapack(data, threadData, sysNumber, aux_x);
481 
482  /* check if solution process was successful, if not use alternative tearing set if available (dynamic tearing)*/
483  if (!success && linsys->strictTearingFunctionCall != NULL){
484  debugString(LOG_DT, "Solving the casual tearing set failed! Now the strict tearing set is used.");
485  success = linsys->strictTearingFunctionCall(data, threadData);
486  if (success){
487  success=2;
488  linsys->failed = 0;
489  }
490  else{
491  linsys->failed = 1;
492  }
493  }
494  else{
495  /* if there is no alternative tearing set, use fallback solver */
496  if (!success){
497  if (linsys->failed){
498  logLevel = LOG_LS;
499  } else {
500  logLevel = LOG_STDOUT;
501  }
502  warningStreamPrint(logLevel, 0, "The default linear solver fails, the fallback solver with total pivoting is started at time %f. That might raise performance issues, for more information use -lv LOG_LS.", data->localData[0]->timeValue);
503  success = solveTotalPivot(data, threadData, sysNumber, aux_x);
504  linsys->failed = 1;
505  }else{
506  linsys->failed = 0;
507  }
508  }
509  break;
510 
511  default:
512  throwStreamPrint(threadData, "unrecognized dense linear solver (%d)", data->simulationInfo->lsMethod);
513  }
514  }
515  linsys->solved = success;
516 
517  linsys->totalTime += rt_ext_tp_tock(&(linsys->totalTimeClock));
518  linsys->numberOfCall++;
519 
520  retVal = check_linear_solution(data, 1, sysNumber);
521 
522  TRACE_POP
523  return retVal;
524 }

566 int check_linear_solution(DATA *data, int printFailingSystems, int sysNumber)
567 {
568  TRACE_PUSH
569  LINEAR_SYSTEM_DATA* linsys = data->simulationInfo->linearSystemData;
570  long j, i = sysNumber;
571 
572  if(linsys[i].solved == 0)
573  {
574  int index = linsys[i].equationIndex, indexes[2] = {1,index};
575  if (!printFailingSystems)
576  {
577  TRACE_POP
578  return 1;
579  }
580  warningStreamPrintWithEquationIndexes(LOG_STDOUT, 1, indexes, "Solving linear system %d fails at time %g. For more information use -lv LOG_LS.", index, data->localData[0]->timeValue);
581 
582  for(j=0; j<modelInfoGetEquation(&data->modelData->modelDataXml, (linsys[i]).equationIndex).numVar; ++j) {
583  int done=0;
584  long k;
585  const MODEL_DATA *mData = data->modelData;
586  for(k=0; k<mData->nVariablesReal && !done; ++k)
587  {
588  if (!strcmp(mData->realVarsData[k].info.name, modelInfoGetEquation(&data->modelData->modelDataXml, (linsys[i]).equationIndex).vars[j]))
589  {
590  done = 1;
591  warningStreamPrint(LOG_LS, 0, "[%ld] Real %s(start=%g, nominal=%g)", j+1,
592  mData->realVarsData[k].info.name,
593  mData->realVarsData[k].attribute.start,
594  mData->realVarsData[k].attribute.nominal);
595  }
596  }
597  if (!done)
598  {
599  warningStreamPrint(LOG_LS, 0, "[%ld] Real %s(start=?, nominal=?)", j+1, modelInfoGetEquation(&data->modelData->modelDataXml, (linsys[i]).equationIndex).vars[j]);
600  }
601  }
602  messageCloseWarning(LOG_STDOUT);
603 
604  TRACE_POP
605  return 1;
606  }
607 
608  if(linsys[i].solved == 2)
609  {
610  linsys[i].solved = 1;
611  return 2;
612  }
613 
614  TRACE_POP
615  return 0;
616 }
