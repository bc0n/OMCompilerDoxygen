335 /*
336  * -----------------------------------------------------------------
337  * cvDlsDenseDQJac
338  * -----------------------------------------------------------------
339  * This routine generates a dense difference quotient approximation to
340  * the Jacobian of f(t,y). It assumes that a dense matrix of type
341  * DlsMat is stored column-wise, and that elements within each column
342  * are contiguous. The address of the jth column of J is obtained via
343  * the macro DENSE_COL and this pointer is associated with an N_Vector
344  * using the N_VGetArrayPointer/N_VSetArrayPointer functions.
345  * Finally, the actual computation of the jth column of the Jacobian is
346  * done with a call to N_VLinearSum.
347  * -----------------------------------------------------------------
348  */
349 
350 int cvDlsDenseDQJac(long int N, realtype t,
351  N_Vector y, N_Vector fy,
352  DlsMat Jac, void *data,
353  N_Vector tmp1, N_Vector tmp2, N_Vector tmp3)
354 {
355  realtype fnorm, minInc, inc, inc_inv, yjsaved, srur;
356  realtype *tmp2_data, *y_data, *ewt_data;
357  N_Vector ftemp, jthCol;
358  long int j;
359  int retval = 0;
360 
361  CVodeMem cv_mem;
362  CVDlsMem cvdls_mem;
363 
364  /* data points to cvode_mem */
365  cv_mem = (CVodeMem) data;
366  cvdls_mem = (CVDlsMem) lmem;
367 
368  /* Save pointer to the array in tmp2 */
369  tmp2_data = N_VGetArrayPointer(tmp2);
370 
371  /* Rename work vectors for readibility */
372  ftemp = tmp1;
373  jthCol = tmp2;
374 
375  /* Obtain pointers to the data for ewt, y */
376  ewt_data = N_VGetArrayPointer(ewt);
377  y_data = N_VGetArrayPointer(y);
378 
379  /* Set minimum increment based on uround and norm of f */
380  srur = SUNRsqrt(uround);
381  fnorm = N_VWrmsNorm(fy, ewt);
382  minInc = (fnorm != ZERO) ?
383  (MIN_INC_MULT * SUNRabs(h) * uround * N * fnorm) : ONE;
384 
385  for (j = 0; j < N; j++) {
386 
387  /* Generate the jth col of J(tn,y) */
388 
389  N_VSetArrayPointer(DENSE_COL(Jac,j), jthCol);
390 
391  yjsaved = y_data[j];
392  inc = SUNMAX(srur*SUNRabs(yjsaved), minInc/ewt_data[j]);
393  y_data[j] += inc;
394 
395  retval = f(t, y, ftemp, user_data);
396  nfeDQ++;
397  if (retval != 0) break;
398 
399  y_data[j] = yjsaved;
400 
401  inc_inv = ONE/inc;
402  N_VLinearSum(inc_inv, ftemp, -inc_inv, fy, jthCol);
403 
404  DENSE_COL(Jac,j) = N_VGetArrayPointer(jthCol);
405  }
406 
407  /* Restore original array pointer in tmp2 */
408  N_VSetArrayPointer(tmp2_data, tmp2);
409 
410  return(retval);
411 }

413 /*
414  * -----------------------------------------------------------------
415  * cvDlsBandDQJac
416  * -----------------------------------------------------------------
417  * This routine generates a banded difference quotient approximation to
418  * the Jacobian of f(t,y). It assumes that a band matrix of type
419  * DlsMat is stored column-wise, and that elements within each column
420  * are contiguous. This makes it possible to get the address of a column
421  * of J via the macro BAND_COL and to write a simple for loop to set
422  * each of the elements of a column in succession.
423  * -----------------------------------------------------------------
424  */
425 
426 int cvDlsBandDQJac(long int N, long int mupper, long int mlower,
427  realtype t, N_Vector y, N_Vector fy,
428  DlsMat Jac, void *data,
429  N_Vector tmp1, N_Vector tmp2, N_Vector tmp3)
430 {
431  N_Vector ftemp, ytemp;
432  realtype fnorm, minInc, inc, inc_inv, srur;
433  realtype *col_j, *ewt_data, *fy_data, *ftemp_data, *y_data, *ytemp_data;
434  long int group, i, j, width, ngroups, i1, i2;
435  int retval = 0;
436 
437  CVodeMem cv_mem;
438  CVDlsMem cvdls_mem;
439 
440  /* data points to cvode_mem */
441  cv_mem = (CVodeMem) data;
442  cvdls_mem = (CVDlsMem) lmem;
443 
444  /* Rename work vectors for use as temporary values of y and f */
445  ftemp = tmp1;
446  ytemp = tmp2;
447 
448  /* Obtain pointers to the data for ewt, fy, ftemp, y, ytemp */
449  ewt_data = N_VGetArrayPointer(ewt);
450  fy_data = N_VGetArrayPointer(fy);
451  ftemp_data = N_VGetArrayPointer(ftemp);
452  y_data = N_VGetArrayPointer(y);
453  ytemp_data = N_VGetArrayPointer(ytemp);
454 
455  /* Load ytemp with y = predicted y vector */
456  N_VScale(ONE, y, ytemp);
457 
458  /* Set minimum increment based on uround and norm of f */
459  srur = SUNRsqrt(uround);
460  fnorm = N_VWrmsNorm(fy, ewt);
461  minInc = (fnorm != ZERO) ?
462  (MIN_INC_MULT * SUNRabs(h) * uround * N * fnorm) : ONE;
463 
464  /* Set bandwidth and number of column groups for band differencing */
465  width = mlower + mupper + 1;
466  ngroups = SUNMIN(width, N);
467 
468  /* Loop over column groups. */
469  for (group=1; group <= ngroups; group++) {
470 
471  /* Increment all y_j in group */
472  for(j=group-1; j < N; j+=width) {
473  inc = SUNMAX(srur*SUNRabs(y_data[j]), minInc/ewt_data[j]);
474  ytemp_data[j] += inc;
475  }
476 
477  /* Evaluate f with incremented y */
478 
479  retval = f(tn, ytemp, ftemp, user_data);
480  nfeDQ++;
481  if (retval != 0) break;
482 
483  /* Restore ytemp, then form and load difference quotients */
484  for (j=group-1; j < N; j+=width) {
485  ytemp_data[j] = y_data[j];
486  col_j = BAND_COL(Jac,j);
487  inc = SUNMAX(srur*SUNRabs(y_data[j]), minInc/ewt_data[j]);
488  inc_inv = ONE/inc;
489  i1 = SUNMAX(0, j-mupper);
490  i2 = SUNMIN(j+mlower, N-1);
491  for (i=i1; i <= i2; i++)
492  BAND_COL_ELEM(col_j,i,j) = inc_inv * (ftemp_data[i] - fy_data[i]);
493  }
494  }
495 
496  return(retval);
497 }
