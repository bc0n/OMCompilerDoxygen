230 static void setAMatrix(modelica_integer* newEnable, modelica_integer nCandidates, modelica_integer nStates, VAR_INFO* Ainfo, VAR_INFO** states, VAR_INFO** statecandidates, DATA *data)
231 {
232  TRACE_PUSH
233  modelica_integer col;
234  modelica_integer row=0;
235  /* clear old values */
236  unsigned int aid = Ainfo->id - data->modelData->integerVarsData[0].info.id;
237  modelica_integer *A = &(data->localData[0]->integerVars[aid]);
238  memset(A, 0, nCandidates*nStates*sizeof(modelica_integer));
239 
240  for(col=0; col<nCandidates; col++)
241  {
242  if(newEnable[col]==2)
243  {
244  unsigned int firstrealid = data->modelData->realVarsData[0].info.id;
245  unsigned int id = statecandidates[col]->id-firstrealid;
246  unsigned int sid = states[row]->id-firstrealid;
247  infoStreamPrint(LOG_DSS, 0, "select %s", statecandidates[col]->name);
248  /* set A[row, col] */
249  A[row*nCandidates + col] = 1;
250  /* reinit state */
251  data->localData[0]->realVars[sid] = data->localData[0]->realVars[id];
252  row++;
253  }
254  }
255  TRACE_POP
256 }

273 static int comparePivot(modelica_integer *oldPivot, modelica_integer *newPivot, modelica_integer nCandidates, modelica_integer nDummyStates, modelica_integer nStates, VAR_INFO* A, VAR_INFO** states, VAR_INFO** statecandidates, DATA *data, int switchStates)
274 {
275  TRACE_PUSH
276  modelica_integer i;
277  int ret = 0;
278  modelica_integer* oldEnable = (modelica_integer*) calloc(nCandidates, sizeof(modelica_integer));
279  modelica_integer* newEnable = (modelica_integer*) calloc(nCandidates, sizeof(modelica_integer));
280 
281  for(i=0; i<nCandidates; i++)
282  {
283  modelica_integer entry = (i < nDummyStates) ? 1: 2;
284  newEnable[ newPivot[i] ] = entry;
285  oldEnable[ oldPivot[i] ] = entry;
286  }
287 
288  for(i=0; i<nCandidates; i++)
289  {
290  if(newEnable[i] != oldEnable[i])
291  {
292  if(switchStates)
293  {
294  infoStreamPrint(LOG_DSS, 1, "select new states at time %f", data->localData[0]->timeValue);
295  setAMatrix(newEnable, nCandidates, nStates, A, states, statecandidates, data);
296  messageClose(LOG_DSS);
297  }
298  ret = -1;
299  break;
300  }
301  }
302 
303  free(oldEnable);
304  free(newEnable);
305 
306  TRACE_POP
307  return ret;
308 }

359 int stateSelection(DATA *data, threadData_t *threadData, char reportError, int switchStates)
360 {
361  TRACE_PUSH
362  long i=0;
363  long j=0;
364  int globalres=0;
365  long k=0;
366  long l=0;
367 
368  /* go through all the state sets */
369  for(i=0; i<data->modelData->nStateSets; i++)
370  {
371  int res=0;
372  STATE_SET_DATA *set = &(data->simulationInfo->stateSetData[i]);
373  modelica_integer* oldColPivot = (modelica_integer*) malloc(set->nCandidates * sizeof(modelica_integer));
374  modelica_integer* oldRowPivot = (modelica_integer*) malloc(set->nDummyStates * sizeof(modelica_integer));
375 
376  /* debug */
377  if(ACTIVE_STREAM(LOG_DSS))
378  {
379  infoStreamPrint(LOG_DSS, 1, "StateSelection Set %ld at time = %f", i, data->localData[0]->timeValue);
380  printStateSelectionInfo(data, set);
381  messageClose(LOG_DSS);
382  }
383  /* generate jacobian, stored in set->J */
384  getAnalyticalJacobianSet(data, threadData, i);
385 
386  /* call pivoting function to select the states */
387  memcpy(oldColPivot, set->colPivot, set->nCandidates*sizeof(modelica_integer));
388  memcpy(oldRowPivot, set->rowPivot, set->nDummyStates*sizeof(modelica_integer));
389  if((pivot(set->J, set->nDummyStates, set->nCandidates, set->rowPivot, set->colPivot) != 0) && reportError)
390  {
391  /* error, report the matrix and the time */
392  char *buffer = (char*)malloc(sizeof(char)*data->simulationInfo->analyticJacobians[set->jacobianIndex].sizeCols*10);
393 
394  warningStreamPrint(LOG_DSS, 1, "jacobian %dx%d [id: %ld]", data->simulationInfo->analyticJacobians[set->jacobianIndex].sizeRows, data->simulationInfo->analyticJacobians[set->jacobianIndex].sizeCols, set->jacobianIndex);
395  for(i=0; i < data->simulationInfo->analyticJacobians[set->jacobianIndex].sizeRows; i++)
396  {
397  buffer[0] = 0;
398  for(j=0; j < data->simulationInfo->analyticJacobians[set->jacobianIndex].sizeCols; j++)
399  sprintf(buffer, "%s%.5e ", buffer, set->J[i*data->simulationInfo->analyticJacobians[set->jacobianIndex].sizeCols+j]);
400  warningStreamPrint(LOG_DSS, 0, "%s", buffer);
401  }
402  free(buffer);
403 
404  for(i=0; i<set->nCandidates; i++)
405  warningStreamPrint(LOG_DSS, 0, "%s", set->statescandidates[i]->name);
406  messageClose(LOG_DSS);
407 
408  throwStreamPrint(threadData, "Error, singular Jacobian for dynamic state selection at time %f\nUse -lv LOG_DSS_JAC to get the Jacobian", data->localData[0]->timeValue);
409  }
410  /* if we have a new set throw event for reinitialization
411  and set the A matrix for set.x=A*(states) */
412  res = comparePivot(oldColPivot, set->colPivot, set->nCandidates, set->nDummyStates, set->nStates, set->A, set->states, set->statescandidates, data, switchStates);
413  if(!switchStates)
414  {
415  memcpy(set->colPivot, oldColPivot, set->nCandidates*sizeof(modelica_integer));
416  memcpy(set->rowPivot, oldRowPivot, set->nDummyStates*sizeof(modelica_integer));
417  }
418  if(res)
419  globalres = 1;
420 
421  free(oldColPivot);
422  free(oldRowPivot);
423  }
424 
425  TRACE_POP
426  return globalres;
427 }
