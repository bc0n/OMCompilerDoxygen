53 zmq::session_base_t *zmq::session_base_t::create (class io_thread_t *io_thread_,
54  bool active_, class socket_base_t *socket_, const options_t &options_,
55  address_t *addr_)
56 {
57  session_base_t *s = NULL;
58  switch (options_.type) {
59  case ZMQ_REQ:
60  s = new (std::nothrow) req_session_t (io_thread_, active_,
61  socket_, options_, addr_);
62  break;
63  case ZMQ_RADIO:
64  s = new (std::nothrow) radio_session_t (io_thread_, active_,
65  socket_, options_, addr_);
66  break;
67  case ZMQ_DISH:
68  s = new (std::nothrow) dish_session_t (io_thread_, active_,
69  socket_, options_, addr_);
70  break;
71  case ZMQ_DEALER:
72  case ZMQ_REP:
73  case ZMQ_ROUTER:
74  case ZMQ_PUB:
75  case ZMQ_XPUB:
76  case ZMQ_SUB:
77  case ZMQ_XSUB:
78  case ZMQ_PUSH:
79  case ZMQ_PULL:
80  case ZMQ_PAIR:
81  case ZMQ_STREAM:
82  case ZMQ_SERVER:
83  case ZMQ_CLIENT:
84  case ZMQ_GATHER:
85  case ZMQ_SCATTER:
86  case ZMQ_DGRAM:
87  s = new (std::nothrow) session_base_t (io_thread_, active_,
88  socket_, options_, addr_);
89  break;
90  default:
91  errno = EINVAL;
92  return NULL;
93  }
94  alloc_assert (s);
95  return s;

209 void zmq::session_base_t::clean_pipes ()
210 {
211  zmq_assert (pipe != NULL);
212 
213  // Get rid of half-processed messages in the out pipe. Flush any
214  // unflushed messages upstream.
215  pipe->rollback ();
216  pipe->flush ();
217 
218  // Remove any half-read message from the in pipe.
219  while (incomplete_in) {
220  msg_t msg;
221  int rc = msg.init ();
222  errno_assert (rc == 0);
223  rc = pull_msg (&msg);
224  errno_assert (rc == 0);
225  rc = msg.close ();
226  errno_assert (rc == 0);
227  }
228 }

230 void zmq::session_base_t::pipe_terminated (pipe_t *pipe_)
231 {
232  // Drop the reference to the deallocated pipe if required.
233  zmq_assert (pipe_ == pipe
234  || pipe_ == zap_pipe
235  || terminating_pipes.count (pipe_) == 1);
236 
237  if (pipe_ == pipe) {
238  // If this is our current pipe, remove it
239  pipe = NULL;
240  if (has_linger_timer) {
241  cancel_timer (linger_timer_id);
242  has_linger_timer = false;
243  }
244  }
245  else
246  if (pipe_ == zap_pipe)
247  zap_pipe = NULL;
248  else
249  // Remove the pipe from the detached pipes set
250  terminating_pipes.erase (pipe_);
251 
252  if (!is_terminating () && options.raw_socket) {
253  if (engine) {
254  engine->terminate ();
255  engine = NULL;
256  }
257  terminate ();
258  }
259 
260  // If we are waiting for pending messages to be sent, at this point
261  // we are sure that there will be no more messages and we can proceed
262  // with termination safely.
263  if (pending && !pipe && !zap_pipe && terminating_pipes.empty ()) {
264  pending = false;
265  own_t::process_term (0);
266  }
267 }

318 int zmq::session_base_t::zap_connect ()
319 {
320  zmq_assert (zap_pipe == NULL);
321 
322  endpoint_t peer = find_endpoint ("inproc://zeromq.zap.01");
323  if (peer.socket == NULL) {
324  errno = ECONNREFUSED;
325  return -1;
326  }
327  if (peer.options.type != ZMQ_REP
328  && peer.options.type != ZMQ_ROUTER
329  && peer.options.type != ZMQ_SERVER) {
330  errno = ECONNREFUSED;
331  return -1;
332  }
333 
334  // Create a bi-directional pipe that will connect
335  // session with zap socket.
336  object_t *parents [2] = {this, peer.socket};
337  pipe_t *new_pipes [2] = {NULL, NULL};
338  int hwms [2] = {0, 0};
339  bool conflates [2] = {false, false};
340  int rc = pipepair (parents, new_pipes, hwms, conflates);
341  errno_assert (rc == 0);
342 
343  // Attach local end of the pipe to this socket object.
344  zap_pipe = new_pipes [0];
345  zap_pipe->set_nodelay ();
346  zap_pipe->set_event_sink (this);
347 
348  send_bind (peer.socket, new_pipes [1], false);
349 
350  // Send empty identity if required by the peer.
351  if (peer.options.recv_identity) {
352  msg_t id;
353  rc = id.init ();
354  errno_assert (rc == 0);
355  id.set_flags (msg_t::identity);
356  if (zap_pipe->write (&id))
357  zap_pipe->flush ();
358  else
359  return -1;
360  }
361 
362  return 0;
363 }

373 void zmq::session_base_t::process_attach (i_engine *engine_)
374 {
375  zmq_assert (engine_ != NULL);
376 
377  // Create the pipe if it does not exist yet.
378  if (!pipe && !is_terminating ()) {
379  object_t *parents [2] = {this, socket};
380  pipe_t *pipes [2] = {NULL, NULL};
381 
382  bool conflate = options.conflate &&
383  (options.type == ZMQ_DEALER ||
384  options.type == ZMQ_PULL ||
385  options.type == ZMQ_PUSH ||
386  options.type == ZMQ_PUB ||
387  options.type == ZMQ_SUB);
388 
389  int hwms [2] = {conflate? -1 : options.rcvhwm,
390  conflate? -1 : options.sndhwm};
391  bool conflates [2] = {conflate, conflate};
392  int rc = pipepair (parents, pipes, hwms, conflates);
393  errno_assert (rc == 0);
394 
395  // Plug the local end of the pipe.
396  pipes [0]->set_event_sink (this);
397 
398  // Remember the local end of the pipe.
399  zmq_assert (!pipe);
400  pipe = pipes [0];
401 
402  // Ask socket to plug into the remote end of the pipe.
403  send_bind (socket, pipes [1]);
404  }
405 
406  // Plug in the engine.
407  zmq_assert (!engine);
408  engine = engine_;
409  engine->plug (io_thread, this);
410 }

412 void zmq::session_base_t::engine_error (
413  zmq::stream_engine_t::error_reason_t reason)
414 {
415  // Engine is dead. Let's forget about it.
416  engine = NULL;
417 
418  // Remove any half-done messages from the pipes.
419  if (pipe)
420  clean_pipes ();
421 
422  zmq_assert (reason == stream_engine_t::connection_error
423  || reason == stream_engine_t::timeout_error
424  || reason == stream_engine_t::protocol_error);
425 
426  switch (reason) {
427  case stream_engine_t::timeout_error:
428  case stream_engine_t::connection_error:
429  if (active)
430  reconnect ();
431  else
432  terminate ();
433  break;
434  case stream_engine_t::protocol_error:
435  terminate ();
436  break;
437  }
438 
439  // Just in case there's only a delimiter in the pipe.
440  if (pipe)
441  pipe->check_read ();
442 
443  if (zap_pipe)
444  zap_pipe->check_read ();
445 }

447 void zmq::session_base_t::process_term (int linger_)
448 {
449  zmq_assert (!pending);
450 
451  // If the termination of the pipe happens before the term command is
452  // delivered there's nothing much to do. We can proceed with the
453  // standard termination immediately.
454  if (!pipe && !zap_pipe && terminating_pipes.empty ()) {
455  own_t::process_term (0);
456  return;
457  }
458 
459  pending = true;
460 
461  if (pipe != NULL) {
462  // If there's finite linger value, delay the termination.
463  // If linger is infinite (negative) we don't even have to set
464  // the timer.
465  if (linger_ > 0) {
466  zmq_assert (!has_linger_timer);
467  add_timer (linger_, linger_timer_id);
468  has_linger_timer = true;
469  }
470 
471  // Start pipe termination process. Delay the termination till all messages
472  // are processed in case the linger time is non-zero.
473  pipe->terminate (linger_ != 0);
474 
475  // TODO: Should this go into pipe_t::terminate ?
476  // In case there's no engine and there's only delimiter in the
477  // pipe it wouldn't be ever read. Thus we check for it explicitly.
478  if (!engine)
479  pipe->check_read ();
480  }
481 
482  if (zap_pipe != NULL)
483  zap_pipe->terminate (false);
484 }

498 void zmq::session_base_t::reconnect ()
499 {
500  // For delayed connect situations, terminate the pipe
501  // and reestablish later on
502  if (pipe && options.immediate == 1
503  && addr->protocol != "pgm" && addr->protocol != "epgm"
504  && addr->protocol != "norm" && addr->protocol != "udp") {
505  pipe->hiccup ();
506  pipe->terminate (false);
507  terminating_pipes.insert (pipe);
508  pipe = NULL;
509 
510  if (has_linger_timer) {
511  cancel_timer (linger_timer_id);
512  has_linger_timer = false;
513  }
514  }
515 
516  reset ();
517 
518  // Reconnect.
519  if (options.reconnect_ivl != -1)
520  start_connecting (true);
521 
522  // For subscriber sockets we hiccup the inbound pipe, which will cause
523  // the socket object to resend all the subscriptions.
524  if (pipe && (options.type == ZMQ_SUB || options.type == ZMQ_XSUB || options.type == ZMQ_DISH))
525  pipe->hiccup ();
526 }

528 void zmq::session_base_t::start_connecting (bool wait_)
529 {
530  zmq_assert (active);
531 
532  // Choose I/O thread to run connecter in. Given that we are already
533  // running in an I/O thread, there must be at least one available.
534  io_thread_t *io_thread = choose_io_thread (options.affinity);
535  zmq_assert (io_thread);
536 
537  // Create the connecter object.
538 
539  if (addr->protocol == "tcp") {
540  if (!options.socks_proxy_address.empty()) {
541  address_t *proxy_address = new (std::nothrow)
542  address_t ("tcp", options.socks_proxy_address, this->get_ctx ());
543  alloc_assert (proxy_address);
544  socks_connecter_t *connecter =
545  new (std::nothrow) socks_connecter_t (
546  io_thread, this, options, addr, proxy_address, wait_);
547  alloc_assert (connecter);
548  launch_child (connecter);
549  }
550  else {
551  tcp_connecter_t *connecter = new (std::nothrow)
552  tcp_connecter_t (io_thread, this, options, addr, wait_);
553  alloc_assert (connecter);
554  launch_child (connecter);
555  }
556  return;
557  }
558 
559 #if !defined ZMQ_HAVE_WINDOWS && !defined ZMQ_HAVE_OPENVMS
560  if (addr->protocol == "ipc") {
561  ipc_connecter_t *connecter = new (std::nothrow) ipc_connecter_t (
562  io_thread, this, options, addr, wait_);
563  alloc_assert (connecter);
564  launch_child (connecter);
565  return;
566  }
567 #endif
568 #if defined ZMQ_HAVE_TIPC
569  if (addr->protocol == "tipc") {
570  tipc_connecter_t *connecter = new (std::nothrow) tipc_connecter_t (
571  io_thread, this, options, addr, wait_);
572  alloc_assert (connecter);
573  launch_child (connecter);
574  return;
575  }
576 #endif
577 
578  if (addr->protocol == "udp") {
579  zmq_assert (options.type == ZMQ_DISH || options.type == ZMQ_RADIO || options.type == ZMQ_DGRAM);
580 
581  udp_engine_t* engine = new (std::nothrow) udp_engine_t (options);
582  alloc_assert (engine);
583 
584  bool recv = false;
585  bool send = false;
586 
587  if (options.type == ZMQ_RADIO) {
588  send = true;
589  recv = false;
590  }
591  else if (options.type == ZMQ_DISH) {
592  send = false;
593  recv = true;
594  }
595  else if (options.type == ZMQ_DGRAM) {
596  send = true;
597  recv = true;
598  }
599 
600  int rc = engine->init (addr, send, recv);
601  errno_assert (rc == 0);
602 
603  send_attach (this, engine);
604 
605  return;
606  }
607 
608 #ifdef ZMQ_HAVE_OPENPGM
609 
610  // Both PGM and EPGM transports are using the same infrastructure.
611  if (addr->protocol == "pgm" || addr->protocol == "epgm") {
612 
613  zmq_assert (options.type == ZMQ_PUB || options.type == ZMQ_XPUB
614  || options.type == ZMQ_SUB || options.type == ZMQ_XSUB);
615 
616  // For EPGM transport with UDP encapsulation of PGM is used.
617  bool const udp_encapsulation = addr->protocol == "epgm";
618 
619  // At this point we'll create message pipes to the session straight
620  // away. There's no point in delaying it as no concept of 'connect'
621  // exists with PGM anyway.
622  if (options.type == ZMQ_PUB || options.type == ZMQ_XPUB) {
623 
624  // PGM sender.
625  pgm_sender_t *pgm_sender = new (std::nothrow) pgm_sender_t (
626  io_thread, options);
627  alloc_assert (pgm_sender);
628 
629  int rc = pgm_sender->init (udp_encapsulation, addr->address.c_str ());
630  errno_assert (rc == 0);
631 
632  send_attach (this, pgm_sender);
633  }
634  else {
635 
636  // PGM receiver.
637  pgm_receiver_t *pgm_receiver = new (std::nothrow) pgm_receiver_t (
638  io_thread, options);
639  alloc_assert (pgm_receiver);
640 
641  int rc = pgm_receiver->init (udp_encapsulation, addr->address.c_str ());
642  errno_assert (rc == 0);
643 
644  send_attach (this, pgm_receiver);
645  }
646 
647  return;
648  }
649 #endif
650 
651 #ifdef ZMQ_HAVE_NORM
652  if (addr->protocol == "norm") {
653  // At this point we'll create message pipes to the session straight
654  // away. There's no point in delaying it as no concept of 'connect'
655  // exists with NORM anyway.
656  if (options.type == ZMQ_PUB || options.type == ZMQ_XPUB) {
657 
658  // NORM sender.
659  norm_engine_t* norm_sender = new (std::nothrow) norm_engine_t(io_thread, options);
660  alloc_assert (norm_sender);
661 
662  int rc = norm_sender->init (addr->address.c_str (), true, false);
663  errno_assert (rc == 0);
664 
665  send_attach (this, norm_sender);
666  }
667  else { // ZMQ_SUB or ZMQ_XSUB

671  alloc_assert (norm_receiver);
672 
673  int rc = norm_receiver->init (addr->address.c_str (), false, true);
674  errno_assert (rc == 0);
675

677  }
678  return;
679  }
680 #endif // ZMQ_HAVE_NORM
681 
682 #if defined ZMQ_HAVE_VMCI
683  if (addr->protocol == "vmci") {
684  vmci_connecter_t *connecter = new (std::nothrow) vmci_connecter_t (
685  io_thread, this, options, addr, wait_);
686  alloc_assert (connecter);
687  launch_child (connecter);
688  return;

691 
692  zmq_assert (false);
