206 int CVSlsSetSparseJacFnB(void *cvode_mem, int which, CVSlsSparseJacFnB jacB)
207 {
208  CVodeMem cv_mem;
209  CVadjMem ca_mem;
210  CVodeBMem cvB_mem;
211  CVSlsMemB cvslsB_mem;
212  void *cvodeB_mem;
213  int flag;
214 
215  /* Check if cvode_mem exists */
216  if (cvode_mem == NULL) {
217  cvProcessError(NULL, CVSLS_MEM_NULL, "CVSSLS", "CVSlsSetSparseJacFnB", MSGSP_CVMEM_NULL);
218  return(CVSLS_MEM_NULL);
219  }
220  cv_mem = (CVodeMem) cvode_mem;
221 
222  /* Was ASA initialized? */
223  if (cv_mem->cv_adjMallocDone == FALSE) {
224  cvProcessError(cv_mem, CVSLS_NO_ADJ, "CVSSLS", "CVSlsSetSparseJacFnB", MSGSP_NO_ADJ);
225  return(CVSLS_NO_ADJ);
226  }
227  ca_mem = cv_mem->cv_adj_mem;
228 
229  /* Check which */
230  if ( which >= ca_mem->ca_nbckpbs ) {
231  cvProcessError(cv_mem, CVSLS_ILL_INPUT, "CVSSLS", "CVSlsSetSparseJacFnB", MSGSP_BAD_WHICH);
232  return(CVSLS_ILL_INPUT);
233  }
234 
235  /* Find the CVodeBMem entry in the linked list corresponding to which */
236  cvB_mem = ca_mem->cvB_mem;
237  while (cvB_mem != NULL) {
238  if ( which == cvB_mem->cv_index ) break;
239  cvB_mem = cvB_mem->cv_next;
240  }
241 
242  cvodeB_mem = (void *) (cvB_mem->cv_mem);
243 
244  if (cvB_mem->cv_lmem == NULL) {
245  cvProcessError(cv_mem, CVSLS_LMEMB_NULL, "CVSSLS", "CVSlsSetSparseJacFnB", MSGSP_LMEMB_NULL);
246  return(CVSLS_LMEMB_NULL);
247  }
248  cvslsB_mem = (CVSlsMemB) (cvB_mem->cv_lmem);
249 
250  cvslsB_mem->s_djacB = jacB;
251 
252  if (jacB != NULL) {
253  flag = CVSlsSetSparseJacFn(cvodeB_mem, cvSlsSparseJacBWrapper);
254  } else {
255  flag = CVSlsSetSparseJacFn(cvodeB_mem, NULL);
256  }
257 
258  return(flag);
259 }

330 static int cvSlsSparseJacBWrapper(realtype t,
331  N_Vector yB, N_Vector fyB,
332  SlsMat JB, void *cvode_mem,
333  N_Vector tmp1B, N_Vector tmp2B, N_Vector tmp3B)
334 {
335  CVodeMem cv_mem;
336  CVadjMem ca_mem;
337  CVodeBMem cvB_mem;
338  CVSlsMemB cvslsB_mem;
339  int retval, flag;
340 
341  cv_mem = (CVodeMem) cvode_mem;
342 
343  ca_mem = cv_mem->cv_adj_mem;
344 
345  cvB_mem = ca_mem->ca_bckpbCrt;
346 
347  cvslsB_mem = (CVSlsMemB) (cvB_mem->cv_lmem);
348 
349  /* Forward solution from interpolation */
350  flag = ca_mem->ca_IMget(cv_mem, t, ca_mem->ca_ytmp, NULL);
351  if (flag != CV_SUCCESS) {
352  cvProcessError(cv_mem, -1, "CVSSLS", "cvSlsSparseJacBWrapper", MSGSP_BAD_TINTERP);
353  return(-1);
354  }
355 
356  /* Call user's adjoint dense djacB routine (of type CVSlsSparseJacFnB) */
357  retval = cvslsB_mem->s_djacB(t, ca_mem->ca_ytmp, yB, fyB, JB,
358  cvB_mem->cv_user_data,
359  tmp1B, tmp2B, tmp3B);
360 
361  return(retval);
362 }
