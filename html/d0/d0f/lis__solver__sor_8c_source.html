78 LIS_INT lis_sor_malloc_work(LIS_SOLVER solver)
79 {
80  LIS_VECTOR *work;
81  LIS_INT i,j,worklen,err;
82 
83  LIS_DEBUG_FUNC_IN;
84 
85  worklen = NWORK;
86  work = (LIS_VECTOR *)lis_malloc( worklen*sizeof(LIS_VECTOR),"lis_sor_malloc_work::work" );
87  if( work==NULL )
88  {
89  LIS_SETERR_MEM(worklen*sizeof(LIS_VECTOR));
90  return LIS_ERR_OUT_OF_MEMORY;
91  }
92  if( solver->precision==LIS_PRECISION_DEFAULT )
93  {
94  for(i=0;i<worklen;i++)
95  {
96  err = lis_vector_duplicate(solver->A,&work[i]);
97  if( err ) break;
98  }
99  }
100  else
101  {
102  for(i=0;i<worklen;i++)
103  {
104  err = lis_vector_duplicateex(LIS_PRECISION_QUAD,solver->A,&work[i]);
105  if( err ) break;
106  }
107  }
108  if( i<worklen )
109  {
110  for(j=0;j<i;j++) lis_vector_destroy(work[j]);
111  lis_free(work);
112  return err;
113  }
114  solver->worklen = worklen;
115  solver->work = work;
116 
117  LIS_DEBUG_FUNC_OUT;
118  return LIS_SUCCESS;
119 }

123 LIS_INT lis_sor(LIS_SOLVER solver)
124 {
125  LIS_MATRIX A;
126  LIS_PRECON M;
127  LIS_VECTOR b,x;
128  LIS_VECTOR r,t,s;
129  LIS_SCALAR w;
130  LIS_REAL bnrm2, nrm2, tol;
131  LIS_INT iter,maxiter,n,output,conv;
132  double times,ptimes;
133 
134  LIS_INT err;
135 
136  LIS_DEBUG_FUNC_IN;
137 
138  A = solver->A;
139  M = solver->precon;
140  b = solver->b;
141  x = solver->x;
142  n = A->n;
143  maxiter = solver->options[LIS_OPTIONS_MAXITER];
144  output = solver->options[LIS_OPTIONS_OUTPUT];
145  conv = solver->options[LIS_OPTIONS_CONV_COND];
146  tol = solver->params[LIS_PARAMS_RESID-LIS_OPTIONS_LEN];
147  w = 1.0 / solver->params[LIS_PARAMS_W-LIS_OPTIONS_LEN];
148  ptimes = 0.0;
149 
150  r = solver->work[0];
151  t = solver->work[1];
152  s = solver->work[2];
153 
154  lis_vector_nrm2(b,&bnrm2);
155  bnrm2 = 1.0 / bnrm2;
156 
157  err = lis_matrix_split(A);
158  if( err ) return err;
159  if( A->use_wd!=LIS_SOLVER_SOR )
160  {
161  if( !A->WD )
162  {
163  err = lis_matrix_diag_duplicate(A->D,&A->WD);
164  if( err ) return err;
165  }
166  lis_matrix_diag_copy(A->D,A->WD);
167  lis_matrix_diag_scale(w,A->WD);
168  lis_matrix_diag_inverse(A->WD);
169  A->use_wd = LIS_SOLVER_SOR;
170  }
171 
172  for( iter=1; iter<=maxiter; iter++ )
173  {
174  /* x += (D/w-L)^{-1}(b - Ax) */
175  times = lis_wtime();
176  lis_psolve(solver,x,s);
177  ptimes += lis_wtime() - times;
178  LIS_MATVEC(A,s,t);
179 /* LIS_MATVEC(A,x,t);*/
180  lis_vector_axpyz(-1,t,b,r);
181  lis_vector_nrm2(r,&nrm2);
182  lis_matrix_solve(A,r,t,LIS_MATRIX_LOWER);
183  lis_vector_axpy(1,t,x);
184 
185  /* convergence check */
186  nrm2 = nrm2 * bnrm2;
187 
188  if( output )
189  {
190  if( output & LIS_PRINT_MEM ) solver->residual[iter] = nrm2;
191  if( output & LIS_PRINT_OUT && A->my_rank==0 ) lis_print_rhistory(iter,nrm2);
192  }
193 
194  if( tol >= nrm2 )
195  {
196  times = lis_wtime();
197  lis_psolve(solver,x,s);
198  ptimes += lis_wtime() - times;
199  lis_vector_copy(s,x);
200  solver->retcode = LIS_SUCCESS;
201  solver->iter = iter;
202  solver->resid = nrm2;
203  solver->ptimes = ptimes;
204  LIS_DEBUG_FUNC_OUT;
205  return LIS_SUCCESS;
206  }
207  }
208 
209  lis_psolve(solver,x,s);
210  lis_vector_copy(s,x);
211  solver->retcode = LIS_MAXITER;
212  solver->iter = iter;
213  solver->resid = nrm2;
214  LIS_DEBUG_FUNC_OUT;
215  return LIS_MAXITER;
216 }
