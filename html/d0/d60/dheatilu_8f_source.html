9 C Example program for DDASKR.
10 C DAE system derived from the discretized heat equation on a square.
11 C
12 C This is the double precision version.
13 C-----------------------------------------------------------------------
14 C
15 C This program solves a DAE system that arises from the heat equation,
16 C du/dt = u + u
17 C xx yy
18 C posed on the 2-D unit square with zero Dirichlet boundary conditions.
19 C An M+2 by M+2 mesh is set on the square, with uniform spacing 1/(M+1).
20 C The spatial deriviatives are represented by standard central finite
21 C difference approximations. At each interior point of the mesh,
22 C the discretized PDE becomes an ODE for the discrete value of u.
23 C At each point on the boundary, we pose the equation u = 0. The
24 C discrete values of u form a vector U, ordered first by x, then by y.
25 C The result is a DAE system G(t,U,U') = 0 of size NEQ = (M+2)*(M+2).
26 C
27 C Initial conditions are posed as u = 16x(1-x)y(1-y) at t = 0.
28 C The problem is solved by DDASKR on the time interval t .le. 10.24.
29 C
30 C The root functions are R1(U) = max(u) - 0.1, R2(U) = max(u) - 0.01.
31 C
32 C The Krylov linear system solution method, with preconditioning, is
33 C selected. The preconditioner is a sparse matrix with half-bandwidths
34 C equal to 1, i.e. a tridiagonal matrix. (The true half-bandwidths
35 C are equal to M+2.) This corresponds to ignoring the y-direction
36 C coupling in the ODEs, for purposes of preconditioning. The extra
37 C iterations resulting from this approximation are offset by the lower
38 C storage and linear system solution costs for a tridiagonal matrix.
39 C
40 C The routines DJACILU and DPSOLILU that generate and solve the sparse
41 C preconditioner are provided in a separate file for general use.
42 C
43 C The output times are t = .01 * 2**n (n = 0,...,10). The maximum of
44 C abs(u) over the mesh, and various performance statistics, are printed.
45 C
46 C For details and test results on this problem, see the reference:
47 C Peter N. Brown, Alan C. Hindmarsh, and Linda R. Petzold,
48 C Using Krylov Methods in the Solution of Large-Scale Differential-
49 C Algebraic Systems, SIAM J. Sci. Comput., 15 (1994), pp. 1467-1488.

299  SUBROUTINE uinit (U, UPRIME, RPAR, IPAR)
300 C
301 C This routine computes and loads the vector of initial values.
302 C The initial U values are given by the polynomial u = 16x(1-x)y(1-y).
303 C The initial UPRIME values are set to zero. (DDASKR corrects these
304 C during the first time step.)
305 C
306  IMPLICIT DOUBLE PRECISION (a-h,o-z)
307  dimension u(*), uprime(*), rpar(4), ipar(34)
308 C
309  neq = ipar(33)
310  m = ipar(34)
311  dx = rpar(3)
312 C
313  DO 20 k = 0,m+1
314  yk = k*dx
315  ioff = (m+2)*k
316  DO 10 j = 0,m+1
317  xj = j*dx
318  i = ioff + j + 1
319  u(i) = 16.0d0*xj*(1.0d0-xj)*yk*(1.0d0-yk)
320  10 CONTINUE
321  20 CONTINUE
322  DO 30 i = 1,neq
323  30 uprime(i) = 0.0d0
324  RETURN
325 C------------ End of Subroutine UINIT --------------------------------
326  END

328  SUBROUTINE resh (T, U, UPRIME, CJ, DELTA, IRES, RPAR, IPAR)
329 C
330 C This is the user-supplied RES subroutine for this example.
331 C It computes the residuals for the 2-D discretized heat equation,
332 C with zero boundary values.
333 C
334  IMPLICIT DOUBLE PRECISION (a-h,o-z)
335  dimension u(*), uprime(*), delta(*), rpar(4), ipar(34)
336 C
337 C Set problem constants using IPAR and RPAR.
338  neq = ipar(33)
339  m = ipar(34)
340  coeff = rpar(4)
341  m2 = m + 2
342 C
343 C Load U into DELTA, in order to set boundary values.
344  DO 10 i = 1,neq
345  10 delta(i) = u(i)
346 C
347 C Loop over interior points, and load residual values.
348  DO 30 k = 1,m
349  ioff = m2*k
350  DO 20 j = 1,m
351  i = ioff + j + 1
352  temx = u(i-1) + u(i+1)
353  temy = u(i-m2) + u(i+m2)
354  delta(i) = uprime(i) - (temx + temy - 4.0d0*u(i))*coeff
355  20 CONTINUE
356  30 CONTINUE
357 C
358  RETURN
