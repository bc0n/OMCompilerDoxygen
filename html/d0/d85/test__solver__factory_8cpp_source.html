78  // Test creation of a Belos solver, using Belos::SolverFactory.
79  //
80  // Template parameters:
81  //
82  // SolverFactoryType: the Belos::SolverFactory specialization
83  // SolverBaseType: the Belos::SolverManager specialization
84  // SolverImplType: the Belos::SolverManager subclass which the
85  // factory is supposed to produce.
86  //
87  // Function parameters:
88  //
89  // factory: A Belos::SolverFactory instance. Passing it in as a
90  // reference lets us reuse the same factory instance for multiple
91  // tests. This saves the (small) overhead of creating the factory
92  // each time, and also (more importantly) ensures that the
93  // factory's create() method gets tested for multiple calls to the
94  // same factory instance. The reference is nonconst because the
95  // create() method is nonconst.
96  //
97  // solverName: Name of the solver to create.
98  //
99  // out: An output stream reference to which to write verbose output
100  // (verbose == true). When building with MPI, this should be a
101  // black-hole stream (Teuchos::oblackholestream) reference on
102  // every MPI process except the process with Rank 0.
103  //
104  // verbose: Whether to write verbose output.
105  template<class SolverFactoryType, class SolverBaseType, class SolverImplType>
106  void
107  testCreatingSolver (SolverFactoryType& factory,
108  const std::string& solverName,
109  std::ostream& out,
110  const bool verbose)
111  {
112  using Teuchos::ParameterList;
113  using Teuchos::parameterList;
114  using Teuchos::RCP;
115  using Teuchos::rcp_dynamic_cast;
116  using Teuchos::TypeNameTraits;
117  using std::endl;
118 
119  if (verbose) {
120  out << "Creating solver named: \"" << solverName << "\"" << endl;
121  }
122  RCP<ParameterList> solverParams = parameterList ();
123  RCP<SolverBaseType> solver = factory.create (solverName, solverParams);
124  TEUCHOS_TEST_FOR_EXCEPTION(solver.is_null(), std::logic_error,
125  "Failed to create solver with valid name \""
126  << solverName << "\".");
127  if (verbose) {
128  out << "Solver description: " << solver->description() << endl;
129  }
130  TEUCHOS_TEST_FOR_EXCEPTION(rcp_dynamic_cast<SolverImplType> (solver).is_null(),
131  std::logic_error, "Solver does not have the correct type. Type should be "
132  << TypeNameTraits<SolverImplType>::name() << ".");
133  }

135  int
136  selectVerbosity (const bool verbose, const bool debug)
137  {
138  // NOTE Calling this a "MsgType" (its correct type) or even an
139  // "enum MsgType" confuses the compiler.
140  int theType = Belos::Errors; // default (always print errors)
141  if (verbose)
142  {
143  // "Verbose" also means printing out Debug messages (as well
144  // as everything else).
145  theType = theType |
146  Belos::Warnings |
147  Belos::IterationDetails |
148  Belos::OrthoDetails |
149  Belos::FinalSummary |
150  Belos::TimingDetails |
151  Belos::StatusTestDetails |
152  Belos::Debug;
153  }
154  if (debug)
155  // "Debug" doesn't necessarily mean the same thing as
156  // "Verbose". We interpret "Debug" to mean printing out
157  // messages marked as Debug (as well as Error messages).
158  theType = theType | Belos::Debug;
159  return theType;
160  }

163  Teuchos::RCP<Epetra_Operator>
164  makeMatrix (const Teuchos::RCP<Epetra_Comm>& comm,
165  Teuchos::RCP<const Epetra_Map>& domainMap,
166  Teuchos::RCP<const Epetra_Map>& rangeMap,
167  const int nx = 30)
168  {
169  using Teuchos::ParameterList;
170  using Teuchos::RCP;
171  using Teuchos::rcp;
172  using Teuchos::rcp_implicit_cast;
173 
174  ParameterList GaleriList;
175  GaleriList.set("n", nx * nx);
176  GaleriList.set("nx", nx);
177  GaleriList.set("ny", nx);
178  RCP<const Epetra_Map> rowMap = rcp (Galeri::CreateMap("Linear", *comm, GaleriList));
179  // "&*rowMap" turns an RCP<Epetra_Map> into a raw pointer
180  // (Epetra_Map*), which is what Galeri::CreateCrsMatrix() wants.
181  RCP<Epetra_RowMatrix> A =
182  rcp (Galeri::CreateCrsMatrix("Laplace2D", &*rowMap, GaleriList));
183  TEUCHOS_TEST_FOR_EXCEPTION(A.is_null(), std::runtime_error,
184  "Galeri returned a null operator A.");
185  domainMap = rowMap;
186  rangeMap = rowMap;
187  return rcp_implicit_cast<Epetra_Operator> (A);
188  }
