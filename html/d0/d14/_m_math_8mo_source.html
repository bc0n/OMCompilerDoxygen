59 public function addRational "adds two rationals"
60  input Rational r1;
61  input Rational r2;
62  output Rational r;
63 algorithm
64  r := match(r1,r2)
65  local Integer i1,i2,i3,i4,ri1,ri2,d;
66  case(RATIONAL(i1,i2),RATIONAL(i3,i4)) equation
67  ri1 = i1*i4 + i3*i2;
68  ri2 = i2*i4;
69  d = intGcd(ri1,ri2);
70  ri1 = intDiv(ri1, d);
71  ri2 = intDiv(ri2, d);
72  then normalizeZero(RATIONAL(ri1,ri2));
73  end match;
74 end addRational;
