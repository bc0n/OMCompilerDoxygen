42 // This driver reads a problem from a Harwell-Boeing (HB) file.
43 // The right-hand-side from the HB file is used instead of random vectors.
44 // The initial guesses are all set to zero.
45 //
46 // NOTE: No preconditioner is used in this case.
47 //

60 int main(int argc, char *argv[]) {

82  int maxiters = -1; // maximum number of iterations for solver to use

86  Teuchos::CommandLineProcessor cmdp(false,true);

88  cmdp.setOption("frequency",&frequency,"Solvers frequency for printing residuals (#iters).");
89  cmdp.setOption("tol",&tol,"Relative residual tolerance used byfixed point solver.");
90  cmdp.setOption("filename",&filename,"Filename for Harwell-Boeing test matrix.");
91  cmdp.setOption("num-rhs",&numrhs,"Number of right-hand sides to be solved for.");
92  cmdp.setOption("max-iters",&maxiters,"Maximum number of iterations per linear system (-1 := adapted to problem).");
93  if (cmdp.parse(argc,argv) != Teuchos::CommandLineProcessor::PARSE_SUCCESSFUL) {
94  return -1;
95  }
96  if (!verbose)
97  frequency = -1; // reset frequency if test is not verbose
98  //
99  // Get the problem
100  //
101  int MyPID;

103  RCP<Epetra_MultiVector> B, X;
104  int return_val =Belos::createEpetraProblem(filename,NULL,&A,&B,&X,&MyPID);
105  if(return_val != 0) return return_val;
106  proc_verbose = ( verbose && (MyPID==0) );
107 
108  // Scale down the rhs
109  std::vector<double> norm( MVT::GetNumberVecs(*B));
110  MVT::MvNorm(*B,norm);
111  for(int i=0; i< MVT::GetNumberVecs(*B); i++)
112  norm[i] = 1.0 / norm[i];
113  MVT::MvScale(*B,norm);
114 
115  // Drastically bump the diagonal and then rescale so FP can converge
116  Epetra_Vector diag(A->RowMap());
117  A->ExtractDiagonalCopy(diag);
118  for(int i=0; i<diag.MyLength(); i++)
119  diag[i]=diag[i]*1e4;
120  A->ReplaceDiagonalValues(diag);
121 
122  for(int i=0; i<diag.MyLength(); i++)
123  diag[i] = 1.0/sqrt(diag[i]);
124  A->LeftScale(diag);
125  A->RightScale(diag);
126 
127  //
128  // Solve using Belos
129  //
130 
131  //
132  // *****Construct initial guess and random right-hand-sides *****
133  //

137  B = rcp( new Epetra_MultiVector( A->Map(), numrhs ) );

162  //

172  //

174  = rcp( new Belos::FixedPointSolMgr<double,MV,OP>(rcp(&problem,false), rcp(&belosList,false)));
175  //
176  // **********Print out information about problem*******************
177  //
178  if (proc_verbose) {
179  std::cout << std::endl << std::endl;
180  std::cout << "Dimension of matrix: " << NumGlobalElements << std::endl;
181  std::cout << "Number of right-hand sides: " << numrhs << std::endl;
182  std::cout << "Max number of iterations: " << maxiters << std::endl;
183  std::cout << "Relative residual tolerance: " << tol << std::endl;
184  std::cout << std::endl;
185  }
186  //
187  // Perform solve
188  //
189  Belos::ReturnType ret = newSolver->solve();
190  //
191  // Compute actual residuals.
192  //
193  bool badRes = false;
194  std::vector<double> actual_resids( numrhs );
195  std::vector<double> rhs_norm( numrhs );
196  Epetra_MultiVector resid(A->Map(), numrhs);
197  OPT::Apply( *A, *X, resid );
198  MVT::MvAddMv( -1.0, resid, 1.0, *B, resid );
199  MVT::MvNorm( resid, actual_resids );
200  MVT::MvNorm( *B, rhs_norm );
201  if (proc_verbose) {
202  std::cout<< "---------- Actual Residuals (normalized) ----------"<<std::endl<<std::endl;
203  for ( int i=0; i<numrhs; i++) {
204  double actRes = actual_resids[i]/rhs_norm[i];
205  std::cout<<"Problem "<<i<<" : \t"<< actRes <<std::endl;
206  if (actRes > tol) badRes = true;
207  }
208  }
209 
210  success = ret==Belos::Converged && !badRes;
211 
212  if (success) {
213  if (proc_verbose)
214  std::cout << std::endl << "End Result: TEST PASSED" << std::endl;
215  } else {
216  if (proc_verbose)
217  std::cout << std::endl << "End Result: TEST FAILED" << std::endl;
218  }
219  }
220  TEUCHOS_STANDARD_CATCH_STATEMENTS(verbose, std::cerr, success);
221 
222  return ( success ? EXIT_SUCCESS : EXIT_FAILURE );
