338 /*
339  * -----------------------------------------------------------------
340  * idaDlsDenseDQJac
341  * -----------------------------------------------------------------
342  * This routine generates a dense difference quotient approximation to
343  * the Jacobian F_y + c_j*F_y'. It assumes that a dense matrix of type
344  * DlsMat is stored column-wise, and that elements within each column
345  * are contiguous. The address of the jth column of J is obtained via
346  * the macro LAPACK_DENSE_COL and this pointer is associated with an N_Vector
347  * using the N_VGetArrayPointer/N_VSetArrayPointer functions.
348  * Finally, the actual computation of the jth column of the Jacobian is
349  * done with a call to N_VLinearSum.
350  * -----------------------------------------------------------------
351  */
352 int idaDlsDenseDQJac(long int N, realtype tt, realtype c_j,
353  N_Vector yy, N_Vector yp, N_Vector rr,
354  DlsMat Jac, void *data,
355  N_Vector tmp1, N_Vector tmp2, N_Vector tmp3)
356 {
357  realtype inc, inc_inv, yj, ypj, srur, conj;
358  realtype *tmp2_data, *y_data, *yp_data, *ewt_data, *cns_data = NULL;
359  N_Vector rtemp, jthCol;
360  long int j;
361  int retval = 0;
362 
363  IDAMem IDA_mem;
364  IDADlsMem idadls_mem;
365 
366  /* data points to IDA_mem */
367  IDA_mem = (IDAMem) data;
368  idadls_mem = (IDADlsMem) lmem;
369 
370  /* Save pointer to the array in tmp2 */
371  tmp2_data = N_VGetArrayPointer(tmp2);
372 
373  /* Rename work vectors for readibility */
374  rtemp = tmp1;
375  jthCol = tmp2;
376 
377  /* Obtain pointers to the data for ewt, yy, yp. */
378  ewt_data = N_VGetArrayPointer(ewt);
379  y_data = N_VGetArrayPointer(yy);
380  yp_data = N_VGetArrayPointer(yp);
381  if(constraints!=NULL) cns_data = N_VGetArrayPointer(constraints);
382 
383  srur = SUNRsqrt(uround);
384 
385  for (j=0; j < N; j++) {
386 
387  /* Generate the jth col of J(tt,yy,yp) as delta(F)/delta(y_j). */
388 
389  /* Set data address of jthCol, and save y_j and yp_j values. */
390  N_VSetArrayPointer(DENSE_COL(Jac,j), jthCol);
391  yj = y_data[j];
392  ypj = yp_data[j];
393 
394  /* Set increment inc to y_j based on sqrt(uround)*abs(y_j), with
395  adjustments using yp_j and ewt_j if this is small, and a further
396  adjustment to give it the same sign as hh*yp_j. */
397 
398  inc = SUNMAX( srur * SUNMAX( SUNRabs(yj), SUNRabs(hh*ypj) ) , ONE/ewt_data[j] );
399 
400  if (hh*ypj < ZERO) inc = -inc;
401  inc = (yj + inc) - yj;
402 
403  /* Adjust sign(inc) again if y_j has an inequality constraint. */
404  if (constraints != NULL) {
405  conj = cns_data[j];
406  if (SUNRabs(conj) == ONE) {if((yj+inc)*conj < ZERO) inc = -inc;}
407  else if (SUNRabs(conj) == TWO) {if((yj+inc)*conj <= ZERO) inc = -inc;}
408  }
409 
410  /* Increment y_j and yp_j, call res, and break on error return. */
411  y_data[j] += inc;
412  yp_data[j] += c_j*inc;
413 
414  retval = res(tt, yy, yp, rtemp, user_data);
415 
416  nreDQ++;
417  if (retval != 0) break;
418 
419  /* Construct difference quotient in jthCol */
420  inc_inv = ONE/inc;
421  N_VLinearSum(inc_inv, rtemp, -inc_inv, rr, jthCol);
422 
423  DENSE_COL(Jac,j) = N_VGetArrayPointer(jthCol);
424 
425  /* reset y_j, yp_j */
426  y_data[j] = yj;
427  yp_data[j] = ypj;
428  }
429 
430  /* Restore original array pointer in tmp2 */
431  N_VSetArrayPointer(tmp2_data, tmp2);
432 
433  return(retval);
434 
435 }

437 /*
438  * -----------------------------------------------------------------
439  * idaDlsBandDQJac
440  * -----------------------------------------------------------------
441  * This routine generates a banded difference quotient approximation JJ
442  * to the DAE system Jacobian J. It assumes that a band matrix of type
443  * BandMat is stored column-wise, and that elements within each column
444  * are contiguous. The address of the jth column of JJ is obtained via
445  * the macros BAND_COL and BAND_COL_ELEM. The columns of the Jacobian are
446  * constructed using mupper + mlower + 1 calls to the res routine, and
447  * appropriate differencing.
448  * The return value is either IDABAND_SUCCESS = 0, or the nonzero value returned
449  * by the res routine, if any.
450  */
451 
452 int idaDlsBandDQJac(long int N, long int mupper, long int mlower,
453  realtype tt, realtype c_j,
454  N_Vector yy, N_Vector yp, N_Vector rr,
455  DlsMat Jac, void *data,
456  N_Vector tmp1, N_Vector tmp2, N_Vector tmp3)
457 {
458  realtype inc, inc_inv, yj, ypj, srur, conj, ewtj;
459  realtype *y_data, *yp_data, *ewt_data, *cns_data = NULL;
460  realtype *ytemp_data, *yptemp_data, *rtemp_data, *r_data, *col_j;
461 
462  N_Vector rtemp, ytemp, yptemp;
463  long int group, i, j, i1, i2, width, ngroups;
464  int retval = 0;
465 
466  IDAMem IDA_mem;
467  IDADlsMem idadls_mem;
468 
469  /* data points to IDA_mem */
470  IDA_mem = (IDAMem) data;
471  idadls_mem = (IDADlsMem) lmem;
472 
473  rtemp = tmp1; /* Rename work vector for use as the perturbed residual. */
474 
475  ytemp = tmp2; /* Rename work vector for use as a temporary for yy. */
476 
477 
478  yptemp= tmp3; /* Rename work vector for use as a temporary for yp. */
479 
480  /* Obtain pointers to the data for all eight vectors used. */
481 
482  ewt_data = N_VGetArrayPointer(ewt);
483  r_data = N_VGetArrayPointer(rr);
484  y_data = N_VGetArrayPointer(yy);
485  yp_data = N_VGetArrayPointer(yp);
486 
487  rtemp_data = N_VGetArrayPointer(rtemp);
488  ytemp_data = N_VGetArrayPointer(ytemp);
489  yptemp_data = N_VGetArrayPointer(yptemp);
490 
491  if (constraints != NULL) cns_data = N_VGetArrayPointer(constraints);
492 
493  /* Initialize ytemp and yptemp. */
494 
495  N_VScale(ONE, yy, ytemp);
496  N_VScale(ONE, yp, yptemp);
497 
498  /* Compute miscellaneous values for the Jacobian computation. */
499 
500  srur = SUNRsqrt(uround);
501  width = mlower + mupper + 1;
502  ngroups = SUNMIN(width, N);
503 
504  /* Loop over column groups. */
505  for (group=1; group <= ngroups; group++) {
506 
507  /* Increment all yy[j] and yp[j] for j in this group. */
508 
509  for (j=group-1; j<N; j+=width) {
510  yj = y_data[j];
511  ypj = yp_data[j];
512  ewtj = ewt_data[j];
513 
514  /* Set increment inc to yj based on sqrt(uround)*abs(yj), with
515  adjustments using ypj and ewtj if this is small, and a further
516  adjustment to give it the same sign as hh*ypj. */
517 
518  inc = SUNMAX( srur * SUNMAX( SUNRabs(yj), SUNRabs(hh*ypj) ) , ONE/ewtj );
519 
520  if (hh*ypj < ZERO) inc = -inc;
521  inc = (yj + inc) - yj;
522 
523  /* Adjust sign(inc) again if yj has an inequality constraint. */
524 
525  if (constraints != NULL) {
526  conj = cns_data[j];
527  if (SUNRabs(conj) == ONE) {if((yj+inc)*conj < ZERO) inc = -inc;}
528  else if (SUNRabs(conj) == TWO) {if((yj+inc)*conj <= ZERO) inc = -inc;}
529  }
530 
531  /* Increment yj and ypj. */
532 
533  ytemp_data[j] += inc;
534  yptemp_data[j] += cj*inc;
535  }
536 
537  /* Call res routine with incremented arguments. */
538 
539  retval = res(tt, ytemp, yptemp, rtemp, user_data);
540  nreDQ++;
541  if (retval != 0) break;
542 
543  /* Loop over the indices j in this group again. */
544 
545  for (j=group-1; j<N; j+=width) {
546 
547  /* Reset ytemp and yptemp components that were perturbed. */
548 
549  yj = ytemp_data[j] = y_data[j];
550  ypj = yptemp_data[j] = yp_data[j];
551  col_j = BAND_COL(Jac, j);
552  ewtj = ewt_data[j];
553 
554  /* Set increment inc exactly as above. */
555 
556  inc = SUNMAX( srur * SUNMAX( SUNRabs(yj), SUNRabs(hh*ypj) ) , ONE/ewtj );
557  if (hh*ypj < ZERO) inc = -inc;
558  inc = (yj + inc) - yj;
559  if (constraints != NULL) {
560  conj = cns_data[j];
561  if (SUNRabs(conj) == ONE) {if((yj+inc)*conj < ZERO) inc = -inc;}
562  else if (SUNRabs(conj) == TWO) {if((yj+inc)*conj <= ZERO) inc = -inc;}
563  }
564 
565  /* Load the difference quotient Jacobian elements for column j. */
566 
567  inc_inv = ONE/inc;
568  i1 = SUNMAX(0, j-mupper);
569  i2 = SUNMIN(j+mlower,N-1);
570 
571  for (i=i1; i<=i2; i++)
572  BAND_COL_ELEM(col_j,i,j) = inc_inv*(rtemp_data[i]-r_data[i]);
573  }
574 
575  }
576 
577  return(retval);
578 
579 }
