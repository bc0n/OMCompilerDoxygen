17 /* ******************************************************************** */
18 /* Structure used for ML_MSR_getrows and ML_CSR_getrows */
19 /* */
20 /* For CSR matrices, we have */
21 /* A(i,j) where 0 <= i < N | j = columns[k] , A(i,j) = values[k] */
22 /* and 0 <= j < N | where rowptr[i] <= k <rowptr[i+1] */
23 /* */
24 /* For MSR matrices, we have */
25 /* A(i,j) where 0 <= i < N,| j = columns[k] , A(i,j) = values[k] */
26 /* i != j , | where columns[i] <= k < columns[i+1]*/
27 /* and 0 <= j < N | */
28 /* A(i,i) where 0 <= i < N,| A(i,j) = values[i] */
29 /* -------------------------------------------------------------------- */
