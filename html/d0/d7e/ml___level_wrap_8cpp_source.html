105 int ML_Epetra::LevelWrap::GenerateSmoother() {
106 
107  string PreOrPostSmoother = List_.get("smoother: pre or post","both");
108  if(PreOrPostSmoother == "post") pre_or_post = ML_POSTSMOOTHER;
109  else if(PreOrPostSmoother == "pre") pre_or_post = ML_PRESMOOTHER;
110  else if(PreOrPostSmoother == "both") pre_or_post = ML_BOTH;
111 
112  if (use_mlsmoother_) { // ML smoothing is experimental
113  ML_Create(&ml_subproblem_,1);
114 #ifdef HAVE_MPI
115  const Epetra_MpiComm *epcomm = dynamic_cast<const Epetra_MpiComm*>(&(A0_->Comm()));
116  // Get the MPI communicator, as it may not be MPI_COMM_W0RLD, and update the ML comm object
117  if (epcomm) ML_Comm_Set_UsrComm(ml_subproblem_->comm,epcomm->Comm());
118 #endif
119  int numMyRows = A0_->NumMyRows();
120  int N_ghost = A0_->NumMyCols() - numMyRows;
121  if (N_ghost < 0) N_ghost = 0;
122  const Epetra_Operator *Z = const_cast<Epetra_CrsMatrix*>(&*A0_);
123  const Epetra_RowMatrix *Arow=dynamic_cast<const Epetra_RowMatrix*>(Z);
124  ML_Init_Amatrix(ml_subproblem_,0,numMyRows, numMyRows,(void *) Arow);
125  ml_subproblem_->Amat[0].type = ML_TYPE_ROW_MATRIX;
126  ml_subproblem_->Amat[0].N_nonzeros = A0_->NumMyNonzeros();
127  ML_Set_Amatrix_Getrow(ml_subproblem_, 0, ML_Epetra_RowMatrix_getrow,
128  ML_Epetra_comm_wrapper, numMyRows+N_ghost);
129  ML_Set_Amatrix_Matvec(ml_subproblem_, 0, ML_Epetra_matvec);
130  int Nparts = List_.sublist("smoother: ifpack list").get("partitioner: local parts",-1);
131  if (Nparts == -1) { printf("must supply partitioner: local parts\n"); return(-1); }
132 
133  ml_subproblem_->Amat[0].num_PDEs = numMyRows/Nparts;
134  ML_Gen_Smoother_BlockGaussSeidel(ml_subproblem_, 0, ML_PRESMOOTHER,
135  1, 1.0 , ml_subproblem_->Amat[0].num_PDEs);
136  }
137  else Smoother_=rcp(ML_Gen_Smoother_Ifpack_Epetra(const_cast<Epetra_CrsMatrix*>(&*A0_),0,List_,"LevelWrap Smoother (level 0): ",verbose_));
138  return 0;

143 int ML_Epetra::LevelWrap::ComputePreconditioner(const bool CheckFiltering){
144 #ifdef ML_TIMING
145  double t_time,t_diff;
146  StartTimer(&t_time);
147 #endif
148  int output_level=List_.get("ML output",0);
149  if(output_level>0) verbose_=true;
150 
151  // Sanity check
152  if(!A0_->DomainMap().SameAs(P0_->RangeMap())) return -1;
153  if(!use_pt_ && !A0_->RangeMap().SameAs(R0_->DomainMap())) return -2;
154 
155  //********************
156  // Setup smoother
157  //********************
158  int rv = GenerateSmoother();
159  if(rv) return rv;
160 
161  //********************
162  // Setup A1
163  //********************
164  Epetra_CrsMatrix * A1=List_.get("user coarse matrix",(Epetra_CrsMatrix *)0);
165  if(A1){
166  // User A1
167  user_A1_=true;
168  A1_=rcp<Epetra_CrsMatrix>(A1,false);
169  }
170  else{
171  // Do RAP
172 #ifdef HAVE_ML_EPETRAEXT
173  if(use_pt_) Epetra_PtAP(*A0_,*P0_,A1,false);
174 #else
175  if(use_pt_) ML_Epetra_PtAP(*A0_,*P0_,A1,false);
176 #endif
177  else ML_Epetra_RAP(*A0_,*P0_,*R0_,A1,false);
178 
179  A1_=rcp<Epetra_CrsMatrix>(A1);
180  }
181 
182  //********************
183  // Setup Coarse Solver
184  //********************
185  Teuchos::ParameterList & CoarseList=List_.sublist("levelwrap: coarse list");
186  A1prec_=rcp<MultiLevelPreconditioner>(new MultiLevelPreconditioner(*A1_,CoarseList,true));
187 
188 #ifdef ML_TIMING
189  StopTimer(&t_time,&t_diff);
190  /* Output */
191  ML_Comm *comm_;
192  ML_Comm_Create(&comm_);
193  int printl=ML_Get_PrintLevel();
194  ML_Set_PrintLevel(output_level);
195  ReportTimer(t_diff,"ML_LevelWrap::ComputePreconditioner",comm_);
196  ML_Set_PrintLevel(printl);
197  ML_Comm_Destroy(&comm_);
198  // NumConstructions_++;
199  // ConstructionTime_+=t_diff;
200 #endif
201  return 0;
202 }

219 // Apply the preconditioner w/ RHS B and get result X
220 int ML_Epetra::LevelWrap::ApplyInverse(const Epetra_MultiVector& B, Epetra_MultiVector& X_) const{
221 #ifdef ML_TIMING
222  double t_time,t_diff;
223  StartTimer(&t_time);
224 #endif
225 
226  // Sanity Checks
227  if (!B.Map().SameAs(OperatorDomainMap())) return -1;
228  if (!X_.Map().SameAs(OperatorRangeMap())) return -1;
229  if (!X_.Map().SameAs(B.Map())) return -1;
230  if (B.NumVectors() != X_.NumVectors()) return -1;
231 
232  // Build new work vector X
233  Epetra_MultiVector X(X_.Map(),X_.NumVectors(),true);
234  Epetra_MultiVector tmp0(X_.Map(),X_.NumVectors(),true);
235  Epetra_MultiVector tmp1(P0_->DomainMap(),X_.NumVectors(),true);
236  Epetra_MultiVector tmp2(P0_->DomainMap(),X_.NumVectors(),true);
237 
238  double *Bptr, *Xptr;
239  int BLDA, XLDA;
240 
241  // Pre Smoother
242  if(pre_or_post==ML_BOTH || pre_or_post==ML_PRESMOOTHER){
243  if (use_mlsmoother_) { // ML smoothing is experimental
244  B.ExtractView(&Bptr,&BLDA);
245  X.ExtractView(&Xptr,&XLDA);
246  ML_Smoother_Apply(&(ml_subproblem_->pre_smoother[0]),
247  X.MyLength(), Xptr,
248  X.MyLength(), Bptr, ML_ZERO);
249  }
250  else Smoother_->ApplyInverse(B,X);
251 
252  A0_->Apply(X,tmp0);
253  tmp0.Update(1.0,B,-1.0);
254  }
255  else tmp0 = B;
256 
257  // Form coarse residual
258  if(use_pt_) P0_->Multiply(true,tmp0,tmp1);
259  else R0_->Multiply(false,tmp0,tmp1);
260 
261  // Solve coarse problem
262  A1prec_->ApplyInverse(tmp1,tmp2);
263 
264  // Update solution
265  P0_->Multiply(false,tmp2,tmp0);
266  X.Update(1.0,tmp0,1.0);
267 
268  // Post Smoother
269  if(pre_or_post==ML_BOTH || pre_or_post==ML_POSTSMOOTHER){
270  if (use_mlsmoother_) { // ML smoothing is experimental
271  Epetra_MultiVector tmp3(X_.Map(),X_.NumVectors(),true);
272 
273  A0_->Apply(X,tmp0);
274  tmp0.Update(1.0,B,-1.0);
275  tmp3.PutScalar(0.0);
276  tmp0.ExtractView(&Bptr,&BLDA);
277  tmp3.ExtractView(&Xptr,&XLDA);
278 
279  ML_Smoother_Apply(&(ml_subproblem_->pre_smoother[0]),X.MyLength(),Xptr,
280  X.MyLength(), Bptr, ML_ZERO);
281  X.Update(1.0,tmp3,1.0);
282  }
283  else Smoother_->ApplyInverse(B,X);
284  }
285 
286  // Copy to output
287  X_=X;
288 
289 #ifdef ML_TIMING
290  StopTimer(&t_time,&t_diff);
291  /* Output */
292  ML_Comm *comm_;
293  ML_Comm_Create(&comm_);
294  // ApplicationTime_+= t_diff;
295  /* if(FirstApplication_){
296  FirstApplication_=false;
297  FirstApplicationTime_=ApplicationTime_;
298  }*//*end if*/
299  ML_Comm_Destroy(&comm_);
300 #endif
301 
302  return 0;
303 }
