277 /* Sort an array of HeapSects by start address. */
278 /* Unfortunately at least some versions of */
279 /* Linux qsort end up calling malloc by way of sysconf, and hence can't */
280 /* be used in the collector. Hence we roll our own. Should be */
281 /* reasonably fast if the array is already mostly sorted, as we expect */
282 /* it to be. */
283 static void sort_heap_sects(struct HeapSect *base, size_t number_of_elements)
284 {
285  signed_word n = (signed_word)number_of_elements;
286  signed_word nsorted = 1;
287 
288  while (nsorted < n) {
289  signed_word i;
290 
291  while (nsorted < n &&
292  (word)base[nsorted-1].hs_start < (word)base[nsorted].hs_start)
293  ++nsorted;
294  if (nsorted == n) break;
295  GC_ASSERT((word)base[nsorted-1].hs_start > (word)base[nsorted].hs_start);
296  i = nsorted - 1;
297  while (i >= 0 && (word)base[i].hs_start > (word)base[i+1].hs_start) {
298  struct HeapSect tmp = base[i];
299  base[i] = base[i+1];
300  base[i+1] = tmp;
301  --i;
302  }
303  GC_ASSERT((word)base[nsorted-1].hs_start < (word)base[nsorted].hs_start);
304  ++nsorted;
305  }
306 }

308 STATIC void GC_register_map_entries(char *maps)
309 {
310  char *prot;
311  char *buf_ptr = maps;
312  ptr_t start, end;
313  unsigned int maj_dev;
314  ptr_t least_ha, greatest_ha;
315  unsigned i;
316 
317  GC_ASSERT(I_HOLD_LOCK());
318  sort_heap_sects(GC_our_memory, GC_n_memory);
319  least_ha = GC_our_memory[0].hs_start;
320  greatest_ha = GC_our_memory[GC_n_memory-1].hs_start
321  + GC_our_memory[GC_n_memory-1].hs_bytes;
322 
323  for (;;) {
324  buf_ptr = GC_parse_map_entry(buf_ptr, &start, &end, &prot,
325  &maj_dev, 0);
326  if (NULL == buf_ptr)
327  break;
328  if (prot[1] == 'w') {
329  /* This is a writable mapping. Add it to */
330  /* the root set unless it is already otherwise */
331  /* accounted for. */
332  if ((word)start <= (word)GC_stackbottom
333  && (word)end >= (word)GC_stackbottom) {
334  /* Stack mapping; discard */
335  continue;
336  }
337 # ifdef THREADS
338  /* This may fail, since a thread may already be */
339  /* unregistered, but its thread stack may still be there. */
340  /* That can fail because the stack may disappear while */
341  /* we're marking. Thus the marker is, and has to be */
342  /* prepared to recover from segmentation faults. */
343 
344  if (GC_segment_is_thread_stack(start, end)) continue;
345 
346  /* FIXME: NPTL squirrels */
347  /* away pointers in pieces of the stack segment that we */
348  /* don't scan. We work around this */
349  /* by treating anything allocated by libpthread as */
350  /* uncollectible, as we do in some other cases. */
351  /* A specifically identified problem is that */
352  /* thread stacks contain pointers to dynamic thread */
353  /* vectors, which may be reused due to thread caching. */
354  /* They may not be marked if the thread is still live. */
355  /* This specific instance should be addressed by */
356  /* INCLUDE_LINUX_THREAD_DESCR, but that doesn't quite */
357  /* seem to suffice. */
358  /* We currently trace entire thread stacks, if they are */
359  /* are currently cached but unused. This is */
360  /* very suboptimal for performance reasons. */
361 # endif
362  /* We no longer exclude the main data segment. */
363  if ((word)end <= (word)least_ha
364  || (word)start >= (word)greatest_ha) {
365  /* The easy case; just trace entire segment */
366  GC_add_roots_inner(start, end, TRUE);
367  continue;
368  }
369  /* Add sections that don't belong to us. */
370  i = 0;
371  while ((word)(GC_our_memory[i].hs_start
372  + GC_our_memory[i].hs_bytes) < (word)start)
373  ++i;
374  GC_ASSERT(i < GC_n_memory);
375  if ((word)GC_our_memory[i].hs_start <= (word)start) {
376  start = GC_our_memory[i].hs_start
377  + GC_our_memory[i].hs_bytes;
378  ++i;
379  }
380  while (i < GC_n_memory
381  && (word)GC_our_memory[i].hs_start < (word)end
382  && (word)start < (word)end) {
383  if ((word)start < (word)GC_our_memory[i].hs_start)
384  GC_add_roots_inner(start,
385  GC_our_memory[i].hs_start, TRUE);
386  start = GC_our_memory[i].hs_start
387  + GC_our_memory[i].hs_bytes;
388  ++i;
389  }
390  if ((word)start < (word)end)
391  GC_add_roots_inner(start, end, TRUE);
392  } else if (prot[0] == '-' && prot[1] == '-' && prot[2] == '-') {
393  /* Even roots added statically might disappear partially */
394  /* (e.g. the roots added by INCLUDE_LINUX_THREAD_DESCR). */
395  GC_remove_roots_subregion(start, end);
396  }
397  }

481 STATIC int GC_register_dynlib_callback(struct dl_phdr_info * info,
482  size_t size, void * ptr)
483 {
484  const ElfW(Phdr) * p;
485  ptr_t start, end;
486  int i;
487 
488  /* Make sure struct dl_phdr_info is at least as big as we need. */
489  if (size < offsetof (struct dl_phdr_info, dlpi_phnum)
490  + sizeof (info->dlpi_phnum))
491  return -1;
492 
493  p = info->dlpi_phdr;
494  for (i = 0; i < (int)info->dlpi_phnum; i++, p++) {
495  if (p->p_type == PT_LOAD) {
496  GC_has_static_roots_func callback = GC_has_static_roots;
497  if ((p->p_flags & PF_W) == 0) continue;
498 
499  start = (ptr_t)p->p_vaddr + info->dlpi_addr;
500  end = start + p->p_memsz;
501  if (callback != 0 && !callback(info->dlpi_name, start, p->p_memsz))
502  continue;
503 # ifdef PT_GNU_RELRO
504 # if CPP_WORDSZ == 64
505  /* TODO: GC_push_all eventually does the correct */
506  /* rounding to the next multiple of ALIGNMENT, so, most */
507  /* probably, we should remove the corresponding assertion */
508  /* check in GC_add_roots_inner along with this code line. */
509  /* start pointer value may require aligning. */
510  start = (ptr_t)((word)start & ~(word)(sizeof(word) - 1));
511 # endif
512  if (n_load_segs >= MAX_LOAD_SEGS) {
513  if (!load_segs_overflow) {
514  WARN("Too many PT_LOAD segments;"
515  " registering as roots directly...\n", 0);
516  load_segs_overflow = TRUE;
517  }
518  GC_add_roots_inner(start, end, TRUE);
519  } else {
520  load_segs[n_load_segs].start = start;
521  load_segs[n_load_segs].end = end;
522  load_segs[n_load_segs].start2 = 0;
523  load_segs[n_load_segs].end2 = 0;
524  ++n_load_segs;
525  }
526 # else
527  GC_add_roots_inner(start, end, TRUE);
528 # endif /* !PT_GNU_RELRO */
529  }
530  }
531 
532 # ifdef PT_GNU_RELRO
533  p = info->dlpi_phdr;
534  for (i = 0; i < (int)info->dlpi_phnum; i++, p++) {
535  if (p->p_type == PT_GNU_RELRO) {
536  /* This entry is known to be constant and will eventually be */
537  /* remapped as read-only. However, the address range covered */
538  /* by this entry is typically a subset of a previously */
539  /* encountered "LOAD" segment, so we need to exclude it. */
540  int j;
541 
542  start = (ptr_t)p->p_vaddr + info->dlpi_addr;
543  end = start + p->p_memsz;
544  for (j = n_load_segs; --j >= 0; ) {
545  if ((word)start >= (word)load_segs[j].start
546  && (word)start < (word)load_segs[j].end) {
547  if (load_segs[j].start2 != 0) {
548  WARN("More than one GNU_RELRO segment per load one\n",0);
549  } else {
550  GC_ASSERT((word)end <= (word)load_segs[j].end);
551  /* Remove from the existing load segment */
552  load_segs[j].end2 = load_segs[j].end;
553  load_segs[j].end = start;
554  load_segs[j].start2 = end;
555  }
556  break;
557  }
558  if (0 == j && 0 == GC_has_static_roots)
559  WARN("Failed to find PT_GNU_RELRO segment"
560  " inside PT_LOAD region\n", 0);
561  /* No warning reported in case of the callback is present */
562  /* because most likely the segment has been excluded. */
563  }
564  }
565  }
566 # endif
567 
568  *(int *)ptr = 1; /* Signal that we were called */
569  return 0;
