50  bool IndexPCalculator::ComputeP()
51  {
52  DBG_START_METH("IndexPCalculator::ComputeP", dbg_verbosity);
53  bool retval = true;
54 
55  // 1. check whether all columns needed by data_A() are in map cols_ - we suppose data_A is IndexSchurData
56  const std::vector<Index>* p2col_idx = dynamic_cast<const IndexSchurData*>(GetRawPtr(data_A()))->GetColIndices();
57  Index col;
58  Number* col_values = NULL;
59  Index curr_dim, curr_schur_row=0;
60  SmartPtr<const DenseVector> comp_vec;
61  const Number* comp_values;
62  std::map< Index, SmartPtr<PColumn> >::iterator find_it;
63  SmartPtr<IteratesVector> col_vec = IpData().curr()->MakeNewIteratesVector();
64  SmartPtr<IteratesVector> sol_vec = col_vec->MakeNewIteratesVector();
65  for (std::vector<Index>::const_iterator col_it=p2col_idx->begin(); col_it!=p2col_idx->end(); ++col_it){
66  col = *col_it;
67 
68  find_it = cols_.find(col);
69  if (find_it==cols_.end()) {
70  // column is in data_A but not in P-matrix ->create
71  data_A()->GetRow(curr_schur_row, *col_vec);
72  retval = Solver()->Solve(sol_vec, ConstPtr(col_vec));
73  DBG_ASSERT(retval);
74 
75  /* This part is for displaying norm2(I_z*K^(-1)*I_1) */
76  DBG_PRINT((dbg_verbosity,"\ncurr_schur_row=%d, ",curr_schur_row));
77  DBG_PRINT((dbg_verbosity,"norm2(z)=%23.16e\n",sol_vec->x()->Nrm2()));
78  /* end displaying norm2 */
79 
80  DBG_ASSERT(col_values== NULL);
81  col_values = new Number[nrows_];
82  curr_dim = 0;
83  for (Index j=0; j<sol_vec->NComps(); ++j) {
84  comp_vec = dynamic_cast<const DenseVector*>(GetRawPtr(sol_vec->GetComp(j)));
85  comp_values = comp_vec->Values();
86  IpBlasDcopy(comp_vec->Dim(), comp_values, 1, col_values+curr_dim,1);
87  curr_dim += comp_vec->Dim();
88  }
89  cols_[col] = new PColumn(nrows_, col_values);
90  col_values = NULL;
91  }
92  curr_schur_row++;
93  }
94 
95  return retval;
96  }

98  bool IndexPCalculator::GetSchurMatrix(const SmartPtr<const SchurData>& B, SmartPtr<Matrix>& S)
99  {
100  DBG_START_METH("IndexPCalculator::GetSchurMatrix", dbg_verbosity);
101  bool retval = true;
102 
103  Number* S_values;
104  if (!IsValid(S)) {
105  if ( B==data_A() ) {
106  SmartPtr<DenseSymMatrixSpace> S_sym_space = new DenseSymMatrixSpace(B->GetNRowsAdded());
107  SmartPtr<DenseSymMatrix> dS = new DenseSymMatrix(GetRawPtr(S_sym_space));
108  S_values = dS->Values();
109  S = GetRawPtr(dS);
110  }
111  else {
112  SmartPtr<DenseGenMatrixSpace> S_sym_space = new DenseGenMatrixSpace(B->GetNRowsAdded(), B->GetNRowsAdded());
113  SmartPtr<DenseGenMatrix> dS = new DenseGenMatrix(GetRawPtr(S_sym_space));
114  S_values = dS->Values();
115  S = GetRawPtr(dS);
116  }
117  }
118  else {
119  // Try DenseGenMatrix - if NULL, try DenseSymMatrix
120  SmartPtr<DenseGenMatrix> dS_gen = dynamic_cast<DenseGenMatrix*>(GetRawPtr(S));
121  if (!IsValid(dS_gen)) {
122  SmartPtr<DenseSymMatrix> dS_sym = dynamic_cast<DenseSymMatrix*>(GetRawPtr(S));
123  S_values = dS_sym->Values();
124  }
125  else {
126  S_values = dS_gen->Values();
127  }
128  }
129  /*
130  DenseGenMatrix* dS = static_cast<DenseGenMatrix*>(&S);
131  DBG_ASSERT(dynamic_cast<const DenseGenMatrix*>(&S));
132  */
133  // Check whether data_A was changed from the outside
134  if (ncols_!=data_A()->GetNRowsAdded()) {
135  ncols_ = data_A()->GetNRowsAdded();
136  ComputeP();
137  }
138  /*
139  DBG_ASSERT(dS->NRows()==dS->NCols());
140  DBG_ASSERT(dS->NRows()==data_A()->GetNRowsAdded());
141  */
142  std::vector<Index> indices;
143  std::vector<Number> factors;
144 
145  // Compute S = B^T*P from indices, factors and P
146  const std::vector<Index>* data_A_idx = dynamic_cast<const IndexSchurData*>(GetRawPtr(data_A()))->GetColIndices();
147  const std::vector<Index>* data_B_idx = dynamic_cast<const IndexSchurData*>(GetRawPtr(B))->GetColIndices();
148  Index col_count = 0;
149  for (std::vector<Index>::const_iterator a_it=data_A_idx->begin(); a_it!=data_A_idx->end(); ++a_it) {
150  cols_[*a_it]->GetSchurMatrixRows(data_B_idx, S_values+col_count*ncols_);
151  col_count++;
152  }
153  return retval;
154  }
