175 public function treeAdd "author: PA
176  Copied from generic implementation. Changed that no hashfunction is passed
177  since a string (ComponentRef) can not be uniquely mapped to an int. Therefore we need to compare two strings
178  to get a unique ordering.
179 
180  Actually, hashing is still important in order to speed up comparison of strings... So it was re-added in a
181  good way, see function keyCompareNinjaSecretHashTricks"
182  input BinTree inBinTree;
183  input Key inKey;
184  input Value inValue;
185  output BinTree outBinTree;
186 algorithm
187  outBinTree := matchcontinue (inBinTree,inKey,inValue)
188  local
189  Key rkey;
190  Value rval;
191  Option<BinTree> left,right;
192  BinTree t_1,t,right_1,left_1;
193  Option<TreeValue> optVal;
194 
195  case (TREENODE(value = NONE(),leftSubTree = NONE(),rightSubTree = NONE()),_,_)
196  then
197  TREENODE(SOME(TREEVALUE(inKey,inValue)),NONE(),NONE());
198 
199  case (TREENODE(value = SOME(TREEVALUE(rkey,_)),leftSubTree = left,rightSubTree = right),_,_)
200  equation
201  0 = keyCmp(rkey,inKey);
202  then
203  TREENODE(SOME(TREEVALUE(rkey,inValue)),left,right);
204 
205  case (TREENODE(value = optVal as SOME(TREEVALUE(rkey,_)),leftSubTree = left,rightSubTree = (SOME(t))),_,_)
206  equation
207  1 = keyCmp(rkey,inKey);
208  t_1 = treeAdd(t, inKey, inValue);
209  then
210  TREENODE(optVal,left,SOME(t_1));
211 
212  case (TREENODE(value = optVal as SOME(TREEVALUE(rkey,_)),leftSubTree = left,rightSubTree = (NONE())),_,_)
213  equation
214  1 = keyCmp(rkey,inKey);
215  right_1 = treeAdd(TREENODE(NONE(),NONE(),NONE()), inKey, inValue);
216  then
217  TREENODE(optVal,left,SOME(right_1));
218 
219  case (TREENODE(value = optVal as SOME(TREEVALUE(rkey,_)),leftSubTree = (SOME(t)),rightSubTree = right),_,_)
220  equation
221  -1 = keyCmp(rkey,inKey);
222  t_1 = treeAdd(t, inKey, inValue);
223  then
224  TREENODE(optVal,SOME(t_1),right);
225 
226  case (TREENODE(value = optVal as SOME(TREEVALUE(rkey,_)),leftSubTree = (NONE()),rightSubTree = right),_,_)
227  equation
228  -1 = keyCmp(rkey,inKey);
229  left_1 = treeAdd(TREENODE(NONE(),NONE(),NONE()), inKey, inValue);
230  then
231  TREENODE(optVal,SOME(left_1),right);
232 
233  else
234  equation
235  Error.addMessage(Error.INTERNAL_ERROR,{"- BinaryTreeInt.treeAdd failed\n"});
236  then
237  fail();
238  end matchcontinue;
239 end treeAdd;
