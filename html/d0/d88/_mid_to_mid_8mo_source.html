64 function longJmpGoto
65  "
66  Replace a longjmp within a function call with goto.
67  longjmp within function calls isn't allowed so
68  this is necessary to form correct c form midcode.
69  "
70  input MidCode.Function oldFunction;
71  output MidCode.Function newFunction;
72 protected
73  list<MidCode.Block> newBody, oldBody;
74  MidCode.Block newBlock, oldBlock;
75  Integer node,jump;
76  list<Integer> jumps;
77  list<Integer> nodes_tmp, checkedNodes;
78  list<tuple<list<Integer>,Integer>> tasks, tasks_tmp; // [ ([Int], Int) ]
79 algorithm
80  /*
81  do depth first search
82  keep a stack of jmp_targets
83  when encoutering a PUSHJMP: push the target label onto jmp_targets
84  when encountering a LONGJMP and jmp_targets in't empty: replace with goto the top of jmp_targets
85  when encountering a POPJMP: pop jmp_targets
86  */
87  /*
88  How to depth first
89  maybe not use program stack ~1000 blocks might cause stack overflow
90  so jmp_targets is a list
91  while loop with empty check on nodes
92  start wih entryId-block
93  keep list of used ids, alreadyChecked
94  need to look up blockId in Function structure
95  find the successors of current node, add all that are not in alreadyChecked
96  build a new list of blocks with terminators maybe replaced
97  replace the body part of function
98  */
99  /*
100  Leave the buffer variables in the function still declared.
101  */
102  oldBody := oldFunction.body;
103  newBody := {};
104  checkedNodes := {oldFunction.entryId};
105  tasks := {({},oldFunction.entryId)};
106 
107  while not listEmpty(tasks) loop
108  ((jumps,node) :: tasks) := tasks; // pop
109  oldBlock := lookupId(oldBody,node); // O(length(oldBody))
110  newBlock := oldBlock; // don't change the block by defualt
111  if isPushJmp(oldBlock.terminator) then
112  jumps := (listHead(getSuccessors(oldBlock)) :: jumps); // push
113  elseif isLongJmp(oldBlock.terminator) and not listEmpty(jumps) then
114  (jump :: _) := jumps; // peek
115  newBlock := MidCode.BLOCK(id=oldBlock.id,stmts=oldBlock.stmts, terminator=MidCode.GOTO(jump));
116  elseif isPopJmp(oldBlock.terminator) then
117  (_ :: jumps) := jumps; // pop
118  end if;
119  newBody := newBlock :: newBody;
120 
121  nodes_tmp := List.setDifference(getSuccessors(oldBlock), checkedNodes);
122  checkedNodes := listAppend(nodes_tmp, checkedNodes);
123  tasks_tmp := list( (jumps,node_tmp) for node_tmp in nodes_tmp );
124  tasks := listAppend(tasks_tmp, tasks);
125  end while;
126  newBody := listReverse(newBody);
127 
128  newFunction := MidCode.FUNCTION(
129  name=oldFunction.name
130  ,locals=oldFunction.locals
131  ,localBufs=oldFunction.localBufs
132  ,localBufPtrs=oldFunction.localBufPtrs
133  ,inputs=oldFunction.inputs
134  ,outputs=oldFunction.outputs
135  ,body=newBody
136  ,entryId=oldFunction.entryId
137  ,exitId=oldFunction.exitId
138  );
139 end longJmpGoto;

141 function lookupId
142  input list<MidCode.Block> blocks;
143  input Integer id;
144  output MidCode.Block block_;
145 protected
146  list<MidCode.Block> blocks_local;
147  MidCode.Block block_local;
148 algorithm
149  block_ := match blocks
150  case (block_local :: _) guard (block_local.id == id) then block_local;
151  case (_ :: blocks_local) then lookupId(blocks_local, id);

162 function getSuccessors
163  input MidCode.Block block_;
164  output list<Integer> neighbours;
165 protected
166  Integer l0,l1;
167  list<tuple<Integer,Integer>> switchList;
168 algorithm
169  neighbours := match block_.terminator
170  case MidCode.GOTO(l0) then {l0};
171  case MidCode.BRANCH(_,l0,l1) then {l0,l1};
172  case MidCode.CALL(_,_,_,_,l0) then {l0};
173  case MidCode.RETURN() then {};
174  case MidCode.SWITCH(_,switchList) then list(tupleSnd(x) for x in switchList);
175  case MidCode.LONGJMP() then {};
176  case MidCode.PUSHJMP(_,_,l0) then {l0};
177  case MidCode.POPJMP(_,l0) then {l0};
178  end match;
179 end getSuccessors;
