64 int zmq::req_t::xsend (msg_t *msg_)
65 {
66  // If we've sent a request and we still haven't got the reply,
67  // we can't send another request unless the strict option is disabled.
68  if (receiving_reply) {
69  if (strict) {
70  errno = EFSM;
71  return -1;
72  }
73 
74  receiving_reply = false;
75  message_begins = true;
76  }
77 
78  // First part of the request is the request identity.
79  if (message_begins) {
80  reply_pipe = NULL;
81 
82  if (request_id_frames_enabled) {
83  request_id++;
84 
85  // Copy request id before sending (see issue #1695 for details).
86  uint32_t *request_id_copy = (uint32_t *) malloc (sizeof (uint32_t));
87  zmq_assert (request_id_copy);
88 
89  *request_id_copy = request_id;
90 
91  msg_t id;
92  int rc = id.init_data (request_id_copy, sizeof (uint32_t),
93  free_id, NULL);
94  errno_assert (rc == 0);
95  id.set_flags (msg_t::more);
96 
97  rc = dealer_t::sendpipe (&id, &reply_pipe);
98  if (rc != 0)
99  return -1;
100  }
101 
102  msg_t bottom;
103  int rc = bottom.init ();
104  errno_assert (rc == 0);
105  bottom.set_flags (msg_t::more);
106 
107  rc = dealer_t::sendpipe (&bottom, &reply_pipe);
108  if (rc != 0)
109  return -1;
110  zmq_assert (reply_pipe);
111 
112  message_begins = false;
113 
114  // Eat all currently available messages before the request is fully
115  // sent. This is done to avoid:
116  // REQ sends request to A, A replies, B replies too.
117  // A's reply was first and matches, that is used.
118  // An hour later REQ sends a request to B. B's old reply is used.
119  msg_t drop;
120  while (true) {
121  rc = drop.init ();
122  errno_assert (rc == 0);
123  rc = dealer_t::xrecv (&drop);
124  if (rc != 0)
125  break;
126  drop.close ();
127  }
128  }
129 
130  bool more = msg_->flags () & msg_t::more ? true : false;
131 
132  int rc = dealer_t::xsend (msg_);
133  if (rc != 0)
134  return rc;
135 
136  // If the request was fully sent, flip the FSM into reply-receiving state.
137  if (!more) {
138  receiving_reply = true;
139  message_begins = true;
140  }
141 
142  return 0;
143 }

145 int zmq::req_t::xrecv (msg_t *msg_)
146 {
147  // If request wasn't send, we can't wait for reply.
148  if (!receiving_reply) {
149  errno = EFSM;
150  return -1;
151  }
152 
153  // Skip messages until one with the right first frames is found.
154  while (message_begins) {
155  // If enabled, the first frame must have the correct request_id.
156  if (request_id_frames_enabled) {
157  int rc = recv_reply_pipe (msg_);
158  if (rc != 0)
159  return rc;
160 
161  if (unlikely (!(msg_->flags () & msg_t::more) ||
162  msg_->size () != sizeof (request_id) ||
163  *static_cast<uint32_t *> (msg_->data ()) != request_id)) {
164  // Skip the remaining frames and try the next message
165  while (msg_->flags () & msg_t::more) {
166  rc = recv_reply_pipe (msg_);
167  errno_assert (rc == 0);
168  }
169  continue;
170  }
171  }
172 
173  // The next frame must be 0.
174  // TODO: Failing this check should also close the connection with the peer!
175  int rc = recv_reply_pipe (msg_);
176  if (rc != 0)
177  return rc;
178 
179  if (unlikely (!(msg_->flags () & msg_t::more) || msg_->size () != 0)) {
180  // Skip the remaining frames and try the next message
181  while (msg_->flags () & msg_t::more) {
182  rc = recv_reply_pipe (msg_);
183  errno_assert (rc == 0);
184  }
185  continue;
186  }
187 
188  message_begins = false;
189  }
190 
191  int rc = recv_reply_pipe (msg_);
192  if (rc != 0)
193  return rc;
194 
195  // If the reply is fully received, flip the FSM into request-sending state.
196  if (!(msg_->flags () & msg_t::more)) {
197  receiving_reply = false;
198  message_begins = true;
199  }
200 
201  return 0;
202 }

222 int zmq::req_t::xsetsockopt (int option_, const void *optval_, size_t optvallen_)
223 {
224  bool is_int = (optvallen_ == sizeof (int));
225  int value = 0;
226  if (is_int)
227  memcpy (&value, optval_, sizeof (int));
228 
229  switch (option_) {
230  case ZMQ_REQ_CORRELATE:
231  if (is_int && value >= 0) {
232  request_id_frames_enabled = (value != 0);
233  return 0;
234  }
235  break;
236 
237  case ZMQ_REQ_RELAXED:
238  if (is_int && value >= 0) {
239  strict = (value == 0);
240  return 0;
241  }
242  break;
243 
244  default:
245  break;
246  }
247 
248  return dealer_t::xsetsockopt (option_, optval_, optvallen_);
249 }

259 {

261  pipe_t *pipe = NULL;
262  int rc = dealer_t::recvpipe (msg_, &pipe);

282 int zmq::req_session_t::push_msg (msg_t *msg_)
283 {
284  switch (state) {
285  case bottom:
286  if (msg_->flags () == msg_t::more) {
287  // In case option ZMQ_CORRELATE is on, allow request_id to be
288  // transfered as first frame (would be too cumbersome to check
289  // whether the option is actually on or not).
290  if (msg_->size () == sizeof (uint32_t)) {
291  state = request_id;
292  return session_base_t::push_msg (msg_);
293  }
294  else if (msg_->size () == 0) {
295  state = body;
296  return session_base_t::push_msg (msg_);
297  }
298  }
299  break;
300  case request_id:
301  if (msg_->flags () == msg_t::more && msg_->size () == 0) {
302  state = body;
303  return session_base_t::push_msg (msg_);
304  }
305  break;
306  case body:
307  if (msg_->flags () == msg_t::more)
308  return session_base_t::push_msg (msg_);
309  if (msg_->flags () == 0) {
310  state = bottom;
311  return session_base_t::push_msg (msg_);
312  }
313  break;
314  }
315  errno = EFAULT;
316  return -1;
317 }
