68 static void convertProfileData(const char *outputPath, const char *prefix, int numFnsAndBlocks)
69 {
70  const char* fullFileName;
71  if (0 > GC_asprintf(&fullFileName, "%s%s", outputPath, prefix)) {
72  throwStreamPrint(NULL, "modelinfo.c: Error: can not allocate memory.");
73  }
74  size_t len = strlen(fullFileName);
75  char *inBinaryInt = (char*)malloc(sizeof(char)*(len + 14));
76  char *inBinaryReal = (char*)malloc(sizeof(char)*(len + 15));
77  FILE *finInt, *finReal, *fout;
78  int fail = 0;
79  size_t intRowSize = sizeof(uint32_t)*(1+numFnsAndBlocks);
80  size_t realRowSize = sizeof(double)*(2+numFnsAndBlocks);
81  uint32_t *intData;
82  double *realData;
83  /* There seems to be a very small performance gain from using mmap here.
84  * But it is not worse than in-memory and may be paged out on low memory. */
85  omc_mmap_write intMap,realMap;
86 
87  memcpy(inBinaryInt,fullFileName,len);
88  memcpy(inBinaryReal,fullFileName,len);
89  strcpy(inBinaryInt + len, "_prof.intdata");
90  strcpy(inBinaryReal + len, "_prof.realdata");
91 
92  intMap = omc_mmap_open_write(inBinaryInt,0);
93  intData = (uint32_t*)intMap.data;
94  assert(0 == intMap.size % intRowSize);
95  matrix_transpose_uint32(intData, 1+numFnsAndBlocks, intMap.size / intRowSize);
96  omc_mmap_close_write(intMap);
97 
98  realMap = omc_mmap_open_write(inBinaryReal,0);
99  realData = (double*)realMap.data;
100  assert(0 == realMap.size % realRowSize);
101  matrix_transpose(realData, 2+numFnsAndBlocks, realMap.size / realRowSize);
102  omc_mmap_close_write(realMap);
103 
104  free(inBinaryInt);
105  free(inBinaryReal);
106 }

108 static void printPlotCommand(FILE *plt, const char *plotFormat, const char *title, const char *outputPath, const char *prefix, int numFnsAndBlocks, int i, int id, const char *idPrefix) {
109  const char *format = "plot \"%s%s_prof.realdata\" binary format=\"%%%ddouble\" using 1:($%d>1e-9 ? $%d : 1e-30) w l lw %d\n";
110  const char *formatCount = "plot \"%s%s_prof.intdata\" binary format=\"%%%duint32\" using %d w l lw %d\n";
111  unsigned long nmin = 0, nmax = 0;
112  double ymin = 0.0, ymax = 0.0;
113  double ygraphmin = 1e-30, ygraphmax = 0.0;
114  if(!plt) return;
115  if(i >= 0) {
116  /*
117  * Note: We set the yrange here because otherwise we get warnings if the
118  * number of calls are the same in every time step (i.e. autoscale thinks
119  * the range is [3:3])
120  */
121  nmin = rt_ncall_min(SIM_TIMER_FIRST_FUNCTION + i);
122  nmax = rt_ncall_max(SIM_TIMER_FIRST_FUNCTION + i);
123  ymin = nmin==0 ? -0.01 : nmin*0.95;
124  ymax = nmax==0 ? 0.01 : nmax*1.05;
125  ygraphmax = rt_max_accumulated(SIM_TIMER_FIRST_FUNCTION + i) * 1.01 + 1e-30;
126  }
127  /* SVG thumbnail */
128  fputs("set terminal svg\n", plt);
129  fprintf(plt, "unset xtics\n");
130  fprintf(plt, "unset ytics\n");
131  fprintf(plt, "unset border\n");
132  fprintf(plt, "set output \"%s%s_prof.%s%d.thumb.svg\"\n", outputPath, prefix, idPrefix, id);
133  fprintf(plt, "set title\n");
134  fprintf(plt, "set xlabel\n");
135  fprintf(plt, "set ylabel\n");
136  fprintf(plt, "set log y\n");
137 
138  if(i>=0)
139  {
140  fprintf(plt, "set yrange [*:%g]\n", ygraphmax);
141  }
142  else
143  {
144  fprintf(plt, "set yrange [*:*]\n");
145  }
146  /* time */
147  fprintf(plt, format, outputPath, prefix, 2+numFnsAndBlocks, 3+i, 3+i, 4);
148  fprintf(plt, "set nolog xy\n");
149  /* count */
150  if(i >= 0) {
151  fprintf(plt, "unset ytics\n");
152  if(nmin == nmax)
153  {
154  fprintf(plt, "set yrange [%g:%g]\n", ymin, ymax);
155  }
156  else
157  {
158  fprintf(plt, "set yrange [*:*]\n");
159  }
160  fprintf(plt, "set output \"%s%s_prof.%s%d_count.thumb.svg\"\n", outputPath, prefix, idPrefix, id);
161  fprintf(plt, formatCount, outputPath, prefix, 1+numFnsAndBlocks, 2+i, 4);
162  fprintf(plt, "set ytics\n");
163  }
164 
165  /* SVG */
166  fprintf(plt, "set xtics\n");
167  fprintf(plt, "set ytics\n");
168  fprintf(plt, "set border\n");
169  fprintf(plt, "set terminal %s\n", plotFormat);
170  fprintf(plt, "set title \"%s\"\n", title);
171  fprintf(plt, "set xlabel \"Global step at time\"\n");
172  fprintf(plt, "set ylabel \"Execution time [s]\"\n");
173  fprintf(plt, "set output \"%s%s_prof.%s%d.%s\"\n", outputPath, prefix, idPrefix, id, plotFormat);
174  fprintf(plt, "set log y\n");
175  if(i>=0)
176  {
177  fprintf(plt, "set yrange [*:%g]\n", ygraphmax);
178  }
179  else
180  {
181  fprintf(plt, "set yrange [*:*]\n");
182  }
183  fprintf(plt, format, outputPath, prefix, 2+numFnsAndBlocks, 3+i, 3+i, 2);
184  /* count */
185  fprintf(plt, "set nolog xy\n");
186  if(i >= 0)
187  {
188  if(nmin == nmax)
189  {
190  fprintf(plt, "set yrange [%g:%g]\n", ymin, ymax);
191  }
192  else
193  {
194  fprintf(plt, "set yrange [*:*]\n");
195  }
196  fprintf(plt, "set xlabel \"Global step number\"\n");
197  fprintf(plt, "set ylabel \"Execution count\"\n");
198  fprintf(plt, "set output \"%s_prof.%s%d_count.%s\"\n", prefix, idPrefix, id, plotFormat);
199  fprintf(plt, formatCount, outputPath, prefix, 1+numFnsAndBlocks, 2+i, 2);
200  }
201 }

203 static void printStrXML(FILE *fout, const char *str)
204 {
205  while(*str) {
206  switch (*str) {
207  case '<': fputs("&lt;",fout);break;
208  case '>': fputs("&gt;",fout);break;
209  case '\'': fputs("&apos;",fout);break;

213  }
214  str++;
215  }

272 static int getVarIdByName(DATA *data, const char* varName)
273 {
274  int i;
275  for(i=0;i<data->modelData->nVariablesReal;++i)
276  if (strcmp(varName, data->modelData->realVarsData[i].info.name) == 0)
277  return data->modelData->realVarsData[i].info.id;
278 
279  for(i=0;i<data->modelData->nParametersReal;++i)
280  if (strcmp(varName, data->modelData->realParameterData[i].info.name) == 0)
281  return data->modelData->realParameterData[i].info.id;
282 
283  for(i=0;i<data->modelData->nVariablesInteger;++i)
284  if (strcmp(varName, data->modelData->integerVarsData[i].info.name) == 0)
285  return data->modelData->integerVarsData[i].info.id;
286 
287  for(i=0;i<data->modelData->nParametersInteger;++i)
288  if (strcmp(varName, data->modelData->integerParameterData[i].info.name) == 0)
289  return data->modelData->integerParameterData[i].info.id;
290 
291  for(i=0;i<data->modelData->nVariablesBoolean;++i)
292  if (strcmp(varName, data->modelData->booleanVarsData[i].info.name) == 0)
293  return data->modelData->booleanVarsData[i].info.id;
294 
295  for(i=0;i<data->modelData->nParametersBoolean;++i)
296  if (strcmp(varName, data->modelData->booleanParameterData[i].info.name) == 0)
297  return data->modelData->booleanParameterData[i].info.id;
298 
299  for(i=0;i<data->modelData->nVariablesString;++i)
300  if (strcmp(varName, data->modelData->stringVarsData[i].info.name) == 0)
301  return data->modelData->stringVarsData[i].info.id;
302 
303  for(i=0;i<data->modelData->nParametersString;++i)
304  if (strcmp(varName, data->modelData->stringParameterData[i].info.name) == 0)
305  return data->modelData->stringParameterData[i].info.id;
306 
307  return 0;
308 }

549 void escapeJSON(FILE *file, const char *data)
550 {
551  while (*data) {
552  switch (*data) {
553  case '\"': if (fputs("\\\"",file)<0) ERROR_WRITE();break;
554  case '\\': if (fputs("\\\\",file)<0) ERROR_WRITE();break;
555  case '\n': if (fputs("\\n",file)<0) ERROR_WRITE();break;
556  case '\b': if (fputs("\\b",file)<0) ERROR_WRITE();break;
557  case '\f': if (fputs("\\f",file)<0) ERROR_WRITE();break;
558  case '\r': if (fputs("\\r",file)<0) ERROR_WRITE();break;
559  case '\t': if (fputs("\\t",file)<0) ERROR_WRITE();break;
560  default:
561  if (*data < ' ') { /* Escape other control characters */
562  if (fprintf(file, "\\u%04x",*data)<0) ERROR_WRITE();
563  } else {
564  if (putc(*data,file)<0) ERROR_WRITE();
565  }
566  }
567  data++;
568  }
569 }
