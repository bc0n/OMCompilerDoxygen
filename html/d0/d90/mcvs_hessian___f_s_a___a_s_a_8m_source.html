2 %mcvsHessian_FSA_ASA - CVODES Hessian calculation example (FSA over ASA)
3 % The following is a simple example illustrating the use
4 % of simultaneous FSA and ASA computations in order to
5 % evaluate the Hessian of a functional depending on the
6 % ODE solution.
7 %
8 % The forward problem consists of the following 3 equations
9 %
10 % dy1/dt = - p1 * y1^2 - y3;
11 % dy2/dt = - y2;
12 % dy3/dt = - p2^2 * y2 * y3;
13 %
14 % depending on the parameters p1 = 1.0 and p2 = 2.0.
15 %
16 % The initial conditions are y1(0) = y2(0) = y390) = 1.0
17 %
18 % The functional considered is
19 %
20 % 2

184 function [ySd, flag, new_data] = rhsSfn(t,y,yd,yS,data)
185 % Sensitivity right-hand side function
186 
187 p1 = data.p(1);
188 p2 = data.p(2);
189 
190 s = yS(:,1);
191 
192 fys1 = - 2.0*p1*y(1) * s(1) - s(3);
193 fys2 = - s(2);
194 fys3 = - p2*p2*y(3) * s(2) - p2*p2*y(2) * s(3);
195 
196 ySd(1,1) = fys1 - y(1)*y(1);
