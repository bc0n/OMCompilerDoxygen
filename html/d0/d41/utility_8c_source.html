40 modelica_real real_int_pow(threadData_t *threadData, modelica_real base, modelica_integer n)
41 {
42  modelica_real result = 1.0;
43  modelica_integer m = n < 0;
44  FILE_INFO info = omc_dummyFileInfo;
45  if(m)
46  {
47  if(base == 0.0)
48  omc_assert(threadData, info, "Model error. 0^(%i) is not defined", n);
49  n = -n;
50  }
51  while(n != 0)
52  {
53  if((n % 2) != 0)
54  {
55  result *= base;
56  n--;
57  }
58  base *= base;
59  n /= 2;
60  }
61  return m ? (1 / result) : result;
62 }

69 extern int OpenModelica_regexImpl(const char* str, const char* re, const int maxn, int extended, int ignoreCase, void*(*mystrdup)(const char*), void **outMatches)
70 {
71  regex_t myregex;
72  int nmatch=0,i,rc,res;
73  int flags = (extended ? REG_EXTENDED : 0) | (ignoreCase ? REG_ICASE : 0) | (maxn ? 0 : REG_NOSUB);
74 #if !defined(_MSC_VER)
75  regmatch_t matches[maxn < 1 ? 1 : maxn];
76 #else
77  /* Stupid compiler */
78  regmatch_t *matches;
79  matches = (regmatch_t*)malloc(maxn*sizeof(regmatch_t));
80  assert(matches != NULL);
81 #endif
82  memset(&myregex, 1, sizeof(regex_t));
83  rc = regcomp(&myregex, re, flags);
84  if (rc && maxn == 0) {
85 #if defined(_MSC_VER)
86  free(matches);
87 #endif
88  return 0;
89  }
90  if (rc) {
91  char err_buf[2048] = {0};
92  int len = 0;
93  len += snprintf(err_buf+len,2040-len,"Failed to compile regular expression: %s with error: ", re);
94  regerror(rc, &myregex, err_buf+len, 2048-len);
95  regfree(&myregex);
96  if (maxn) {
97  outMatches[0] = mystrdup(err_buf);
98  for (i=1; i<maxn; i++)
99  outMatches[i] = mystrdup("");
100  }
101 #if defined(_MSC_VER)
102  free(matches);
103 #endif
104  return 0;
105  }
106  res = regexec(&myregex, str, maxn, matches, 0);
107  if (!maxn)
108  nmatch += res == 0 ? 1 : 0;
109  else if (maxn) {
110  char *dup = strdup(str);
111  for (i=0; i<maxn; i++) {
112  if (!res && matches[i].rm_so != -1) {
113  memcpy(dup, str + matches[i].rm_so, matches[i].rm_eo - matches[i].rm_so);
114  dup[matches[i].rm_eo - matches[i].rm_so] = '\0';
115  outMatches[nmatch++] = mystrdup(dup);
116  }
117  }
118  for (i=nmatch; i<maxn; i++) {
119  outMatches[i] = mystrdup("");
120  }
121  free(dup);
122  }
123 
124  regfree(&myregex);
125 #if defined(_MSC_VER)
126  free(matches);
127 #endif
128  return nmatch;
129 }

163 static modelica_string uriToFilenameRegularPaths(modelica_string uri_om, const char *uri, char buf[PATH_MAX], const char *origUri, const char *resourcesDir)
164 {
165  FILE_INFO info = omc_dummyFileInfo;
166  struct stat stat_buf;
167  size_t len;
168  int uriExists = 0==stat(uri, &stat_buf);
169  if (resourcesDir) {
170  if (strlen(resourcesDir)+strlen(uri)+2 < PATH_MAX) {
171  sprintf(buf, "%s/%s", resourcesDir, uri);
172  if (!uriExists || 0==stat(buf, &stat_buf)) {
173  /* The path with resources prepended either exists or the path without resources does not exist
174  * So re-run uriToFilenameRegularPaths with resourcesDir prepended to the URI
175  */
176  char newbuf[PATH_MAX];
177  return uriToFilenameRegularPaths(NULL, buf, newbuf, origUri, NULL);
178  }
179  } else {
180  omc_assert_warning(info, "Path longer than PATH_MAX: %s/%s", resourcesDir, uri);
181  }
182  }
183  if (uriExists) {
184  if (resourcesDir) {
185  omc_assert_warning(info, PATH_NOT_IN_FMU_RESOURCES, uri);
186  }
187  /* This is a file, directory, etc. Can't use open to check this. */
188  if (0==realpath(uri, buf)) {
189  /* Unexpected; we know the file exists, but realpath failed. Just return the URI */
190  omc_assert_warning(info, "realpath failed for existing path %s: %s", uri, strerror(errno));
191  return uri_om ? uri_om : mmc_mk_scon(uri);
192  }
193  /* Use the realpath result */
194  if (S_ISDIR(stat_buf.st_mode)) {
195  /* Make directories end with a / if the original URI ends with a / */
196  len = strlen(buf);
197  if (buf[len-1]!='/' && origUri[strlen(origUri)-1]=='/') {
198  if (len+1 >= PATH_MAX) {
199  /* Can't fit the path; just return the original URI */
200  omc_assert_warning(info, "Path longer than PATH_MAX: %s/, returning %s", buf, buf);
201  return uri_om ? uri_om : mmc_mk_scon(uri);
202  }
203  strcpy(buf+len, "/");
204  }
205  }
206  return (0==strcmp(uri, buf) && uri_om) ? uri_om : mmc_mk_scon(buf);
207  }
208 
209  if (uri[0]=='/') {
210  /* Absolute path */
211  return uri_om ? uri_om : mmc_mk_scon(uri);
212  }
213  if (0==realpath("./", buf)) {
214  /* Failed to resolve ./ */
215  omc_assert_warning(info, "realpath failed to resolve ./");
216  return uri_om ? uri_om : mmc_mk_scon(uri);
217  }
218  len = strlen(buf);
219  if (len+strlen(uri)+1 >= PATH_MAX) {
220  /* Can't fit the path; just return the original URI */
221  omc_assert_warning(info, "Path longer than PATH_MAX: %s/%s, returning %s", buf, uri, uri);
222  return uri_om ? uri_om : mmc_mk_scon(uri);
223  }
224  /* Copy the rest of the URI onto the buffer */
225  if (buf[len-1]!='/') {
226  buf[len++]='/';
227  }
228  strcpy(buf+len, uri);
229  return mmc_mk_scon(buf);
230 }

261 extern modelica_string OpenModelica_uriToFilename_impl(threadData_t *threadData, modelica_string uri_om, const char *resourcesDir)
262 {
263  FILE_INFO info = omc_dummyFileInfo;
264  char buf[PATH_MAX];
265  const char *uri = MMC_STRINGDATA(uri_om);
266  modelica_string dir;
267  if (0==strncasecmp(uri, "modelica://", 11)) {
268  struct stat stat_buf;
269  uri += 11;
270  getIdent(uri, buf, &uri);
271  if (0 == *buf) {
272  omc_assert(threadData, info, "Malformed URI (couldn't get a class name): %s", MMC_STRINGDATA(uri_om));
273  MMC_THROW();
274  }
275  dir = lookupDirectoryFromName(buf, threadData->localRoots[LOCAL_ROOT_URI_LOOKUP]);
276  if (dir==NULL || MMC_STRLEN(dir)==0) {
277  omc_assert(threadData, info, "Failed to lookup URI (is the package loaded?) %s", MMC_STRINGDATA(uri_om));
278  MMC_THROW();
279  }
280  if (resourcesDir) {
281  if (strlen(dir)+2+strlen(resourcesDir) >= PATH_MAX) {
282  omc_assert_warning(info, "Path longer than PATH_MAX: %s/%s, ignoring the resourcesDir", dir, resourcesDir);
283  } else {
284  int dirExists = 0==stat(dir, &stat_buf);
285  sprintf(buf, "%s/%s", dir, resourcesDir);
286  if (!dirExists || 0==stat(buf, &stat_buf)) {
287  dir = buf;
288  } else {
289  omc_assert_warning(info, PATH_NOT_IN_FMU_RESOURCES, dir);
290  }
291  }
292  }
293  /* We found where the package is stored */
294  while (1) {
295  if (*uri == '.') {
296  uri++;
297  } else {
298  break;
299  }
300  getIdent(uri, buf, &uri);
301  if (0 == *buf) {
302  if (*uri == '.') {
303  omc_assert(threadData, info, "Malformed URI (double dot in class name): %s", MMC_STRINGDATA(uri_om));
304  MMC_THROW();
305  }
306  break; /* / or end of string */
307  }
308  if (MMC_STRLEN(dir)+strlen(buf)+1 >= PATH_MAX) {
309  omc_assert(threadData, info, "Failed to resolve URI; path longer than PATH_MAX(%d): %s", PATH_MAX, MMC_STRINGDATA(uri_om));
310  MMC_THROW();
311  }
312  /* Move the found ident last in the path */
313  strcpy(buf+MMC_STRLEN(dir)+1, buf);
314  /* Copy the old directory in there */
315  strcpy(buf, MMC_STRINGDATA(dir));
316  buf[MMC_STRLEN(dir)]='/';
317  if (!(0==stat(buf, &stat_buf) && S_ISDIR(stat_buf.st_mode))) {
318  break;
319  }
320  dir = mmc_mk_scon(buf);
321  }
322  while (*uri && *(uri++) != '/') /* Ignore */;
323  if (0 == strlen(uri)) {
324  /* realpath, etc */
325  return uriToFilenameRegularPaths(dir, MMC_STRINGDATA(dir), buf, MMC_STRINGDATA(uri_om), NULL);
326  }
327  if (MMC_STRLEN(dir)+strlen(uri-1) >= PATH_MAX) {
328  return mmc_emptystring;
329  }
330  strcpy(buf, MMC_STRINGDATA(dir));
331  strcpy(buf+MMC_STRLEN(dir), uri-1);
332  dir = mmc_mk_scon(buf);
333  return uriToFilenameRegularPaths(dir, MMC_STRINGDATA(dir), buf, MMC_STRINGDATA(uri_om), NULL);
334  }
335  if (0==strncasecmp(uri, "file://", 7)) {
336  return uriToFilenameRegularPaths(NULL, uri+7, buf, MMC_STRINGDATA(uri_om), resourcesDir);
337  }
338  if (strstr(uri, "://")) {
339  omc_assert(threadData, info, "Unknown URI schema: %s", MMC_STRINGDATA(uri_om));
340  MMC_THROW();
341  }
342  return uriToFilenameRegularPaths(uri_om, uri, buf, MMC_STRINGDATA(uri_om), resourcesDir);
343 }
