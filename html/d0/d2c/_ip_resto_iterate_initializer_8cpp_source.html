44  bool RestoIterateInitializer::SetInitialIterates()
45  {
46  DBG_START_METH("RestoIterateInitializer::SetInitialIterates",
47  dbg_verbosity);
48 
49  // Get a grip on the restoration phase NLP and obtain the pointers
50  // to the original NLP data
51  SmartPtr<RestoIpoptNLP> resto_ip_nlp =
52  static_cast<RestoIpoptNLP*> (&IpNLP());
53  SmartPtr<IpoptNLP> orig_ip_nlp =
54  static_cast<IpoptNLP*> (&resto_ip_nlp->OrigIpNLP());
55  SmartPtr<IpoptData> orig_ip_data =
56  static_cast<IpoptData*> (&resto_ip_nlp->OrigIpData());
57  SmartPtr<IpoptCalculatedQuantities> orig_ip_cq =
58  static_cast<IpoptCalculatedQuantities*> (&resto_ip_nlp->OrigIpCq());
59 
60  // Set the value of the barrier parameter
61  Number resto_mu;
62  resto_mu = Max(orig_ip_data->curr_mu(),
63  orig_ip_cq->curr_c()->Amax(),
64  orig_ip_cq->curr_d_minus_s()->Amax());
65  IpData().Set_mu(resto_mu);
66  Jnlst().Printf(J_DETAILED, J_INITIALIZATION,
67  "Initial barrier parameter resto_mu = %e\n", resto_mu);
68 
70  // Initialize primal varialbes //
72 
73  // initialize the data structures in the restoration phase NLP
74  IpData().InitializeDataStructures(IpNLP(), false, false, false,
75  false, false);
76 
77  SmartPtr<Vector> new_x = IpData().curr()->x()->MakeNew();
78  SmartPtr<CompoundVector> Cnew_x =
79  static_cast<CompoundVector*> (GetRawPtr(new_x));
80  DBG_ASSERT(dynamic_cast<CompoundVector*> (GetRawPtr(new_x)));
81 
82  // Set the trial x variables from the original NLP
83  Cnew_x->GetCompNonConst(0)->Copy(*orig_ip_data->curr()->x());
84 
85  // Compute the initial values for the n and p variables for the
86  // equality constraints
87  Number rho = resto_ip_nlp->Rho();
88  DBG_PRINT((1,"rho = %e\n", rho));
89  SmartPtr<Vector> nc = Cnew_x->GetCompNonConst(1);
90  SmartPtr<Vector> pc = Cnew_x->GetCompNonConst(2);
91  SmartPtr<const Vector> cvec = orig_ip_cq->curr_c();
92  DBG_PRINT_VECTOR(2, "cvec", *cvec);
93  SmartPtr<Vector> a = nc->MakeNew();
94  SmartPtr<Vector> b = nc->MakeNew();
95  a->Set(resto_mu/(2.*rho));
96  a->Axpy(-0.5, *cvec);
97  b->Copy(*cvec);
98  b->Scal(resto_mu/(2.*rho));
99  DBG_PRINT_VECTOR(2, "a", *a);
100  DBG_PRINT_VECTOR(2, "b", *b);
101  solve_quadratic(*a, *b, *nc);
102  pc->Copy(*cvec);
103  pc->Axpy(1., *nc);
104  DBG_PRINT_VECTOR(2, "nc", *nc);
105  DBG_PRINT_VECTOR(2, "pc", *pc);
106 
107  // initial values for the n and p variables for the inequality
108  // constraints
109  SmartPtr<Vector> nd = Cnew_x->GetCompNonConst(3);
110  SmartPtr<Vector> pd = Cnew_x->GetCompNonConst(4);
111  cvec = orig_ip_cq->curr_d_minus_s();
112  a = nd->MakeNew();
113  b = nd->MakeNew();
114  a->Set(resto_mu/(2.*rho));
115  a->Axpy(-0.5, *cvec);
116  b->Copy(*cvec);
117  b->Scal(resto_mu/(2.*rho));
118  solve_quadratic(*a, *b, *nd);
119  pd->Copy(*cvec);
120  pd->Axpy(1., *nd);
121  DBG_PRINT_VECTOR(2, "nd", *nd);
122  DBG_PRINT_VECTOR(2, "pd", *pd);
123 
124  // Leave the slacks unchanged
125  SmartPtr<Vector> new_s = IpData().curr()->s()->MakeNew();
126  SmartPtr<CompoundVector> Cnew_s =
127  static_cast<CompoundVector*> (GetRawPtr(new_s));
128  DBG_ASSERT(dynamic_cast<CompoundVector*> (GetRawPtr(new_s)));
129  DBG_ASSERT(Cnew_s->NComps() == 1);
130 
131  // Set the trial s variables from the original NLP
132  Cnew_s->GetCompNonConst(0)->Copy(*orig_ip_data->curr()->s());
133 
134  // Now set the primal trial variables
135  DBG_PRINT_VECTOR(2,"new_s",*new_s);
136  DBG_PRINT_VECTOR(2,"new_x",*new_x);
137  SmartPtr<IteratesVector> trial = IpData().curr()->MakeNewContainer();
138  trial->Set_primal(*new_x, *new_s);
139  IpData().set_trial(trial);
140 
141  DBG_PRINT_VECTOR(2, "resto_c", *IpCq().trial_c());
142  DBG_PRINT_VECTOR(2, "resto_d_minus_s", *IpCq().trial_d_minus_s());
143 
145  // Initialize bound multipliers //
147 
148  SmartPtr<Vector> new_z_L = IpData().curr()->z_L()->MakeNew();
149  SmartPtr<CompoundVector> Cnew_z_L =
150  static_cast<CompoundVector*> (GetRawPtr(new_z_L));
151  DBG_ASSERT(dynamic_cast<CompoundVector*> (GetRawPtr(new_z_L)));
152  SmartPtr<Vector> new_z_U = IpData().curr()->z_U()->MakeNew();
153  SmartPtr<CompoundVector> Cnew_z_U =
154  static_cast<CompoundVector*> (GetRawPtr(new_z_U));
155  DBG_ASSERT(dynamic_cast<CompoundVector*> (GetRawPtr(new_z_U)));
156  DBG_ASSERT(Cnew_z_U->NComps() == 1);
157  SmartPtr<Vector> new_v_L = IpData().curr()->v_L()->MakeNew();
158  SmartPtr<CompoundVector> Cnew_v_L =
159  static_cast<CompoundVector*> (GetRawPtr(new_v_L));
160  DBG_ASSERT(dynamic_cast<CompoundVector*> (GetRawPtr(new_v_L)));
161  DBG_ASSERT(Cnew_v_L->NComps() == 1);
162  SmartPtr<Vector> new_v_U = IpData().curr()->v_U()->MakeNew();
163  SmartPtr<CompoundVector> Cnew_v_U =
164  static_cast<CompoundVector*> (GetRawPtr(new_v_U));
165  DBG_ASSERT(dynamic_cast<CompoundVector*> (GetRawPtr(new_v_U)));
166  DBG_ASSERT(Cnew_v_U->NComps() == 1);
167 
168  // multipliers for the original bounds are
169  SmartPtr<const Vector> orig_z_L = orig_ip_data->curr()->z_L();
170  SmartPtr<const Vector> orig_z_U = orig_ip_data->curr()->z_U();
171  SmartPtr<const Vector> orig_v_L = orig_ip_data->curr()->v_L();
172  SmartPtr<const Vector> orig_v_U = orig_ip_data->curr()->v_U();
173 
174  // Set the new multipliers to the min of the penalty parameter Rho
175  // and their current value
176  SmartPtr<Vector> Cnew_z_L0 = Cnew_z_L->GetCompNonConst(0);
177  Cnew_z_L0->Set(rho);
178  Cnew_z_L0->ElementWiseMin(*orig_z_L);
179  SmartPtr<Vector> Cnew_z_U0 = Cnew_z_U->GetCompNonConst(0);
180  Cnew_z_U0->Set(rho);
181  Cnew_z_U0->ElementWiseMin(*orig_z_U);
182  SmartPtr<Vector> Cnew_v_L0 = Cnew_v_L->GetCompNonConst(0);
183  Cnew_v_L0->Set(rho);
184  Cnew_v_L0->ElementWiseMin(*orig_v_L);
185  SmartPtr<Vector> Cnew_v_U0 = Cnew_v_U->GetCompNonConst(0);
186  Cnew_v_U0->Set(rho);
187  Cnew_v_U0->ElementWiseMin(*orig_v_U);
188 
189  // Set the multipliers for the p and n bounds to the "primal" multipliers
190  SmartPtr<Vector> Cnew_z_L1 = Cnew_z_L->GetCompNonConst(1);
191  Cnew_z_L1->Set(resto_mu);
192  Cnew_z_L1->ElementWiseDivide(*nc);
193  SmartPtr<Vector> Cnew_z_L2 = Cnew_z_L->GetCompNonConst(2);
194  Cnew_z_L2->Set(resto_mu);
195  Cnew_z_L2->ElementWiseDivide(*pc);
196  SmartPtr<Vector> Cnew_z_L3 = Cnew_z_L->GetCompNonConst(3);
197  Cnew_z_L3->Set(resto_mu);
198  Cnew_z_L3->ElementWiseDivide(*nd);
199  SmartPtr<Vector> Cnew_z_L4 = Cnew_z_L->GetCompNonConst(4);
200  Cnew_z_L4->Set(resto_mu);
201  Cnew_z_L4->ElementWiseDivide(*pd);
202 
203  // Set those initial values to be the trial values in Data
204  trial = IpData().trial()->MakeNewContainer();
205  trial->Set_bound_mult(*new_z_L, *new_z_U, *new_v_L, *new_v_U);
206  IpData().set_trial(trial);
207 
209  // Initialize equality constraint multipliers //
211 
212  DefaultIterateInitializer::least_square_mults(
213  Jnlst(), IpNLP(), IpData(), IpCq(),
214  resto_eq_mult_calculator_, constr_mult_init_max_);
215 
216  // upgrade the trial to the current point
217  IpData().AcceptTrialPoint();
218 
219  DBG_PRINT_VECTOR(2, "y_c", *IpData().curr()->y_c());
220  DBG_PRINT_VECTOR(2, "y_d", *IpData().curr()->y_d());
221 
222  DBG_PRINT_VECTOR(2, "z_L", *IpData().curr()->z_L());
223  DBG_PRINT_VECTOR(2, "z_U", *IpData().curr()->z_U());
224  DBG_PRINT_VECTOR(2, "v_L", *IpData().curr()->v_L());
225  DBG_PRINT_VECTOR(2, "v_U", *IpData().curr()->v_U());
226 
227  return true;
