60 static void free_printimpl(void *data)
61 {
62  int i;
63  print_members *members = (print_members *) data;
64  if (data == NULL) return;
65  if (members->savedBuffers) {
66  for (i=0; i<MAXSAVEDBUFFERS; i++) {
67  if (members->savedBuffers[i]) {
68  free(members->savedBuffers[i]);
69  members->savedBuffers[i] = NULL;
70  }
71  }
72  free(members->savedBuffers);
73  }
74  if (members->buf != NULL) free(members->buf);
75  if (members->errorBuf != NULL) free(members->errorBuf);
76  if (members->savedCurSize != NULL) free(members->savedCurSize);
77  if (members->savedNfilled != NULL) free(members->savedNfilled);
78  free(members);
79 }

130 {

132  char *new_buf;
133  int new_size;
134  if (cursize == 0) {

136  if (new_buf == NULL) { return -1; }
137  new_buf[0]='\0';
138  cursize = INITIAL_BUFSIZE;
139  } else {
140  //fprintf(stderr,"increasing buffer from %d to %d \n",cursize,((int)(cursize * GROWTH_FACTOR)));
141  new_buf = (char*)malloc((new_size =(int) (cursize * GROWTH_FACTOR))*sizeof(char));
142  if (new_buf == NULL) { return -1; }
143  memcpy(new_buf,buf,cursize);
144  cursize = new_size;

146  if (buf) {

391 #if defined(__MINGW32__) || defined(_MSC_VER)
392  const char *fileOpenMode = "wt"; /* on Windows do translation so that \n becomes \r\n */
393 #else

395 #endif

397  FILE * file = NULL;
398  regex_t re_begin,re_end;
399  regmatch_t matches[3];
400  mmc_uint_t nlines = 6 /* We start at 6 because we write 6 lines before the first line */, modelicaLine = 0;
401  /* What we try to match: */
402  /*#modelicaLine [/path/to/a.mo:4:3-4:12]*/
403  /*#endModelicaLine*/
404  /* which on Windows is (N is the drive letter): */
405  /*#modelicaLine [N:/path/to/a.mo:4:3-4:12]*/
406  /*#endModelicaLine*/
407 #if defined(__MINGW32__) || defined(_MSC_VER)
408  const char *re_str[2] = {"^ */[*]#modelicaLine .(.:/[^:]*):([0-9]*):[0-9]*-[0-9]*:[0-9]*.[*]/$", "^ */[*]#endModelicaLine[*]/$"};
409 #else /* real OSes */
410  const char *re_str[2] = {"^ */[*]#modelicaLine .([^:]*):([0-9]*):[0-9]*-[0-9]*:[0-9]*.[*]/$", "^ */[*]#endModelicaLine[*]/$"};
411 #endif
412  char *modelicaFileName = NULL;
413  char* strtmp = NULL;
414 
415  if (str!=NULL && nfilled!=0) {
416  str[nfilled] = '\0';
417 
418  /* First, compile the regular expressions */
419  if (regcomp(&re_begin, re_str[0], REG_EXTENDED) || regcomp(&re_end, re_str[1], 0)) {
420  c_add_message(NULL,21, /* WRITING_FILE_ERROR */
421  ErrorType_scripting,
422  ErrorLevel_error,
423  gettext("Error compiling regular expression: %s or %s."),
424  re_str,
425  2);
426  return 1;
427  }
428  }
429 
430  /* check if we have something to write */
431  /* open the file */
432  /* adrpo: 2010-09-22 open the file in BINARY mode as otherwise \r\n becomes \r\r\n! */
433 #if defined(__APPLE_CC__)||defined(__MINGW32__)||defined(__MINGW64__)
434  unlink(filename);
435 #endif
436  file = fopen(filename,fileOpenMode);
437  if (file == NULL) {
438  const char *c_tokens[1]={filename};
439  c_add_message(NULL,21, /* WRITING_FILE_ERROR */
440  ErrorType_scripting,
441  ErrorLevel_error,
442  gettext("Error writing to file %s."),
443  c_tokens,
444  1);
445  regfree(&re_begin);
446  regfree(&re_end);
447  return 1;
448  }
449  if (str == NULL || str[0]=='\0') {
450  /* nothing to write to file, just close it and return ! */
451  fclose(file);
452  return 1;
453  }
454 #if defined(__MINGW32__) || defined(_MSC_VER)
455  /* on Windows change the backslashes to forward slashes */
456  strtmp = _replace(filename, "\\", "/");
457 #endif
458  fprintf(file,"#ifdef OMC_BASE_FILE\n"
459  " #define OMC_FILE OMC_BASE_FILE\n"
460  "#else\n"
461  " #define OMC_FILE \"%s\"\n"
462  "#endif\n",
463 #if defined(__MINGW32__) || defined(_MSC_VER)
464  strtmp);
465  GC_free(strtmp);
466 #else
467  filename);
468 #endif
469  do {
470  next = strchr(str,'\n');
471  if (!next) {
472  fprintf(file,"%s",str);
473  break;
474  }
475  *next++ = '\0';
476  if (0==regexec(&re_begin, str, 3, matches, 0)) {
477  str[matches[1].rm_eo] = '\0';
478  str[matches[2].rm_eo] = '\0';
479  modelicaFileName = str + matches[1].rm_so;
480  modelicaLine = strtol(str + matches[2].rm_so, NULL, 10);
481 #if defined(__MINGW32__) || defined(_MSC_VER)
482  /* on Windows change the backslashes to forward slashes */
483  modelicaFileName = _replace(modelicaFileName, "\\", "/");
484 #endif
485  } else if (0==regexec(&re_end, str, 3, matches, 0)) {
486  if (modelicaFileName) { /* There is sometimes #endModlicaLine without a matching #modelicaLine */
487 #if defined(__MINGW32__) || defined(_MSC_VER)
488  GC_free(modelicaFileName);
489 #endif
490  modelicaFileName = NULL;
491  fprintf(file,"#line %ld OMC_FILE\n", nlines++);
492  }
493  } else if (modelicaFileName) {
494  fprintf(file,"#line %ld \"%s\"\n", modelicaLine, modelicaFileName);
495  fprintf(file,"%s\n", str);
496  nlines+=2;
497  } else {
498  fprintf(file,"%s\n", str);
499  nlines++;
500  }
501  str = next;
502  } while (1);
503 #if defined(__MINGW32__) || defined(_MSC_VER)

505  GC_free(modelicaFileName);
506  }
507 #endif
508  /* We do destructive updates on the print buffer; hide our tracks */
509  *buf = 0;
510  nfilled = 0;
511  regfree(&re_begin);
512  regfree(&re_end);
