45 /*
46  * -----------------------------------------------------------------
47  * Function : IDASptfqmr
48  * -----------------------------------------------------------------
49  * A call to the IDASptfqmr function links the main integrator with
50  * the IDASPTFQMR linear solver module. Its parameters are as
51  * follows:
52  *
53  * IDA_mem is the pointer to memory block returned by IDACreate.
54  *
55  * maxl is the maximum Krylov subspace dimension, an
56  * optional input. Pass 0 to use the default value.
57  * Otherwise pass a positive integer.
58  *
59  * The return values of IDASptfqmr are:
60  * IDASPILS_SUCCESS if successful
61  * IDASPILS_MEM_NULL if the IDAS memory was NULL
62  * IDASPILS_MEM_FAIL if there was a memory allocation failure
63  * IDASPILS_ILL_INPUT if there was illegal input.
64  * The above constants are defined in idas_spils.h
65  *
66  * -----------------------------------------------------------------
67  */
68
