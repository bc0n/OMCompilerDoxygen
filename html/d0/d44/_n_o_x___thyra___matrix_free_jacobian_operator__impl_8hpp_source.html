198 template<typename Scalar>
199 void
200 MatrixFreeJacobianOperator<Scalar>::applyImpl(const ::Thyra::EOpTransp M_trans,
201  const ::Thyra::MultiVectorBase< Scalar > &thyra_mv_y,
202  const Teuchos::Ptr< ::Thyra::MultiVectorBase< Scalar > > &thyra_mv_u,
203  const Scalar alpha,
204  const Scalar beta) const
205 {
206  TEUCHOS_ASSERT(setup_called_);
207 
208  // Use a directional derivative to approximate u = Jy
209 
210  /*
211  * delta = scalar perturbation
212  * x = solution vector used to evaluate f
213  * f = function evaluation (RHS)
214  * y = vector that J is applied to
215  *
216  * f(x + delta * y) - f(x)
217  * u = Jy = -----------------------
218  * delta
219  */
220  TEUCHOS_ASSERT(thyra_mv_y.domain()->dim() == 1);
221  TEUCHOS_ASSERT(thyra_mv_u->domain()->dim() == 1);
222  const Teuchos::RCP<const ::Thyra::VectorBase<Scalar> > y_ptr = thyra_mv_y.col(0);
223  const Teuchos::RCP< ::Thyra::VectorBase<Scalar> > u_ptr = thyra_mv_u->col(0);
224  const ::Thyra::VectorBase<Scalar>& y = *y_ptr;
225  ::Thyra::VectorBase<Scalar>& u = *u_ptr;
226  typename Teuchos::ScalarTraits<Scalar>::magnitudeType norm_2_x = ::Thyra::norm(*x_base_);
227  typename Teuchos::ScalarTraits<Scalar>::magnitudeType norm_2_y = ::Thyra::norm(y);
228 
229  // Make sure the y-norm is not zero, otherwise we can get an inf perturbation from divide by zero
230  if (norm_2_y == 0.0) {
231  norm_2_y = 1.0;
232  ::Thyra::assign(Teuchos::ptrFromRef(u),0.0);
233  return;
234  }
235 
236  if (perturbation_type_ == SalingerLOCA) {
237  delta_ = lambda_ * (lambda_ + norm_2_x / norm_2_y);
238  }
239  else if (perturbation_type_ == KelleySalingerPawlowski) {
240  Scalar inner_prod_x_y = ::Thyra::inner(*x_base_,y);
241 
242  if (inner_prod_x_y==Teuchos::ScalarTraits<Scalar>::zero())
243  inner_prod_x_y = 1.0e-12;
244 
245  delta_ = lambda_ * (1.0e-12 / lambda_ + std::fabs(inner_prod_x_y) / (norm_2_y * norm_2_y) ) * inner_prod_x_y / std::fabs(inner_prod_x_y);
246  }
247  else if (perturbation_type_ == KnollKeyes) {
248  delta_ = lambda_ * ::Thyra::norm_1(*x_base_) / (Teuchos::as<double>(x_base_->space()->dim()) * norm_2_y) + lambda_;
249  }
250  else {
251  delta_ = user_defined_delta_;
252  }
253 
254  // perturbed solution: x_p = delta * y + x_0
255  ::Thyra::V_StVpV(x_perturb_.ptr(),delta_,y,*x_base_);
256 
257  if (is_null(in_args_)) {
258  in_args_ = Teuchos::rcp(new ::Thyra::ModelEvaluatorBase::InArgs<Scalar>(model_->createInArgs()));
259  in_args_->setArgs(model_->getNominalValues());
260  }
261 
262  in_args_->set_x(x_perturb_);
263 
264  if (is_null(out_args_))
265  out_args_ = Teuchos::rcp(new ::Thyra::ModelEvaluatorBase::OutArgs<Scalar>(model_->createOutArgs()));
266 
267  out_args_->set_f( ::Thyra::ModelEvaluatorBase::Evaluation< ::Thyra::VectorBase<Scalar> >(f_perturb_, ::Thyra::ModelEvaluatorBase::EVAL_TYPE_APPROX_DERIV) );
268 
269  // f_p = f(delta * y + x)
270  model_->evalModel(*in_args_,*out_args_);
271 
272  // to be safe, remove arguments
273  in_args_->set_x(Teuchos::null);
274  out_args_->set_f(Teuchos::null);
275 
276  Scalar inv_delta = Teuchos::ScalarTraits<Scalar>::one() / delta_;
277 
278  if ( difference_type_ == Centered ) {
279  // perturbed solution: x_p = -delta * y + x_0
280  ::Thyra::V_StVpV(x_perturb_.ptr(),-delta_,y,*x_base_);
281 
282  out_args_->set_f(f2_perturb_);
283 
284  // f_p2 = f(-delta * y + x)
285  model_->evalModel(*in_args_,*out_args_);
286 
287  // to be safe, remove arguments
288  in_args_->set_x(Teuchos::null);
289  out_args_->set_f(Teuchos::null);
290 
291  ::Thyra::V_StVpStV(ptrFromRef(u),inv_delta,*f_perturb_,-inv_delta,*f2_perturb_);
292  }
293  else {
294  ::Thyra::V_StVpStV(ptrFromRef(u),inv_delta,*f_perturb_,-inv_delta,*f_base_);
295  }
296 
297 }
