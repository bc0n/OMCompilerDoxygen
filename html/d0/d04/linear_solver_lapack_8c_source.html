105 int getAnalyticalJacobianLapack(DATA* data, threadData_t *threadData, double* jac, int sysNumber)
106 {
107  int i,j,k,l,ii,currentSys = sysNumber;
108  LINEAR_SYSTEM_DATA* systemData = &(((DATA*)data)->simulationInfo->linearSystemData[currentSys]);
109 
110  const int index = systemData->jacobianIndex;
111  ANALYTIC_JACOBIAN* jacobian = &(data->simulationInfo->analyticJacobians[systemData->jacobianIndex]);
112 
113  memset(jac, 0, (systemData->size)*(systemData->size)*sizeof(double));
114 
115  for(i=0; i < jacobian->sparsePattern.maxColors; i++)
116  {
117  /* activate seed variable for the corresponding color */
118  for(ii=0; ii < jacobian->sizeCols; ii++)
119  if(jacobian->sparsePattern.colorCols[ii]-1 == i)
120  jacobian->seedVars[ii] = 1;
121 
122  ((systemData->analyticalJacobianColumn))(data, threadData, jacobian, systemData->parentJacobian);
123 
124  for(j = 0; j < jacobian->sizeCols; j++)
125  {
126  if(jacobian->seedVars[j] == 1)
127  {
128  ii = jacobian->sparsePattern.leadindex[j];
129  while(ii < jacobian->sparsePattern.leadindex[j+1])
130  {
131  l = jacobian->sparsePattern.index[ii];
132  k = j*jacobian->sizeRows + l;
133  jac[k] = -jacobian->resultVars[l];
134  ii++;
135  };
136  }
137  /* de-activate seed variable for the corresponding color */
138  if(jacobian->sparsePattern.colorCols[j]-1 == i)
139  jacobian->seedVars[j] = 0;
140  }
141  }
142 
143  return 0;
144 }

164 int solveLapack(DATA *data, threadData_t *threadData, int sysNumber, double* aux_x)
165 {
166  void *dataAndThreadData[2] = {data, threadData};
167  int i, iflag = 1;
168  LINEAR_SYSTEM_DATA* systemData = &(data->simulationInfo->linearSystemData[sysNumber]);
169  DATA_LAPACK* solverData = (DATA_LAPACK*)systemData->solverData[0];
170 
171  int success = 1;
172 
173  /* We are given the number of the linear system.
174  * We want to look it up among all equations. */
175  int eqSystemNumber = systemData->equationIndex;
176  int indexes[2] = {1,eqSystemNumber};
177  _omc_scalar residualNorm = 0;
178  double tmpJacEvalTime;
179  int reuseMatrixJac = (data->simulationInfo->currentContext == CONTEXT_SYM_JACOBIAN && data->simulationInfo->currentJacobianEval > 0);
180 
181  infoStreamPrintWithEquationIndexes(LOG_LS, 0, indexes, "Start solving Linear System %d (size %d) at time %g with Lapack Solver",
182  eqSystemNumber, (int) systemData->size,
183  data->localData[0]->timeValue);
184 
185 
186  /* set data */
187  _omc_setVectorData(solverData->x, aux_x);
188  _omc_setVectorData(solverData->b, systemData->b);
189  _omc_setMatrixData(solverData->A, systemData->A);
190 
191  rt_ext_tp_tick(&(solverData->timeClock));
192  if (0 == systemData->method) {
193 
194  if (!reuseMatrixJac){
195  /* reset matrix A */
196  memset(systemData->A, 0, (systemData->size)*(systemData->size)*sizeof(double));
197 
198  /* update matrix A */
199  systemData->setA(data, threadData, systemData);
200  }
201 
202  /* update vector b (rhs) */
203  systemData->setb(data, threadData, systemData);
204  } else {
205 
206  if (!reuseMatrixJac){
207  /* calculate jacobian -> matrix A*/
208  if(systemData->jacobianIndex != -1){
209  getAnalyticalJacobianLapack(data, threadData, solverData->A->data, sysNumber);
210  } else {
211  assertStreamPrint(threadData, 1, "jacobian function pointer is invalid" );
212  }
213  }
214 
215  /* calculate vector b (rhs) */
216  _omc_copyVector(solverData->work, solverData->x);
217  wrapper_fvec_lapack(solverData->work, solverData->b, &iflag, dataAndThreadData, sysNumber);
218  }
219  tmpJacEvalTime = rt_ext_tp_tock(&(solverData->timeClock));
220  systemData->jacobianTime += tmpJacEvalTime;
221  infoStreamPrint(LOG_LS_V, 0, "### %f time to set Matrix A and vector b.", tmpJacEvalTime);
222 
223  /* Log A*x=b */
224  if(ACTIVE_STREAM(LOG_LS_V)){
225  _omc_printVector(solverData->x, "Vector old x", LOG_LS_V);
226  _omc_printMatrix(solverData->A, "Matrix A", LOG_LS_V);
227  _omc_printVector(solverData->b, "Vector b", LOG_LS_V);
228  }
229 
230  rt_ext_tp_tick(&(solverData->timeClock));
231 
232  /* if reuseMatrixJac use also previous factorization */
233  if (!reuseMatrixJac)
234  {
235  /* Solve system */
236  dgesv_((int*) &systemData->size,
237  (int*) &solverData->nrhs,
238  solverData->A->data,
239  (int*) &systemData->size,
240  solverData->ipiv,
241  solverData->b->data,
242  (int*) &systemData->size,
243  &solverData->info);
244 
245  } /* further Jacobian evaluations */
246  else
247  {
248  char trans = 'N';
249  /* Solve system */
250  dgetrs_(&trans,
251  (int*) &systemData->size,
252  (int*) &solverData->nrhs,
253  solverData->A->data,
254  (int*) &systemData->size,
255  solverData->ipiv,
256  solverData->b->data,
257  (int*) &systemData->size,
258  &solverData->info);
259  }
260 
261 
262  infoStreamPrint(LOG_LS_V, 0, "Solve System: %f", rt_ext_tp_tock(&(solverData->timeClock)));
263 
264  if(solverData->info < 0)
265  {
266  warningStreamPrint(LOG_LS, 0, "Error solving linear system of equations (no. %d) at time %f. Argument %d illegal.", (int)systemData->equationIndex, data->localData[0]->timeValue, (int)solverData->info);
267  success = 0;
268  }
269  else if(solverData->info > 0)
270  {
271  warningStreamPrint(LOG_LS, 0,
272  "Failed to solve linear system of equations (no. %d) at time %f, system is singular for U[%d, %d].",
273  (int)systemData->equationIndex, data->localData[0]->timeValue, (int)solverData->info+1, (int)solverData->info+1);
274 
275  success = 0;
276 
277  /* debug output */
278  if (ACTIVE_STREAM(LOG_LS)){
279  _omc_printMatrix(solverData->A, "Matrix U", LOG_LS);
280 
281  _omc_printVector(solverData->b, "Output vector x", LOG_LS);
282  }
283  }
284 
285  if (1 == success){
286 
287  if (1 == systemData->method){
288  /* take the solution */
289  solverData->x = _omc_addVectorVector(solverData->x, solverData->work, solverData->b); // x = xold(work) + xnew(b)
290 
291  /* update inner equations */
292  wrapper_fvec_lapack(solverData->x, solverData->work, &iflag, dataAndThreadData, sysNumber);
293  residualNorm = _omc_euclideanVectorNorm(solverData->work);
294 
295  if ((isnan(residualNorm)) || (residualNorm>1e-4)){
296  warningStreamPrint(LOG_LS, 0,
297  "Failed to solve linear system of equations (no. %d) at time %f. Residual norm is %.15g.",
298  (int)systemData->equationIndex, data->localData[0]->timeValue, residualNorm);
299  success = 0;
300  }
301  } else {
302  /* take the solution */
303  _omc_copyVector(solverData->x, solverData->b);
304  }
305 
306  if (ACTIVE_STREAM(LOG_LS_V)){
307  infoStreamPrint(LOG_LS_V, 1, "Residual Norm %.15g of solution x:", residualNorm);
308  infoStreamPrint(LOG_LS_V, 0, "System %d numVars %d.", eqSystemNumber, modelInfoGetEquation(&data->modelData->modelDataXml,eqSystemNumber).numVar);
309 
310  for(i = 0; i < systemData->size; ++i) {
311  infoStreamPrint(LOG_LS_V, 0, "[%d] %s = %.15g", i+1, modelInfoGetEquation(&data->modelData->modelDataXml,eqSystemNumber).vars[i], aux_x[i]);
312  }
313 
314  messageClose(LOG_LS_V);
315  }
316  }
317 
318  return success;
319 }
