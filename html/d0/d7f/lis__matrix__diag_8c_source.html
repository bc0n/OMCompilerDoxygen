93 LIS_INT lis_matrix_diag_create(LIS_INT local_n, LIS_INT global_n, LIS_Comm comm, LIS_MATRIX_DIAG *D)
94 {
95  LIS_INT nprocs,my_rank;
96  LIS_INT is,ie;
97  LIS_INT *ranges;
98  #ifdef USE_MPI
99  LIS_INT i;
100  #endif
101 
102  LIS_DEBUG_FUNC_IN;
103 
104  *D = NULL;
105 
106  if( global_n>0 && local_n>global_n )
107  {
108  LIS_SETERR2(LIS_ERR_ILL_ARG,"local n(=%d) is larger than global n(=%d)\n",local_n,global_n);
109  return LIS_ERR_ILL_ARG;
110  }
111  if( local_n<0 || global_n<0 )
112  {
113  LIS_SETERR2(LIS_ERR_ILL_ARG,"local n(=%d) or global n(=%d) are less than 0\n",local_n,global_n);
114  return LIS_ERR_ILL_ARG;
115  }
116  if( local_n==0 && global_n==0 )
117  {
118  LIS_SETERR2(LIS_ERR_ILL_ARG,"local n(=%d) and global n(=%d) are 0\n",local_n,global_n);
119  return LIS_ERR_ILL_ARG;
120  }
121 
122  *D = (LIS_MATRIX_DIAG)lis_malloc( sizeof(struct LIS_MATRIX_DIAG_STRUCT),"lis_matrix_diag_create::D" );
123  if( NULL==*D )
124  {
125  LIS_SETERR_MEM(sizeof(struct LIS_MATRIX_DIAG_STRUCT));
126  return LIS_OUT_OF_MEMORY;
127  }
128  lis_matrix_diag_init(D);
129 
130  #ifdef USE_MPI
131  MPI_Comm_size(comm,&nprocs);
132  MPI_Comm_rank(comm,&my_rank);
133  ranges = (LIS_INT *)lis_malloc( (nprocs+1)*sizeof(LIS_INT),"lis_matrix_diag_create::ranges" );
134  if( ranges==NULL )
135  {
136  LIS_SETERR_MEM((nprocs+1)*sizeof(LIS_INT));
137  lis_matrix_diag_destroy(*D);
138  *D = NULL;
139  return LIS_OUT_OF_MEMORY;
140  }
141  #else
142  nprocs = 1;
143  my_rank = 0;
144  ranges = NULL;
145  #endif
146 
147  #ifdef USE_MPI
148  MPI_Allreduce(&local_n,&i,1,LIS_MPI_INT,MPI_SUM,comm);
149  if( i==0 )
150  #else
151  if( local_n==0 )
152  #endif
153  {
154  #ifdef USE_MPI
155  LIS_GET_ISIE(my_rank,nprocs,global_n,is,ie);
156  local_n = ie-is;
157  MPI_Allgather(&ie,1,LIS_MPI_INT,&ranges[1],1,LIS_MPI_INT,comm);
158  ranges[0] = 0;
159  #else
160  local_n = global_n;
161  is = 0;
162  ie = global_n;
163  #endif
164  }
165  else
166  {
167  #ifdef USE_MPI
168  MPI_Allgather(&local_n,1,LIS_MPI_INT,&ranges[1],1,LIS_MPI_INT,comm);
169  ranges[0] = 0;
170  for(i=0;i<nprocs;i++)
171  {
172  ranges[i+1] += ranges[i];
173  }
174  global_n = ranges[nprocs];
175  is = ranges[my_rank];
176  ie = ranges[my_rank+1];
177  #else
178  global_n = local_n;
179  is = 0;
180  ie = local_n;
181  #endif
182  }
183  (*D)->ranges = ranges;
184 
185  (*D)->value = (LIS_SCALAR *)lis_malloc( local_n*sizeof(LIS_SCALAR),"lis_matrix_diag_create::D->value" );
186  if( NULL==(*D)->value )
187  {
188  LIS_SETERR_MEM(local_n*sizeof(LIS_SCALAR));
189  lis_matrix_diag_destroy(*D);
190  *D = NULL;
191  return LIS_OUT_OF_MEMORY;
192  }
193 
194  (*D)->n = local_n;
195  (*D)->nr = local_n;
196  (*D)->gn = global_n;
197  (*D)->np = local_n;
198  (*D)->comm = comm;
199  (*D)->my_rank = my_rank;
200  (*D)->nprocs = nprocs;
201  (*D)->is = is;
202  (*D)->ie = ie;
203 
204  LIS_DEBUG_FUNC_OUT;
205  return LIS_SUCCESS;
206 }

254  *Dout = NULL;
255 
256  *Dout = (LIS_MATRIX_DIAG)lis_malloc( sizeof(struct LIS_MATRIX_DIAG_STRUCT),"lis_matrix_diag_duplicate::Dout" );

267  {

279  if( Din->bns==NULL )

284  LIS_SETERR_MEM(Din->bn*Din->bn*Din->nr*sizeof(LIS_SCALAR));
285  lis_matrix_diag_destroy(*Dout);
286  *Dout = NULL;
287  return LIS_OUT_OF_MEMORY;
288  }
289  (*Dout)->bn = Din->bn;
290  }
291  else
292  {
293  nr = Din->nr;
294  (*Dout)->bns = (LIS_INT *)lis_malloc( nr*sizeof(LIS_INT),"lis_matrix_diag_duplicate::Dout->bns" );
295  if( NULL==(*Dout)->bns )
296  {
297  LIS_SETERR_MEM(nr*sizeof(LIS_INT));
298  lis_matrix_diag_destroy(*Dout);
299  *Dout = NULL;
300  return LIS_OUT_OF_MEMORY;
301  }
302  (*Dout)->v_value = (LIS_SCALAR **)lis_malloc( nr*sizeof(LIS_SCALAR *),"lis_matrix_diag_duplicate::Dout->value" );
303  if( NULL==(*Dout)->v_value )
304  {
305  LIS_SETERR_MEM(nr*sizeof(LIS_SCALAR *));
306  lis_matrix_diag_destroy(*Dout);
307  *Dout = NULL;
308  return LIS_OUT_OF_MEMORY;
309  }
310  bnmax = 0;
311  for(i=0;i<nr;i++)
312  {
313  t = Din->bns[i];
314  bnmax = _max(bnmax,t);
315  (*Dout)->bns[i] = t;
316  (*Dout)->v_value[i] = (LIS_SCALAR *)malloc( t*t*sizeof(LIS_SCALAR));
317  }
318  (*Dout)->bn = bnmax;
319  (*Dout)->nr = nr;
320  }
321 
322  (*Dout)->n = Din->n;
323  (*Dout)->nr = Din->nr;
324  (*Dout)->gn = Din->gn;
325  (*Dout)->np = Din->np;
326  (*Dout)->comm = Din->comm;
327  (*Dout)->my_rank = Din->my_rank;
328  (*Dout)->nprocs = Din->nprocs;
329  (*Dout)->is = Din->is;
330  (*Dout)->ie = Din->ie;
331  (*Dout)->origin = Din->origin;
332 
333  LIS_DEBUG_FUNC_OUT;
334  return LIS_SUCCESS;
335 }

343  LIS_INT i,k,t,bnmax;
344  #ifdef USE_MPI
345  LIS_INT *ranges;
346  #endif
347 
348  LIS_DEBUG_FUNC_IN;
349 
350  err = lis_matrix_check(Ain,LIS_MATRIX_CHECK_ALL);
351  if( err ) return err;
352 
353  nprocs = Ain->nprocs;
354  *Dout = NULL;
355 
356  *Dout = (LIS_MATRIX_DIAG)lis_malloc( sizeof(struct LIS_MATRIX_DIAG_STRUCT),"lis_matrix_diag_duplicateM::Dout" );
357  if( NULL==*Dout )
358  {
359  LIS_SETERR_MEM(sizeof(struct LIS_MATRIX_DIAG_STRUCT));
360  return LIS_OUT_OF_MEMORY;
361  }
362  lis_matrix_diag_init(Dout);
363 
364  #ifdef USE_MPI
365  ranges = (LIS_INT *)lis_malloc( (nprocs+1)*sizeof(LIS_INT),"lis_matrix_diag_duplicateM::ranges" );
366  if( ranges==NULL )
367  {
368  LIS_SETERR_MEM((nprocs+1)*sizeof(LIS_INT));
369  lis_matrix_diag_destroy(*Dout);
370  *Dout = NULL;
371  return LIS_OUT_OF_MEMORY;
372  }
373  for(i=0;i<nprocs+1;i++) ranges[i] = Ain->ranges[i];
374  (*Dout)->ranges = ranges;
375  #else
376  (*Dout)->ranges = NULL;
377  #endif
378 
379  switch(Ain->matrix_type)
380  {
381  case LIS_MATRIX_BSR:
382  case LIS_MATRIX_BSC:
383  k = Ain->nr*Ain->bnr*Ain->bnc;
384  (*Dout)->value = (LIS_SCALAR *)lis_malloc( k*sizeof(LIS_SCALAR),"lis_matrix_diag_duplicateM::Dout->value" );

393  (*Dout)->nr = Ain->nr;
394  break;
395  case LIS_MATRIX_VBR:
396  nr = Ain->nr;
397  (*Dout)->bns = (LIS_INT *)lis_malloc( nr*sizeof(LIS_INT),"lis_matrix_diag_duplicateM::Dout->bns" );
398  if( NULL==(*Dout)->bns )
399  {

408  LIS_SETERR_MEM(nr*sizeof(LIS_SCALAR *));
409  lis_matrix_diag_destroy(*Dout);
410  *Dout = NULL;
411  return LIS_OUT_OF_MEMORY;
412  }
413  bnmax = 0;
414  for(i=0;i<nr;i++)
415  {
416  t = Ain->row[i+1] - Ain->row[i];
417  bnmax = _max(bnmax,t);
418  (*Dout)->bns[i] = t;
419  (*Dout)->v_value[i] = (LIS_SCALAR *)malloc( t*t*sizeof(LIS_SCALAR));
420  }
421  (*Dout)->bn = bnmax;
422  (*Dout)->nr = nr;
423  break;
424  default:

426  if( NULL==(*Dout)->value )
427  {
428  LIS_SETERR_MEM(Ain->np*sizeof(LIS_SCALAR));
429  lis_matrix_diag_destroy(*Dout);
430  *Dout = NULL;
431  return LIS_OUT_OF_MEMORY;
432  }
433  (*Dout)->nr = Ain->n;
434  break;
435  }
436 
437  (*Dout)->n = Ain->n;
438  (*Dout)->gn = Ain->gn;
439  (*Dout)->np = Ain->np;
440  (*Dout)->comm = Ain->comm;
441  (*Dout)->my_rank = Ain->my_rank;
442  (*Dout)->nprocs = Ain->nprocs;
443  (*Dout)->is = Ain->is;
444  (*Dout)->ie = Ain->ie;
445  (*Dout)->origin = Ain->origin;
446 
447  LIS_DEBUG_FUNC_OUT;
448  return LIS_SUCCESS;
449 }

593 #undef __FUNC__
594 #define __FUNC__ "lis_matrix_diag_copy"
595 LIS_INT lis_matrix_diag_copy(LIS_MATRIX_DIAG X, LIS_MATRIX_DIAG Y)
596 {
597  LIS_INT i,k,n,nr,bn;
598  LIS_SCALAR *x,*y;
599 
600  LIS_DEBUG_FUNC_IN;
601 
602  x = X->value;
603  y = Y->value;
604  n = X->n;
605  nr = X->nr;
606  if( n!=Y->n )
607  {
608  LIS_SETERR(LIS_ERR_ILL_ARG,"length of diagonal matrix X and Y is not equal\n");
609  return LIS_ERR_ILL_ARG;
610  }
611  if( X->bns )
612  {
613  #ifdef _OPENMP
614  #pragma omp parallel for private(i)
615  #endif
616  for(i=0;i<nr;i++)
617  {
618  bn = X->bns[i]*X->bns[i];
619  memcpy(Y->v_value[i],X->v_value[i],bn*sizeof(LIS_SCALAR));
620  }
621  }
