182 static int symbolic_initialization(DATA *data, threadData_t *threadData)
183 {
184  TRACE_PUSH
185  int retVal;
186  FILE *pFile = NULL;
187  long i;
188  MODEL_DATA *mData = data->modelData;
189  int homotopySupport = 0;
190  int solveWithGlobalHomotopy;
191  int adaptiveGlobal;
192  int kinsol = 0;
193 
194 #if !defined(OMC_MINIMAL_RUNTIME)
195  kinsol = (data->simulationInfo->nlsMethod == NLS_KINSOL);
196 #endif
197 
198 #if !defined(OMC_NUM_NONLINEAR_SYSTEMS) || OMC_NUM_NONLINEAR_SYSTEMS>0
199  for(i=0; i<mData->nNonLinearSystems; i++) {
200  if (data->simulationInfo->nonlinearSystemData[i].homotopySupport) {
201  homotopySupport = 1;
202  break;
203  }
204  }
205 #endif
206  adaptiveGlobal = data->callback->useHomotopy == 2;
207  solveWithGlobalHomotopy = homotopySupport
208  && ((data->callback->useHomotopy == 1 && init_lambda_steps > 1) || adaptiveGlobal);
209 
210 #if !defined(OMC_NDELAY_EXPRESSIONS) || OMC_NDELAY_EXPRESSIONS>0
211  /* initial sample and delay before initial the system */
212  initDelay(data, data->simulationInfo->startTime);
213 #endif
214  /* initialize all relations that are ZeroCrossings */
215  storePreValues(data);
216  overwriteOldSimulationData(data);
217 
218  /* If there is no homotopy in the model or local homotopy is activated
219  or homotopy is disabled by runtime flag '-ils=<lambda_steps>',
220  solve WITHOUT HOMOTOPY or LOCAL HOMOTOPY. */
221  if (!solveWithGlobalHomotopy){
222  data->simulationInfo->lambda = 1.0;
223  data->callback->functionInitialEquations(data, threadData);
224 
225  /* If there is homotopy in the model and global homotopy is activated
226  and homotopy on first try is deactivated,
227  TRY TO SOLVE WITHOUT HOMOTOPY FIRST.
228  TO-DO: For the adaptive global approach, provide a separate DAE with
229  the original unmanipulated systems for trying without homotopy */
230  } else if (!omc_flag[FLAG_HOMOTOPY_ON_FIRST_TRY]) {
231  /* try */
232 #ifndef OMC_EMCC
233  MMC_TRY_INTERNAL(simulationJumpBuffer)
234 #endif
235  if (adaptiveGlobal && kinsol) {
236  infoStreamPrint(LOG_INIT, 0, "Automatically set -homotopyOnFirstTry, because trying without homotopy first is not supported for the adaptive global approach in combination with KINSOL.");
237  } else {
238  if (adaptiveGlobal)
239  data->callback->useHomotopy = 1;
240  data->simulationInfo->lambda = 1.0;
241  infoStreamPrint(LOG_INIT, 0, "Try to solve the initialization problem without homotopy first.");
242  data->callback->functionInitialEquations(data, threadData);
243  solveWithGlobalHomotopy = 0;
244  }
245 
246  /* catch */
247 #ifndef OMC_EMCC
248  MMC_CATCH_INTERNAL(simulationJumpBuffer)
249 #endif
250  if (adaptiveGlobal)
251  data->callback->useHomotopy = 2;
252  if(solveWithGlobalHomotopy) {
253  if (!kinsol)
254  warningStreamPrint(LOG_ASSERT, 0, "Failed to solve the initialization problem without homotopy method. If homotopy is available the homotopy method is used now.");
255  omc_flag[FLAG_HOMOTOPY_ON_FIRST_TRY] = 1;
256  setAllParamsToStart(data);
257  setAllVarsToStart(data);
258  data->callback->updateBoundParameters(data, threadData);
259  data->callback->updateBoundVariableAttributes(data, threadData);
260  }
261  }
262 
263  /* If there is homotopy in the model and the equidistant global homotopy approach is activated
264  and solving without homotopy failed or is not wanted,
265  use EQUIDISTANT GLOBAL HOMOTOPY METHOD. */
266  if (data->callback->useHomotopy == 1 && solveWithGlobalHomotopy)
267  {
268  long step;
269  char buffer[4096];
270 
271  infoStreamPrint(LOG_INIT, 0, "Global homotopy with equidistant step size started.");
272 
273 #if !defined(OMC_NO_FILESYSTEM)
274  const char sep[] = ",";
275  if(ACTIVE_STREAM(LOG_INIT))
276  {
277  sprintf(buffer, "%s_equidistant_global_homotopy.csv", mData->modelFilePrefix);
278  infoStreamPrint(LOG_INIT, 0, "The homotopy path will be exported to %s.", buffer);
279  pFile = fopen(buffer, "wt");
280  fprintf(pFile, "\"sep=%s\"\n%s", sep, "\"lambda\"");
281  for(i=0; i<mData->nVariablesReal; ++i)
282  fprintf(pFile, "%s\"%s\"", sep, mData->realVarsData[i].info.name);
283  fprintf(pFile, "\n");
284  }
285 #endif
286 
287  infoStreamPrint(LOG_INIT, 1, "homotopy process\n---------------------------");
288  for(step=0; step<init_lambda_steps; ++step)
289  {
290  data->simulationInfo->lambda = ((double)step)/(init_lambda_steps-1);
291  infoStreamPrint(LOG_INIT, 0, "homotopy parameter lambda = %g", data->simulationInfo->lambda);
292 
293  if(data->simulationInfo->lambda > 1.0) {
294  data->simulationInfo->lambda = 1.0;
295  }
296 
297  if(0 == step)
298  data->callback->functionInitialEquations_lambda0(data, threadData);
299  else
300  data->callback->functionInitialEquations(data, threadData);
301 
302  infoStreamPrint(LOG_INIT, 0, "homotopy parameter lambda = %g done\n---------------------------", data->simulationInfo->lambda);
303 
304 #if !defined(OMC_NO_FILESYSTEM)
305  if(ACTIVE_STREAM(LOG_INIT))
306  {
307  fprintf(pFile, "%.16g", data->simulationInfo->lambda);
308  for(i=0; i<mData->nVariablesReal; ++i)
309  fprintf(pFile, "%s%.16g", sep, data->localData[0]->realVars[i]);
310  fprintf(pFile, "\n");
311  }
312 #endif
313  }
314  data->simulationInfo->homotopySteps += init_lambda_steps;
315  messageClose(LOG_INIT);
316 
317 #if !defined(OMC_NO_FILESYSTEM)
318  if(ACTIVE_STREAM(LOG_INIT))
319  fclose(pFile);
320 #endif
321  }
322 
323  /* If there is homotopy in the model and the adaptive global homotopy approach is activated
324  and solving without homotopy failed or is not wanted,
325  use ADAPTIVE GLOBAL HOMOTOPY APPROACH. */
326  if (adaptiveGlobal && solveWithGlobalHomotopy)
327  {
328  infoStreamPrint(LOG_INIT, 0, "Global homotopy with adaptive step size started.");
329  infoStreamPrint(LOG_INIT, 1, "homotopy process\n---------------------------");
330 
331  // Solve lambda0-DAE
332  data->simulationInfo->lambda = 0;
333  infoStreamPrint(LOG_INIT, 0, "solve simplified lambda0-DAE");
334  data->callback->functionInitialEquations_lambda0(data, threadData);
335  infoStreamPrint(LOG_INIT, 0, "solving simplified lambda0-DAE done\n---------------------------");
336 
337  // Run along the homotopy path and solve the actual system
338  data->callback->functionInitialEquations(data, threadData);
339 
340  messageClose(LOG_INIT);
341  }
342 
343  storeRelations(data);
344 
345  /* check for over-determined systems */
346  retVal = data->callback->functionRemovedInitialEquations(data, threadData);
347 
348  TRACE_POP
349  return retVal;
350 }

362 static char *mapToDymolaVars(const char *varname)
363 {
364  unsigned int varnameSize = strlen(varname);
365  unsigned int level = 0;
366  unsigned int i=0, j=0, pos=0;
367  char* newVarname = NULL;
368  unsigned int newVarnameSize = 0;
369 
370  newVarnameSize = varnameSize;
371  for(i=0; i<varnameSize; i++)
372  {
373  if(varname[i] == '[')
374  level++;
375  else if(varname[i] == ']')
376  level--;
377 
378  if(level > 0 && varname[i] == ',' && varname[i+1] != ' ')
379  newVarnameSize++;
380  }
381 
382  newVarname = (char*)malloc((newVarnameSize+1) * sizeof(char));
383  for(i=0,j=0; i<newVarnameSize; i++,j++)
384  {
385  if(varname[j] == '[')
386  level++;
387  else if(varname[j] == ']')
388  level--;
389 
390  newVarname[i] = varname[j];
391  if(level > 0 && varname[j] == ',' && varname[j+1] != ' ')
392  {
393  i++;
394  newVarname[i] = ' ';
395  }
396  }
397  newVarname[newVarnameSize] = '\0';
398 
399  while(!memcmp((const void*)newVarname, (const void*)"der(", 4*sizeof(char)))
400  {
401  for(pos=newVarnameSize; pos>=4; pos--)
402  if(newVarname[pos] == '.')
403  break;
404 
405  if(pos == 3)
406  break;
407 
408  memcpy((void*)newVarname, (const void*)(newVarname+4), (pos-3)*sizeof(char));
409  memcpy((void*)(newVarname+pos-3), (const void*)"der(", 4*sizeof(char));
410  }
411 
412  return newVarname;
413 }

560 void initSample(DATA* data, threadData_t *threadData, double startTime, double stopTime)
561 {
562  TRACE_PUSH
563  long i;
564 
565  data->callback->function_initSample(data, threadData); /* set-up sample */
566  data->simulationInfo->nextSampleEvent = NAN; /* should never be reached */
567  for(i=0; i<data->modelData->nSamples; ++i) {
568  if(startTime < data->modelData->samplesInfo[i].start) {
569  data->simulationInfo->nextSampleTimes[i] = data->modelData->samplesInfo[i].start;
570  } else {
571  data->simulationInfo->nextSampleTimes[i] = data->modelData->samplesInfo[i].start + ceil((startTime-data->modelData->samplesInfo[i].start) / data->modelData->samplesInfo[i].interval) * data->modelData->samplesInfo[i].interval;
572  }
573 
574  if((i == 0) || (data->simulationInfo->nextSampleTimes[i] < data->simulationInfo->nextSampleEvent)) {
575  data->simulationInfo->nextSampleEvent = data->simulationInfo->nextSampleTimes[i];
576  }
577  }
578 
579  if(stopTime < data->simulationInfo->nextSampleEvent) {
580  debugStreamPrint(LOG_EVENTS, 0, "there are no sample-events");
581  } else {
582  debugStreamPrint(LOG_EVENTS, 0, "first sample-event at t = %g", data->simulationInfo->nextSampleEvent);
583  }
584 
585  TRACE_POP
