444 static int ML_LocalReorder_with_METIS( int Nrows, int xadj[], int adjncy[] ,
445  int Nparts, indextype part[], int level,
446  ML_Comm *comm )
447 {
448 
449  int i, j, k, count;
450  int row_coarsest, col_coarsest, col;
451  int MaxNnzRow;
452  indextype * xadj2 = NULL, * adjncy2 = NULL;
453  indextype * perm = NULL, * iperm = NULL;
454  indextype options[8];
455  size_t used_mem;
456  int bandwidth_orig, bandwidth_perm;
457  int mypid = comm->ML_mypid;
458  double t0;
459 #if defined(HAVE_ML_METIS) && (METIS_VER_MAJOR < 5)
460  int nbytes, nbytes_max;
461 #endif
462 #ifdef METIS_DEBUG
463  FILE *fp;
464  char filename[80];
465 #endif
466 
467  /* ------------------- execution begins --------------------------------- */
468 
469  t0 = GetClock();
470 
471  if ( mypid == 0 && ML_Get_PrintLevel() > 7 ) {
472 
473  printf("Entering METIS reordering (level %d)\n",
474  level );
475 
476  }
477 
478  /* FIXME: DELELE ME?
479  * if( Nparts < 3 ) return Nparts; do nothing if 2 nodes only... */
480 
481  /* ********************************************************************** */
482  /* In order to compute the fill-in reducin ordering, I need to form the */
483  /* graph of the matrix on the next level. This matrix is still not formed */
484  /* so I have to construct it. It is a LOCAL matrix, and numerical entries */
485  /* are not relevant. I need some memory for this. As I don't know exactly */
486  /* the number of nonzeros in the coarsert matrix, I use the one of the */
487  /* finest level (cannot be more) */
488  /* ********************************************************************** */
489 
490  MaxNnzRow = 0;
491  for (i = 0 ; i < Nrows ; i++) {
492  if ((xadj[i+1] - xadj[i])> MaxNnzRow)
493  MaxNnzRow = xadj[i+1] - xadj[i];
494  }
495 
496  MaxNnzRow *= MaxNnzRow;
497 
498  xadj2 = (indextype *)ML_allocate(sizeof(indextype) * (Nparts+1));
499  adjncy2 = (indextype *)ML_allocate(sizeof(indextype) * MaxNnzRow * Nparts);
500 
501  if((xadj2 == NULL) || (adjncy2 == NULL)) {
502  fprintf( stderr,
503  "*ML*ERR* not nough memory to allocated %d and %d bytex\n"
504  "*ML*ERR* (file %s, line %d)\n",
505  (int)sizeof(indextype) * Nparts,
506  (int)sizeof(indextype) * MaxNnzRow,
507  __FILE__,
508  __LINE__ );
509  exit( EXIT_FAILURE );
510  }
511 
512  /* ********************************************************************** */
513  /* I cycle over the rows of the finest level. For each nonzero (an edge */
514  /* for the METIS format), I look for the corresponding aggregate for form */
515  /* the local row, added to the coarset level matrix. This matrix will be */
516  /* passed to the METIS function to compute orderings, and the new ordering*/
517  /* will be used to enumerate the local aggregates. */
518  /* ********************************************************************** */
519 
520  for (i = 0 ; i < Nparts + 1 ; i++)
521  xadj2[i] = 0;
522  for (i = 0 ; i < Nparts * MaxNnzRow ; i++)
523  adjncy2[i] = -1;
524 
525  /* cycle over the rows */
526 
527  for (i = 0 ; i < Nrows ; i++) {
528 
529  row_coarsest = part[i];
530 
531  /* cycle over the columns of the finest level */
532  for (j = xadj[i] ; j < xadj[i+1] ; j++) {
533 
534  col = adjncy[j];
535 
536  col_coarsest = part[col];
537 
538  /* discard element diagonal */
539  if (col_coarsest != row_coarsest) {
540 
541  for (k= 0 ; k < xadj2[row_coarsest+1] ; k++) {
542 
543  if (adjncy2[row_coarsest*MaxNnzRow+k] == col_coarsest) {
544  k = -1;
545  break;
546  }
547  if (adjncy2[row_coarsest*MaxNnzRow+k] == -1) break;
548 
549  }
550  if (k == -1) continue;
551  else if (k == MaxNnzRow) {
552  fprintf( stderr,
553  "*ML*ERR* something went wrong: k=%d, MaxNnzRow=%d\n"
554  "*ML*ERR* (file %s, line %d)\n",
555  k, MaxNnzRow,
556  __FILE__,
557  __LINE__ ) ;
558  exit( EXIT_FAILURE );
559  }
560  adjncy2[row_coarsest*MaxNnzRow+k] = col_coarsest;
561  xadj2[row_coarsest+1]++;
562  }
563 
564  }
565 
566  }
567 
568  /* compress the crs format */
569 
570  for( i=1 ; i<Nparts+1 ; i++ ) xadj2[i] += xadj2[i-1];
571 
572  /* count is the position of the first available zero position */
573 
574  count = 0;
575 
576  while( adjncy2[count] != -1 && count < Nparts*MaxNnzRow ) {
577  count++;
578  }
579 
580  for( i=count+1 ; i<Nparts*MaxNnzRow ; i++ ) {
581  /* find the next nonzero position */
582  if( adjncy2[i] == -1 ) continue;
583  adjncy2[count] = adjncy2[i];
584  adjncy2[i] = -1;
585 
586  while( adjncy2[count] != -1 ) {
587  count++;
588  }
589  }
590 
591  used_mem = sizeof(indextype) * xadj2[Nparts];
592 
593  if( count != xadj2[Nparts] ) {
594  fprintf( stderr,
595  "*ML*ERR* something went wrong: count=%d, xadj2[Nparts]=%d\n"
596  "*ML*ERR* (file %s line %d)\n",
597  count,
598  xadj2[Nparts],
599  __FILE__,
600  __LINE__ ) ;
601  exit( EXIT_FAILURE );
602  }
603 
604  adjncy2 = (indextype *) realloc( adjncy2, used_mem );
605 
606  perm = (indextype *) ML_allocate( sizeof(indextype) * Nparts );
607  iperm = (indextype *) ML_allocate( sizeof(indextype) * Nparts );
608 
609  if( perm == NULL || iperm == NULL ) {
610  fprintf( stderr,
611  "*ML*ERR* not nough memory to allocated %d bytex\n"
612  "*ML*ERR* (file %s, line %d)\n",
613  (int)sizeof(indextype) * Nparts,
614  __FILE__,
615  __LINE__ );
616  exit( EXIT_FAILURE );
617  }
618 
619  options[0] = 0; /* default values */
620  /* those are instead possible non-default values. No idea about how
621  to pick them up...
622  options[0] = 1;
623  options[1] = 2;
624  options[2] = 2;
625  options[3] = 1;
626  options[4] = 0;
627  options[5] = 1;
628  options[6] = 0;
629  options[7] = 1;
630  */
631 
632  /* ********************************************************************** */
633  /* estimate memory required by reordering */
634  /* ********************************************************************** */
635 
636  j = 0;
637  i = 4;
638 #if defined(HAVE_ML_METIS) && (METIS_VER_MAJOR < 5)
639  METIS_EstimateMemory( &Nrows, xadj, adjncy, &j,
640  &i, &nbytes );
641 
642  nbytes_max = ML_gmax_int( nbytes, comm );
643  nbytes = ML_gsum_int( nbytes, comm);
644 
645  if( mypid == 0 && ML_Get_PrintLevel() > 7 ) {
646 
647  printf("METIS reordering (level %d) estimated required mem = %d Kb\n"
648  "METIS reordering (level %d) max estimated mem = %d Kb\n",
649  level,
650  nbytes,
651  level,
652  nbytes_max );
653  }
654 #endif
655 
656  /* ********************************************************************** */
657  /* the METIS manual conseils to use NodeNS and not EdgeND. I have tested */
658  /* EdgeNS (in one scalar case), and it does not seem too different from */
659  /* NodeND. No idea about CPU times. */
660  /* iperm is a vector defined as: row i of unreordered matrix is row */
661  /* iperm[i] of reordered matrix */
662  /* ********************************************************************** */
663 
664  i = 0; /* offset, C or FORTRAN */
665 #ifdef HAVE_ML_METIS
666  METIS_NodeND( &Nparts, xadj2, adjncy2, &i, options, perm, iperm );
667 #else
668  fprintf( stderr,
669  "*ERR*ML* This function has been compiled without -DHAVE_ML_METIS\n"
670  "*ERR*ML* To use METIS, please add --with-ml_metis to your\n"
671  "*ERR*ML* configure script (recall to specify include dir and\n"
672  "*ERR*ML* location of the METIS lib; see configure --help\n"
673  "*ERR*ML* for more defailts).\n"
674  "*ERR*ML* (file %s, line %d)\n",
675  __FILE__,
676  __LINE__);
677  exit( EXIT_FAILURE );
678 #endif
679 
680  /* ********************************************************************** */
681  /* replace previous ordering in part with this fill-in reducing one */
682  /* Also, compute the bandwidth before and after reordering */
683  /* It seesm to me that the bandwidth is INCREASED after reordering. */
684  /* However, the number of zero elements (as given by OCTAVE's LU) is */
685  /* increased --- In the small test I run, from 88% to 93%. */
686  /* ********************************************************************** */
687 
688  bandwidth_orig = 0;
689  bandwidth_perm = 0;
690  for( i=0 ; i<Nparts ; i++ ) {
691  for( j=xadj2[i] ; j<xadj2[i+1] ; j++ ) {
692  k = abs(adjncy2[j] - i);
693  if( k>bandwidth_orig ) bandwidth_orig = k;
694  k = abs(iperm[adjncy2[j]] - iperm[i]);
695  if( k>bandwidth_perm ) bandwidth_perm = k;
696  }
697  }
698 
699  if ( mypid == 0 && ML_Get_PrintLevel() > 8 ) {
700 
701  printf("METIS reordering (level %d) Bandwidth before = %d, after = %d\n",
702  level,
703  bandwidth_orig, bandwidth_perm);
704 
705  }
706 
707  for( i=0 ; i<Nrows ; i++ ) {
708  j = part[i];
709  part[i] = iperm[j];
710  }
711 
712  /* ------------------- that's all folks --------------------------------- */
713 
714  ML_free( xadj2 ) ;
715  ML_free( adjncy2 );
716  ML_free( perm );
717  ML_free( iperm );
718 
719  t0 = GetClock() - t0;
720 
721  if ( mypid == 0 && ML_Get_PrintLevel() > 8 ) {
722 
723  printf("METIS (level %d) Time required for reordering = %e (s)\n",
724  level,
725  t0 );
726 
727  }
728 
729  return 0;
730
