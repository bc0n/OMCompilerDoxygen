55 /*
56  * Find row with greatest time that is smaller than or equal to 'time'
57  * Conditions:
58  * the buffer in 'delayStruct' is not empty
59  * 'time' is smaller than the last entry in 'delayStruct'
60  */
61 static int findTime(double time, RINGBUFFER *delayStruct)
62 {
63  int start = 0;
64  int end = ringBufferLength(delayStruct);
65  double t;
66 
67 
68  infoStreamPrint(LOG_EVENTS_V, 0, "findTime %e", time);
69  do
70  {
71  int i = (start + end) / 2;
72  t = ((TIME_AND_VALUE*)getRingData(delayStruct, i))->t;
73  infoStreamPrint(LOG_EVENTS_V, 0, "time(%d, %d)[%d] = %e", start, end, i, t);
74  if(t > time)
75  end = i;
76  else
77  start = i;
78  }while(t != time && end > start + 1);
79  infoStreamPrint(LOG_EVENTS_V, 0, "return time[%d, %d] = %e", start, end, t);
80  return (start);
81 }

107 double delayImpl(DATA* data, threadData_t *threadData, int exprNumber, double exprValue, double time, double delayTime, double delayMax)
108 {
109  RINGBUFFER* delayStruct = data->simulationInfo->delayStructure[exprNumber];
110  int length = ringBufferLength(delayStruct);
111 
112  infoStreamPrint(LOG_EVENTS_V, 0, "delayImpl: exprNumber = %d, exprValue = %g, time = %g, delayTime = %g", exprNumber, exprValue, time, delayTime);
113 
114  /* Check for errors */
115 
116  assertStreamPrint(threadData, 0 <= exprNumber, "invalid exprNumber = %d", exprNumber);
117  assertStreamPrint(threadData, exprNumber < data->modelData->nDelayExpressions, "invalid exprNumber = %d", exprNumber);
118 
119  if(time <= data->simulationInfo->tStart)
120  {
121  infoStreamPrint(LOG_EVENTS_V, 0, "delayImpl: Entered at time < starting time: %g.", exprValue);
122  return (exprValue);
123  }
124 
125  if(delayTime < 0.0)
126  {
127  throwStreamPrint(threadData, "Negative delay requested %g", delayTime);
128  }
129 
130  if(length == 0)
131  {
132  /* This occurs in the initialization phase */
133  infoStreamPrint(LOG_EVENTS, 0, "delayImpl: Missing initial value, using argument value %g instead.", exprValue);
134  return (exprValue);
135  }
136 
137  /*
138  * Returns: expr(time?delayTime) for time>time.start + delayTime and
139  * expr(time.start) for time <= time.start + delayTime.
140  * The arguments, i.e., expr, delayTime and delayMax, need to be subtypes of Real.
141  * DelayMax needs to be additionally a parameter expression.
142  * The following relation shall hold: 0 <= delayTime <= delayMax,
143  * otherwise an error occurs. If delayMax is not supplied in the argument list,
144  * delayTime need to be a parameter expression. See also Section 3.7.2.1.
145  * For non-scalar arguments the function is vectorized according to Section 10.6.12.
146  */
147  if(time <= data->simulationInfo->tStart + delayTime)
148  {
149  double res = ((TIME_AND_VALUE*)getRingData(delayStruct, 0))->value;
150  infoStreamPrint(LOG_EVENTS_V, 0, "findTime: time <= tStart + delayTime: [%d] = %g",exprNumber, res);
151  return res;
152  }
153  else
154  {
155  /* return expr(time-delayTime) */
156  double timeStamp = time - delayTime;
157  double time0, time1, value0, value1;
158  int i;
159 
160  assertStreamPrint(threadData, 0.0 <= delayTime, "Negative delay requested: delayTime = %g", delayTime);
161 
162  /* find the row for the lower limit */
163  if(timeStamp > ((TIME_AND_VALUE*)getRingData(delayStruct, length - 1))->t)
164  {
165  infoStreamPrint(LOG_EVENTS_V, 0, "delayImpl: find the row %g = %g", timeStamp, ((TIME_AND_VALUE*)getRingData(delayStruct, length - 1))->t);
166  /* delay between the last accepted time step and the current time */
167  time0 = ((TIME_AND_VALUE*)getRingData(delayStruct, length - 1))->t;
168  value0 = ((TIME_AND_VALUE*)getRingData(delayStruct, length - 1))->value;
169  time1 = time;
170  value1 = exprValue;
171  infoStreamPrint(LOG_EVENTS_V, 0, "delayImpl: times %g and %g", time0, time1);
172  infoStreamPrint(LOG_EVENTS_V, 0, "delayImpl: values %g and %g", value0, value1);
173  }
174  else
175  {
176  i = findTime(timeStamp, delayStruct);
177  assertStreamPrint(threadData, i < length, "%d = i < length = %d", i, length);
178  time0 = ((TIME_AND_VALUE*)getRingData(delayStruct, i))->t;
179  value0 = ((TIME_AND_VALUE*)getRingData(delayStruct, i))->value;
180 
181  /* was it the last value? */
182  if(i+1 == length)
183  {
184  return value0;
185  }
186  time1 = ((TIME_AND_VALUE*)getRingData(delayStruct, i+1))->t;
187  value1 = ((TIME_AND_VALUE*)getRingData(delayStruct, i+1))->value;
188  }
189  /* was it an exact match?*/
190  if(time0 == timeStamp){
191  infoStreamPrint(LOG_EVENTS_V, 0, "delayImpl: Exact match at %g = %g", timeStamp, value0);
192 
193  return value0;
194  } else if(time1 == timeStamp) {
195  infoStreamPrint(LOG_EVENTS_V, 0, "delayImpl: Exact match at %g = %g", timeStamp, value1);
196 
197  return value1;
198  } else {
199  /* linear interpolation */
200  double timedif = time1 - time0;
201  double dt0 = time1 - timeStamp;
202  double dt1 = timeStamp - time0;
203  double retVal = (value0 * dt0 + value1 * dt1) / timedif;
204  infoStreamPrint(LOG_EVENTS_V, 0, "delayImpl: Linear interpolation of %g between %g and %g", timeStamp, time0, time1);
205  infoStreamPrint(LOG_EVENTS_V, 0, "delayImpl: Linear interpolation of %g value: %g and %g = %g", timeStamp, value0, value1, retVal);
206  return retVal;
207  }
208  }

210 }
