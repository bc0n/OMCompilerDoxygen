68 void FMU2Logger::writeInternal(string msg, LogCategory cat, LogLevel lvl,
69  LogStructure ls)
70 {
71  LogCategoryFMU category;
72  fmi2Status status;
73 
74  if (ls == LS_END)
75  return;
76 
77  // determine FMI status and category from LogLevel
78  switch (lvl) {
79  case LL_ERROR:
80  status = fmi2Error;
81  category = logStatusError;
82  break;
83  case LL_WARNING:
84  status = fmi2Warning;
85  category = logStatusWarning;
86  break;
87  default:
88  status = fmi2OK;
89  category = logStatusWarning;
90  }
91 
92  // override FMU category with matching LogCategory
93  switch (cat) {
94  case LC_NLS:
95  category = logNonlinearSystems;
96  break;
97  case LC_EVENTS:
98  category = logEvents;
99  break;
100  }
101 
102  // call FMU log function
103  FMU2_LOG(_wrapper, status, category, msg.c_str());
104 }

144 fmi2Status FMU2Wrapper::setDebugLogging(fmi2Boolean loggingOn,
145  size_t nCategories,
146  const fmi2String categories[])
147 {
148  fmi2Status ret = fmi2OK;
149 
150  if (_logger == NULL) {
151  LogSettings logSettings = _globalSettings.getLogSettings();
152  FMU2Logger::initialize(this, logSettings, loggingOn);
153  _logger = Logger::getInstance();
154  }
155  else {
156  _logger->setEnabled(loggingOn);
157  }
158  if (nCategories == 0) {
159  _logCategories = loggingOn? 0xFFFF: 0x0000;
160  _logger->setAll(loggingOn? LL_DEBUG: LL_ERROR);
161  }
162  else {
163  int i, j, nSupported = sizeof(_LogCategoryFMUNames) / sizeof(fmi2String);
164  for (i = 0; i < nCategories; i++) {
165  if (strcmp(categories[i], "logAll") == 0) {
166  _logCategories = loggingOn? 0xFFFF: 0x0000;
167  _logger->setAll(loggingOn? LL_DEBUG: LL_ERROR);
168  continue;
169  }
170  for (j = 0; j < nSupported; j++) {
171  if (strcmp(categories[i], _LogCategoryFMUNames[j]) == 0) {
172  if (loggingOn)
173  _logCategories |= (1 << j);
174  else
175  _logCategories &= ~(1 << j);
176  switch (j) {
177  case logEvents:
178  _logger->set(LC_EVENTS, loggingOn? LL_DEBUG: LL_ERROR);
179  break;
180  case logNonlinearSystems:
181  _logger->set(LC_NLS, loggingOn? LL_DEBUG: LL_ERROR);
182  break;
183  }
184  break;
185  }
186  }
187  // warn about unsupported log category
188  if (j == nSupported) {
189  unsigned int logCategories_bak = _logCategories;
190  _logCategories = 0xFFFF;
191  FMU2_LOG(this, fmi2Warning, logStatusWarning,
192  "Unsupported log category \"%s\"", categories[i]);
193  _logCategories = logCategories_bak;
194  ret = fmi2Warning;
195  }
196  }
197  }
198  return ret;
199 }

440 fmi2Status FMU2Wrapper::newDiscreteStates(fmi2EventInfo *eventInfo)
441 {
442  if (_needUpdate || _nclockTick > 0) {
443  if (_nclockTick > 0)
444  _model->setClock(_clockTick, _clockSubactive);
445  updateModel();
446  if (_nclockTick > 0) {
447  // reset clocks
448  std::fill(_clockTick, _clockTick + _model->getDimClock(), false);
449  std::fill(_clockSubactive, _clockSubactive + _model->getDimClock(), false);
450  _nclockTick = 0;
451  }
452  }
453  // Check if an Zero Crossings happend
454  double f[NUMBER_OF_EVENT_INDICATORS + 1];
455  bool events[NUMBER_OF_EVENT_INDICATORS + 1];
456  _model->getZeroFunc(f);
457  for (int i = 0; i < NUMBER_OF_EVENT_INDICATORS; i++)
458  events[i] = f[i] >= 0;
459  // Handle Zero Crossings if nessesary
460  bool state_vars_reinitialized = _model->handleSystemEvents(events);
461  //time events
462  eventInfo->nextEventTime = _model->computeNextTimeEvents(_model->getTime());
463  if ((eventInfo->nextEventTime != 0.0) && (eventInfo->nextEventTime != std::numeric_limits<double>::max()))
464  eventInfo->nextEventTimeDefined = fmi2True;
465  else
466  eventInfo->nextEventTimeDefined = fmi2False;
467  // everything is done
468  eventInfo->newDiscreteStatesNeeded = fmi2False;
469  eventInfo->terminateSimulation = fmi2False;
470  eventInfo->nominalsOfContinuousStatesChanged = state_vars_reinitialized;
471  eventInfo->valuesOfContinuousStatesChanged = state_vars_reinitialized;
472  return fmi2OK;
473 }
