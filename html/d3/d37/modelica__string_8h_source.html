103 static inline void* mmc_mk_scon(const char *s)
104 {
105  size_t nbytes = strlen(s);
106  size_t header = MMC_STRINGHDR(nbytes);
107  size_t nwords = MMC_HDRSLOTS(header) + 1;
108  struct mmc_string *p;
109  void *res;
110  if (nbytes == 0) return mmc_emptystring;
111  if (nbytes == 1) {
112  unsigned char c = *s;
113  return mmc_strings_len1[(unsigned int)c];
114  }
115  p = (struct mmc_string *) mmc_check_out_of_memory(omc_alloc_interface.malloc_atomic(nwords*sizeof(void*)));
116  p->header = header;
117  memcpy(p->data, s, nbytes+1); /* including terminating '\0' */
118  res = MMC_TAGPTR(p);
119  MMC_CHECK_STRING(res);
120 #ifdef MMC_MK_DEBUG
121  fprintf(stderr, "STRING slots: %u size: %d str: %s\n", MMC_HDRSLOTS(header), nbytes, s); fflush(NULL);
122 #endif
123  return res;
124 }
