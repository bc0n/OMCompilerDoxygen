73 void zmq::object_t::process_command (command_t &cmd_)
74 {
75  switch (cmd_.type) {
76 
77  case command_t::activate_read:
78  process_activate_read ();
79  break;
80 
81  case command_t::activate_write:
82  process_activate_write (cmd_.args.activate_write.msgs_read);
83  break;
84 
85  case command_t::stop:
86  process_stop ();
87  break;
88 
89  case command_t::plug:
90  process_plug ();
91  process_seqnum ();
92  break;
93 
94  case command_t::own:
95  process_own (cmd_.args.own.object);
96  process_seqnum ();
97  break;
98 
99  case command_t::attach:
100  process_attach (cmd_.args.attach.engine);
101  process_seqnum ();
102  break;
103 
104  case command_t::bind:
105  process_bind (cmd_.args.bind.pipe);
106  process_seqnum ();
107  break;
108 
109  case command_t::hiccup:
110  process_hiccup (cmd_.args.hiccup.pipe);
111  break;
112 
113  case command_t::pipe_term:
114  process_pipe_term ();
115  break;
116 
117  case command_t::pipe_term_ack:
118  process_pipe_term_ack ();
119  break;
120 
121  case command_t::pipe_hwm:
122  process_pipe_hwm (cmd_.args.pipe_hwm.inhwm, cmd_.args.pipe_hwm.outhwm);
123  break;
124 
125  case command_t::term_req:
126  process_term_req (cmd_.args.term_req.object);
127  break;
128 
129  case command_t::term:
130  process_term (cmd_.args.term.linger);
131  break;
132 
133  case command_t::term_ack:
134  process_term_ack ();
135  break;
136 
137  case command_t::reap:
138  process_reap (cmd_.args.reap.socket);
139  break;
140 
141  case command_t::reaped:
142  process_reaped ();
143  break;
144 
145  case command_t::inproc_connected:
146  process_seqnum ();
147  break;
148 
149  case command_t::done:
150  default:
151  zmq_assert (false);
152  }
153 }
