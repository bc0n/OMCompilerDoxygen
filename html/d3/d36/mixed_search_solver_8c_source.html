107 modelica_boolean nextVar(modelica_boolean *b, int n) {
108  /*number of "1" */
109  int n1 = 0;
110  int i;
111  int last;
112  for(i = 0; i < n; i++){
113  if(b[i] == 1)
114  n1++;
115  }
116  /*index of last element with "1"*/
117  last = n - 1;
118  while(last >= 0 && !b[last])
119  last--;
120  if(n1 == n) /*exit - all combination were already generated*/
121  return 0;
122  else if(last == -1) { /* 0000 -> 1000 */
123  b[0] = 1;
124  return 1;
125  } else if(last < n - 1) { /* e.g. 1010 -> 1001 */
126  b[last] = 0;
127  b[last + 1] = 1;
128  return 1;
129  } else { /*at the end of the array is "1"*/
130  /*detect position of last ocurenc of sequence 10 */
131  int ip = n - 2; /*actual position in array*/
132  int nr1 = 1; /*count of "1"*/
133  while(ip >= 0) {
134  if(b[ip] && !b[ip + 1]) { /*we found*/
135  nr1++;
136  break;
137  } else if(b[ip]) { /*we didn't find, but 1 - increase nr1*/
138  nr1++;
139  ip--;
140  } else { /*we didnt't find, 0*/
141  ip--;
142  }
143  }
144  if(ip >= 0) { /*e.g. 1001 -> 0110*/
145  int pn = ip + nr1;
146  b[ip] = 0;
147  for(i = ip + 1; i <= pn; i++)
148  b[i] = 1;
149  for(i = pn + 1; i <= n - 1; i++)
150  b[i] = 0;
151  return 1;
152  } else {
153  for(i = 0; i <= n1; i++)
154  b[i] = 1;
155  for(i = n1 + 1; i <= n - 1; i++)
156  b[i] = 0;
157  return 1;
158  }
159  }
160 }

169 int solveMixedSearch(DATA *data, int sysNumber)
170 {
171  MIXED_SYSTEM_DATA* systemData = &(data->simulationInfo->mixedSystemData[sysNumber]);
172  DATA_SEARCHMIXED_SOLVER* solverData = (DATA_SEARCHMIXED_SOLVER*)systemData->solverData;
173 
174  int eqSystemNumber = systemData->equationIndex;
175 
176  int found_solution = 0;
177  /*
178  * We are given the number of the non-linear system.
179  * We want to look it up among all equations.
180  */
181  int i, ix;
182 
183  int stepCount = 0;
184  int mixedIterations = 0;
185  int success = 0;
186 
187  debugStreamPrint(LOG_NLS, 1, "\n#### Start solver mixed equation system at time %f.", data->localData[0]->timeValue);
188 
189  memset(solverData->stateofSearch, 0, systemData->size);
190 
191  /* update pre iteration vars */
192  /* update iteration vars */
193  for(i=0;i<systemData->size;++i)
194  solverData->iterationVarsPre[i] = *(systemData->iterationVarsPtr[i]);
195 
196  do
197  {
198  /* update pre iteration vars */
199  for(i=0;i<systemData->size;++i)
200  solverData->iterationVars[i] = *(systemData->iterationVarsPtr[i]);
201 
202  /* solve continuous equation part
203  * and update iteration variables in model
204  */
205  systemData->solveContinuousPart(data);
206  systemData->updateIterationExps(data);
207 
208  /* set new values of boolean variable */
209  for(i=0;i<systemData->size;++i)
210  solverData->iterationVars2[i] = *(systemData->iterationVarsPtr[i]);
211 
212 
213  found_solution = systemData->continuous_solution;
214  debugStreamPrint(LOG_NLS, 0, "#### continuous system solution status = %d", found_solution);
215 
216  /* restart if any relation has changed */
217  if(checkRelations(data))
218  {
219  updateRelationsPre(data);
220  systemData->updateIterationExps(data);
221  debugStreamPrint(LOG_NLS, 0, "#### System relation changed restart iteration");
222  if(mixedIterations++ > 200)
223  found_solution = -4; /* mixedIterations++ > 200 */
224  }
225 
226  if(found_solution == -1)
227  {
228  /* system of equations failed */
229  found_solution = -2;
230  debugStreamPrint(LOG_NLS, 0, "#### NO SOLUTION ");
231  }
232  else
233  {
234  found_solution = 1;
235  for(i = 0; i < systemData->size; i++)
236  {
237  debugStreamPrint(LOG_NLS, 0, " check iterationVar[%d] = %d <-> %d", i, solverData->iterationVars[i], solverData->iterationVars2[i]);
238  if(solverData->iterationVars[i] != solverData->iterationVars2[i])
239  {
240  found_solution = 0;
241  break;
242  }
243  }
244  debugStreamPrint(LOG_NLS, 0, "#### SOLUTION = %c", found_solution ? 'T' : 'F');
245  }
246 
247  if(!found_solution )
248  {
249  /* try next set of values*/
250  if(nextVar(solverData->stateofSearch, systemData->size))
251  {
252  debugStreamPrint(LOG_NLS, 0, "#### set next STATE ");
253  for(i = 0; i < systemData->size; i++)
254  *(systemData->iterationVarsPtr[i]) = *(systemData->iterationPreVarsPtr[i]) != solverData->stateofSearch[i];
255 
256  /* debug output */
257  if(ACTIVE_STREAM(LOG_NLS))
258  {
259  const char * __name;
260  for(i = 0; i < systemData->size; i++)
261  {
262  ix = (systemData->iterationVarsPtr[i]-data->localData[0]->booleanVars);
263  __name = data->modelData->booleanVarsData[ix].info.name;
264  debugStreamPrint(LOG_NLS, 0, "%s changed : %d -> %d", __name, solverData->iterationVars[i], *(systemData->iterationVarsPtr[i]));
265  }
266  }
267  }
268  else
269  {
270  /* while the initialization it's okay not a solution */
271  if(!data->simulationInfo->initial)
272  {
273  warningStreamPrint(LOG_STDOUT, 0,
274  "Error solving mixed equation system with index %d at time %e",
275  eqSystemNumber, data->localData[0]->timeValue);
276  }
277  data->simulationInfo->needToIterate = 1;
278  found_solution = -1;
279  /*TODO: "break simulation?"*/
280  }
281  }
282  /* we found a solution*/
283  if(found_solution == 1)
284  {
285  success = 1;
286  if(ACTIVE_STREAM(LOG_NLS))
287  {
288  const char * __name;
289  debugStreamPrint(LOG_NLS, 0, "#### SOLUTION FOUND! (system %d)", eqSystemNumber);
290  for(i = 0; i < systemData->size; i++)
291  {
292  ix = (systemData->iterationVarsPtr[i]-data->localData[0]->booleanVars);
293  __name = data->modelData->booleanVarsData[ix].info.name;
294  debugStreamPrint(LOG_NLS, 0, "%s = %d pre(%s)= %d", __name, *systemData->iterationVarsPtr[i], __name,
295  *systemData->iterationPreVarsPtr[i]);
296  }
297  }
298  }
299 
300  stepCount++;
301  mixedIterations++;
302 
303  }while(!found_solution);
304 
305  messageClose(LOG_NLS);
306  debugStreamPrint(LOG_NLS, 0, "#### Finished mixed equation system in steps %d.\n", stepCount);
307  return success;
308 }
