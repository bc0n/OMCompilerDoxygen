46 /*
47  * -----------------------------------------------------------------
48  *
49  * SUMMARY
50  *
51  * These routines provide a band matrix preconditioner based on
52  * difference quotients of the ODE right-hand side function f.
53  * The user supplies parameters
54  * mu = upper half-bandwidth (number of super-diagonals)
55  * ml = lower half-bandwidth (number of sub-diagonals)
56  * The routines generate a band matrix of bandwidth ml + mu + 1
57  * and use this to form a preconditioner for use with the Krylov
58  * linear solver in CVSP*. Although this matrix is intended to
59  * approximate the Jacobian df/dy, it may be a very crude
60  * approximation. The true Jacobian need not be banded, or its
61  * true bandwith may be larger than ml + mu + 1, as long as the
62  * banded approximation generated here is sufficiently accurate
63  * to speed convergence as a preconditioner.
64  *
65  * Usage:
66  * The following is a summary of the usage of this module.
67  * Details of the calls to CVodeCreate, CVodeInit, CVSp*,
68  * and CVode are available in the User Guide.
69  * To use these routines, the sequence of calls in the user
70  * main program should be as follows:
71  *
72  * #include <cvodes/cvodes_bandpre.h>
73  * #include <nvector_serial.h>
74  * ...
75  * Set y0
76  * ...
77  * cvode_mem = CVodeCreate(...);
78  * ier = CVodeInit(...);
79  * ...
80  * flag = CVSptfqmr(cvode_mem, pretype, maxl);
81  * -or-
82  * flag = CVSpgmr(cvode_mem, pretype, maxl);
83  * -or-
84  * flag = CVSpbcg(cvode_mem, pretype, maxl);
85  * ...
86  * flag = CVBandPrecInit(cvode_mem, N, mu, ml);
87  * ...
88  * flag = CVode(...);
89  * ...
90  * Free y0
91  * ...
92  * CVodeFree(&cvode_mem);
93  *
94  * Notes:
95  * (1) Include this file for the CVBandPrecData type definition.
96  * (2) In the CVBandPrecInit call, the arguments N is the
97  * problem dimension.
98  * (3) In the CVBPSp* call, the user is free to specify
99  * the input pretype and the optional input maxl.
100  * -----------------------------------------------------------------
101  */
102

104 /*
105  * -----------------------------------------------------------------
106  * Function : CVBandPrecInit
107  * -----------------------------------------------------------------
108  * CVBandPrecInit allocates and initializes the BANDPRE preconditioner
109  * module. This functino must be called AFTER one of the SPILS linear
110  * solver modules has been attached to the CVODE integrator.
111  *
112  * The parameters of CVBandPrecInit are as follows:
113  *
114  * cvode_mem is the pointer to CVODE memory returned by CVodeCreate.
115  *
116  * N is the problem size.
117  *
118  * mu is the upper half bandwidth.
119  *
120  * ml is the lower half bandwidth.
121  *
122  * The return value of CVBandPrecInit is one of:
123  * CVSPILS_SUCCESS if no errors occurred
124  * CVSPILS_MEM_NULL if the integrator memory is NULL
125  * CVSPILS_LMEM_NULL if the linear solver memory is NULL
126  * CVSPILS_ILL_INPUT if an input has an illegal value
127  * CVSPILS_MEM_FAIL if a memory allocation request failed
128  *
129  * NOTE: The band preconditioner assumes a serial implementation
130  * of the NVECTOR package. Therefore, CVBandPrecInit will
131  * first test for a compatible N_Vector internal
132  * representation by checking for required functions.
133  * -----------------------------------------------------------------
134  */
135
