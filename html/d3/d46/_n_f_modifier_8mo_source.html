255  function patchElementModFinal
256  // TODO: This would be cheaper to do in SCodeUtil when creating the
257  // modifiers, but it breaks the old instantiation.
258  "This function makes modifiers applied to final elements final, e.g. for
259  'final Real x(start = 1.0)' it will mark '(start = 1.0)' as final. This is
260  done so that we only need to check for final violations while merging
261  modifiers."
262  input SCode.Prefixes prefixes;
263  input SourceInfo info;
264  input output SCode.Mod mod;
265  algorithm
266  if SCode.finalBool(SCode.prefixesFinal(prefixes)) then
267  mod := match mod

269  algorithm
270  mod.finalPrefix := SCode.Final.FINAL();
271  then
272  mod;
273 
274  case SCode.Mod.REDECL()
275  algorithm
276  mod.finalPrefix := SCode.Final.FINAL();
277  then

279 
280  else SCode.Mod.MOD(SCode.Final.FINAL(), SCode.Each.NOT_EACH(), {}, NONE(), info);
281  end match;
282  end if;

392  function merge
393  input Modifier outerMod;
394  input Modifier innerMod;
395  input String name = "";
396  output Modifier mergedMod;
397  algorithm
398  mergedMod := match(outerMod, innerMod)
399  local
400  ModTable.Tree submods;
401  Binding binding;
402 
403  // One of the modifiers is NOMOD, return the other.
404  case (NOMOD(), _) then innerMod;
405  case (_, NOMOD()) then outerMod;
406 
407  // Two modifiers, merge bindings and submodifiers.
408  case (MODIFIER(), MODIFIER())
409  algorithm
410  checkFinalOverride(innerMod.finalPrefix, outerMod, innerMod.info);
411  binding := if Binding.isBound(outerMod.binding) then
412  outerMod.binding else innerMod.binding;
413  submods := ModTable.join(innerMod.subModifiers, outerMod.subModifiers, merge);
414  then
415  MODIFIER(outerMod.name, outerMod.finalPrefix, outerMod.eachPrefix, binding, submods, outerMod.info);
416 
417  case (REDECLARE(), MODIFIER())
418  algorithm
419  outerMod.mod := merge(outerMod.mod, innerMod);
420  then
421  outerMod;
422 
423  case (MODIFIER(), REDECLARE())
424  algorithm
425  innerMod.mod := merge(outerMod, innerMod.mod);
426  then
427  innerMod;
428 
429  case (REDECLARE(), _) then outerMod;
430  case (_, REDECLARE()) then innerMod;
431 
432  else
433  algorithm
434  Error.addMessage(Error.INTERNAL_ERROR,
435  {"Mod.mergeMod failed on unknown mod."});
436  then
437  fail();

439  end match;

561  function mergeLocal
562  "Merges two modifiers in the same scope, i.e. like a(x(y = 1), x(z = 2)).
563  This is allowed as long as the two modifiers doesn't modify the same
564  element, otherwise it's an error."
565  input Modifier mod1;
566  input Modifier mod2;
567  input String name = "";
568  input ModifierScope scope;
569  input list<String> prefix = {};
570  output Modifier mod;
571  protected
572  String comp_name;
573  algorithm
574  mod := match (mod1, mod2)
575  // The second modifier has no binding, use the binding from the first.
576  case (MODIFIER(), MODIFIER(binding = Binding.UNBOUND()))
577  algorithm
578  mod1.subModifiers := ModTable.join(mod1.subModifiers, mod2.subModifiers,
579  function mergeLocal(scope = scope, prefix = mod1.name :: prefix));
580  then
581  mod1;
582 
583  // The first modifier has no binding, use the binding from the second.
584  case (MODIFIER(binding = Binding.UNBOUND()), MODIFIER())
585  algorithm
586  mod2.subModifiers := ModTable.join(mod2.subModifiers, mod1.subModifiers,
587  function mergeLocal(scope = scope, prefix = mod1.name :: prefix));
588  then
589  mod2;
590 
591  // Both modifiers modify the same element, give duplicate modification error.
592  else
593  algorithm
594  comp_name := stringDelimitList(listReverse(Modifier.name(mod1) :: prefix), ".");
595  Error.addMultiSourceMessage(Error.DUPLICATE_MODIFICATIONS,
596  {comp_name, ModifierScope.toString(scope)},
597  {Modifier.info(mod1), Modifier.info(mod2)});
598  then
599  fail();
600 
601  end match;
602  end mergeLocal;
