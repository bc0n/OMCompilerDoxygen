69 int base_array_ok(const base_array_t *a)
70 {
71  int i;
72  if(a == NULL) {
73  fprintf(stderr, "base_array.c: array is NULL!\n"); fflush(stderr);
74  return 0;
75  }
76  if(a->ndims < 0) {
77  fprintf(stderr, "base_array.c: the number of array dimensions are < 0!\n"); fflush(stderr);
78  return 0;
79  }
80  if(a->dim_size == NULL) {
81  fprintf(stderr, "base_array.c: array dimensions sizes are NULL!\n"); fflush(stderr);
82  return 0;
83  }
84  for(i = 0; i < a->ndims; ++i) {
85  if(a->dim_size[i] < 0) {
86  fprintf(stderr, "base_array.c: array dimension size for dimension %d is %d < 0!\n", i, (int) a->dim_size[i]); fflush(stderr);
87  return 0;
88  }
89  }
90  return 1;
91 }

171 int index_spec_fit_base_array(const index_spec_t *s, const base_array_t *a)
172 {
173  int i, j;
174 
175  if(s->ndims != a->ndims) {
176  fprintf(stderr, "index spec dimensions and array dimensions do not agree %d != %d\n",
177  (int)s->ndims, (int)a->ndims); fflush(stderr);
178  return 0;
179  }
180  for(i = 0; i < s->ndims; ++i) {
181  if(s->dim_size[i] == 0) {
182  if((s->index[i][0] <= 0) || (s->index[i][0] > a->dim_size[i])) {
183  fprintf(stderr,
184  "scalar s->index[%d][0] == %d incorrect, a->dim_size[%d] == %d\n",
185  i, (int) s->index[i][0], i, (int) a->dim_size[i]); fflush(stderr);
186  return 0;
187  }
188  }
189 
190  if(s->index[i] != NULL)
191  {
192  for(j = 0; j < s->dim_size[i]; ++j) {
193  if((s->index[i][j] <= 0) || (s->index[i][j] > a->dim_size[i])) {
194  fprintf(stderr,
195  "array s->index[%d][%d] == %d incorrect, a->dim_size[%d] == %d\n",
196  i, j, (int) s->index[i][j], i, (int) a->dim_size[i]); fflush(stderr);
197  return 0;
198  }
199  }
200  }
201  }
202 
203  return 1;
204 }

265 size_t calc_base_index_spec(int ndims, const _index_t *idx_vec,
266  const base_array_t *arr, const index_spec_t *spec)
267 {
268  /* idx_vec is zero based */
269  /* spec is one based */
270  int i;
271  int d2;
272  size_t index = 0;
273 
274  assert(base_array_ok(arr));
275  assert(index_spec_ok(spec));
276  assert(index_spec_fit_base_array(spec, arr));
277  assert((ndims == arr->ndims) && (ndims == spec->ndims));
278 
279  index = 0;
280  for(i = 0; i < ndims; ++i) {
281  int d = idx_vec[i];
282  if(spec->index[i] != NULL) {
283  d2 = spec->index[i][d] - 1;
284  } else {
285  d2 = d;
286  }
287  index = (index * arr->dim_size[i]) + d2;
288  }
289 
290  return index;
291 }

331  omc_assert(NULL, info, "Dimension %d has bounds 1..%d, got array subscript %d", i+1, dims[i], subs[i]+1);
332  }
333  index = (index * dims[i]) + subs[i];
334  }
335 
336 
337  return index;
