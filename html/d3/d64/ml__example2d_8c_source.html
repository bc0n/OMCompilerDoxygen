445 /* Set the initial guess and the right hand side where the right hand side
446  * is obtained by doing a matrix-vector multiplication.
447  *
448  * Author: Ray Tuminaro, Div 1422, SNL
449  * Date : 3/15/95
450  *
451  * Parameters
452  *
453  * update_index == On input, ordering of update and external
454  * locally on this processor. For example
455  * 'update_index[i]' gives the index location
456  * of the block which has the global index
457  * 'update[i]'.
458  * update == On input, list of pts to be updated on this node
459  * data_org == On input, indicates how data is set on this node.
460  * For example, data_org[] contains information on
461  * how many unknowns are internal, external and
462  * border unknowns as well as which points need
463  * to be communicated. See User's Guide for more
464  * details.
465  * val, indx, == On input, holds matrix nonzeros. See User's Guide
466  * bindx, rpntr, for more details.
467  * cpntr, bpntr
468  * x == On output, 'x' is allocated and set to all zeros.
469  * b == On output, 'b' is allocated and is set to the
470  * result of a matrix-vector product.
471  ******************************************************************************/
472 void init_guess_and_rhs(int update_index[], int update[], double *x[],double
473  *b[],int data_org[], double val[], int indx[], int
474  bindx[], int rpntr[], int cpntr[], int bpntr[], int
475  proc_config[])

483  num = data_org[AZ_N_internal] + data_org[AZ_N_border];
484 
485  /* allocate vectors */
486 
487  i = num + data_org[AZ_N_external];
488  *x = (double *) calloc(i, sizeof(double));
489  *b = (double *) calloc(i, sizeof(double));
490  if ((*b == NULL) && (i != 0)) {

492  }

494  /*initialize 'x' to a function which will be used in matrix-vector product */
495 
496  for (i = 0; i < temp; i++)
497  (*x)[i] = (double) (update[i])/(double)(num_PDE_eqns);
498  /* for (i = 0; i < temp; i++) (*x)[i] = (double) 1; */
499 
500  for (i = 0; i < temp/num_PDE_eqns; i++)
501  for (j = 0; j < num_PDE_eqns; j++)
502  (*x)[i*num_PDE_eqns+j] = (*x)[i*num_PDE_eqns];
503 
504  /* Reorder 'x' so that it conforms to the transformed matrix */
505 
506  AZ_reorder_vec(*x,data_org,update_index,rpntr);
507 
508  /* take out the constant vector. Used for the */

510

512  Amat.bpntr = bpntr; Amat.bindx = bindx;
513  Amat.indx = indx; Amat.val = val;

516  if (data_org[AZ_matrix_type] == AZ_MSR_MATRIX)
517  Amat.matvec = AZ_MSR_matvec_mult;
518  else if (data_org[AZ_matrix_type] == AZ_VBR_MATRIX)
519  Amat.matvec = AZ_VBR_matvec_mult;
520 
521  Amat.matvec(*x, *b, &Amat, proc_config);
522 
523  for (i = 0; i < num; i++) (*x)[i] = 0.0;

863  pr_error("Error: nprocs should be a square (%d).\n",nprocs_1d);

866  mypid_y = mypid / nprocs_1d;

869  /* process the grid */
870  /* --------------------------------------------------------------- */
871 
872  nelmnt_1d = (int) pow( (double) N_elements, 0.500001 );
873 
874  /* partition the elements along the xy-axis */
875 
876  nelmnt_part_xy = nelmnt_1d / nprocs_1d;
877  if (nelmnt_part_xy * nprocs_1d != nelmnt_1d) {
878  pr_error("Error: nelmnt_part not good. %d %d\n",nelmnt_part_xy,nelmnt_1d);
879  }
880  nelmnt_local = nelmnt_part_xy * nelmnt_part_xy;
881 
882  /* initialize the mesh */
883 
884  ML_GridAGX_Create( meshp );
885  mesh = (*meshp);
886  h = 1.0 / (double) nelmnt_1d;
887  ML_GridAGX_Set_Dimension(mesh, 2);
888  ML_GridAGX_Set_NElmnts(mesh, nelmnt_local, 4);
889 
890  /* generate the global element numbers of the local elements */
891 
892  ML_memory_alloc((void**) &big_intlist, nelmnt_local * sizeof(ml_big_int), "AP1");
893  estart_x = nelmnt_part_xy * mypid_x;
894  estart_y = nelmnt_part_xy * mypid_y;
895  eend_x = estart_x + nelmnt_part_xy - 1;
896  eend_y = estart_y + nelmnt_part_xy - 1;
897  icnt = 0;
898 
899  for (k = estart_y; k <= eend_y; k++)
900  for (j = estart_x; j <= eend_x; j++)
901  big_intlist[icnt++] = j + k * nelmnt_1d;
902  ML_GridAGX_Load_ElmntGlobalNum(mesh, nelmnt_local, big_intlist);
903  ML_memory_free( (void **) &big_intlist);
904 
905  /* generate the local computational grid */
906 
907  nnode_x = nelmnt_part_xy + 1;
908  nnode_y = nelmnt_part_xy + 1;
909  ML_memory_alloc((void**) &square, nnode_x * sizeof(int*), "AP2");
910  for ( i = 0; i < nnode_x; i++ )
911  ML_memory_alloc((void**) &(square[i]), nnode_y * sizeof(int), "AP3");
912  icnt = 0;
913  x_begin = 1;
914  y_begin = 1;
915  if ( estart_x == 0 ) x_begin--;
916  if ( estart_y == 0 ) y_begin--;
917  for ( k = y_begin; k < nnode_y; k++ )
918  for ( j = x_begin; j < nnode_x; j++ )
919  square[j][k] = icnt++;
920  if ( x_begin == 1 || y_begin == 1 ) square[0][0] = icnt++;
921  if ( x_begin == 1 ) {
922  for ( k = 1; k < nnode_y; k++ ) square[0][k] = icnt++;
923  }
924  if ( y_begin == 1 ) {
925  for ( j = 1; j < nnode_x; j++ ) square[j][0] = icnt++;
926  }
927 
928  /* generate the global node IDs */
929 
930  nnode_local_x = nnode_x - 1;
931  nnode_local_y = nnode_y - 1;
932  if (mypid_x == 0 ) nnode_local_x++;
933  if (mypid_y == 0 ) nnode_local_y++;

938  (*update) = (int*) malloc(nnode_expand * sizeof(int));
939 
940  nstart_x = mypid_x * nnode_local_x;
941  nstart_y = mypid_y * nnode_local_y;
942  for (k = 0; k < nnode_y; k++)
943  for (j = 0; j < nnode_x; j++) {
944  index = square[j][k];
945  offset = j + nstart_x + (k+nstart_y)*nnode_1d;
946  (*update)[index] = offset;
947  }
948

950  ML_GridAGX_Load_VertGlobalNum(mesh, nnode_expand, *update);
951 
952  /* generate coordinates of local nodes */
953 
954  ML_memory_alloc((void**) &coord, 2*nnode_expand*sizeof(double),"AP6");
955  for (k = 0; k < nnode_y; k++)

957  index = square[j][k];

967

970  for (j = estart_x; j <= eend_x; j++) {
971  intlist[0] = square[j-estart_x][k-estart_y];
972  intlist[1] = square[j-estart_x+1][k-estart_y];
973  intlist[2] = square[j-estart_x][k-estart_y+1];

978  for ( i = 0; i < nnode_x; i++ )
979  ML_memory_free( (void**) &(square[i]) );
980  ML_memory_free( (void**) &square);
981  ML_memory_free( (void **) &coord);
