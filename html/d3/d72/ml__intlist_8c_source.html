100  cur_leng = end + (10 * ncnt);

102  ML_memory_alloc( (void**) &itmp, cur_leng * sizeof(int), "IL5");
103  for ( i = 0; i < begin; i++ ) itmp[i] = ilist->members[i];
104  ML_memory_free( (void **) &(ilist->members) );
105  ilist->members = itmp;
106  }
107  for ( i = begin; i < end; i++ ) ilist->members[i] = list[icnt++];
108  ilist->start[leng+1] = ilist->start[leng] + ncnt;
109  ilist->length++;
