31 STATIC MYBOOL includeMDO(MYBOOL *usedpos, int item)
32 {
33 /* Legend: TRUE => A basic slack variable already in the basis
34  FALSE => A column free for being pivoted in
35  AUTOMATIC+TRUE => A row-singleton user column pivoted into the basis
36  AUTOMATIC+FALSE => A column-singleton user column pivoted into the basis */
37 
38  /* Handle case where we are processing all columns */
39  if(usedpos == NULL)
40  return( TRUE );
41 
42  else {
43  /* Otherwise do the selective case */
44  MYBOOL test = usedpos[item];
45 #if 1
46  return( test != TRUE );
47 #else
48  test = test & TRUE;
49  return( test == FALSE );
50 #endif
51  }
52 }

54 STATIC int prepareMDO(lprec *lp, MYBOOL *usedpos, int *colorder, int *data, int *rowmap)
55 /* This routine prepares data structures for colamd(). It is called twice, the first
56  time to count applicable non-zero elements by column, and the second time to fill in
57  the row indexes of the non-zero values from the first call. Note that the colamd()
58  row index base is 0 (which suits lp_solve fine). */
59 {
60  int i, ii, j, k, kk;
61  int nrows = lp->rows+1, ncols = colorder[0];
62  int offset = 0, Bnz = 0, Tnz;
63  MYBOOL dotally = (MYBOOL) (rowmap == NULL);
64  MATrec *mat = lp->matA;
65  REAL hold;
66  REAL *value;
67  int *rownr;
68 
69  if(dotally)
70  data[0] = 0;
71 
72  Tnz = nrows - ncols;
73  for(j = 1; j <= ncols; j++) {
74  kk = colorder[j];
75 
76  /* Process slacks */
77  if(kk <= lp->rows) {
78  if(includeMDO(usedpos, kk)) {
79  if(!dotally)
80  data[Bnz] = rowmap[kk]+offset;
81  Bnz++;
82  }
83  Tnz++;
84  }
85  /* Process user columns */
86  else {
87  k = kk - lp->rows;
88  i = mat->col_end[k-1];
89  ii= mat->col_end[k];
90  Tnz += ii-i;
91 #ifdef Paranoia
92  if(i >= ii)
93  lp->report(lp, SEVERE, "prepareMDO: Encountered empty basic column %d\n", k);
94 #endif
95 
96  /* Detect if we need to do phase 1 adjustments of zero-valued OF variable */
97  rownr = &COL_MAT_ROWNR(i);
98  value = &COL_MAT_VALUE(i);
99  hold = 0;
100  if((*rownr > 0) && includeMDO(usedpos, 0) && modifyOF1(lp, kk, &hold, 1.0)) {
101  if(!dotally)
102  data[Bnz] = offset;
103  Bnz++;
104  }
105  /* Loop over all NZ-variables */
106  for(; i < ii;
107  i++, value += matValueStep, rownr += matRowColStep) {
108  if(!includeMDO(usedpos, *rownr))
109  continue;
110  /* See if we need to change phase 1 OF value */
111  if(*rownr == 0) {
112  hold = *value;
113  if(!modifyOF1(lp, kk, &hold, 1.0))
114  continue;
115  }
116  /* Tally uneliminated constraint row values */
117  if(!dotally)
118  data[Bnz] = rowmap[*rownr]+offset;
119  Bnz++;
120  }
121  }
122  if(dotally)
123  data[j] = Bnz;
124  }
125  return( Tnz );
126 }

128 STATIC MYBOOL verifyMDO(lprec *lp, int *col_end, int *row_nr, int rowmax, int colmax)
129 {
130  int i, j, n, err = 0;
131 
132  for(i = 1; i <= colmax; i++) {
133  n = 0;
134  for(j = col_end[i-1]; (j < col_end[i]) && (err == 0); j++, n++) {
135  if(row_nr[j] < 0 || row_nr[j] > rowmax)
136  err = 1;
137  if(n > 0 && row_nr[j] <= row_nr[j-1])
138  err = 2;
139  n++;
140  }
141  }
142  if(err != 0)
143  lp->report(lp, SEVERE, "verifyMDO: Invalid MDO input structure generated (error %d)\n", err);
144  return( (MYBOOL) (err == 0) );
145 }

157 int __WINAPI getMDO(lprec *lp, MYBOOL *usedpos, int *colorder, int *size, MYBOOL symmetric)
158 {
159  int error = FALSE;
160  int nrows = lp->rows+1, ncols = colorder[0];
161  int i, j, kk, n;
162  int *col_end, *row_map = NULL;
163  int Bnz, Blen, *Brows = NULL;
164  int stats[COLAMD_STATS];
165  double knobs[COLAMD_KNOBS];
166 
167  /* Tally the non-zero counts of the unused columns/rows of the
168  basis matrix and store corresponding "net" starting positions */
169  allocINT(lp, &col_end, ncols+1, FALSE);
170  n = prepareMDO(lp, usedpos, colorder, col_end, NULL);
171  Bnz = col_end[ncols];
172 
173  /* Check that we have unused basic columns, otherwise skip analysis */
174  if(ncols == 0 || Bnz == 0)
175  goto Transfer;
176 
177  /* Get net number of rows and fill mapper */
178  allocINT(lp, &row_map, nrows, FALSE);
179  nrows = 0;
180  for(i = 0; i <= lp->rows; i++) {
181  row_map[i] = i-nrows;
182  /* Increment eliminated row counter if necessary */
183  if(!includeMDO(usedpos, i))
184  nrows++;
185  }
186  nrows = lp->rows+1 - nrows;
187 
188  /* Store row indeces of non-zero values in the basic columns */
189  Blen = colamd_recommended(Bnz, nrows, ncols);
190  allocINT(lp, &Brows, Blen, FALSE);
191  prepareMDO(lp, usedpos, colorder, Brows, row_map);
192 #ifdef Paranoia
193  verifyMDO(lp, col_end, Brows, nrows, ncols);
194 #endif
195 
196  /* Compute the MDO */
197 #if 1
198  colamd_set_defaults(knobs);
199  knobs [COLAMD_DENSE_ROW] = 0.2+0.2 ; /* default changed for UMFPACK */
200  knobs [COLAMD_DENSE_COL] = knobs [COLAMD_DENSE_ROW];
201  if(symmetric && (nrows == ncols)) {
202  MEMCOPY(colorder, Brows, ncols + 1);
203  error = !symamd(nrows, colorder, col_end, Brows, knobs, stats, mdo_calloc, mdo_free);
204  }
205  else
206  error = !colamd(nrows, ncols, Blen, Brows, col_end, knobs, stats);
207 #else
208  if(symmetric && (nrows == ncols)) {
209  MEMCOPY(colorder, Brows, ncols + 1);
210  error = !symamd(nrows, colorder, col_end, Brows, knobs, stats, mdo_calloc, mdo_free);
211  }
212  else
213  error = !colamd(nrows, ncols, Blen, Brows, col_end, (double *) NULL, stats);
214 #endif
215 
216  /* Transfer the estimated optimal ordering, adjusting for index offsets */
217 Transfer:
218  if(error)
219  error = stats[COLAMD_STATUS];
220  else {
221  MEMCOPY(Brows, colorder, ncols + 1);
222  for(j = 0; j < ncols; j++) {
223  kk = col_end[j];
224  n = Brows[kk+1];
225  colorder[j+1] = n;
226  }
227  }
228 
229  /* Free temporary vectors */
230  FREE(col_end);
231  if(row_map != NULL)
232  FREE(row_map);
233  if(Brows != NULL)
234  FREE(Brows);
235 
236  if(size != NULL)
237  *size = ncols;
238  return( error );
239 }
