126 {
127 #if defined _POSIX_THREAD_PRIORITY_SCHEDULING && _POSIX_THREAD_PRIORITY_SCHEDULING >= 0
128  int policy = 0;
129  struct sched_param param;
130 
131 #if _POSIX_THREAD_PRIORITY_SCHEDULING == 0 && defined _SC_THREAD_PRIORITY_SCHEDULING
132  if (sysconf(_SC_THREAD_PRIORITY_SCHEDULING) < 0) {
133  return;
134  }
135 #endif
136  int rc = pthread_getschedparam(descriptor, &policy, &param);
137  posix_assert (rc);
138 
139  if(priority_ != -1)
140  {
141  param.sched_priority = priority_;
142  }
143 
144  if(schedulingPolicy_ != -1)
145  {
146  policy = schedulingPolicy_;
147  }
148 
149 #ifdef __NetBSD__
150  if(policy == SCHED_OTHER) param.sched_priority = -1;
151 #endif
152 
153  rc = pthread_setschedparam(descriptor, policy, &param);
154 
155 #ifdef __FreeBSD_kernel__
156  // If this feature is unavailable at run-time, don't abort.
157  if(rc == ENOSYS) return;
158 #endif
159 
160  posix_assert (rc);
161 #else
162 
163  LIBZMQ_UNUSED (priority_);
164  LIBZMQ_UNUSED (schedulingPolicy_);
