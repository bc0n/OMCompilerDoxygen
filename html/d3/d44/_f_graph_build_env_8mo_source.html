189 public function mkModNode
190  input Name inName "a name for this mod so we can call it from sub-mods";
191  input SCode.Mod inMod;
192  input Ref inParentRef;
193  input Kind inKind;
194  input Graph inGraph;
195  output Graph outGraph;
196 algorithm
197  outGraph := matchcontinue(inName, inMod, inParentRef, inKind, inGraph)
198  local
199  Name name;
200  Graph g;
201  Node n;
202  Ref nr;
203  SCode.Element e;
204  list<SCode.SubMod> sm;
205  Option<Absyn.Exp> b;
206 
207  // no mods
208  case (_, SCode.NOMOD(), _, _, g) then g;
209 
210  // no binding no sub-mods
211  case (_, SCode.MOD(subModLst = {}, binding = NONE()), _, _, g)
212  then
213  g;
214 
215  // just a binding
216  case (name, SCode.MOD(subModLst = {}, binding = b as SOME(_)), _, _, g)
217  equation
218  (g, n) = FGraph.node(g, name, {inParentRef}, FCore.MO(inMod));
219  nr = FNode.toRef(n);
220  FNode.addChildRef(inParentRef, name, nr);
221  g = mkBindingNode(b, nr, inKind, g);
222  then
223  g;
224 
225  // yeha, some mods for sure and a possible binding
226  case (name, SCode.MOD(subModLst = sm, binding = b), _, _, g)
227  equation
228  (g, n) = FGraph.node(g, name, {inParentRef}, FCore.MO(inMod));
229  nr = FNode.toRef(n);
230  FNode.addChildRef(inParentRef, name, nr);
231  g = mkSubMods(sm, nr, inKind, g);
232  g = mkBindingNode(b, nr, inKind, g);
233  then
234  g;
235 
236  // ouch, a redeclare :)
237  case (name, SCode.REDECL(element = e), _, _, g)
238  equation
239  (g, n) = FGraph.node(g, name, {inParentRef}, FCore.MO(inMod));
240  nr = FNode.toRef(n);
241  FNode.addChildRef(inParentRef, name, nr);
242  g = mkElementNode(e, nr, inKind, g);
243  then
244  g;
245 
246  // something bad happened!
247  case (name, _, _, _, g)
248  equation
249  print("FGraphBuildEnv.mkModNode failed with: " + name + " mod: " + SCodeDump.printModStr(inMod, SCodeDump.defaultOptions) + "\n");
250  then
251  g;
252 
253  end matchcontinue;
254 end mkModNode;

313 protected function mkClassChildren
314 "Extends the graph with a class's components."
315  input SCode.ClassDef inClassDef;
316  input Ref inParentRef;
317  input Kind inKind;
318  input Graph inGraph;
319  output Graph outGraph;
320 algorithm
321  outGraph := matchcontinue(inClassDef, inParentRef, inKind, inGraph)
322  local
323  list<SCode.Element> el;
324  Graph g;
325  SCode.Element c;
326  SCode.ClassDef cdef;
327  Node n;
328  Ref nr;
329  Absyn.TypeSpec ts;
330  Absyn.Path p;
331  Name name;
332  SCode.Mod m;
333  Absyn.ArrayDim ad;
334  list<SCode.Equation> eqs, ieqs;
335  list<SCode.AlgorithmSection> als, ials;
336  list<SCode.ConstraintSection> constraintLst;
337  list<Absyn.NamedArg> clsattrs;
338  Option<SCode.ExternalDecl> externalDecl;
339  list<Absyn.Path> pathlst;
340 
341  case (SCode.PARTS(
342  elementLst = el,
343  normalEquationLst = eqs,
344  initialEquationLst = ieqs,
345  normalAlgorithmLst = als,
346  initialAlgorithmLst = ials,
347  constraintLst = constraintLst,
348  clsattrs = clsattrs,
349  externalDecl = externalDecl
350  ), _, _, g)
351  equation
352  g = List.fold2(el, mkElementNode, inParentRef, inKind, g);
353  g = mkEqNode(FNode.eqNodeName, eqs, inParentRef, inKind, g);
354  g = mkEqNode(FNode.ieqNodeName, ieqs, inParentRef, inKind, g);
355  g = mkAlNode(FNode.alNodeName, als, inParentRef, inKind, g);
356  g = mkAlNode(FNode.ialNodeName, ials, inParentRef, inKind, g);
357  g = mkOptNode(FNode.optNodeName, constraintLst, clsattrs, inParentRef, inKind, g);
358  g = mkExternalNode(FNode.edNodeName, externalDecl, inParentRef, inKind, g);
359  then
360  g;
361 
362  case (SCode.CLASS_EXTENDS(composition = cdef, modifications = m), _, _, g)
363  equation
364  g = mkClassChildren(cdef, inParentRef, inKind, g);
365  g = mkModNode(FNode.modNodeName, m, inParentRef, inKind, g);
366  g = mkRefNode(FNode.refNodeName, {}, inParentRef, g);
367  then
368  g;
369 
370  case (SCode.DERIVED(typeSpec = ts, modifications = m), _, _, g)
371  equation
372  _ = Absyn.typeSpecPath(ts);
373  nr = inParentRef;
374  g = mkModNode(FNode.modNodeName, m, nr, inKind, g);
375  ad = Absyn.typeSpecDimensions(ts);
376  g = mkDimsNode(FNode.tydimsNodeName, SOME(ad), nr, inKind, g);
377  g = mkRefNode(FNode.refNodeName, {}, nr, g);
378  then
379  g;
380 
381  case (SCode.OVERLOAD(_), _, _, g)
382  equation
383  then
384  g;
385 
386  case (SCode.PDER(_, _), _, _, g)
387  equation
388  then
389  g;
390 
391  else inGraph;
392 
393  end matchcontinue;
394 end mkClassChildren;

396 public function mkElementNode
397 "Extends the graph with an element."
398  input SCode.Element inElement;
399  input Ref inParentRef;
400  input Kind inKind;
401  input Graph inGraph;
402  output Graph outGraph;
403 algorithm
404  outGraph := match(inElement, inParentRef, inKind, inGraph)
405  local
406  Graph g;
407  SCode.Ident name;
408  Absyn.Path p;
409  Node n;
410  Absyn.TypeSpec ts;
411  Ref nr;
412  SCode.Mod m;
413 
414  // component
415  case (SCode.COMPONENT(), _, _, g)
416  equation
417  g = mkCompNode(inElement, inParentRef, inKind, g);
418  then
419  g;
420 
421  // class
422  case (SCode.CLASS(), _, _, g)
423  equation
424  g = mkClassNode(inElement, Prefix.NOPRE(), DAE.NOMOD(), inParentRef, inKind, g);
425  then
426  g;
427 
428  case (SCode.EXTENDS(baseClassPath = p, modifications = m), _, _, g)
429  equation
430  // the extends is saved as a child with the extends name
431  name = FNode.mkExtendsName(p);
432  (g, n) = FGraph.node(g, name, {inParentRef}, FCore.EX(inElement, DAE.NOMOD()));
433  nr = FNode.toRef(n);
434  FNode.addChildRef(inParentRef, name, nr);
435  g = mkModNode(FNode.modNodeName, m, nr, inKind, g);
436  g = mkRefNode(FNode.refNodeName, {}, nr, g);
437  then
438  g;
439 
440  case (SCode.IMPORT(), _, _, g)
441  equation
442  g = mkImportNode(inElement, inParentRef, inKind, g);
443  then
444  g;
445 
446  case (SCode.DEFINEUNIT(), _, _, g)
447  equation
448  g = mkUnitsNode(inElement, inParentRef, inKind, g);
449  then
450  g;
451 
452  end match;
453 end mkElementNode;
