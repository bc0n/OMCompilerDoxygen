66  bool
67  InexactDoglegNormalStep::ComputeNormalStep(SmartPtr<Vector>& normal_x,
68  SmartPtr<Vector>& normal_s)
69 
70  {
71  DBG_START_METH("InexactDoglegNormalStep::ComputeNormalStep",
72  dbg_verbosity);
73 
74  // test if we should increase the trust region factor
75  if (!last_tr_inactive_ && InexData().full_step_accepted()) {
76  if (curr_omega_ >= omega_max_) {
77  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM,
78  "Trust region radius factor would be increased, but it is already at its upper limit %e.\n", curr_omega_);
79  IpData().Append_info_string("O");
80  }
81  else {
82  Number omega_old = curr_omega_;
83  curr_omega_ = Min(omega_max_, 10.*curr_omega_);
84  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM,
85  "Increasing trust region factor from %e to %e\n.", omega_old, curr_omega_);
86  IpData().Append_info_string("o");
87  }
88  }
89  last_tr_inactive_ = false;
90 
91  // TODO if (IpCq().curr_primal_infeasibility(NORM_2) == 0.) {
92  if (IpCq().curr_primal_infeasibility(NORM_2) <= 1e-12 ) {
93  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM, "Dogleg step: We are at a feasible point, the normal step is set to zero.\n");
94  normal_x = IpData().curr()->x()->MakeNew();
95  normal_s = IpData().curr()->s()->MakeNew();
96  normal_x->Set(0.);
97  normal_s->Set(0.);
98  last_tr_inactive_ = true;
99  return true;
100  }
101 
103 
104  // Compute the negative of the steepest descent direction.
105  // (scaled constraint Jacobian transpose times constraint values)
106  SmartPtr<const Vector> curr_jac_cdT_times_curr_cdminuss =
107  InexCq().curr_jac_cdT_times_curr_cdminuss();
108  SmartPtr<const Vector> curr_slack_scaled_d_minus_s =
109  InexCq().curr_slack_scaled_d_minus_s();
110 
111  DBG_PRINT_VECTOR(1, "curr_jac_cdT_times_curr_cdminuss", *curr_jac_cdT_times_curr_cdminuss);
112  DBG_PRINT_VECTOR(1, "curr_slack_scaled_d_minus_s", *curr_slack_scaled_d_minus_s);
113 
114  // Compute the norm of the (scaled) gradient of the objective
115  // function (A^T c)
116  Number v_ATc_norm = InexCq().curr_scaled_Ac_norm();
117 
118  // Compute A * A^T * c
119  SmartPtr<const Vector> vec_AATc_c =
120  IpCq().curr_jac_c_times_vec(*curr_jac_cdT_times_curr_cdminuss);
121  SmartPtr<Vector> vec_AATc_d = curr_slack_scaled_d_minus_s->MakeNewCopy();
122  vec_AATc_d->ElementWiseMultiply(*InexCq().curr_scaling_slacks());
123  DBG_PRINT_VECTOR(1, "curr_scaling_slacks", *InexCq().curr_scaling_slacks());
124  DBG_PRINT_VECTOR(1, "vec_AATc_d", *vec_AATc_d);
125  vec_AATc_d->AddOneVector(1., *IpCq().curr_jac_d_times_vec(*curr_jac_cdT_times_curr_cdminuss) , 1.);
126  DBG_PRINT_VECTOR(1, "IpCq().curr_jac_d_times_vec(*curr_jac_cdT_times_curr_cdminuss)", *IpCq().curr_jac_d_times_vec(*curr_jac_cdT_times_curr_cdminuss));
127  DBG_PRINT_VECTOR(1, "vec_AATc_c", *vec_AATc_c);
128  DBG_PRINT_VECTOR(1, "vec_AATc_d", *vec_AATc_d);
129  Number AATc_norm = IpCq().CalcNormOfType(NORM_2, *vec_AATc_c, *vec_AATc_d);
130 
131  // Compute the step size for the Cauchy step
132  Number alpha_cs = Min(curr_omega_, v_ATc_norm*v_ATc_norm/(AATc_norm*AATc_norm));
133  Jnlst().Printf(J_MOREDETAILED, J_SOLVE_PD_SYSTEM,
134  "Dogleg step: Cauchy step size alpha_cs = %e\n", alpha_cs);
135  DBG_PRINT((1, "alpha_cs = %e v_ATc_norm = %e AATc_norm = %e\n", alpha_cs, v_ATc_norm,AATc_norm));
136 
137  // Finally get the Cauchy step
138  SmartPtr<Vector> v_cauchy_x =
139  curr_jac_cdT_times_curr_cdminuss->MakeNewCopy();
140  SmartPtr<Vector> v_cauchy_s =
141  curr_slack_scaled_d_minus_s->MakeNewCopy();
142  v_cauchy_x->Scal(-alpha_cs);
143  v_cauchy_s->Scal(alpha_cs);
144 
145  // output
146  if (Jnlst().ProduceOutput(J_MOREVECTOR, J_SOLVE_PD_SYSTEM)) {
147  Jnlst().Printf(J_MOREVECTOR, J_SOLVE_PD_SYSTEM,
148  "Dogleg step: Cauchy step:\n");
149  v_cauchy_x->Print(Jnlst(), J_MOREVECTOR, J_SOLVE_PD_SYSTEM, "v_cauchy_x");
150  v_cauchy_s->Print(Jnlst(), J_MOREVECTOR, J_SOLVE_PD_SYSTEM, "v_cauchy_s");
151  }
152 
153  // Compute the objective function reduction of the normal problem
154  // for the Cauchy step
155  SmartPtr<const Vector> curr_c = IpCq().curr_c();
156  SmartPtr<const Vector> curr_d_minus_s = IpCq().curr_d_minus_s();
157  SmartPtr<Vector> inf_c = curr_c->MakeNew();
158  SmartPtr<Vector> inf_d = curr_d_minus_s->MakeNew();
159  inf_c->AddTwoVectors(1., *curr_c, -alpha_cs, *vec_AATc_c, 0.);
160  inf_d->AddTwoVectors(1., *curr_d_minus_s, -alpha_cs, *vec_AATc_d, 0.);
161  Number c_Avc_norm_cauchy = IpCq().CalcNormOfType(NORM_2, *inf_c, *inf_d);
162  if (IsValid(normal_tester_)) {
163  normal_tester_->Set_c_Avc_norm_cauchy(c_Avc_norm_cauchy);
164  }
165  Number objred_normal_cs = 0.5*(IpCq().CalcNormOfType(NORM_2, *curr_c, *curr_d_minus_s)-c_Avc_norm_cauchy);
166  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM,
167  "Dogleg: Reduction of normal problem objective function by Cauchy step = %23.16e\n", objred_normal_cs);
168 
169  // If the Cauchy step already hits the trust region, we are done
170  if (alpha_cs == curr_omega_) {
171  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM, "Dogleg step: Cauchy step already hits trust region.\n");
172  normal_x = v_cauchy_x;
173  normal_s = v_cauchy_s;
174  // unscale the slack-based scaling
175  normal_s->ElementWiseMultiply(*InexCq().curr_scaling_slacks());
176  IpData().Append_info_string("Nc ");
177  return true;
178  }
179  // ToDo: We don't need this if we do a proper check for Newton step below
180  SmartPtr<Vector> v_cauchy_x_bak = v_cauchy_x->MakeNewCopy();
181  SmartPtr<Vector> v_cauchy_s_bak = v_cauchy_s->MakeNewCopy();
182 
184 
185  SmartPtr<Vector> v_newton_x = v_cauchy_x->MakeNew();
186  SmartPtr<Vector> v_newton_s = v_cauchy_s->MakeNew();
187  bool retval =
188  newton_step_->ComputeNewtonNormalStep(*v_newton_x, *v_newton_s);
189  if (!retval) {
190  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM, "Dogleg step: Newton step could not be calculated, return Cauchy step.\n");
191  normal_x = v_cauchy_x_bak;
192  normal_s = v_cauchy_s_bak;
193  // unscale the slack-based scaling
194  normal_s->ElementWiseMultiply(*InexCq().curr_scaling_slacks());
195  IpData().Append_info_string("NF ");
196  return true;
197  }
198  // output
199  if (Jnlst().ProduceOutput(J_MOREVECTOR, J_SOLVE_PD_SYSTEM)) {
200  Jnlst().Printf(J_MOREVECTOR, J_SOLVE_PD_SYSTEM,
201  "Dogleg step: Newton step:\n");
202  v_newton_x->Print(Jnlst(), J_MOREVECTOR, J_SOLVE_PD_SYSTEM, "v_newton_x");
203  v_newton_s->Print(Jnlst(), J_MOREVECTOR, J_SOLVE_PD_SYSTEM, "v_newton_s");
204  }
205 
207 
208  // Compute the trust region radius
209  const Number tr_radius = curr_omega_ * v_ATc_norm;
210 
211  // norm of the Newton step
212  Number v_newton_norm = IpCq().CalcNormOfType(NORM_2,
213  *v_newton_x, *v_newton_s);
214  Jnlst().Printf(J_MOREDETAILED, J_SOLVE_PD_SYSTEM,
215  "Norm of Newton step = %e, trust region radius = %e\n",
216  v_newton_norm, tr_radius);
217  if (v_newton_norm <= tr_radius) {
218  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM, "Dogleg step: Newton step is within trust region.\n");
219  normal_x = v_newton_x;
220  normal_s = v_newton_s;
221  last_tr_inactive_ = true;
222  IpData().Append_info_string("Nn ");
223  }
224  else {
225  Number v_cauchy_norm = IpCq().CalcNormOfType(NORM_2,
226  *v_cauchy_x, *v_cauchy_s);
227  Number v_cs_dot_n = v_newton_x->Dot(*v_cauchy_x) +
228  v_newton_s->Dot(*v_cauchy_s);
229  Number a = v_newton_norm*v_newton_norm - 2*v_cs_dot_n +
230  v_cauchy_norm*v_cauchy_norm;
231  Number b = 2*(v_cs_dot_n - v_newton_norm*v_newton_norm);
232  Number c = v_newton_norm*v_newton_norm - tr_radius*tr_radius;
233  Number lambda = (-b-sqrt(b*b-4.*a*c))/(2.*a);
234 
235  DBG_PRINT((1, "v_cauchy_norm = %e v_cs_dot_n = %e v_newton_norm = %e\n",v_cauchy_norm,v_cs_dot_n,v_newton_norm));
236 
237  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM, "Dogleg step: Using convex combination of Cauchy and Newton step with factor lambda = %e\n", lambda);
238  v_cauchy_x->AddOneVector(1.-lambda, *v_newton_x, lambda);
239  v_cauchy_s->AddOneVector(1.-lambda, *v_newton_s, lambda);
240  normal_x = v_cauchy_x;
241  normal_s = v_cauchy_s;
242  IpData().Append_info_string("Nd ");
243 
244  DBG_PRINT((1, "v_normal^2 = %e\n",normal_x->Dot(*normal_x)+normal_s->Dot(*normal_s)));
245  }
246 
247  DBG_PRINT_VECTOR(1, "normal_x scaled", *normal_x);
248  DBG_PRINT_VECTOR(1, "normal_s scaled", *normal_s);
249 
250  // Compute the unscaled steps
251  normal_s->ElementWiseMultiply(*InexCq().curr_scaling_slacks());
252  v_cauchy_s_bak->ElementWiseMultiply(*InexCq().curr_scaling_slacks());
253 
254  // We now check if the Dogleg step, shorted by the
255  // fraction-to-the-boundary rule, gives at least as much progress
256  // as the Cauchy step, also shortened by the
257  // fraction-to-the-boundary rule. If not, we throw away the
258  // Newton step component.
259 
260  // TODO: Implement efficiently
261  const Number tau = IpData().curr_tau();
262  Number ftb_cauchy =
263  IpCq().primal_frac_to_the_bound(tau, *v_cauchy_x_bak, *v_cauchy_s_bak);
264  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM,
265  "Dogleg: Fraction-to-the-bounary step size for Cauchy step = %23.16e\n", ftb_cauchy);
266  inf_c = IpCq().curr_jac_c_times_vec(*v_cauchy_x_bak)->MakeNewCopy();
267  inf_c->AddOneVector(1., *curr_c, ftb_cauchy);
268  inf_d = curr_d_minus_s->MakeNewCopy();
269  inf_d->AddTwoVectors(-ftb_cauchy, *v_cauchy_s_bak,
270  ftb_cauchy, *IpCq().curr_jac_d_times_vec(*v_cauchy_x_bak) , 1.);
271  Number objred_ftb_cauchy = 0.5*(IpCq().CalcNormOfType(NORM_2, *curr_c, *curr_d_minus_s)-IpCq().CalcNormOfType(NORM_2, *inf_c, *inf_d));
272  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM,
273  "Dogleg: Reduction of normal problem objective function by ftb cauchy step = %23.16e\n", objred_ftb_cauchy);
274 
275  Number ftb_dogleg =
276  IpCq().primal_frac_to_the_bound(tau, *normal_x, *normal_s);
277  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM,
278  "Dogleg: Fraction-to-the-bounary step size for Dogleg step = %23.16e\n", ftb_dogleg);
279  inf_c = IpCq().curr_jac_c_times_vec(*normal_x)->MakeNewCopy();
280  inf_c->AddOneVector(1., *curr_c, ftb_dogleg);
281  inf_d = curr_d_minus_s->MakeNewCopy();
282  inf_d->AddTwoVectors(-ftb_dogleg, *normal_s,
283  ftb_dogleg, *IpCq().curr_jac_d_times_vec(*normal_x) , 1.);
284  Number objred_ftb_dogleg = 0.5*(IpCq().CalcNormOfType(NORM_2, *curr_c, *curr_d_minus_s)-IpCq().CalcNormOfType(NORM_2, *inf_c, *inf_d));
285  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM,
286  "Dogleg: Reduction of normal problem objective function by ftb dogleg step = %23.16e\n", objred_ftb_dogleg);
287 
288  Number rhs = 10.*objred_ftb_dogleg;
289  Number lhs = objred_ftb_cauchy;
290  Number BasVal = curr_c->Nrm2()+curr_d_minus_s->Nrm2();
291  bool ok = Compare_le(lhs, rhs, BasVal);
292  if (!ok) {
293  Jnlst().Printf(J_DETAILED, J_SOLVE_PD_SYSTEM, "Dogleg step: Dogleg step makes less progress than Cauchy step, resetting to Cauchy step.\n");
294  normal_x = v_cauchy_x_bak;
295  normal_s = v_cauchy_s_bak;
296  IpData().Append_info_string("NR ");
297  }
298 
299  return true;
300  }
