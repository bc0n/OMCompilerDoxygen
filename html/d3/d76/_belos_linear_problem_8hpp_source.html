637  template <class ScalarType, class MV, class OP>
638  void LinearProblem<ScalarType,MV,OP>::setLSIndex(const std::vector<int>& index)
639  {
640  // Set new linear systems using the indices in index.
641  rhsIndex_ = index;
642 
643  // Compute the new block linear system.
644  // ( first clean up old linear system )
645  curB_ = Teuchos::null;
646  curX_ = Teuchos::null;
647 
648  // Create indices for the new linear system.
649  int validIdx = 0, ivalidIdx = 0;
650  blocksize_ = rhsIndex_.size();
651  std::vector<int> vldIndex( blocksize_ );
652  std::vector<int> newIndex( blocksize_ );
653  std::vector<int> iIndex( blocksize_ );
654  for (int i=0; i<blocksize_; ++i) {
655  if (rhsIndex_[i] > -1) {
656  vldIndex[validIdx] = rhsIndex_[i];
657  newIndex[validIdx] = i;
658  validIdx++;
659  }
660  else {
661  iIndex[ivalidIdx] = i;
662  ivalidIdx++;
663  }
664  }
665  vldIndex.resize(validIdx);
666  newIndex.resize(validIdx);
667  iIndex.resize(ivalidIdx);
668  num2Solve_ = validIdx;
669 
670  // Create the new linear system using index
671  if (num2Solve_ != blocksize_) {
672  newIndex.resize(num2Solve_);
673  vldIndex.resize(num2Solve_);
674  //
675  // First create multivectors of blocksize.
676  // Fill the RHS with random vectors LHS with zero vectors.
677  curX_ = MVT::Clone( *X_, blocksize_ );
678  MVT::MvInit(*curX_);
679  Teuchos::RCP<MV> tmpCurB = MVT::Clone( *B_, blocksize_ );
680  MVT::MvRandom(*tmpCurB);
681  //
682  // Now put in the part of B into curB
683  Teuchos::RCP<const MV> tptr = MVT::CloneView( *B_, vldIndex );
684  MVT::SetBlock( *tptr, newIndex, *tmpCurB );
685  curB_ = tmpCurB;
686  //
687  // Now put in the part of X into curX
688  tptr = MVT::CloneView( *X_, vldIndex );
689  MVT::SetBlock( *tptr, newIndex, *curX_ );
690  //
691  solutionUpdated_ = false;
692  }
693  else {
694  curX_ = MVT::CloneViewNonConst( *X_, rhsIndex_ );
695  curB_ = MVT::CloneView( *B_, rhsIndex_ );
696  }
697  //
698  // Increment the number of linear systems that have been loaded into this object.
699  //
700  lsNum_++;
701  }

704  template <class ScalarType, class MV, class OP>
705  void LinearProblem<ScalarType,MV,OP>::setCurrLS()
706  {
707  //
708  // We only need to copy the solutions back if the linear systems of
709  // interest are less than the block size.
710  //
711  if (num2Solve_ < blocksize_) {
712  //
713  // Get a view of the current solutions and correction std::vector.
714  //
715  int validIdx = 0;
716  std::vector<int> newIndex( num2Solve_ );
717  std::vector<int> vldIndex( num2Solve_ );
718  for (int i=0; i<blocksize_; ++i) {
719  if ( rhsIndex_[i] > -1 ) {
720  vldIndex[validIdx] = rhsIndex_[i];
721  newIndex[validIdx] = i;
722  validIdx++;
723  }
724  }
725  Teuchos::RCP<MV> tptr = MVT::CloneViewNonConst( *curX_, newIndex );
726  MVT::SetBlock( *tptr, vldIndex, *X_ );
727  }
728  //
729  // Clear the current vectors of this linear system so that any future calls
730  // to get the vectors for this system return null pointers.
731  //
732  curX_ = Teuchos::null;
733  curB_ = Teuchos::null;
734  rhsIndex_.resize(0);
735  }

738  template <class ScalarType, class MV, class OP>
739  Teuchos::RCP<MV>
740  LinearProblem<ScalarType,MV,OP>::
741  updateSolution (const Teuchos::RCP<MV>& update,
742  bool updateLP,
743  ScalarType scale)
744  {
745  using Teuchos::RCP;
746  using Teuchos::null;
747 
748  RCP<MV> newSoln;
749  if (update.is_null())
750  { // The caller didn't supply an update vector, so we assume
751  // that the current solution curX_ is unchanged, and return a
752  // pointer to it.
753  newSoln = curX_;
754  }
755  else // the update vector is NOT null.
756  {
757  if (updateLP) // The caller wants us to update the linear problem.
758  {
759  if (RP_.is_null())
760  { // There is no right preconditioner.
761  // curX_ := curX_ + scale * update.
762  MVT::MvAddMv( 1.0, *curX_, scale, *update, *curX_ );
763  }
764  else
765  { // There is indeed a right preconditioner, so apply it
766  // before computing the new solution.
767  RCP<MV> rightPrecUpdate =
768  MVT::Clone (*update, MVT::GetNumberVecs (*update));
769  {
770 #ifdef BELOS_TEUCHOS_TIME_MONITOR
771  Teuchos::TimeMonitor PrecTimer (*timerPrec_);
772 #endif
773  OPT::Apply( *RP_, *update, *rightPrecUpdate );
774  }
775  // curX_ := curX_ + scale * rightPrecUpdate.
776  MVT::MvAddMv( 1.0, *curX_, scale, *rightPrecUpdate, *curX_ );
777  }
778  solutionUpdated_ = true;
779  newSoln = curX_;
780  }
781  else
782  { // Rather than updating our stored current solution curX_,
783  // we make a copy and compute the new solution in the
784  // copy, without modifying curX_.
785  newSoln = MVT::Clone (*update, MVT::GetNumberVecs (*update));
786  if (RP_.is_null())
787  { // There is no right preconditioner.
788  // newSoln := curX_ + scale * update.
789  MVT::MvAddMv( 1.0, *curX_, scale, *update, *newSoln );
790  }
791  else
792  { // There is indeed a right preconditioner, so apply it
793  // before computing the new solution.
794  RCP<MV> rightPrecUpdate =
795  MVT::Clone (*update, MVT::GetNumberVecs (*update));
796  {
797 #ifdef BELOS_TEUCHOS_TIME_MONITOR
798  Teuchos::TimeMonitor PrecTimer(*timerPrec_);
799 #endif
800  OPT::Apply( *RP_, *update, *rightPrecUpdate );
801  }
802  // newSoln := curX_ + scale * rightPrecUpdate.
803  MVT::MvAddMv( 1.0, *curX_, scale, *rightPrecUpdate, *newSoln );
804  }
805  }
806  }
807  return newSoln;
808  }

831  template <class ScalarType, class MV, class OP>
832  bool
833  LinearProblem<ScalarType,MV,OP>::
834  setProblem (const Teuchos::RCP<MV> &newX,
835  const Teuchos::RCP<const MV> &newB)
836  {
837  // Create timers if the haven't been created yet.
838  if (timerOp_ == Teuchos::null) {
839  std::string opLabel = label_ + ": Operation Op*x";
840 #ifdef BELOS_TEUCHOS_TIME_MONITOR
841  timerOp_ = Teuchos::TimeMonitor::getNewCounter( opLabel );
842 #endif
843  }
844  if (timerPrec_ == Teuchos::null) {
845  std::string precLabel = label_ + ": Operation Prec*x";
846 #ifdef BELOS_TEUCHOS_TIME_MONITOR
847  timerPrec_ = Teuchos::TimeMonitor::getNewCounter( precLabel );
848 #endif
849  }
850 
851  // Set the linear system using the arguments newX and newB
852  if (newX != Teuchos::null)
853  X_ = newX;
854  if (newB != Teuchos::null)
855  B_ = newB;
856 
857  // Invalidate the current linear system indices and multivectors.
858  rhsIndex_.resize(0);
859  curX_ = Teuchos::null;
860  curB_ = Teuchos::null;
861 
862  // If we didn't set a matrix A, a left-hand side X, or a
863  // right-hand side B, then we didn't set the problem.
864  if (A_ == Teuchos::null || X_ == Teuchos::null || B_ == Teuchos::null) {
865  isSet_ = false;
866  return isSet_;
867  }
868 
869  // Reset whether the solution has been updated. (We're just
870  // setting the problem now, so of course we haven't updated the
871  // solution yet.)
872  solutionUpdated_ = false;
873 
874  // Compute the initial residuals.
875  if(Teuchos::is_null(R0_user_)) {
876  if (R0_==Teuchos::null || MVT::GetNumberVecs( *R0_ )!=MVT::GetNumberVecs( *B_ )) {
877  R0_ = MVT::Clone( *B_, MVT::GetNumberVecs( *B_ ) );
878  }
879  computeCurrResVec( &*R0_, &*X_, &*B_ );
880 
881  if (LP_!=Teuchos::null) {
882  if (PR0_==Teuchos::null || (PR0_==R0_) || (MVT::GetNumberVecs(*PR0_)!=MVT::GetNumberVecs(*B_))) {
883  PR0_ = MVT::Clone( *B_, MVT::GetNumberVecs( *B_ ) );
884  }
885  {
886 #ifdef BELOS_TEUCHOS_TIME_MONITOR
887  Teuchos::TimeMonitor PrecTimer(*timerPrec_);
888 #endif
889  OPT::Apply( *LP_, *R0_, *PR0_ );
890  }
891  }
892  else {
893  PR0_ = R0_;
894  }
895  }
896  else { // User specified the residuals
897  if (MVT::GetNumberVecs( *R0_user_ )!=MVT::GetNumberVecs( *B_ )) {
898  Teuchos::RCP<MV> helper = MVT::Clone( *B_, MVT::GetNumberVecs( *B_ ) );
899  computeCurrResVec( &*helper, &*X_, &*B_ );
900  R0_user_ = helper;
901  }
902 
903  if (LP_!=Teuchos::null) {
904  if (PR0_user_==Teuchos::null || (PR0_user_==R0_) || (MVT::GetNumberVecs(*PR0_user_)!=MVT::GetNumberVecs(*B_))) {
905  Teuchos::RCP<MV> helper = MVT::Clone( *B_, MVT::GetNumberVecs( *B_ ) );
906  {
907 #ifdef BELOS_TEUCHOS_TIME_MONITOR
908  Teuchos::TimeMonitor PrecTimer(*timerPrec_);
909 #endif
910  OPT::Apply( *LP_, *R0_user_, *helper );
911  }
912  PR0_user_ = helper;
913  }
914  }
915  else {
916  PR0_user_ = R0_user_;
917  }
918  }
919 
920  // The problem has been set and is ready for use.
921  isSet_ = true;
922 
923  // Return isSet.
924  return isSet_;
925  }

967  template <class ScalarType, class MV, class OP>
968  void LinearProblem<ScalarType,MV,OP>::apply( const MV& x, MV& y ) const
969  {
970  using Teuchos::null;
971  using Teuchos::RCP;
972 
973  const bool leftPrec = LP_ != null;
974  const bool rightPrec = RP_ != null;
975 
976  // We only need a temporary vector for intermediate results if
977  // there is a left or right preconditioner. We really should just
978  // keep this temporary vector around instead of allocating it each
979  // time.
980  RCP<MV> ytemp = (leftPrec || rightPrec) ? MVT::Clone (y, MVT::GetNumberVecs (y)) : null;
981 
982  //
983  // No preconditioning.
984  //
985  if (!leftPrec && !rightPrec){
986 #ifdef BELOS_TEUCHOS_TIME_MONITOR
987  Teuchos::TimeMonitor OpTimer(*timerOp_);
988 #endif
989  OPT::Apply( *A_, x, y );
990  }
991  //
992  // Preconditioning is being done on both sides
993  //
994  else if( leftPrec && rightPrec )
995  {
996  {
997 #ifdef BELOS_TEUCHOS_TIME_MONITOR
998  Teuchos::TimeMonitor PrecTimer(*timerPrec_);
999 #endif
1000  OPT::Apply( *RP_, x, y );
1001  }
1002  {
1003 #ifdef BELOS_TEUCHOS_TIME_MONITOR
1004  Teuchos::TimeMonitor OpTimer(*timerOp_);
1005 #endif
1006  OPT::Apply( *A_, y, *ytemp );
