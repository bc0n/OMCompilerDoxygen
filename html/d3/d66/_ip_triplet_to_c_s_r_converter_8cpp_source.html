56  Index TripletToCSRConverter::InitializeConverter(Index dim, Index nonzeros,
57  const Index* airn,
58  const Index* ajcn)
59  {
60  DBG_START_METH("TSymLinearSolver::InitializeStructure",
61  dbg_verbosity);
62 
63  DBG_ASSERT(dim>0);
64  DBG_ASSERT(nonzeros>0);
65 
66  delete[] ia_;
67  delete[] ja_;
68  delete[] ipos_first_;
69  delete[] ipos_double_triplet_;
70  delete[] ipos_double_compressed_;
71 
72  dim_ = dim;
73  nonzeros_triplet_ = nonzeros;
74 
75  // Create a list with all triplet entries
76  std::vector<TripletEntry> entry_list(nonzeros);
77  std::vector<TripletEntry>::iterator list_iterator = entry_list.begin();
78  for (Index i=0; i<nonzeros; i++) {
79  list_iterator->Set(airn[i], ajcn[i], i);
80  list_iterator++;
81  }
82  DBG_ASSERT(list_iterator == entry_list.end());
83 
84  if (DBG_VERBOSITY()>=2) {
85  for (Index i=0; i<nonzeros; i++) {
86  DBG_PRINT((2, "airn[%5d] = %5d acjn[%5d] = %5d\n", i, airn[i], i, ajcn[i]));
87  }
88  }
89 
90  // sort the list
91  std::sort(entry_list.begin(), entry_list.end());
92 
93  // Now got through the list and compute ipos_ arrays and the
94  // number of elements in the compressed format
95  Index* ja_tmp = new Index[nonzeros]; // overestimate memory requirement
96  Index* rc_tmp = NULL;
97  if (hf_ == Full_Format) {
98  rc_tmp = new Index[dim_+1];
99  }
100  ia_ = new Index[dim_+1];
101  Index* ipos_first_tmp = new Index[nonzeros]; // overestimate memory requirement
102  Index* ipos_double_triplet_tmp = new Index[nonzeros]; // overestimate memory requirement
103  Index* ipos_double_compressed_tmp = new Index[nonzeros]; // overestimate memory requirement
104 
105  Index nonzeros_compressed_full = 0;
106  nonzeros_compressed_ = 0;
107  Index cur_row = 1;
108 
109  if (hf_ == Full_Format) {
110  // Zero row counts
111  for (Index i=0; i<dim_+1; i++) {
112  rc_tmp[i] = 0;
113  }
114  }
115 
116  // Take care of possible empty rows
117  list_iterator = entry_list.begin();
118  while (cur_row < list_iterator->IRow()) {
119  ia_[cur_row-1] = 0;
120  cur_row++;
121  }
122  ia_[cur_row-1] = 0;
123  ja_tmp[0] = list_iterator->JCol();
124  ipos_first_tmp[0] = list_iterator->PosTriplet();
125  if (hf_ == Full_Format) {
126  // Count in both lower and upper triangles. Count diagonal only once.
127  nonzeros_compressed_full++;
128  rc_tmp[cur_row-1]++;
129  if (cur_row!=list_iterator->JCol()) {
130  nonzeros_compressed_full++;
131  rc_tmp[list_iterator->JCol()-1]++;
132  }
133  }
134 
135  list_iterator++;
136  Index idouble = 0;
137  Index idouble_full = 0;
138  while (list_iterator != entry_list.end()) {
139  Index irow = list_iterator->IRow();
140  Index jcol = list_iterator->JCol();
141  if (cur_row == irow && ja_tmp[nonzeros_compressed_] == jcol) {
142  // This element appears repeatedly, add to the double list
143  ipos_double_triplet_tmp[idouble] = list_iterator->PosTriplet();
144  ipos_double_compressed_tmp[idouble] = nonzeros_compressed_;
145  idouble++;
146  idouble_full++;
147  if (hf_==Full_Format && irow!=jcol) {
148  idouble_full++;
149  }
150  }
151  else {
152  // This is a new element
153  if (hf_==Full_Format) {
154  // Count in both lower and upper triangles. Count diagonal only once.
155  nonzeros_compressed_full++;
156  rc_tmp[jcol-1]++;
157  if (irow!=jcol) {
158  nonzeros_compressed_full++;
159  rc_tmp[irow-1]++;
160  }
161  }
162  nonzeros_compressed_++;
163  ja_tmp[nonzeros_compressed_] = jcol;
164  ipos_first_tmp[nonzeros_compressed_] = list_iterator->PosTriplet();
165  if (cur_row != irow) {
166  // this is in a new row
167 
168  ia_[cur_row] = nonzeros_compressed_;
169  cur_row++;
170  }
171  }
172 
173  list_iterator++;
174  }
175  nonzeros_compressed_++;
176  for (Index i=cur_row; i<=dim_; i++) {
177  ia_[i] = nonzeros_compressed_;
178  }
179  DBG_ASSERT(idouble == nonzeros_triplet_-nonzeros_compressed_);
180 
181  // Now copy the ja_tmp array to the (shorter) final one and make
182  // sure that the correct offset is used
183  if (hf_==Triangular_Format) {
184  ja_ = new Index[nonzeros_compressed_];
185  if (offset_==0) {
186  for (Index i=0; i<nonzeros_compressed_; i++) {
187  ja_[i] = ja_tmp[i] - 1;
188  }
189  }
190  else {
191  for (Index i=0; i<nonzeros_compressed_; i++) {
192  ja_[i] = ja_tmp[i];
193  }
194  for (Index i=0; i<=dim_; i++) {
195  ia_[i] = ia_[i] + 1;
196  }
197  }
198  delete[] ja_tmp;
199 
200  // Reallocate memory for the "first" array
201  ipos_first_ = new Index[nonzeros_compressed_];
202  for (Index i=0; i<nonzeros_compressed_; i++) {
203  ipos_first_[i] = ipos_first_tmp[i];
204  }
205  delete[] ipos_first_tmp;
206 
207  // Reallocate memory for the "double" arrays
208  ipos_double_triplet_ = new Index[idouble];
209  ipos_double_compressed_ = new Index[idouble];
210  for (Index i=0; i<idouble; i++) {
211  ipos_double_triplet_[i] = ipos_double_triplet_tmp[i];
212  ipos_double_compressed_[i] = ipos_double_compressed_tmp[i];
213  }
214  delete[] ipos_double_triplet_tmp;
215  delete[] ipos_double_compressed_tmp;
216  num_doubles_ = nonzeros_triplet_ - nonzeros_compressed_;
217  }
218  else { // hf_==Full_Format
219 
220  // Setup ia_tmp to contain insert position for column i as ia_tmp[i+1]
221  Index *ia_tmp = new Index[dim_+1];
222  ia_tmp[0] = 0;
223  ia_tmp[1] = 0;
224  for (Index i=1; i<dim_; i++) {
225  ia_tmp[i+1] = ia_tmp[i] + rc_tmp[i-1];
226  }
227  delete[] rc_tmp;
228 
229  // Loop over elements of matrix, copying them and duplicating as required
230  ja_ = new Index[nonzeros_compressed_full];
231  ipos_first_ = new Index[nonzeros_compressed_full];
232  ipos_double_triplet_ = new Index[idouble_full];
233  ipos_double_compressed_ = new Index[idouble_full];
234  Index jd1=0; // Entry into ipos_double_compressed_tmp
235  Index jd2=0; // Entry into ipos_double_compressed_
236  for (Index i=0; i<dim_; i++) {
237  for (Index j=ia_[i]; j<ia_[i+1]; j++) {
238  Index jrow = ja_tmp[j]-1;
239  ja_[ia_tmp[i+1]] = jrow + offset_;
240  ipos_first_[ia_tmp[i+1]] = ipos_first_tmp[j];
241  while (jd1<idouble && j==ipos_double_compressed_tmp[jd1]) {
242  ipos_double_triplet_[jd2] = ipos_double_triplet_tmp[jd1];
243  ipos_double_compressed_[jd2] = ia_tmp[i+1];
244  jd2++;
245  if (jrow!=i) {
246  ipos_double_triplet_[jd2] = ipos_double_triplet_tmp[jd1];
247  ipos_double_compressed_[jd2] = ia_tmp[jrow+1];
248  jd2++;
249  }
250  jd1++;
251  }
252  ia_tmp[i+1]++;
253  if (jrow!=i) {
254  ja_[ia_tmp[jrow+1]] = i + offset_;
255  ipos_first_[ia_tmp[jrow+1]] = ipos_first_tmp[j];
256  ia_tmp[jrow+1]++;
257  }
258  }
259  }
260  delete[] ja_tmp;
261  delete[] ipos_first_tmp;
262  delete[] ipos_double_triplet_tmp;
263  delete[] ipos_double_compressed_tmp;
264 
265  // Copy ia_tmp to ia_ with offset_ adjustment
266  for (Index i=0; i<dim_+1; i++) {
267  ia_[i] = ia_tmp[i] + offset_;
268  }
269  delete[] ia_tmp;
270 
271  // Set nonzeros_compressed_ to correct size
272  nonzeros_compressed_ = nonzeros_compressed_full;
273  num_doubles_ = idouble_full;
274  }
275 
276  initialized_ = true;
277 
278  if (DBG_VERBOSITY()>=2) {
279  for (Index i=0; i<=dim_; i++) {
280  DBG_PRINT((2, "ia[%5d] = %5d\n", i, ia_[i]));
281  }
282  for (Index i=0; i<nonzeros_compressed_; i++) {
283  DBG_PRINT((2, "ja[%5d] = %5d ipos_first[%5d] = %5d\n", i, ja_[i], i, ipos_first_[i]));
284  }
285  for (Index i=0; i<nonzeros_triplet_-nonzeros_compressed_; i++) {
286  DBG_PRINT((2, "ipos_double_triplet[%5d] = %5d ipos_double_compressed[%5d] = %5d\n", i, ipos_double_triplet_[i], i, ipos_double_compressed_[i]));
287  }
288  }
289 
290  return nonzeros_compressed_;
291  }
