19  * This is the header file for the IDABBDPRE module, for a
20  * band-block-diagonal preconditioner, i.e. a block-diagonal
21  * matrix with banded blocks, for use with IDA and
22  * IDASPGMR/IDASPBCG/IDASPTFQMR.
23  *
24  * Summary:
25  *
26  * These routines provide a preconditioner matrix that is
27  * block-diagonal with banded blocks. The blocking corresponds
28  * to the distribution of the dependent variable vector y among
29  * the processors. Each preconditioner block is generated from
30  * the Jacobian of the local part (on the current processor) of a
31  * given function G(t,y,y') approximating F(t,y,y'). The blocks
32  * are generated by a difference quotient scheme on each processor
33  * independently. This scheme utilizes an assumed banded structure
34  * with given half-bandwidths, mudq and mldq. However, the banded
35  * Jacobian block kept by the scheme has half-bandwiths mukeep and
36  * mlkeep, which may be smaller.
37  *
38  * The user's calling program should have the following form:
39  *
40  * #include <ida/ida_bbdpre.h>
41  * #include <nvector_parallel.h>
42  * ...
43  * y0 = N_VNew_Parallel(...);
44  * yp0 = N_VNew_Parallel(...);
45  * ...
46  * ida_mem = IDACreate(...);
47  * ier = IDAInit(...);
48  * ...
49  * flag = IDASptfqmr(ida_mem, maxl);
50  * -or-
51  * flag = IDASpgmr(ida_mem, maxl);
52  * -or-
53  * flag = IDASpbcg(ida_mem, maxl);
54  * ...
55  * flag = IDABBDPrecInit(ida_mem, Nlocal, mudq, mldq,
56  * mukeep, mlkeep, dq_rel_yy, Gres, Gcomm);
57  * ...
58  * ier = IDASolve(...);
59  * ...
60  * IDAFree(&ida_mem);
61  *
62  * N_VDestroy(y0);
63  * N_VDestroy(yp0);
64  *
65  * The user-supplied routines required are:
66  *
67  * res is the function F(t,y,y') defining the DAE system to
68  * be solved: F(t,y,y') = 0.
69  *
70  * Gres is the function defining a local approximation
71  * G(t,y,y') to F, for the purposes of the preconditioner.
72  *
73  * Gcomm is the function performing communication needed
74  * for Glocal.
75  *
76  * Notes:
77  *
78  * 1) This header file is included by the user for the definition
79  * of the IBBDPrecData type and for needed function prototypes.
80  *
81  * 2) The IDABBDPrecInit call includes half-bandwidths mudq and
82  * mldq to be used in the approximate Jacobian. They need
83  * not be the true half-bandwidths of the Jacobian of the
84  * local block of G, when smaller values may provide a greater
85  * efficiency. Similarly, mukeep and mlkeep, specifying the
86  * bandwidth kept for the approximate Jacobian, need not be
87  * the true half-bandwidths. Also, mukeep, mlkeep, mudq, and
88  * mldq need not be the same on every processor.
89  *
90  * 3) The actual name of the user's res function is passed to
91  * IDAInit, and the names of the user's Gres and Gcomm
92  * functions are passed to IDABBDPrecInit.
93  *
94  * 4) The pointer to the user-defined data block user_data, which
95  * is set through IDASetUserData is also available to the user
96  * in glocal and gcomm.
97  *
98  * 5) Optional outputs specific to this module are available by
99  * way of routines listed below. These include work space sizes
100  * and the cumulative number of glocal calls. The costs
101  * associated with this module also include nsetups banded LU
102  * factorizations, nsetups gcomm calls, and nps banded
103  * backsolve calls, where nsetups and nps are integrator
104  * optional outputs.
105  * -----------------------------------------------------------------
106  */

117 /*
118  * -----------------------------------------------------------------
119  * Type : IDABBDLocalFn
120  * -----------------------------------------------------------------
121  * The user must supply a function G(t,y,y') which approximates
122  * the function F for the system F(t,y,y') = 0, and which is
123  * computed locally (without interprocess communication).
124  * (The case where G is mathematically identical to F is allowed.)
125  * The implementation of this function must have type IDABBDLocalFn.
126  *
127  * This function takes as input the independent variable value tt,
128  * the current solution vector yy, the current solution
129  * derivative vector yp, and a pointer to the user-defined data
130  * block user_data. It is to compute the local part of G(t,y,y')
131  * and store it in the vector gval. (Providing memory for yy and
132  * gval is handled within this preconditioner module.) It is
133  * expected that this routine will save communicated data in work
134  * space defined by the user, and made available to the
135  * preconditioner function for the problem. The user_data
136  * parameter is the same as that passed by the user to the
137  * IDASetUserdata routine.
138  *
139  * An IDABBDLocalFn Gres is to return an int, defined in the same
140  * way as for the residual function: 0 (success), +1 or -1 (fail).
141  * -----------------------------------------------------------------
142  */
143 
144 typedef int (*IDABBDLocalFn)(long int Nlocal, realtype tt,
145  N_Vector yy, N_Vector yp, N_Vector gval,
146  void *user_data);

148 /*
149  * -----------------------------------------------------------------
150  * Type : IDABBDCommFn
151  * -----------------------------------------------------------------
152  * The user may supply a function of type IDABBDCommFn which
153  * performs all interprocess communication necessary to
154  * evaluate the approximate system function described above.
155  *
156  * This function takes as input the solution vectors yy and yp,
157  * and a pointer to the user-defined data block user_data. The
158  * user_data parameter is the same as that passed by the user to
159  * the IDASetUserData routine.
160  *
161  * The IDABBDCommFn Gcomm is expected to save communicated data in
162  * space defined with the structure *user_data.
163  *
164  * A IDABBDCommFn Gcomm returns an int value equal to 0 (success),
165  * > 0 (recoverable error), or < 0 (unrecoverable error).
166  *
167  * Each call to the IDABBDCommFn is preceded by a call to the system
168  * function res with the same vectors yy and yp. Thus the
169  * IDABBDCommFn gcomm can omit any communications done by res if
170  * relevant to the evaluation of the local function glocal.
171  * A NULL communication function can be passed to IDABBDPrecInit
172  * if all necessary communication was done by res.
173  * -----------------------------------------------------------------
174  */
175

180 /*
181  * -----------------------------------------------------------------
182  * Function : IDABBDPrecInit
183  * -----------------------------------------------------------------
184  * IDABBDPrecInit allocates and initializes the BBD preconditioner.
185  *
186  * The parameters of IDABBDPrecInit are as follows:
187  *
188  * ida_mem is a pointer to the memory blockreturned by IDACreate.
189  *
190  * Nlocal is the length of the local block of the vectors yy etc.
191  * on the current processor.
192  *
193  * mudq, mldq are the upper and lower half-bandwidths to be used
194  * in the computation of the local Jacobian blocks.
195  *
196  * mukeep, mlkeep are the upper and lower half-bandwidths to be
197  * used in saving the Jacobian elements in the local
198  * block of the preconditioner matrix PP.
199  *
200  * dq_rel_yy is an optional input. It is the relative increment
201  * to be used in the difference quotient routine for
202  * Jacobian calculation in the preconditioner. The
203  * default is sqrt(unit roundoff), and specified by
204  * passing dq_rel_yy = 0.
205  *
206  * Gres is the name of the user-supplied function G(t,y,y')
207  * that approximates F and whose local Jacobian blocks
208  * are to form the preconditioner.
209  *
210  * Gcomm is the name of the user-defined function that performs
211  * necessary interprocess communication for the
212  * execution of glocal.
213  *
214  * The return value of IDABBDPrecInit is one of:
215  * IDASPILS_SUCCESS if no errors occurred
216  * IDASPILS_MEM_NULL if the integrator memory is NULL
217  * IDASPILS_LMEM_NULL if the linear solver memory is NULL
218  * IDASPILS_ILL_INPUT if an input has an illegal value
219  * IDASPILS_MEM_FAIL if a memory allocation request failed
220  * -----------------------------------------------------------------
221  */
