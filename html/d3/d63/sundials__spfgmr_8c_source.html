25 SpfgmrMem SpfgmrMalloc(int l_max, N_Vector vec_tmpl)
26 {
27  SpfgmrMem mem;
28  N_Vector *V, *Z, xcor, vtemp;
29  realtype **Hes, *givens, *yg;
30  int k, i;
31 
32  /* Check the input parameters. */
33  if (l_max <= 0) return(NULL);
34 
35  /* Get memory for the Krylov basis vectors V[0], ..., V[l_max]. */
36  V = N_VCloneVectorArray(l_max+1, vec_tmpl);
37  if (V == NULL) return(NULL);
38 
39  /* Get memory for the preconditioned basis vectors Z[0], ..., Z[l_max]. */
40  Z = N_VCloneVectorArray(l_max+1, vec_tmpl);
41  if (Z == NULL) {
42  N_VDestroyVectorArray(V, l_max+1);
43  return(NULL);
44  }
45 
46  /* Get memory for the Hessenberg matrix Hes. */
47  Hes = NULL;
48  Hes = (realtype **) malloc((l_max+1)*sizeof(realtype *));
49  if (Hes == NULL) {
50  N_VDestroyVectorArray(V, l_max+1);
51  N_VDestroyVectorArray(Z, l_max+1);
52  return(NULL);
53  }
54  for (k=0; k<=l_max; k++) {
55  Hes[k] = NULL;
56  Hes[k] = (realtype *) malloc(l_max*sizeof(realtype));
57  if (Hes[k] == NULL) {
58  for (i=0; i<k; i++) {free(Hes[i]); Hes[i] = NULL;}
59  free(Hes); Hes = NULL;
60  N_VDestroyVectorArray(V, l_max+1);
61  N_VDestroyVectorArray(Z, l_max+1);
62  return(NULL);
63  }
64  }
65 
66  /* Get memory for Givens rotation components. */
67  givens = NULL;
68  givens = (realtype *) malloc(2*l_max*sizeof(realtype));
69  if (givens == NULL) {
70  for (i=0; i<=l_max; i++) {free(Hes[i]); Hes[i] = NULL;}
71  free(Hes); Hes = NULL;
72  N_VDestroyVectorArray(V, l_max+1);
73  N_VDestroyVectorArray(Z, l_max+1);
74  return(NULL);
75  }
76 
77  /* Get memory to hold the correction to z_tilde. */
78  xcor = N_VClone(vec_tmpl);
79  if (xcor == NULL) {
80  free(givens); givens = NULL;
81  for (i=0; i<=l_max; i++) {free(Hes[i]); Hes[i] = NULL;}
82  free(Hes); Hes = NULL;
83  N_VDestroyVectorArray(V, l_max+1);
84  N_VDestroyVectorArray(Z, l_max+1);
85  return(NULL);
86  }
87 
88  /* Get memory to hold SPFGMR y and g vectors. */
89  yg = NULL;
90  yg = (realtype *) malloc((l_max+1)*sizeof(realtype));
91  if (yg == NULL) {
92  N_VDestroy(xcor);
93  free(givens); givens = NULL;
94  for (i=0; i<=l_max; i++) {free(Hes[i]); Hes[i] = NULL;}
95  free(Hes); Hes = NULL;
96  N_VDestroyVectorArray(V, l_max+1);
97  N_VDestroyVectorArray(Z, l_max+1);
98  return(NULL);
99  }
100 
101  /* Get an array to hold a temporary vector. */
102  vtemp = N_VClone(vec_tmpl);
103  if (vtemp == NULL) {
104  free(yg); yg = NULL;
105  N_VDestroy(xcor);
106  free(givens); givens = NULL;
107  for (i=0; i<=l_max; i++) {free(Hes[i]); Hes[i] = NULL;}
108  free(Hes); Hes = NULL;
109  N_VDestroyVectorArray(V, l_max+1);
110  N_VDestroyVectorArray(Z, l_max+1);
111  return(NULL);
112  }
113 
114  /* Get memory for an SpfgmrMemRec containing SPFGMR matrices and vectors. */
115  mem = NULL;
116  mem = (SpfgmrMem) malloc(sizeof(SpfgmrMemRec));
117  if (mem == NULL) {
118  N_VDestroy(vtemp);
119  free(yg); yg = NULL;
120  N_VDestroy(xcor);
121  free(givens); givens = NULL;
122  for (i=0; i<=l_max; i++) {free(Hes[i]); Hes[i] = NULL;}
123  free(Hes); Hes = NULL;
124  N_VDestroyVectorArray(V, l_max+1);
125  N_VDestroyVectorArray(Z, l_max+1);
126  return(NULL);
127  }
128 
129  /* Set the fields of mem. */
130  mem->l_max = l_max;
131  mem->V = V;
132  mem->Z = Z;
133  mem->Hes = Hes;
134  mem->givens = givens;
135  mem->xcor = xcor;
136  mem->yg = yg;
137  mem->vtemp = vtemp;
138 
139  /* Return the pointer to SPFGMR memory. */
140  return(mem);
141 }

143 /*----------------------------------------------------------------
144  Function : SpfgmrSolve
145  ---------------------------------------------------------------*/
146 int SpfgmrSolve(SpfgmrMem mem, void *A_data, N_Vector x,
147  N_Vector b, int pretype, int gstype, realtype delta,
148  int max_restarts, int maxit, void *P_data,
149  N_Vector s1, N_Vector s2, ATimesFn atimes,
150  PSolveFn psolve, realtype *res_norm, int *nli, int *nps)
151 {
152  N_Vector *V, *Z, xcor, vtemp;
153  realtype **Hes, *givens, *yg;
154  realtype beta, rotation_product, r_norm, s_product, rho;
155  booleantype preOnRight, scale1, scale2, converged;
156  int i, j, k, l, l_max, krydim, ier, ntries;
157 
158  if (mem == NULL) return(SPFGMR_MEM_NULL);
159 
160  /* Initialize some variables */
161  krydim = 0;
162 
163  /* Make local copies of mem variables. */
164  l_max = mem->l_max;
165  V = mem->V;
166  Z = mem->Z;
167  Hes = mem->Hes;
168  givens = mem->givens;
169  xcor = mem->xcor;
170  yg = mem->yg;
171  vtemp = mem->vtemp;
172 
173  *nli = *nps = 0; /* Initialize counters */
174  converged = FALSE; /* Initialize converged flag */
175 
176  /* If maxit is greater than l_max, then set maxit=l_max */
177  if (maxit > l_max) maxit = l_max;
178 
179  /* Check for legal value of max_restarts */
180  if (max_restarts < 0) max_restarts = 0;
181 
182  /* Set preconditioning flag (enabling any preconditioner implies right
183  preconditioning, since FGMRES does not support left preconditioning) */
184  preOnRight = ((pretype == PREC_RIGHT) || (pretype == PREC_BOTH) || (pretype == PREC_LEFT));
185 
186  /* Set scaling flags */
187  scale1 = (s1 != NULL);
188  scale2 = (s2 != NULL);
189 
190  /* Set vtemp to initial (unscaled) residual r_0 = b - A*x_0. */
191  if (N_VDotProd(x, x) == ZERO) {
192  N_VScale(ONE, b, vtemp);
193  } else {
194  ier = atimes(A_data, x, vtemp);
195  if (ier != 0)
196  return((ier < 0) ? SPFGMR_ATIMES_FAIL_UNREC : SPFGMR_ATIMES_FAIL_REC);
197  N_VLinearSum(ONE, b, -ONE, vtemp, vtemp);
198  }
199 
200  /* Apply left scaling to vtemp = r_0 to fill V[0]. */
201  if (scale1) {
202  N_VProd(s1, vtemp, V[0]);
203  } else {
204  N_VScale(ONE, vtemp, V[0]);
205  }
206 
207  /* Set r_norm = beta to L2 norm of V[0] = s1 r_0, and return if small */
208  *res_norm = r_norm = beta = SUNRsqrt(N_VDotProd(V[0], V[0]));
209  if (r_norm <= delta)
210  return(SPFGMR_SUCCESS);
211 
212  /* Initialize rho to avoid compiler warning message */
213  rho = beta;
214 
215  /* Set xcor = 0. */
216  N_VConst(ZERO, xcor);
217 
218  /* Begin outer iterations: up to (max_restarts + 1) attempts. */
219  for (ntries=0; ntries<=max_restarts; ntries++) {
220 
221  /* Initialize the Hessenberg matrix Hes and Givens rotation
222  product. Normalize the initial vector V[0]. */
223  for (i=0; i<=l_max; i++)
224  for (j=0; j<l_max; j++)
225  Hes[i][j] = ZERO;
226  rotation_product = ONE;
227  N_VScale(ONE/r_norm, V[0], V[0]);
228 
229  /* Inner loop: generate Krylov sequence and Arnoldi basis. */
230  for (l=0; l<maxit; l++) {
231 
232  (*nli)++;
233 
234  krydim = l + 1;
235 
236  /* Generate A-tilde V[l], where A-tilde = s1 A P_inv s2_inv. */
237 
238  /* Apply right scaling: vtemp = s2_inv V[l]. */
239  if (scale2) N_VDiv(V[l], s2, vtemp);
240  else N_VScale(ONE, V[l], vtemp);
241 
242  /* Apply right preconditioner: vtemp = Z[l] = P_inv s2_inv V[l]. */
243  if (preOnRight) {
244  N_VScale(ONE, vtemp, V[l+1]);
245  ier = psolve(P_data, V[l+1], vtemp, PREC_RIGHT);
246  (*nps)++;
247  if (ier != 0)
248  return((ier < 0) ? SPFGMR_PSOLVE_FAIL_UNREC : SPFGMR_PSOLVE_FAIL_REC);
249  }
250  N_VScale(ONE, vtemp, Z[l]);
251 
252  /* Apply A: V[l+1] = A P_inv s2_inv V[l]. */
253  ier = atimes(A_data, vtemp, V[l+1]);
254  if (ier != 0)
255  return((ier < 0) ? SPFGMR_ATIMES_FAIL_UNREC : SPFGMR_ATIMES_FAIL_REC);
256 
257  /* Apply left scaling: V[l+1] = s1 A P_inv s2_inv V[l]. */
258  if (scale1) N_VProd(s1, V[l+1], V[l+1]);
259 
260  /* Orthogonalize V[l+1] against previous V[i]: V[l+1] = w_tilde. */
261  if (gstype == CLASSICAL_GS) {
262  if (ClassicalGS(V, Hes, l+1, l_max, &(Hes[l+1][l]),
263  vtemp, yg) != 0)
264  return(SPFGMR_GS_FAIL);
265  } else {
266  if (ModifiedGS(V, Hes, l+1, l_max, &(Hes[l+1][l])) != 0)
267  return(SPFGMR_GS_FAIL);
268  }
269 
270  /* Update the QR factorization of Hes. */
271  if(QRfact(krydim, Hes, givens, l) != 0 )
272  return(SPFGMR_QRFACT_FAIL);
273 
274  /* Update residual norm estimate; break if convergence test passes. */
275  rotation_product *= givens[2*l+1];
276  *res_norm = rho = SUNRabs(rotation_product*r_norm);
277  if (rho <= delta) { converged = TRUE; break; }
278 
279  /* Normalize V[l+1] with norm value from the Gram-Schmidt routine. */
280  N_VScale(ONE/Hes[l+1][l], V[l+1], V[l+1]);
281  }
282 
283  /* Inner loop is done. Compute the new correction vector xcor. */
284 
285  /* Construct g, then solve for y. */
286  yg[0] = r_norm;
287  for (i=1; i<=krydim; i++) yg[i]=ZERO;
288  if (QRsol(krydim, Hes, givens, yg) != 0)
289  return(SPFGMR_QRSOL_FAIL);
290 
291  /* Add correction vector Z_l y to xcor. */
292  for (k=0; k<krydim; k++)
293  N_VLinearSum(yg[k], Z[k], ONE, xcor, xcor);
294 
295  /* If converged, construct the final solution vector x and return. */
296  if (converged) {
297  N_VLinearSum(ONE, x, ONE, xcor, x);
298  return(SPFGMR_SUCCESS);
299  }
300 
301  /* Not yet converged; if allowed, prepare for restart. */
302  if (ntries == max_restarts) break;
303 
304  /* Construct last column of Q in yg. */
305  s_product = ONE;
306  for (i=krydim; i>0; i--) {
307  yg[i] = s_product*givens[2*i-2];
308  s_product *= givens[2*i-1];
309  }
310  yg[0] = s_product;
311 
312  /* Scale r_norm and yg. */
313  r_norm *= s_product;
314  for (i=0; i<=krydim; i++)
315  yg[i] *= r_norm;
316  r_norm = SUNRabs(r_norm);
317 
318  /* Multiply yg by V_(krydim+1) to get last residual vector; restart. */
319  N_VScale(yg[0], V[0], V[0]);
320  for (k=1; k<=krydim; k++)
321  N_VLinearSum(yg[k], V[k], ONE, V[0], V[0]);
322 
323  }
324 
325  /* Failed to converge, even after allowed restarts.
326  If the residual norm was reduced below its initial value, compute
327  and return x anyway. Otherwise return failure flag. */
328  if (rho < beta) {
329  N_VLinearSum(ONE, x, ONE, xcor, x);
330  return(SPFGMR_RES_REDUCED);
331  }
332 
333  return(SPFGMR_CONV_FAIL);
334 }

359 }
