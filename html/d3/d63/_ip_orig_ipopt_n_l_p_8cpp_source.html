141  bool OrigIpoptNLP::Initialize(const Journalist& jnlst,
142  const OptionsList& options,
143  const std::string& prefix)
144  {
145  options.GetNumericValue("bound_relax_factor", bound_relax_factor_, prefix);
146  options.GetBoolValue("honor_original_bounds",
147  honor_original_bounds_, prefix);
148  options.GetBoolValue("warm_start_same_structure",
149  warm_start_same_structure_, prefix);
150  options.GetBoolValue("check_derivatives_for_naninf",
151  check_derivatives_for_naninf_, prefix);
152  Index enum_int;
153  options.GetEnumValue("hessian_approximation", enum_int, prefix);
154  hessian_approximation_ = HessianApproximationType(enum_int);
155  options.GetEnumValue("hessian_approximation_space", enum_int, prefix);
156  hessian_approximation_space_ = HessianApproximationSpace(enum_int);
157 
158  options.GetBoolValue("jac_c_constant", jac_c_constant_, prefix);
159  options.GetBoolValue("jac_d_constant", jac_d_constant_, prefix);
160  options.GetBoolValue("hessian_constant", hessian_constant_, prefix);
161 
162  // Reset the function evaluation counters (for warm start)
163  f_evals_=0;
164  grad_f_evals_=0;
165  c_evals_=0;
166  jac_c_evals_=0;
167  d_evals_=0;
168  jac_d_evals_=0;
169  h_evals_=0;
170 
171  if (!warm_start_same_structure_) {
172  // Reset all caches.
173  grad_f_cache_.Clear();
174  c_cache_.Clear();
175  jac_c_cache_.Clear();
176  d_cache_.Clear();
177  jac_d_cache_.Clear();
178  // If the hessian is constant, we want two hessians to be
179  // cached, one for regular iterations and one for restoration
180  // phase
181  if (hessian_constant_)
182  h_cache_.Clear(2);
183  else
184  h_cache_.Clear(1);
185  }
186 
187  // Reset the cache entries belonging to a dummy dependency. This
188  // is required for repeated solve, since the cache is not updated
189  // if a dimension is zero. It is also required if we choose
190  // jac_[cd]_constant and hessian_constant differently between
191  // runs
192  std::vector<const TaggedObject*> deps(1);
193  deps[0] = NULL;
194  std::vector<Number> sdeps(0);
195  c_cache_.InvalidateResult(deps, sdeps);
196  d_cache_.InvalidateResult(deps, sdeps);
197  jac_c_cache_.InvalidateResult(deps, sdeps);
198  jac_d_cache_.InvalidateResult(deps, sdeps);
199  h_cache_.InvalidateResult(deps, sdeps);
200 
201  if (!nlp_->ProcessOptions(options, prefix)) {
202  return false;
203  }
204 
205  initialized_ = true;
206  return IpoptNLP::Initialize(jnlst, options, prefix);
207  }

209  bool OrigIpoptNLP::InitializeStructures(SmartPtr<Vector>& x,
210  bool init_x,
211  SmartPtr<Vector>& y_c,
212  bool init_y_c,
213  SmartPtr<Vector>& y_d,
214  bool init_y_d,
215  SmartPtr<Vector>& z_L,
216  bool init_z_L,
217  SmartPtr<Vector>& z_U,
218  bool init_z_U,
219  SmartPtr<Vector>& v_L,
220  SmartPtr<Vector>& v_U
221  )
222  {
223  DBG_START_METH("OrigIpoptNLP::InitializeStructures", dbg_verbosity);
224  DBG_ASSERT(initialized_);
225  bool retValue;
226 
227  SmartPtr<Vector> x_L;
228  SmartPtr<Matrix> Px_L;
229  SmartPtr<Vector> x_U;
230  SmartPtr<Matrix> Px_U;
231  SmartPtr<Vector> d_L;
232  SmartPtr<Matrix> Pd_L;
233  SmartPtr<Vector> d_U;
234  SmartPtr<Matrix> Pd_U;
235 
236  if (!warm_start_same_structure_) {
237 
238  retValue = nlp_->GetSpaces(x_space_, c_space_, d_space_,
239  x_l_space_, px_l_space_,
240  x_u_space_, px_u_space_,
241  d_l_space_, pd_l_space_,
242  d_u_space_, pd_u_space_,
243  jac_c_space_, jac_d_space_,
244  h_space_);
245 
246  if (!retValue) {
247  jnlst_->Printf(J_WARNING, J_INITIALIZATION,
248  "GetSpaces method for the NLP returns false.\n");
249  return false;
250  }
251 
252  // Check if the Hessian space is actually a limited-memory
253  // approximation. If so, get the required information from the
254  // NLP and create an appropreate h_space
255  if (hessian_approximation_==LIMITED_MEMORY) {
256  SmartPtr<VectorSpace> approx_vecspace;
257  SmartPtr<Matrix> P_approx;
258  if (hessian_approximation_space_==NONLINEAR_VARS) {
259  nlp_->GetQuasiNewtonApproximationSpaces(approx_vecspace,
260  P_approx);
261  }
262  if (IsValid(approx_vecspace)) {
263  DBG_ASSERT(IsValid(P_approx));
264  h_space_ = new LowRankUpdateSymMatrixSpace(x_space_->Dim(),
265  ConstPtr(P_approx),
266  ConstPtr(approx_vecspace),
267  true);
268  jnlst_->Printf(J_DETAILED, J_INITIALIZATION,
269  "Hessian approximation will be done in smaller space of dimension %d (instead of %d)\n\n",
270  P_approx->NCols(), P_approx->NRows());
271  }
272  else {
273  DBG_ASSERT(IsNull(P_approx));
274  h_space_ = new LowRankUpdateSymMatrixSpace(x_space_->Dim(),
275  ConstPtr(P_approx),
276  ConstPtr(x_space_),
277  true);
278  jnlst_->Printf(J_DETAILED, J_INITIALIZATION,
279  "Hessian approximation will be done in the space of all %d x variables.\n\n",
280  x_space_->Dim());
281  }
282  }
283 
284  // Create the bounds structures
285  x_L = x_l_space_->MakeNew();
286  Px_L = px_l_space_->MakeNew();
287  x_U = x_u_space_->MakeNew();
288  Px_U = px_u_space_->MakeNew();
289  d_L = d_l_space_->MakeNew();
290  Pd_L = pd_l_space_->MakeNew();
291  d_U = d_u_space_->MakeNew();
292  Pd_U = pd_u_space_->MakeNew();
293 
294  retValue = nlp_->GetBoundsInformation(*Px_L, *x_L, *Px_U, *x_U,
295  *Pd_L, *d_L, *Pd_U, *d_U);
296  if (!retValue) {
297  return false;
298  }
299 
300  NLP_scaling()->DetermineScaling(x_space_,
301  c_space_, d_space_,
302  jac_c_space_, jac_d_space_,
303  h_space_,
304  scaled_jac_c_space_, scaled_jac_d_space_,
305  scaled_h_space_,
306  *Px_L, *x_L, *Px_U, *x_U);
307 
308  if (x_space_->Dim() < c_space_->Dim()) {
309  char msg[128];
310  Snprintf(msg, 127, "Too few degrees of freedom: %d equality constriants but only %d variables", c_space_->Dim(), x_space_->Dim());
311  THROW_EXCEPTION(TOO_FEW_DOF, msg);
312  }
313  ASSERT_EXCEPTION(x_space_->Dim() > 0, TOO_FEW_DOF,
314  "Too few degrees of freedom (no free variables)!");
315 
316  // cannot have any null pointers, want zero length vectors
317  // instead of null - this will later need to be changed for _h;
318  retValue = (IsValid(x_space_) && IsValid(c_space_) && IsValid(d_space_)
319  && IsValid(x_l_space_) && IsValid(px_l_space_)
320  && IsValid(x_u_space_) && IsValid(px_u_space_)
321  && IsValid(d_u_space_) && IsValid(pd_u_space_)
322  && IsValid(d_l_space_) && IsValid(pd_l_space_)
323  && IsValid(jac_c_space_) && IsValid(jac_d_space_)
324  && IsValid(h_space_)
325  && IsValid(scaled_jac_c_space_)
326  && IsValid(scaled_jac_d_space_)
327  && IsValid(scaled_h_space_));
328 
329  DBG_ASSERT(retValue && "Model cannot return null vector or matrix prototypes or spaces,"
330  " please return zero length vectors instead");
331  }
332  else {
333  ASSERT_EXCEPTION(IsValid(x_space_), INVALID_WARMSTART,
334  "OrigIpoptNLP called with warm_start_same_structure, but the problem is solved for the first time.");
335  // Create the bounds structures
336  x_L = x_l_space_->MakeNew();
337  Px_L = px_l_space_->MakeNew();
338  x_U = x_u_space_->MakeNew();
339  Px_U = px_u_space_->MakeNew();
340  d_L = d_l_space_->MakeNew();
341  Pd_L = pd_l_space_->MakeNew();
342  d_U = d_u_space_->MakeNew();
343  Pd_U = pd_u_space_->MakeNew();
344 
345  retValue = nlp_->GetBoundsInformation(*Px_L, *x_L, *Px_U, *x_U,
346  *Pd_L, *d_L, *Pd_U, *d_U);
347  if (!retValue) {
348  return false;
349  }
350  }
351 
352  x_L->Print(*jnlst_, J_MOREVECTOR, J_INITIALIZATION,
353  "original x_L unscaled");
354  x_U->Print(*jnlst_, J_MOREVECTOR, J_INITIALIZATION,
355  "original x_U unscaled");
356  d_L->Print(*jnlst_, J_MOREVECTOR, J_INITIALIZATION,
357  "original d_L unscaled");
358  d_U->Print(*jnlst_, J_MOREVECTOR, J_INITIALIZATION,
359  "original d_U unscaled");
360 
361  if (honor_original_bounds_) {
362  SmartPtr<Vector> tmp;
363  tmp = x_L->MakeNewCopy();
364  orig_x_L_ = ConstPtr(tmp);
365  tmp = x_U->MakeNewCopy();
366  orig_x_U_ = ConstPtr(tmp);
367  }
368 
369  relax_bounds(-bound_relax_factor_, *x_L);
370  relax_bounds( bound_relax_factor_, *x_U);
371  relax_bounds(-bound_relax_factor_, *d_L);
372  relax_bounds( bound_relax_factor_, *d_U);
373 
374  x_L_ = ConstPtr(x_L);
375  Px_L_ = ConstPtr(Px_L);
376  x_U_ = ConstPtr(x_U);
377  Px_U_ = ConstPtr(Px_U);
378  d_L_ = ConstPtr(d_L);
379  Pd_L_ = ConstPtr(Pd_L);
380  d_U_ = ConstPtr(d_U);
381  Pd_U_ = ConstPtr(Pd_U);
382 
383  // now create and store the scaled bounds
384  x_L_ = NLP_scaling()->apply_vector_scaling_x_LU(*Px_L_, x_L_, *x_space_);
385  x_U_ = NLP_scaling()->apply_vector_scaling_x_LU(*Px_U_, x_U_, *x_space_);
386  d_L_ = NLP_scaling()->apply_vector_scaling_d_LU(*Pd_L_, d_L_, *d_space_);
387  d_U_ = NLP_scaling()->apply_vector_scaling_d_LU(*Pd_U_, d_U_, *d_space_);
388 
389  x_L_->Print(*jnlst_, J_VECTOR, J_INITIALIZATION,
390  "modified x_L scaled");
391  x_U_->Print(*jnlst_, J_VECTOR, J_INITIALIZATION,
392  "modified x_U scaled");
393  d_L_->Print(*jnlst_, J_VECTOR, J_INITIALIZATION,
394  "modified d_L scaled");
395  d_U_->Print(*jnlst_, J_VECTOR, J_INITIALIZATION,
396  "modified d_U scaled");
397 
398  // Create the iterates structures
399  x = x_space_->MakeNew();
400  y_c = c_space_->MakeNew();
401  y_d = d_space_->MakeNew();
402  z_L = x_l_space_->MakeNew();
403  z_U = x_u_space_->MakeNew();
404  v_L = d_l_space_->MakeNew();
405  v_U = d_u_space_->MakeNew();
406 
407  retValue = nlp_->GetStartingPoint(GetRawPtr(x), init_x,
408  GetRawPtr(y_c), init_y_c,
409  GetRawPtr(y_d), init_y_d,
410  GetRawPtr(z_L), init_z_L,
411  GetRawPtr(z_U), init_z_U);
412 
413  if (!retValue) {
414  return false;
415  }
416 
417 
418  Number obj_scal = NLP_scaling()->apply_obj_scaling(1.);
419  if (init_x) {
420  x->Print(*jnlst_, J_VECTOR, J_INITIALIZATION, "initial x unscaled");
421  if (NLP_scaling()->have_x_scaling()) {
422  x = NLP_scaling()->apply_vector_scaling_x_NonConst(ConstPtr(x));
423  }
424  }
425  if (init_y_c) {
426  y_c->Print(*jnlst_, J_VECTOR, J_INITIALIZATION, "initial y_c unscaled");
427  if (NLP_scaling()->have_c_scaling()) {
428  y_c = NLP_scaling()->unapply_vector_scaling_c_NonConst(ConstPtr(y_c));
429  }
430  if (obj_scal!=1.) {
431  y_c->Scal(obj_scal);
432  }
433  }
434  if (init_y_d) {
435  y_d->Print(*jnlst_, J_VECTOR, J_INITIALIZATION, "initial y_d unscaled");
436  if (NLP_scaling()->have_d_scaling()) {
437  y_d = NLP_scaling()->unapply_vector_scaling_d_NonConst(ConstPtr(y_d));
438  }
439  if (obj_scal!=1.) {
440  y_d->Scal(obj_scal);
441  }
442  }
443  if (init_z_L) {
444  z_L->Print(*jnlst_, J_VECTOR, J_INITIALIZATION, "initial z_L unscaled");
445  if (NLP_scaling()->have_x_scaling()) {
446  z_L = NLP_scaling()->apply_vector_scaling_x_LU_NonConst(*Px_L_, ConstPtr(z_L), *x_space_);
447  }
448  if (obj_scal!=1.) {
449  z_L->Scal(obj_scal);
450  }
451  }
452  if (init_z_U) {
453  z_U->Print(*jnlst_, J_VECTOR, J_INITIALIZATION, "initial z_U unscaled");
454  if (NLP_scaling()->have_x_scaling()) {
455  z_U = NLP_scaling()->apply_vector_scaling_x_LU_NonConst(*Px_U_, ConstPtr(z_U), *x_space_);
456  }
457  if (obj_scal!=1.) {
458  z_U->Scal(obj_scal);
459  }
460  }
461 
462  return true;
463  }

465  void
466  OrigIpoptNLP::relax_bounds(Number bound_relax_factor, Vector& bounds)
467  {
468  DBG_START_METH("OrigIpoptNLP::relax_bounds", dbg_verbosity);
469  if (bound_relax_factor!=0.) {
470  SmartPtr<Vector> tmp = bounds.MakeNew();
471  tmp->Copy(bounds);
472  tmp->ElementWiseAbs();
473  SmartPtr<Vector> ones = bounds.MakeNew();
474  ones->Set(1.);
475  tmp->ElementWiseMax(*ones);
476  DBG_PRINT((1, "bound_relax_factor = %e", bound_relax_factor));
477  DBG_PRINT_VECTOR(2, "tmp", *tmp);
478  DBG_PRINT_VECTOR(2, "bounds before", bounds);
479  bounds.Axpy(bound_relax_factor, *tmp);
480  DBG_PRINT_VECTOR(2, "bounds after", bounds);
481  }
482  }

484  Number OrigIpoptNLP::f(const Vector& x)
485  {
486  DBG_START_METH("OrigIpoptNLP::f", dbg_verbosity);
487  Number ret = 0.0;
488  DBG_PRINT((2, "x.Tag = %u\n", x.GetTag()));
489  if (!f_cache_.GetCachedResult1Dep(ret, &x)) {
490  f_evals_++;
491  SmartPtr<const Vector> unscaled_x = get_unscaled_x(x);
492  f_eval_time_.Start();
493  bool success = nlp_->Eval_f(*unscaled_x, ret);
494  f_eval_time_.End();
495  DBG_PRINT((1, "success = %d ret = %e\n", success, ret));
496  ASSERT_EXCEPTION(success && IsFiniteNumber(ret), Eval_Error,
497  "Error evaluating the objective function");
498  ret = NLP_scaling()->apply_obj_scaling(ret);
499  f_cache_.AddCachedResult1Dep(ret, &x);
500  }
501 
502  return ret;
503  }

540  SmartPtr<const Vector> OrigIpoptNLP::c(const Vector& x)
541  {
542  SmartPtr<const Vector> retValue;
543  if (c_space_->Dim()==0) {
544  // We do this caching of an empty vector so that the returned
545  // Vector has always the same tag (this might make a difference
546  // in cases where only the constraints are supposed to change...
547  SmartPtr<const Vector> dep = NULL;
548  if (!c_cache_.GetCachedResult1Dep(retValue, GetRawPtr(dep))) {
549  retValue = c_space_->MakeNew();
550  c_cache_.AddCachedResult1Dep(retValue, GetRawPtr(dep));
551  }
552  }
553  else {
554  if (!c_cache_.GetCachedResult1Dep(retValue, x)) {
555  SmartPtr<Vector> unscaled_c = c_space_->MakeNew();
556  c_evals_++;
557  SmartPtr<const Vector> unscaled_x = get_unscaled_x(x);
558  c_eval_time_.Start();
559  bool success = nlp_->Eval_c(*unscaled_x, *unscaled_c);
560  c_eval_time_.End();
561  if (!success || !IsFiniteNumber(unscaled_c->Nrm2())) {
562  if (check_derivatives_for_naninf_ && !IsFiniteNumber(unscaled_c->Nrm2())) {

715  SmartPtr<const SymMatrix> OrigIpoptNLP::h(const Vector& x,
716  Number obj_factor,
717  const Vector& yc,
718  const Vector& yd)
719  {

721  SmartPtr<const SymMatrix> retValue;
722 
723  std::vector<const TaggedObject*> deps(3);
724  if (!hessian_constant_) {
725  deps[0] = &x;
726  deps[1] = &yc;
727  deps[2] = &yd;
728  }
729  else {
730  deps[0] = NULL;
731  deps[1] = NULL;
732  deps[2] = NULL;
733  }
734  std::vector<Number> scalar_deps(1);
735  scalar_deps[0] = obj_factor;
736 
737  if (!h_cache_.GetCachedResult(retValue, deps, scalar_deps)) {
738  h_evals_++;
739  unscaled_h = h_space_->MakeNewSymMatrix();
740 
741  SmartPtr<const Vector> unscaled_x = get_unscaled_x(x);
742  SmartPtr<const Vector> unscaled_yc = NLP_scaling()->apply_vector_scaling_c(&yc);
743  SmartPtr<const Vector> unscaled_yd = NLP_scaling()->apply_vector_scaling_d(&yd);
744  Number scaled_obj_factor = NLP_scaling()->apply_obj_scaling(obj_factor);
745  h_eval_time_.Start();
746  bool success = nlp_->Eval_h(*unscaled_x, scaled_obj_factor, *unscaled_yc, *unscaled_yd, *unscaled_h);
747  h_eval_time_.End();
748  ASSERT_EXCEPTION(success, Eval_Error, "Error evaluating the hessian of the lagrangian");
749  if (check_derivatives_for_naninf_) {
750  if (!unscaled_h->HasValidNumbers()) {
751  jnlst_->Printf(J_WARNING, J_NLP,
752  "The Lagrangian Hessian contains an invalid number\n");
753 
754  unscaled_h->Print(*jnlst_, J_MOREDETAILED, J_MAIN, "unscaled_h");

756  THROW_EXCEPTION(Eval_Error, "The Lagrangian Hessian contains an invalid number");
757  }
758  }
759  retValue = NLP_scaling()->apply_hessian_scaling(ConstPtr(unscaled_h));
760  h_cache_.AddCachedResult(retValue, deps, scalar_deps);
761  }
762 
763  return retValue;

827  void OrigIpoptNLP::FinalizeSolution(SolverReturn status,
828  const Vector& x, const Vector& z_L, const Vector& z_U,
829  const Vector& c, const Vector& d,
830  const Vector& y_c, const Vector& y_d,
831  Number obj_value,
832  const IpoptData* ip_data,
833  IpoptCalculatedQuantities* ip_cq)
834  {
835  DBG_START_METH("OrigIpoptNLP::FinalizeSolution", dbg_verbosity);
836  // need to submit the unscaled solution back to the nlp
837  SmartPtr<const Vector> unscaled_x = get_unscaled_x(x);
838  SmartPtr<const Vector> unscaled_c =
839  NLP_scaling()->unapply_vector_scaling_c(&c);
840  SmartPtr<const Vector> unscaled_d =
841  NLP_scaling()->unapply_vector_scaling_d(&d);
842  const Number unscaled_obj = NLP_scaling()->unapply_obj_scaling(obj_value);
843 
844  SmartPtr<const Vector> unscaled_z_L;
845  SmartPtr<const Vector> unscaled_z_U;
846  SmartPtr<const Vector> unscaled_y_c;
847  SmartPtr<const Vector> unscaled_y_d;
848 
849  // The objective function scaling factor also appears in the constraints
850  Number obj_unscale_factor = NLP_scaling()->unapply_obj_scaling(1.);
851  if (obj_unscale_factor!=1.) {
852  SmartPtr<Vector> tmp = NLP_scaling()->apply_vector_scaling_x_LU_NonConst(*Px_L_, &z_L, *x_space_);
853  tmp->Scal(obj_unscale_factor);
854  unscaled_z_L = ConstPtr(tmp);
855 
856  tmp = NLP_scaling()->apply_vector_scaling_x_LU_NonConst(*Px_U_, &z_U, *x_space_);
857  tmp->Scal(obj_unscale_factor);
858  unscaled_z_U = ConstPtr(tmp);
859 
860  tmp = NLP_scaling()->apply_vector_scaling_c_NonConst(&y_c);
861  tmp->Scal(obj_unscale_factor);
862  unscaled_y_c = ConstPtr(tmp);
863 
864  tmp = NLP_scaling()->apply_vector_scaling_d_NonConst(&y_d);
865  tmp->Scal(obj_unscale_factor);
866  unscaled_y_d = ConstPtr(tmp);
867  }
868  else {
869  unscaled_z_L = NLP_scaling()->apply_vector_scaling_x_LU(*Px_L_, &z_L, *x_space_);
870  unscaled_z_U = NLP_scaling()->apply_vector_scaling_x_LU(*Px_U_, &z_U, *x_space_);
871  unscaled_y_c = NLP_scaling()->apply_vector_scaling_c(&y_c);
872  unscaled_y_d = NLP_scaling()->apply_vector_scaling_d(&y_d);
873  }
874 
875  if (honor_original_bounds_ && (Px_L_->NCols()>0 || Px_U_->NCols()>0)) {
876  // Make sure the user specified bounds are satisfied
877  SmartPtr<Vector> tmp;
878  SmartPtr<Vector> un_x = unscaled_x->MakeNewCopy();
879  if (Px_L_->NCols()>0) {
880  tmp = orig_x_L_->MakeNewCopy();
881  Px_L_->TransMultVector(1., *un_x, 0., *tmp);
882  Px_L_->MultVector(-1., *tmp, 1., *un_x);
883  tmp->ElementWiseMax(*orig_x_L_);
884  Px_L_->MultVector(1., *tmp, 1., *un_x);
885  }
886  if (Px_U_->NCols()>0) {
887  tmp = orig_x_U_->MakeNewCopy();
888  Px_U_->TransMultVector(1., *un_x, 0., *tmp);
889  Px_U_->MultVector(-1., *tmp, 1., *un_x);
890  tmp->ElementWiseMin(*orig_x_U_);
891  Px_U_->MultVector(1., *tmp, 1., *un_x);
892  }
893  unscaled_x = ConstPtr(un_x);
894  }
895 
896  unscaled_x->Print(*jnlst_, J_VECTOR, J_SOLUTION, "final x unscaled");
897  unscaled_y_c->Print(*jnlst_, J_VECTOR, J_SOLUTION, "final y_c unscaled");
898  unscaled_y_d->Print(*jnlst_, J_VECTOR, J_SOLUTION, "final y_d unscaled");
899  unscaled_z_L->Print(*jnlst_, J_VECTOR, J_SOLUTION, "final z_L unscaled");
900  unscaled_z_U->Print(*jnlst_, J_VECTOR, J_SOLUTION, "final z_U unscaled");
901 
902  nlp_->FinalizeSolution(status, *unscaled_x,
903  *unscaled_z_L, *unscaled_z_U,
904  *unscaled_c, *unscaled_d,
905  *unscaled_y_c, *unscaled_y_d,
906  unscaled_obj, ip_data, ip_cq);
907  }
