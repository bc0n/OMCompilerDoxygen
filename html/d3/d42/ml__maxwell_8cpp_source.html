7 // Sample driver for Maxwell equation AMG solver in the ML package. The
8 // software is tested by setting up a 2-dimensional uniform grid example on

393 /********************************************************************/
394 /* Given the (i,j)th cell return the node indices defining the cell */
395 /* */
396 /* NW-------NE */
397 /* | | */
398 /* | | */
399 /* | | */
400 /* SW--------SE */
401 /* */
402 /********************************************************************/
403 
404 int northwest(int i, int j, int n) { return(i + (j)*n); }
405 int southwest(int i, int j, int n) { if (j == 0) j = n;
406  return(i + (j-1)*n);}
407 int southeast(int i, int j, int n) { if (j == 0) j = n;
408  return(i+1 + (j-1)*n);}
409 
410 
411 /********************************************************************/
412 /* Given the (i,j)th cell return the edge indices defining the cell */
413 /* */
414 /* .--north--. */
415 /* | | */
416 /* west east */
417 /* | | */
418 /* .--south--. */
419 /* */
420 /********************************************************************/
421 
422 int north(int i, int j, int n) { return(i + (j)*n);}
423 int south(int i, int j, int n) { return(i + (j-1)*n);}
424 int west(int i, int j, int n) { return(i+(j)*n+n*n -1);}
425 int east(int i, int j, int n) { return(i+1+(j)*n+n*n -1);}
426 
427 /* Given the index of either a 'south' or 'west' edge, return the */
428 /* cell coordinates (i,j) as well as the orientation (horizontal */
429 /* or vertical) of the edge. */
430 int invindex(int index, int *i, int *j, int n, int *hor_or_vert)
431 {
432  *hor_or_vert = HORIZONTAL;
433  if (index >= n*n) {
434  *hor_or_vert = VERTICAL;
435  index -= n*n;
436  }
437  *i = (index%n);
438  *j = (index - (*i))/n;
439  if ( *hor_or_vert == HORIZONTAL) (*j) = ((*j)+1)%n;
440  if ( *hor_or_vert == VERTICAL) (*i) = ((*i)+1)%n;
441  return 1;
442 }

475 /********************************************************************/
476 /* Set up Ke_mat */
477 /* 1) First set it up as an Aztec DMSR matrix */
478 /* This stores the diagonal of row j in Ke_val[j] and stores a */
479 /* pointer to row j's off-diagonal nonzeros in Ke_bindx[j] with */
480 /* column/nonzero entries in Ke_bindx[Ke_bindx[j]:Ke_bindx[j+1]-1] */
481 /* and Ke_val[Ke_bindx[j]:Ke_bindx[j+1]-1]. */
482 /* 2) call AZ_transform to convert global column indices to local */
483 /* indices and to set up Aztec's communication structure. */
484 /* 3) Stuff the arrays into an Aztec matrix. */
485 /*------------------------------------------------------------------*/
486 
487 AZ_MATRIX *user_Ke_build(struct user_partition *Edge_Partition)
488 {
489  double dcenter, doff, sigma = .0001;
490  int ii,jj, horv, i, nx, global_id, nz_ptr, Nlocal_edges;
491 
492  /* Aztec matrix and temp variables */
493 
494  int *Ke_bindx, *Ke_data_org = NULL;
495  double *Ke_val;
496  AZ_MATRIX *Ke_mat;
497  int proc_config[AZ_PROC_SIZE], *cpntr = NULL;
498  int *reordered_glob_edges = NULL, *reordered_edge_externs = NULL;
499 
500  Nlocal_edges = Edge_Partition->Nlocal;
501  nx = (int) sqrt( ((double) Edge_Partition->Nglobal/2) + .00001);
502 
503  Ke_bindx = (int *) malloc((7*Nlocal_edges+1)*sizeof(int));
504  Ke_val = (double *) malloc((7*Nlocal_edges+1)*sizeof(double));
505  Ke_bindx[0] = Nlocal_edges+1;
506 
507  dcenter = 2 + 2.*sigma/((double) ( 3 * nx * nx));
508  doff = -1 + sigma/((double) ( 6 * nx * nx));
509 
510  /* Create a DMSR matrix with global column indices */
511 
512  for (i = 0; i < Nlocal_edges; i++) {
513  global_id = (Edge_Partition->my_global_ids)[i];
514  invindex(global_id, &ii, &jj, nx, &horv);
515  nz_ptr = Ke_bindx[i];
516 
517  Ke_val[i] = dcenter;
518 
519  if (horv == HORIZONTAL) {
520  if (jj != 0) {
521  Ke_bindx[nz_ptr] = north(ii,jj,nx); Ke_val[nz_ptr++] = doff;
522  Ke_bindx[nz_ptr] = east(ii,jj,nx); Ke_val[nz_ptr++] = -1.;
523  if (ii != 0) {Ke_bindx[nz_ptr]=west(ii,jj,nx); Ke_val[nz_ptr++]= 1.;}
524  jj--;
525  }
526  else {
527  Ke_val[i] = 1. + 2.*sigma/((double) ( 3 * nx * nx));
528  jj = nx-1;
529  }
530  Ke_bindx[nz_ptr] = east(ii,jj,nx); Ke_val[nz_ptr++] = 1.;
531  if (ii != 0){ Ke_bindx[nz_ptr]=west(ii,jj,nx); Ke_val[nz_ptr++]=-1.;}
532  if (jj != 0){ Ke_bindx[nz_ptr]=south(ii,jj,nx); Ke_val[nz_ptr++]=doff;}
533  }
534  else {
535  if (ii != 0) {
536  Ke_bindx[nz_ptr] = north(ii,jj,nx); Ke_val[nz_ptr++] = -1.;
537  Ke_bindx[nz_ptr] = east(ii,jj,nx); Ke_val[nz_ptr++] = doff;
538  if (jj != 0) {Ke_bindx[nz_ptr]=south(ii,jj,nx); Ke_val[nz_ptr++]=1.;}
539  ii--;
540  }
541  else {
542  Ke_val[i] = 1 + 2.*sigma/((double) ( 3 * nx * nx));
543  ii = nx-1;
544  }
545  Ke_bindx[nz_ptr] = north(ii,jj,nx); Ke_val[nz_ptr++] = 1.;
546  if (ii != 0) {Ke_bindx[nz_ptr]=west(ii,jj,nx); Ke_val[nz_ptr++]=doff;}
547  if (jj != 0) {Ke_bindx[nz_ptr]=south(ii,jj,nx); Ke_val[nz_ptr++]=-1.;}
548  }
549  Ke_bindx[i+1] = nz_ptr;
550  }
551 
552 
553  /* Transform the global Aztec matrix into a local Aztec matrix. That is, */
554  /* replace global column indices by local indices and set up communication */
555  /* data structure 'Ke_data_org' that will be used for matvec's. */
556 
557  AZ_set_proc_config(proc_config, COMMUNICATOR);
558 
559  AZ_transform_norowreordering(proc_config, &(Edge_Partition->needed_external_ids),
560  Ke_bindx, Ke_val, Edge_Partition->my_global_ids,
561  &reordered_glob_edges, &reordered_edge_externs,
562  &Ke_data_org, Nlocal_edges, 0, 0, 0,
563  &cpntr, AZ_MSR_MATRIX);
