3  This makes it possible to solve Ux=h (ftran) faster for sparse h,
4  since we only run down the rows of U (columns of U') for which
5  the corresponding entry in h is non-zero. */
6 MYBOOL LU1U0(LUSOLrec *LUSOL, LUSOLmat **mat, int *inform)
7 {

10  int *lsumc;
11 
12  /* Assume success */
13  *inform = LUSOL_INFORM_LUSUCCESS;
14 
15  /* Check if there is anything worth doing */
16  if(mat == NULL)
17  return( status );
18  if(*mat != NULL)
19  LUSOL_matfree(mat);
20  NUMU = LUSOL->luparm[LUSOL_IP_RANK_U];
21  LENU = LUSOL->luparm[LUSOL_IP_NONZEROS_U];
22  if((NUMU == 0) || (LENU == 0) ||
23  (LUSOL->luparm[LUSOL_IP_ACCELERATION] == LUSOL_BASEORDER) ||
24  ((LUSOL->luparm[LUSOL_IP_ACCELERATION] & LUSOL_ACCELERATE_U) == 0))
25  return( status );
26

28  lsumc = (int *) LUSOL_CALLOC((LUSOL->n+1), sizeof(*lsumc));
29  if(lsumc == NULL) {
30  *inform = LUSOL_INFORM_NOMEMLEFT;
31  return( status );
32  }
33 
34  /* Compute non-zero counts by permuted column index (order is unimportant) */
35  for(L = 1; L <= LENU; L++) {
36  J = LUSOL->indr[L];
37  lsumc[J]++;
38  }
39 
40  /* Check if we should apply "smarts" before proceeding to the column matrix creation */
41  if((LUSOL->luparm[LUSOL_IP_ACCELERATION] & LUSOL_AUTOORDER) &&
42  ((REAL) sqrt((REAL) NUMU/LENU) > LUSOL->parmlu[LUSOL_RP_SMARTRATIO]))
43  goto Finish;
44 
45  /* We are Ok to create the new matrix object */
46  *mat = LUSOL_matcreate(LUSOL->n, LENU);
47  if(*mat == NULL) {
48  *inform = LUSOL_INFORM_NOMEMLEFT;
49  goto Finish;
50  }
51 
52  /* Cumulate row counts to get vector offsets; first column is leftmost
53  (stick with Fortran array offset for consistency) */
54  (*mat)->lenx[0] = 1;
55  for(K = 1; K <= LUSOL->n; K++) {
56  (*mat)->lenx[K] = (*mat)->lenx[K-1] + lsumc[K];
57  lsumc[K] = (*mat)->lenx[K-1];
58  }
59 
60  /* Map the matrix into column order by permuted index;
61  Note: The first permuted column is located leftmost in the array.
62  The row order is irrelevant, since the indeces will
63  refer to constant / resolved values of V[] during solve. */
64  for(L = 1; L <= LENU; L++) {
65  J = LUSOL->indr[L];
66  LL = lsumc[J]++;
67  (*mat)->a[LL] = LUSOL->a[L];
68  (*mat)->indr[LL] = J;
69  (*mat)->indc[LL] = LUSOL->indc[L];
70  }
71 
72  /* Pack column starting positions, and set mapper from original index to packed */
73  J = 0;
74  for(L = 1; L <= LUSOL->n; L++) {
75  K = LUSOL->iq[L];
76 #if 1 /* Deactivate to produce a full-rank version (implicit unit diagonals) */
77  if((*mat)->lenx[K] > (*mat)->lenx[K-1])
78 #endif
79  {
80  J++;
81  (*mat)->indx[J] = K;
82  }
83  }
84 
85  /* Confirm that everything went well */
86  status = TRUE;
87 
88  /* Clean up */
89 Finish:
90  FREE(lsumc);
91  return( status );
92 }
