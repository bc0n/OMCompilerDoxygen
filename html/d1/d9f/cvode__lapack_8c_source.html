122 /*
123  * -----------------------------------------------------------------
124  * CVLapackDense
125  * -----------------------------------------------------------------
126  * This routine initializes the memory record and sets various function
127  * fields specific to the linear solver module. CVLapackDense first
128  * calls the existing lfree routine if this is not NULL. Then it sets
129  * the cv_linit, cv_lsetup, cv_lsolve, cv_lfree fields in (*cvode_mem)
130  * to be cvLapackDenseInit, cvLapackDenseSetup, cvLapackDenseSolve,
131  * and cvLapackDenseFree, respectively. It allocates memory for a
132  * structure of type CVDlsMemRec and sets the cv_lmem field in
133  * (*cvode_mem) to the address of this structure. It sets setupNonNull
134  * in (*cvode_mem) to TRUE, and the d_jac field to the default
135  * cvDlsDenseDQJac. Finally, it allocates memory for M, pivots, and
136  * savedJ.
137  * The return value is SUCCESS = 0, or LMEM_FAIL = -1.
138  *
139  * NOTE: The dense linear solver assumes a serial implementation
140  * of the NVECTOR package. Therefore, CVLapackDense will first
141  * test for a compatible N_Vector internal representation
142  * by checking that N_VGetArrayPointer and N_VSetArrayPointer
143  * exist.
144  * -----------------------------------------------------------------
145  */
146 int CVLapackDense(void *cvode_mem, int N)
147 {
148  CVodeMem cv_mem;
149  CVDlsMem cvdls_mem;
150 
151  /* Return immediately if cvode_mem is NULL */
152  if (cvode_mem == NULL) {
153  cvProcessError(NULL, CVDLS_MEM_NULL, "CVLAPACK", "CVLapackDense", MSGD_CVMEM_NULL);
154  return(CVDLS_MEM_NULL);
155  }
156  cv_mem = (CVodeMem) cvode_mem;
157 
158  /* Test if the NVECTOR package is compatible with the LAPACK solver */
159  if (tempv->ops->nvgetarraypointer == NULL ||
160  tempv->ops->nvsetarraypointer == NULL) {
161  cvProcessError(cv_mem, CVDLS_ILL_INPUT, "CVLAPACK", "CVLapackDense", MSGD_BAD_NVECTOR);
162  return(CVDLS_ILL_INPUT);
163  }
164 
165  if (lfree !=NULL) lfree(cv_mem);
166 
167  /* Set four main function fields in cv_mem */
168  linit = cvLapackDenseInit;
169  lsetup = cvLapackDenseSetup;
170  lsolve = cvLapackDenseSolve;
171  lfree = cvLapackDenseFree;
172 
173  /* Get memory for CVDlsMemRec */
174  cvdls_mem = NULL;
175  cvdls_mem = (CVDlsMem) malloc(sizeof(struct CVDlsMemRec));
176  if (cvdls_mem == NULL) {
177  cvProcessError(cv_mem, CVDLS_MEM_FAIL, "CVLAPACK", "CVLapackDense", MSGD_MEM_FAIL);
178  return(CVDLS_MEM_FAIL);
179  }
180 
181  /* Set matrix type */
182  mtype = SUNDIALS_DENSE;
183 
184  /* Initialize Jacobian-related data */
185  jacDQ = TRUE;
186  djac = NULL;
187  J_data = NULL;
188 
189  last_flag = CVDLS_SUCCESS;
190  setupNonNull = TRUE;
191 
192  /* Set problem dimension */
193  n = (long int) N;
194 
195  /* Allocate memory for M, pivot array, and savedJ */
196  M = NULL;
197  pivots = NULL;
198  savedJ = NULL;
199 
200  M = NewDenseMat(n, n);
201  if (M == NULL) {
202  cvProcessError(cv_mem, CVDLS_MEM_FAIL, "CVLAPACK", "CVLapackDense", MSGD_MEM_FAIL);
203  free(cvdls_mem); cvdls_mem = NULL;
204  return(CVDLS_MEM_FAIL);
205  }
206  pivots = NewIntArray(N);
207  if (pivots == NULL) {
208  cvProcessError(cv_mem, CVDLS_MEM_FAIL, "CVLAPACK", "CVLapackDense", MSGD_MEM_FAIL);
209  DestroyMat(M);
210  free(cvdls_mem); cvdls_mem = NULL;
211  return(CVDLS_MEM_FAIL);
212  }
213  savedJ = NewDenseMat(n, n);
214  if (savedJ == NULL) {
215  cvProcessError(cv_mem, CVDLS_MEM_FAIL, "CVLAPACK", "CVLapackDense", MSGD_MEM_FAIL);
216  DestroyMat(M);
217  DestroyArray(pivots);
218  free(cvdls_mem); cvdls_mem = NULL;
219  return(CVDLS_MEM_FAIL);
220  }
221 
222  /* Attach linear solver memory to integrator memory */
223  lmem = cvdls_mem;
224 
225  return(CVDLS_SUCCESS);
226 }

228 /*
229  * -----------------------------------------------------------------
230  * CVLapackBand
231  * -----------------------------------------------------------------
232  * This routine initializes the memory record and sets various function
233  * fields specific to the band linear solver module. It first calls
234  * the existing lfree routine if this is not NULL. It then sets the
235  * cv_linit, cv_lsetup, cv_lsolve, and cv_lfree fields in (*cvode_mem)
236  * to be cvLapackBandInit, cvLapackBandSetup, cvLapackBandSolve,
237  * and cvLapackBandFree, respectively. It allocates memory for a
238  * structure of type CVLapackBandMemRec and sets the cv_lmem field in
239  * (*cvode_mem) to the address of this structure. It sets setupNonNull
240  * in (*cvode_mem) to be TRUE, mu to be mupper, ml to be mlower, and
241  * the jacE and jacI field to NULL.
242  * Finally, it allocates memory for M, pivots, and savedJ.
243  * The CVLapackBand return value is CVDLS_SUCCESS = 0,
244  * CVDLS_MEM_FAIL = -1, or CVDLS_ILL_INPUT = -2.
245  *
246  * NOTE: The CVLAPACK linear solver assumes a serial implementation
247  * of the NVECTOR package. Therefore, CVLapackBand will first
248  * test for compatible a compatible N_Vector internal
249  * representation by checking that the function
250  * N_VGetArrayPointer exists.
251  * -----------------------------------------------------------------
252  */
253 int CVLapackBand(void *cvode_mem, int N, int mupper, int mlower)
254 {
255  CVodeMem cv_mem;
256  CVDlsMem cvdls_mem;
257 
258  /* Return immediately if cvode_mem is NULL */
259  if (cvode_mem == NULL) {
260  cvProcessError(NULL, CVDLS_MEM_NULL, "CVLAPACK", "CVLapackBand", MSGD_CVMEM_NULL);
261  return(CVDLS_MEM_NULL);
262  }
263  cv_mem = (CVodeMem) cvode_mem;
264 
265  /* Test if the NVECTOR package is compatible with the BAND solver */
266  if (tempv->ops->nvgetarraypointer == NULL) {
267  cvProcessError(cv_mem, CVDLS_ILL_INPUT, "CVLAPACK", "CVLapackBand", MSGD_BAD_NVECTOR);
268  return(CVDLS_ILL_INPUT);
269  }
270 
271  if (lfree != NULL) lfree(cv_mem);
272 
273  /* Set four main function fields in cv_mem */
274  linit = cvLapackBandInit;
275  lsetup = cvLapackBandSetup;
276  lsolve = cvLapackBandSolve;
277  lfree = cvLapackBandFree;
278 
279  /* Get memory for CVDlsMemRec */
280  cvdls_mem = NULL;
281  cvdls_mem = (CVDlsMem) malloc(sizeof(struct CVDlsMemRec));
282  if (cvdls_mem == NULL) {
283  cvProcessError(cv_mem, CVDLS_MEM_FAIL, "CVLAPACK", "CVLapackBand", MSGD_MEM_FAIL);
284  return(CVDLS_MEM_FAIL);
285  }
286 
287  /* Set matrix type */
288  mtype = SUNDIALS_BAND;
289 
290  /* Initialize Jacobian-related data */
291  jacDQ = TRUE;
292  bjac = NULL;
293  J_data = NULL;
294 
295  last_flag = CVDLS_SUCCESS;
296  setupNonNull = TRUE;
297 
298  /* Load problem dimension */
299  n = (long int) N;
300 
301  /* Load half-bandwiths in cvdls_mem */
302  ml = (long int) mlower;
303  mu = (long int) mupper;
304 
305  /* Test ml and mu for legality */
306  if ((ml < 0) || (mu < 0) || (ml >= n) || (mu >= n)) {
307  cvProcessError(cv_mem, CVDLS_ILL_INPUT, "CVLAPACK", "CVLapackBand", MSGD_BAD_SIZES);
308  free(cvdls_mem); cvdls_mem = NULL;
309  return(CVDLS_ILL_INPUT);
310  }
311 
312  /* Set extended upper half-bandwith for M (required for pivoting) */
313  smu = mu + ml;
314 
315  /* Allocate memory for M, pivot array, and savedJ */
316  M = NULL;
317  pivots = NULL;
318  savedJ = NULL;
319 
320  M = NewBandMat(n, mu, ml, smu);
321  if (M == NULL) {
322  cvProcessError(cv_mem, CVDLS_MEM_FAIL, "CVLAPACK", "CVLapackBand", MSGD_MEM_FAIL);
323  free(cvdls_mem); cvdls_mem = NULL;
324  return(CVDLS_MEM_FAIL);
325  }
326  pivots = NewIntArray(N);
327  if (pivots == NULL) {
328  cvProcessError(cv_mem, CVDLS_MEM_FAIL, "CVLAPACK", "CVLapackBand", MSGD_MEM_FAIL);
329  DestroyMat(M);
330  free(cvdls_mem); cvdls_mem = NULL;
331  return(CVDLS_MEM_FAIL);
332  }
333  savedJ = NewBandMat(n, mu, ml, smu);
334  if (savedJ == NULL) {
335  cvProcessError(cv_mem, CVDLS_MEM_FAIL, "CVLAPACK", "CVLapackBand", MSGD_MEM_FAIL);
336  DestroyMat(M);
337  DestroyArray(pivots);
338  free(cvdls_mem); cvdls_mem = NULL;
339  return(CVDLS_MEM_FAIL);
340  }
341 
342  /* Attach linear solver memory to integrator memory */
343  lmem = cvdls_mem;
344 
345  return(CVDLS_SUCCESS);
346 }

380 /*
381  * cvLapackDenseSetup does the setup operations for the dense linear solver.
382  * It makes a decision whether or not to call the Jacobian evaluation
383  * routine based on various state variables, and if not it uses the
384  * saved copy. In any case, it constructs the Newton matrix M = I - gamma*J
385  * updates counters, and calls the dense LU factorization routine.
386  */
387 static int cvLapackDenseSetup(CVodeMem cv_mem, int convfail,
388  N_Vector yP, N_Vector fctP,
389  booleantype *jcurPtr,
390  N_Vector tmp1, N_Vector tmp2, N_Vector tmp3)
391 {
392  CVDlsMem cvdls_mem;
393  realtype dgamma, fact;
394  booleantype jbad, jok;
395  int ier, retval, one = 1;
396  int intn, lenmat;
397 
398  cvdls_mem = (CVDlsMem) lmem;
399  intn = (int) n;
400  lenmat = M->ldata ;
401 
402  /* Use nst, gamma/gammap, and convfail to set J eval. flag jok */
403  dgamma = SUNRabs((gamma/gammap) - ONE);
404  jbad = (nst == 0) || (nst > nstlj + CVD_MSBJ) ||
405  ((convfail == CV_FAIL_BAD_J) && (dgamma < CVD_DGMAX)) ||
406  (convfail == CV_FAIL_OTHER);
407  jok = !jbad;
408 
409  if (jok) {
410 
411  /* If jok = TRUE, use saved copy of J */
412  *jcurPtr = FALSE;
413  dcopy_f77(&lenmat, savedJ->data, &one, M->data, &one);
414 
415  } else {
416 
417  /* If jok = FALSE, call jac routine for new J value */
418  nje++;
419  nstlj = nst;
420  *jcurPtr = TRUE;
421  SetToZero(M);
422 
423  retval = djac(n, tn, yP, fctP, M, J_data, tmp1, tmp2, tmp3);
424 
425  if (retval == 0) {
426  dcopy_f77(&lenmat, M->data, &one, savedJ->data, &one);
427  } else if (retval < 0) {
428  cvProcessError(cv_mem, CVDLS_JACFUNC_UNRECVR, "CVLAPACK", "cvLapackDenseSetup", MSGD_JACFUNC_FAILED);
429  last_flag = CVDLS_JACFUNC_UNRECVR;
430  return(-1);
431  } else if (retval > 0) {
432  last_flag = CVDLS_JACFUNC_RECVR;
433  return(1);
434  }
435 
436  }
437 
438  /* Scale J by - gamma */
439  fact = -gamma;
440  dscal_f77(&lenmat, &fact, M->data, &one);
441 
442  /* Add identity to get M = I - gamma*J*/
443  AddIdentity(M);
444 
445  /* Do LU factorization of M */
446  dgetrf_f77(&intn, &intn, M->data, &intn, pivots, &ier);
447 
448  /* Return 0 if the LU was complete; otherwise return 1 */
449  last_flag = (long int) ier;
450  if (ier > 0) return(1);
451  return(0);
452 }

454 /*
455  * cvLapackDenseSolve handles the solve operation for the dense linear solver
456  * by calling the dense backsolve routine.
457  */
458 static int cvLapackDenseSolve(CVodeMem cv_mem, N_Vector b, N_Vector weight,
459  N_Vector yC, N_Vector fctC)
460 {
461  CVDlsMem cvdls_mem;
462  realtype *bd, fact;
463  int ier, one = 1;
464  int intn;
465 
466  cvdls_mem = (CVDlsMem) lmem;
467  intn = (int) n;
468 
469  bd = N_VGetArrayPointer(b);
470 
471  dgetrs_f77("N", &intn, &one, M->data, &intn, pivots, bd, &intn, &ier, 1);
472 
473  if (ier > 0) return(1);
474 
475  /* For BDF, scale the correction to account for change in gamma */
476  if ((lmm == CV_BDF) && (gamrat != ONE)) {
477  fact = TWO/(ONE + gamrat);
478  dscal_f77(&intn, &fact, bd, &one);
479  }
480 
481  last_flag = CVDLS_SUCCESS;
482  return(0);
483 }
