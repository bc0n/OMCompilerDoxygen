133 /* clear all mark bits in the header */
134 GC_INNER void GC_clear_hdr_marks(hdr *hhdr)
135 {
136  size_t last_bit;
137 
138 # ifdef AO_HAVE_load
139  /* Atomic access is used to avoid racing with GC_realloc. */
140  last_bit = FINAL_MARK_BIT((size_t)AO_load((volatile AO_t *)&hhdr->hb_sz));
141 # else
142  /* No race as GC_realloc holds the lock while updating hb_sz. */
143  last_bit = FINAL_MARK_BIT((size_t)hhdr->hb_sz);
144 # endif
145 
146  BZERO(hhdr -> hb_marks, sizeof(hhdr->hb_marks));
147  set_mark_bit_from_hdr(hhdr, last_bit);
148  hhdr -> hb_n_marks = 0;
149 }

151 /* Set all mark bits in the header. Used for uncollectible blocks. */
152 GC_INNER void GC_set_hdr_marks(hdr *hhdr)
153 {
154  unsigned i;
155  size_t sz = (size_t)hhdr->hb_sz;
156  unsigned n_marks = (unsigned)FINAL_MARK_BIT(sz);
157 
158 # ifdef USE_MARK_BYTES
159  for (i = 0; i <= n_marks; i += (unsigned)MARK_BIT_OFFSET(sz)) {
160  hhdr -> hb_marks[i] = 1;
161  }
162 # else
163  for (i = 0; i < divWORDSZ(n_marks + WORDSZ); ++i) {
164  hhdr -> hb_marks[i] = ONES;
165  }
166 # endif
167 # ifdef MARK_BIT_PER_OBJ
168  hhdr -> hb_n_marks = n_marks;
169 # else
170  hhdr -> hb_n_marks = HBLK_OBJS(sz);
171 # endif
172 }

189 GC_API void GC_CALL GC_set_mark_bit(const void *p)
190 {
191  struct hblk *h = HBLKPTR(p);
192  hdr * hhdr = HDR(h);
193  word bit_no = MARK_BIT_NO((ptr_t)p - (ptr_t)h, hhdr -> hb_sz);
194 
195  if (!mark_bit_from_hdr(hhdr, bit_no)) {
196  set_mark_bit_from_hdr(hhdr, bit_no);

201 GC_API void GC_CALL GC_clear_mark_bit(const void *p)
202 {
203  struct hblk *h = HBLKPTR(p);
204  hdr * hhdr = HDR(h);
205  word bit_no = MARK_BIT_NO((ptr_t)p - (ptr_t)h, hhdr -> hb_sz);
206 
207  if (mark_bit_from_hdr(hhdr, bit_no)) {
208  size_t n_marks = hhdr -> hb_n_marks;
209 
210  GC_ASSERT(n_marks != 0);
211  clear_mark_bit_from_hdr(hhdr, bit_no);
212  n_marks--;
213 # ifdef PARALLEL_MARK
214  if (n_marks != 0 || !GC_parallel)
215  hhdr -> hb_n_marks = n_marks;
216  /* Don't decrement to zero. The counts are approximate due to */
217  /* concurrency issues, but we need to ensure that a count of */
218  /* zero implies an empty block. */
219 # else
220  hhdr -> hb_n_marks = n_marks;
221 # endif
222  }

292 static void alloc_mark_stack(size_t);

294 /* Perform a small amount of marking. */
295 /* We try to touch roughly a page of memory. */
296 /* Return TRUE if we just finished a mark phase. */
297 /* Cold_gc_frame is an address inside a GC frame that */
298 /* remains valid until all marking is complete. */
299 /* A zero value indicates that it's OK to miss some */
300 /* register values. */
301 /* We hold the allocation lock. In the case of */
302 /* incremental collection, the world may not be stopped.*/
303 #ifdef WRAP_MARK_SOME
304  /* For win32, this is called after we establish a structured */
305  /* exception handler, in case Windows unmaps one of our root */
306  /* segments. See below. In either case, we acquire the */
307  /* allocator lock long before we get here. */
308  STATIC GC_bool GC_mark_some_inner(ptr_t cold_gc_frame)
309 #else
310  GC_INNER GC_bool GC_mark_some(ptr_t cold_gc_frame)
311 #endif
312 {
313  switch(GC_mark_state) {
314  case MS_NONE:
315  break;
316 
317  case MS_PUSH_RESCUERS:
318  if ((word)GC_mark_stack_top
319  >= (word)(GC_mark_stack_limit - INITIAL_MARK_STACK_SIZE/2)) {
320  /* Go ahead and mark, even though that might cause us to */
321  /* see more marked dirty objects later on. Avoid this */
322  /* in the future. */
323  GC_mark_stack_too_small = TRUE;
324  MARK_FROM_MARK_STACK();
325  break;
326  } else {
327  scan_ptr = GC_push_next_marked_dirty(scan_ptr);
328  if (scan_ptr == 0) {
329 # if !defined(GC_DISABLE_INCREMENTAL)
330  GC_COND_LOG_PRINTF("Marked from %lu dirty pages\n",
331  (unsigned long)GC_n_rescuing_pages);
332 # endif
333  GC_push_roots(FALSE, cold_gc_frame);
334  GC_objects_are_marked = TRUE;
335  if (GC_mark_state != MS_INVALID) {
336  GC_mark_state = MS_ROOTS_PUSHED;
337  }
338  }
339  }
340  break;
341 
342  case MS_PUSH_UNCOLLECTABLE:
343  if ((word)GC_mark_stack_top
344  >= (word)(GC_mark_stack + GC_mark_stack_size/4)) {
345 # ifdef PARALLEL_MARK
346  /* Avoid this, since we don't parallelize the marker */
347  /* here. */
348  if (GC_parallel) GC_mark_stack_too_small = TRUE;
349 # endif
350  MARK_FROM_MARK_STACK();
351  break;
352  } else {
353  scan_ptr = GC_push_next_marked_uncollectable(scan_ptr);
354  if (scan_ptr == 0) {
355  GC_push_roots(TRUE, cold_gc_frame);
356  GC_objects_are_marked = TRUE;
357  if (GC_mark_state != MS_INVALID) {
358  GC_mark_state = MS_ROOTS_PUSHED;
359  }
360  }
361  }
362  break;
363 
364  case MS_ROOTS_PUSHED:
365 # ifdef PARALLEL_MARK
366  /* In the incremental GC case, this currently doesn't */
367  /* quite do the right thing, since it runs to */
368  /* completion. On the other hand, starting a */
369  /* parallel marker is expensive, so perhaps it is */
370  /* the right thing? */
371  /* Eventually, incremental marking should run */
372  /* asynchronously in multiple threads, without grabbing */
373  /* the allocation lock. */
374  if (GC_parallel) {
375  GC_do_parallel_mark();
376  GC_ASSERT((word)GC_mark_stack_top < (word)GC_first_nonempty);
377  GC_mark_stack_top = GC_mark_stack - 1;
378  if (GC_mark_stack_too_small) {
379  alloc_mark_stack(2*GC_mark_stack_size);
380  }
381  if (GC_mark_state == MS_ROOTS_PUSHED) {
382  GC_mark_state = MS_NONE;
383  return(TRUE);
384  }
385  break;
386  }
387 # endif
388  if ((word)GC_mark_stack_top >= (word)GC_mark_stack) {
389  MARK_FROM_MARK_STACK();
390  break;
391  } else {
392  GC_mark_state = MS_NONE;
393  if (GC_mark_stack_too_small) {
394  alloc_mark_stack(2*GC_mark_stack_size);
395  }
396  return(TRUE);
397  }
398 
399  case MS_INVALID:
400  case MS_PARTIALLY_INVALID:
401  if (!GC_objects_are_marked) {
402  GC_mark_state = MS_PUSH_UNCOLLECTABLE;
403  break;
404  }
405  if ((word)GC_mark_stack_top >= (word)GC_mark_stack) {
406  MARK_FROM_MARK_STACK();
407  break;
408  }
409  if (scan_ptr == 0 && GC_mark_state == MS_INVALID) {
410  /* About to start a heap scan for marked objects. */
411  /* Mark stack is empty. OK to reallocate. */
412  if (GC_mark_stack_too_small) {
413  alloc_mark_stack(2*GC_mark_stack_size);
414  }
415  GC_mark_state = MS_PARTIALLY_INVALID;
416  }
417  scan_ptr = GC_push_next_marked(scan_ptr);
418  if (scan_ptr == 0 && GC_mark_state == MS_PARTIALLY_INVALID) {
419  GC_push_roots(TRUE, cold_gc_frame);
420  GC_objects_are_marked = TRUE;
421  if (GC_mark_state != MS_INVALID) {
422  GC_mark_state = MS_ROOTS_PUSHED;
423  }
424  }
425  break;
426 
427  default:
428  ABORT("GC_mark_some: bad state");
429  }
430  return(FALSE);
431 }

471  GC_INNER GC_bool GC_mark_some(ptr_t cold_gc_frame)
472  {
473  GC_bool ret_val;
474 
475 # if defined(MSWIN32) || defined(MSWINCE)
476 # ifndef __GNUC__
477  /* Windows 98 appears to asynchronously create and remove */
478  /* writable memory mappings, for reasons we haven't yet */
479  /* understood. Since we look for writable regions to */
480  /* determine the root set, we may try to mark from an */
481  /* address range that disappeared since we started the */
482  /* collection. Thus we have to recover from faults here. */
483  /* This code does not appear to be necessary for Windows */
484  /* 95/NT/2000+. Note that this code should never generate */
485  /* an incremental GC write fault. */
486  /* This code seems to be necessary for WinCE (at least in */
487  /* the case we'd decide to add MEM_PRIVATE sections to */
488  /* data roots in GC_register_dynamic_libraries()). */
489  /* It's conceivable that this is the same issue with */
490  /* terminating threads that we see with Linux and */
491  /* USE_PROC_FOR_LIBRARIES. */
492 
493  __try {
494  ret_val = GC_mark_some_inner(cold_gc_frame);
495  } __except (GetExceptionCode() == EXCEPTION_ACCESS_VIOLATION ?
496  EXCEPTION_EXECUTE_HANDLER : EXCEPTION_CONTINUE_SEARCH) {
497  goto handle_ex;
498  }
499 # if defined(GC_WIN32_THREADS) && !defined(GC_PTHREADS)
500  /* With DllMain-based thread tracking, a thread may have */
501  /* started while we were marking. This is logically equivalent */
502  /* to the exception case; our results are invalid and we have */
503  /* to start over. This cannot be prevented since we can't */
504  /* block in DllMain. */
505  if (GC_started_thread_while_stopped()) goto handle_ex;
506 # endif
507  rm_handler:
508  return ret_val;
509 
510 # else /* __GNUC__ */
511 
512  /* Manually install an exception handler since GCC does */
513  /* not yet support Structured Exception Handling (SEH) on */
514  /* Win32. */
515 
516  ext_ex_regn er;
517 
518 # if GC_GNUC_PREREQ(4, 7) || GC_CLANG_PREREQ(3, 3)
519 # pragma GCC diagnostic push
520  /* Suppress "taking the address of label is non-standard" warning. */
521 # if defined(__clang__) || GC_GNUC_PREREQ(6, 4)
522 # pragma GCC diagnostic ignored "-Wpedantic"
523 # else
524  /* GCC before ~4.8 does not accept "-Wpedantic" quietly. */
525 # pragma GCC diagnostic ignored "-pedantic"
526 # endif
527  er.alt_path = &&handle_ex;
528 # pragma GCC diagnostic pop
529 # else /* pragma diagnostic is not supported */
530  er.alt_path = &&handle_ex;
531 # endif
532  er.ex_reg.handler = mark_ex_handler;
533  __asm__ __volatile__ ("movl %%fs:0, %0" : "=r" (er.ex_reg.prev));
534  __asm__ __volatile__ ("movl %0, %%fs:0" : : "r" (&er));
535  ret_val = GC_mark_some_inner(cold_gc_frame);
536  /* Prevent GCC from considering the following code unreachable */
537  /* and thus eliminating it. */
538  if (er.alt_path == 0)
539  goto handle_ex;
540 # if defined(GC_WIN32_THREADS) && !defined(GC_PTHREADS)
541  if (GC_started_thread_while_stopped())
542  goto handle_ex;
543 # endif
544  rm_handler:
545  /* Uninstall the exception handler */
546  __asm__ __volatile__ ("mov %0, %%fs:0" : : "r" (er.ex_reg.prev));
547  return ret_val;
548 
549 # endif /* __GNUC__ */
550 # else /* !MSWIN32 */
551  /* Here we are handling the case in which /proc is used for root */
552  /* finding, and we have threads. We may find a stack for a */
553  /* thread that is in the process of exiting, and disappears */
554  /* while we are marking it. This seems extremely difficult to */
555  /* avoid otherwise. */
556  if (GC_incremental) {
557  WARN("Incremental GC incompatible with /proc roots\n", 0);
558  /* I'm not sure if this could still work ... */
559  }
560  GC_setup_temporary_fault_handler();
561  if(SETJMP(GC_jmp_buf) != 0) goto handle_ex;
562  ret_val = GC_mark_some_inner(cold_gc_frame);
563  rm_handler:
564  GC_reset_fault_handler();
565  return ret_val;
566 
567 # endif /* !MSWIN32 */
568

570  /* Exception handler starts here for all cases. */
571  {
572  static word warned_gc_no;
573 
574  /* Warn about it at most once per collection. */
575  if (warned_gc_no != GC_gc_no) {
576  warned_gc_no = GC_gc_no;
577  WARN("Caught ACCESS_VIOLATION in marker;"
578  " memory mapping disappeared\n", 0);
579  }
580  }
581  /* We have bad roots on the stack. Discard mark stack. */
582  /* Rescan from marked objects. Redetermine roots. */
583 # ifdef REGISTER_LIBRARIES_EARLY
584  START_WORLD();
585  GC_cond_register_dynamic_libraries();
586  STOP_WORLD();
587 # endif
588  GC_invalidate_mark_state();
589  scan_ptr = 0;
590 
591  ret_val = FALSE;
592  goto rm_handler; /* Back to platform-specific code. */
593  }

619 /*
620  * Mark objects pointed to by the regions described by
621  * mark stack entries between mark_stack and mark_stack_top,
622  * inclusive. Assumes the upper limit of a mark stack entry
623  * is never 0. A mark stack entry never has size 0.
624  * We try to traverse on the order of a hblk of memory before we return.
625  * Caller is responsible for calling this until the mark stack is empty.
626  * Note that this is the most performance critical routine in the
627  * collector. Hence it contains all sorts of ugly hacks to speed
628  * things up. In particular, we avoid procedure calls on the common
629  * path, we take advantage of peculiarities of the mark descriptor
630  * encoding, we optionally maintain a cache for the block address to
631  * header mapping, we prefetch when an object is "grayed", etc.
632  */
633 GC_ATTR_NO_SANITIZE_ADDR GC_ATTR_NO_SANITIZE_MEMORY GC_ATTR_NO_SANITIZE_THREAD
634 GC_INNER mse * GC_mark_from(mse *mark_stack_top, mse *mark_stack,
635  mse *mark_stack_limit)
636 {
637  signed_word credit = HBLKSIZE; /* Remaining credit for marking work */
638  ptr_t current_p; /* Pointer to current candidate ptr. */
639  word current; /* Candidate pointer. */
640  ptr_t limit = 0; /* (Incl) limit of current candidate range. */
641  word descr;
642  ptr_t greatest_ha = (ptr_t)GC_greatest_plausible_heap_addr;
643  ptr_t least_ha = (ptr_t)GC_least_plausible_heap_addr;
644  DECLARE_HDR_CACHE;
645 
646 # define SPLIT_RANGE_WORDS 128 /* Must be power of 2. */
647 
648  GC_objects_are_marked = TRUE;
649  INIT_HDR_CACHE;
650 # ifdef OS2 /* Use untweaked version to circumvent compiler problem */
651  while ((word)mark_stack_top >= (word)mark_stack && credit >= 0)
652 # else
653  while (((((word)mark_stack_top - (word)mark_stack) | (word)credit)
654  & SIGNB) == 0)
655 # endif
656  {
657  current_p = mark_stack_top -> mse_start;
658  descr = mark_stack_top -> mse_descr.w;
659  retry:
660  /* current_p and descr describe the current object. */
661  /* *mark_stack_top is vacant. */
662  /* The following is 0 only for small objects described by a simple */
663  /* length descriptor. For many applications this is the common */
664  /* case, so we try to detect it quickly. */
665  if (descr & ((~(WORDS_TO_BYTES(SPLIT_RANGE_WORDS) - 1)) | GC_DS_TAGS)) {
666  word tag = descr & GC_DS_TAGS;
667 
668  GC_STATIC_ASSERT(GC_DS_TAGS == 0x3);
669  switch(tag) {
670  case GC_DS_LENGTH:
671  /* Large length. */
672  /* Process part of the range to avoid pushing too much on the */
673  /* stack. */
674  GC_ASSERT(descr < (word)GC_greatest_plausible_heap_addr
675  - (word)GC_least_plausible_heap_addr
676  || (word)(current_p + descr)
677  <= (word)GC_least_plausible_heap_addr
678  || (word)current_p >= (word)GC_greatest_plausible_heap_addr);
679 # ifdef PARALLEL_MARK
680 # define SHARE_BYTES 2048
681  if (descr > SHARE_BYTES && GC_parallel
682  && (word)mark_stack_top < (word)(mark_stack_limit - 1)) {
683  word new_size = (descr/2) & ~(word)(sizeof(word)-1);
684 
685  mark_stack_top -> mse_start = current_p;
686  mark_stack_top -> mse_descr.w = new_size + sizeof(word);
687  /* makes sure we handle */
688  /* misaligned pointers. */
689  mark_stack_top++;
690 # ifdef ENABLE_TRACE
691  if ((word)GC_trace_addr >= (word)current_p
692  && (word)GC_trace_addr < (word)(current_p + descr)) {
693  GC_log_printf("GC #%u: large section; start %p, len %lu,"
694  " splitting (parallel) at %p\n",
695  (unsigned)GC_gc_no, (void *)current_p,
696  (unsigned long)descr,
697  (void *)(current_p + new_size));
698  }
699 # endif
700  current_p += new_size;
701  descr -= new_size;
702  goto retry;
703  }
704 # endif /* PARALLEL_MARK */
705  mark_stack_top -> mse_start =
706  limit = current_p + WORDS_TO_BYTES(SPLIT_RANGE_WORDS-1);
707  mark_stack_top -> mse_descr.w =
708  descr - WORDS_TO_BYTES(SPLIT_RANGE_WORDS-1);
709 # ifdef ENABLE_TRACE
710  if ((word)GC_trace_addr >= (word)current_p
711  && (word)GC_trace_addr < (word)(current_p + descr)) {
712  GC_log_printf("GC #%u: large section; start %p, len %lu,"
713  " splitting at %p\n",
714  (unsigned)GC_gc_no, (void *)current_p,
715  (unsigned long)descr, (void *)limit);
716  }
717 # endif
718  /* Make sure that pointers overlapping the two ranges are */
719  /* considered. */
720  limit += sizeof(word) - ALIGNMENT;
721  break;
722  case GC_DS_BITMAP:
723  mark_stack_top--;
724 # ifdef ENABLE_TRACE
725  if ((word)GC_trace_addr >= (word)current_p
726  && (word)GC_trace_addr < (word)(current_p
727  + WORDS_TO_BYTES(WORDSZ-2))) {
728  GC_log_printf("GC #%u: tracing from %p bitmap descr %lu\n",
729  (unsigned)GC_gc_no, (void *)current_p,
730  (unsigned long)descr);
731  }
732 # endif /* ENABLE_TRACE */
733  descr &= ~GC_DS_TAGS;
734  credit -= WORDS_TO_BYTES(WORDSZ/2); /* guess */
735  while (descr != 0) {
736  if ((descr & SIGNB) != 0) {
737  current = *(word *)current_p;
738  FIXUP_POINTER(current);
739  if (current >= (word)least_ha && current < (word)greatest_ha) {
740  PREFETCH((ptr_t)current);
741 # ifdef ENABLE_TRACE
742  if (GC_trace_addr == current_p) {
743  GC_log_printf("GC #%u: considering(3) %p -> %p\n",
744  (unsigned)GC_gc_no, (void *)current_p,
745  (void *)current);
746  }
747 # endif /* ENABLE_TRACE */
748  PUSH_CONTENTS((ptr_t)current, mark_stack_top,
749  mark_stack_limit, current_p);
750  }
751  }
752  descr <<= 1;
753  current_p += sizeof(word);
754  }
755  continue;
756  case GC_DS_PROC:
757  mark_stack_top--;
758 # ifdef ENABLE_TRACE
759  if ((word)GC_trace_addr >= (word)current_p
760  && GC_base(current_p) != 0
761  && GC_base(current_p) == GC_base(GC_trace_addr)) {
762  GC_log_printf("GC #%u: tracing from %p, proc descr %lu\n",
763  (unsigned)GC_gc_no, (void *)current_p,
764  (unsigned long)descr);
765  }
766 # endif /* ENABLE_TRACE */
767  credit -= GC_PROC_BYTES;
768  mark_stack_top = (*PROC(descr))((word *)current_p, mark_stack_top,
769  mark_stack_limit, ENV(descr));
770  continue;
771  case GC_DS_PER_OBJECT:
772  if ((signed_word)descr >= 0) {
773  /* Descriptor is in the object. */
774  descr = *(word *)(current_p + descr - GC_DS_PER_OBJECT);
775  } else {
776  /* Descriptor is in type descriptor pointed to by first */
777  /* word in object. */
778  ptr_t type_descr = *(ptr_t *)current_p;
779  /* type_descr is either a valid pointer to the descriptor */
780  /* structure, or this object was on a free list. */
781  /* If it was anything but the last object on the free list, */
782  /* we will misinterpret the next object on the free list as */
783  /* the type descriptor, and get a 0 GC descriptor, which */
784  /* is ideal. Unfortunately, we need to check for the last */
785  /* object case explicitly. */
786  if (EXPECT(0 == type_descr, FALSE)) {
787  mark_stack_top--;
788  continue;
789  }
790  descr = *(word *)(type_descr
791  - ((signed_word)descr + (GC_INDIR_PER_OBJ_BIAS
792  - GC_DS_PER_OBJECT)));
793  }
794  if (0 == descr) {
795  /* Can happen either because we generated a 0 descriptor */
796  /* or we saw a pointer to a free object. */
797  mark_stack_top--;
798  continue;
799  }
800  goto retry;
801  }
802  } else /* Small object with length descriptor */ {
803  mark_stack_top--;
804 # ifndef SMALL_CONFIG
805  if (descr < sizeof(word))
806  continue;
807 # endif
808 # ifdef ENABLE_TRACE
809  if ((word)GC_trace_addr >= (word)current_p
810  && (word)GC_trace_addr < (word)(current_p + descr)) {
811  GC_log_printf("GC #%u: small object; start %p, len %lu\n",
812  (unsigned)GC_gc_no, (void *)current_p,
813  (unsigned long)descr);
814  }
815 # endif
816  limit = current_p + (word)descr;
817  }
818  /* The simple case in which we're scanning a range. */
819  GC_ASSERT(!((word)current_p & (ALIGNMENT-1)));
820  credit -= limit - current_p;
821  limit -= sizeof(word);
822  {
823 # define PREF_DIST 4
824 
825 # ifndef SMALL_CONFIG
826  word deferred;
827 
828  /* Try to prefetch the next pointer to be examined ASAP. */
829  /* Empirically, this also seems to help slightly without */
830  /* prefetches, at least on linux/X86. Presumably this loop */
831  /* ends up with less register pressure, and gcc thus ends up */
832  /* generating slightly better code. Overall gcc code quality */
833  /* for this loop is still not great. */
834  for(;;) {
835  PREFETCH(limit - PREF_DIST*CACHE_LINE_SIZE);
836  GC_ASSERT((word)limit >= (word)current_p);
837  deferred = *(word *)limit;
838  FIXUP_POINTER(deferred);
839  limit -= ALIGNMENT;
840  if (deferred >= (word)least_ha && deferred < (word)greatest_ha) {
841  PREFETCH((ptr_t)deferred);
842  break;
843  }
844  if ((word)current_p > (word)limit) goto next_object;
845  /* Unroll once, so we don't do too many of the prefetches */
846  /* based on limit. */
847  deferred = *(word *)limit;
848  FIXUP_POINTER(deferred);
849  limit -= ALIGNMENT;
850  if (deferred >= (word)least_ha && deferred < (word)greatest_ha) {
851  PREFETCH((ptr_t)deferred);
852  break;
853  }
854  if ((word)current_p > (word)limit) goto next_object;
855  }
856 # endif
857 
858  while ((word)current_p <= (word)limit) {
859  /* Empirically, unrolling this loop doesn't help a lot. */
860  /* Since PUSH_CONTENTS expands to a lot of code, */
861  /* we don't. */
862  current = *(word *)current_p;
863  FIXUP_POINTER(current);
864  PREFETCH(current_p + PREF_DIST*CACHE_LINE_SIZE);
865  if (current >= (word)least_ha && current < (word)greatest_ha) {
866  /* Prefetch the contents of the object we just pushed. It's */
867  /* likely we will need them soon. */
868  PREFETCH((ptr_t)current);
869 # ifdef ENABLE_TRACE
870  if (GC_trace_addr == current_p) {
871  GC_log_printf("GC #%u: considering(1) %p -> %p\n",
872  (unsigned)GC_gc_no, (void *)current_p,
873  (void *)current);
874  }
875 # endif /* ENABLE_TRACE */
876  PUSH_CONTENTS((ptr_t)current, mark_stack_top,
877  mark_stack_limit, current_p);
878  }
879  current_p += ALIGNMENT;
880  }
881 
882 # ifndef SMALL_CONFIG
883  /* We still need to mark the entry we previously prefetched. */
884  /* We already know that it passes the preliminary pointer */
885  /* validity test. */
886 # ifdef ENABLE_TRACE
887  if (GC_trace_addr == current_p) {
888  GC_log_printf("GC #%u: considering(2) %p -> %p\n",
889  (unsigned)GC_gc_no, (void *)current_p,
890  (void *)deferred);
891  }
892 # endif /* ENABLE_TRACE */
893  PUSH_CONTENTS((ptr_t)deferred, mark_stack_top,
894  mark_stack_limit, current_p);
895  next_object:;
896 # endif
897  }
898  }
899  return mark_stack_top;
900 }

964 /* until we either steal mse high, or we have max entries. */

966 /* *next is replaced by a pointer to the next unscanned mark stack */
967 /* entry. */
968 STATIC mse * GC_steal_mark_stack(mse * low, mse * high, mse * local,
969  unsigned max, mse **next)
970 {
971  mse *p;
972  mse *top = local - 1;
973  unsigned i = 0;
974 
975  GC_ASSERT((word)high >= (word)(low - 1)
976  && (word)(high - low + 1) <= GC_mark_stack_size);
977  for (p = low; (word)p <= (word)high && i <= max; ++p) {
978  word descr = (word)AO_load(&p->mse_descr.ao);
979  if (descr != 0) {
980  /* Must be ordered after read of descr: */
981  AO_store_release_write(&p->mse_descr.ao, 0);
982  /* More than one thread may get this entry, but that's only */
983  /* a minor performance problem. */
984  ++top;
985  top -> mse_descr.w = descr;
986  top -> mse_start = p -> mse_start;
987  GC_ASSERT((descr & GC_DS_TAGS) != GC_DS_LENGTH
988  || descr < (word)GC_greatest_plausible_heap_addr
989  - (word)GC_least_plausible_heap_addr
990  || (word)(p->mse_start + descr)
991  <= (word)GC_least_plausible_heap_addr
992  || (word)p->mse_start
993  >= (word)GC_greatest_plausible_heap_addr);
994  /* If this is a big object, count it as */
995  /* size/256 + 1 objects. */
996  ++i;
997  if ((descr & GC_DS_TAGS) == GC_DS_LENGTH) i += (int)(descr >> 8);
998  }
999  }

1096 /* Mark using the local mark stack until the global mark stack is empty */
1097 /* and there are no active workers. Update GC_first_nonempty to reflect */
1098 /* progress. Caller holds the mark lock. */
1099 /* Caller has already incremented GC_helper_count. We decrement it, */
1100 /* and maintain GC_active_count. */
1101 STATIC void GC_mark_local(mse *local_mark_stack, int id)
1102 {
1103  mse * my_first_nonempty;
1104 
1105  GC_active_count++;
1106  my_first_nonempty = (mse *)AO_load(&GC_first_nonempty);
1107  GC_ASSERT((word)GC_mark_stack <= (word)my_first_nonempty);
1108  GC_ASSERT((word)my_first_nonempty
1109  <= (word)AO_load((volatile AO_t *)&GC_mark_stack_top) + sizeof(mse));
1110  GC_VERBOSE_LOG_PRINTF("Starting mark helper %d\n", id);
1111  GC_release_mark_lock();
1112  for (;;) {
1113  size_t n_on_stack;
1114  unsigned n_to_get;
1115  mse * my_top;
1116  mse * local_top;
1117  mse * global_first_nonempty = (mse *)AO_load(&GC_first_nonempty);
1118 
1119  GC_ASSERT((word)my_first_nonempty >= (word)GC_mark_stack &&
1120  (word)my_first_nonempty <=
1121  (word)AO_load((volatile AO_t *)&GC_mark_stack_top)
1122  + sizeof(mse));
1123  GC_ASSERT((word)global_first_nonempty >= (word)GC_mark_stack);
1124  if ((word)my_first_nonempty < (word)global_first_nonempty) {
1125  my_first_nonempty = global_first_nonempty;
1126  } else if ((word)global_first_nonempty < (word)my_first_nonempty) {
1127  (void)AO_compare_and_swap(&GC_first_nonempty,
1128  (AO_t)global_first_nonempty,
1129  (AO_t)my_first_nonempty);
1130  /* If this fails, we just go ahead, without updating */
1131  /* GC_first_nonempty. */
1132  }
1133  /* Perhaps we should also update GC_first_nonempty, if it */
1134  /* is less. But that would require using atomic updates. */
1135  my_top = (mse *)AO_load_acquire((volatile AO_t *)(&GC_mark_stack_top));
1136  if ((word)my_top < (word)my_first_nonempty) {
1137  GC_acquire_mark_lock();
1138  my_top = GC_mark_stack_top;
1139  /* Asynchronous modification impossible here, */
1140  /* since we hold mark lock. */
1141  n_on_stack = my_top - my_first_nonempty + 1;
1142  if (0 == n_on_stack) {
1143  GC_active_count--;
1144  GC_ASSERT(GC_active_count <= GC_helper_count);
1145  /* Other markers may redeposit objects */
1146  /* on the stack. */
1147  if (0 == GC_active_count) GC_notify_all_marker();
1148  while (GC_active_count > 0
1149  && (word)AO_load(&GC_first_nonempty)
1150  > (word)GC_mark_stack_top) {
1151  /* We will be notified if either GC_active_count */
1152  /* reaches zero, or if more objects are pushed on */
1153  /* the global mark stack. */
1154  GC_wait_marker();
1155  }
