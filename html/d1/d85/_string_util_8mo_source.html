162 public function wordWrap
163  "Breaks the given string into lines which are no longer than the given wrap
164  length. The function tries to break lines at word boundaries, i.e. at spaces,
165  so that words are not split. It also wraps the string at any newline
166  characters it finds. The function also takes two optional parameters to set
167  the delimiter and raggedness.
168 
169  inDelimiter sets the delimiter which is prefixed to all lines except for the
170  first one. The length of this delimiter is taken into account when wrapping
171  the string, so it must be shorter than the wrap length. Otherwise the string
172  will be returned unwrapped. The default is an empty string.
173 
174  inRaggedness determines the allowed raggedness of the lines, given as a ratio
175  between 0 and 1. A raggedness of e.g. 0.2 means that each segment may be at
176  most 20% smaller than the max line length. If a line would be shorter than
177  this, due to a long word, then the function instead hyphenates the last word.
178  This is not done according to any grammatical rules, the words are just
179  broken so that the line is as long as allowed. The default is 0.3.
180 
181  This function operates on ASCII strings, and does not handle UTF-8 strings
182  correctly."
183  input String inString;
184  input Integer inWrapLength;
185  input String inDelimiter = "";
186  input Real inRaggedness = 0.3;
187  output list<String> outStrings = {};
188 protected
189  Integer start_pos = 1, end_pos = inWrapLength;
190  Integer line_len, pos, next_char, char, gap_size, next_gap_size;
191  String str, delim = "";
192  list<String> lines;
193 algorithm
194  // Check that the wrap length is larger than the delimiter, otherwise just
195  // return the string as it is.
196  if stringLength(inDelimiter) >= inWrapLength - 1 then
197  outStrings := {inString};
198  return;
199  end if;
200 
201  // Split the string at newlines.
202  lines := System.strtok(inString, "\n");
203  // Calculate the length of each line, excluding the delimiter.
204  line_len := inWrapLength - stringLength(inDelimiter) - 1;
205  // The gap size is how many characters a line may be shorter than the sought
206  // after line length.
207  gap_size := max(realInt(realMul(line_len, inRaggedness)), 0);
208 
209  // Wrap each line separately.
210  for line in lines loop
211  while end_pos < stringLength(line) loop
212  next_char := stringGetNoBoundsChecking(line, end_pos + 1);
213 
214  if next_char <> CHAR_SPACE and next_char <> CHAR_DASH then
215  // If the next character isn't a space or dash, search backwards for a space.
216  pos := rfindChar(line, CHAR_SPACE, end_pos, end_pos - gap_size);
217 
218  if pos <> NO_POS then
219  // A space was found, break the string here.
220  str := substring(line, start_pos, pos - 1);
221  start_pos := pos + 1;
222  else
223  // No space was found, search for a dash instead.
224  pos := rfindChar(line, CHAR_DASH, end_pos, start_pos + gap_size);
225 
226  if pos > 1 then
227  // A dash was found, check that the previous character is alphabetic.
228  char := stringGetNoBoundsChecking(line, pos - 1);
229  pos := if isAlpha(char) and isAlpha(next_char) then pos else NO_POS;
230  end if;
231 
232  if pos <> NO_POS then
233  // A dash was found, break the string here.
234  str := substring(line, start_pos, pos);
235  start_pos := pos + 1;
236  else
237  // No dash was found, break the word and hyphenate it.
238  str := substring(line, start_pos, end_pos - 1) + "-";
239  start_pos := end_pos;
240  end if;
241  end if;
242  else
243  // The next character is a space or dash, split the string here.
244  str := substring(line, start_pos, end_pos);
245  // Skip the space.
246  start_pos := end_pos + (if next_char == CHAR_SPACE then 2 else 1);
247  end if;
248 
249  // Add the string to the list and continue with the rest of the line.
250  outStrings := (delim + str) :: outStrings;
251  end_pos := start_pos + line_len;
252  delim := inDelimiter;
253  end while;
254 
255  // Add any remainder of the line to the list.
256  if start_pos < stringLength(line) then
257  str := delim + substring(line, start_pos, stringLength(line));
258  outStrings := str :: outStrings;
259  end if;
260 
261  // Continue with the next line.
262  start_pos := 1;
263  end_pos := line_len;
264  delim := inDelimiter;
265  end for;
266 
267  outStrings := listReverseInPlace(outStrings);
268 end wordWrap;

323 function bytesToReadableUnit

327  output String str;
