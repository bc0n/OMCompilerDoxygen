97 {
98  int ierr;
99  if (Op.GetDomainSpace() != Op.GetRangeSpace())
100  ML_THROW("Matrix is not square", -1);
101 
102  ER.Reshape(Op.GetDomainSpace());
103  EI.Reshape(Op.GetDomainSpace());
104 
105  Epetra_LinearProblem Problem;
106  Problem.SetOperator(const_cast<Epetra_RowMatrix*>(Op.GetRowMatrix()));
107  Amesos_Lapack Lapack(Problem);
108 
109  Epetra_Vector ER_Epetra(Op.GetRowMatrix()->RowMatrixRowMap());
110  Epetra_Vector EI_Epetra(Op.GetRowMatrix()->RowMatrixRowMap());
111 
112  ierr = Lapack.GEEV(ER_Epetra, EI_Epetra);
113 
114  if (ierr)
115  ML_THROW("GEEV returned error code = " + GetString(ierr), -1);
116 
117  for (int i = 0 ; i < ER.GetMyLength() ; ++i) {
118  ER(i) = ER_Epetra[i];
119  EI(i) = EI_Epetra[i];
120  }
121 }
