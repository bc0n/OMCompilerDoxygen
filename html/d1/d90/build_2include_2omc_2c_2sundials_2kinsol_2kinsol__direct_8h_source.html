61 /*
62  * -----------------------------------------------------------------
63  * Type: KINDlsDenseJacFn
64  * -----------------------------------------------------------------
65  *
66  * A dense Jacobian approximation function Jac must be of type
67  * KINDlsDenseJacFn. Its parameters are:
68  *
69  * N - problem size.
70  *
71  * u - current iterate (unscaled) [input]
72  *
73  * fu - vector (type N_Vector) containing result of nonlinear
74  * system function evaluated at current iterate:
75  * fu = F(u) [input]
76  *
77  * J - dense matrix (of type DlsMat) that will be loaded
78  * by a KINDlsDenseJacFn with an approximation to the
79  * Jacobian matrix J = (dF_i/dy_j).
80  *
81  * user_data - pointer to user data - the same as the user_data
82  * parameter passed to KINSetFdata.
83  *
84  * tmp1, tmp2 - available scratch vectors (volatile storage)
85  *
86  * A KINDlsDenseJacFn should return 0 if successful, a positive
87  * value if a recoverable error occurred, and a negative value if
88  * an unrecoverable error occurred.
89  *
90  * -----------------------------------------------------------------
91  *
92  * NOTE: The following are two efficient ways to load a dense Jac:
93  * (1) (with macros - no explicit data structure references)
94  * for (j=0; j < Neq; j++) {
95  * col_j = DENSE_COL(Jac,j);
96  * for (i=0; i < Neq; i++) {
97  * generate J_ij = the (i,j)th Jacobian element
98  * col_j[i] = J_ij;
99  * }
100  * }
101  * (2) (without macros - explicit data structure references)
102  * for (j=0; j < Neq; j++) {
103  * col_j = (Jac->data)[j];
104  * for (i=0; i < Neq; i++) {
105  * generate J_ij = the (i,j)th Jacobian element
106  * col_j[i] = J_ij;
107  * }
108  * }
109  * A third way, using the DENSE_ELEM(A,i,j) macro, is much less
110  * efficient in general. It is only appropriate for use in small
111  * problems in which efficiency of access is NOT a major concern.
112  *
113  * -----------------------------------------------------------------
114  */
115

118  N_Vector u, N_Vector fu,
119  DlsMat J, void *user_data,
120  N_Vector tmp1, N_Vector tmp2);

122 /*
123  * -----------------------------------------------------------------
124  * Type: KINDlsBandJacFn
125  * -----------------------------------------------------------------
126  *
127  * A band Jacobian approximation function Jac must have the
128  * prototype given below. Its parameters are:
129  *
130  * N is the problem size
131  *
132  * mupper is the upper half-bandwidth of the approximate banded
133  * Jacobian. This parameter is the same as the mupper parameter
134  * passed by the user to the linear solver initialization function.
135  *
136  * mlower is the lower half-bandwidth of the approximate banded
137  * Jacobian. This parameter is the same as the mlower parameter
138  * passed by the user to the linear solver initialization function.
139  *
140  * u - current iterate (unscaled) [input]
141  *
142  * fu - vector (type N_Vector) containing result of nonlinear
143  * system function evaluated at current iterate:
144  * fu = F(uu) [input]
145  *
146  * J - band matrix (of type DlsMat) that will be loaded by a
147  * KINDlsBandJacFn with an approximation to the Jacobian
148  * matrix Jac = (dF_i/dy_j).
149  *
150  * user_data - pointer to user data - the same as the user_data
151  * parameter passed to KINSetFdata.
152  *
153  * tmp1, tmp2 - available scratch vectors (volatile storage)
154  *
155  * A KINDlsBandJacFn should return 0 if successful, a positive value
156  * if a recoverable error occurred, and a negative value if an
157  * unrecoverable error occurred.
158  *
159  * -----------------------------------------------------------------
160  *
161  * NOTE. Three efficient ways to load J are:
162  *
163  * (1) (with macros - no explicit data structure references)
164  * for (j=0; j < n; j++) {
165  * col_j = BAND_COL(Jac,j);
166  * for (i=j-mupper; i <= j+mlower; i++) {
167  * generate J_ij = the (i,j)th Jacobian element
168  * BAND_COL_ELEM(col_j,i,j) = J_ij;
169  * }
170  * }
171  *
172  * (2) (with BAND_COL macro, but without BAND_COL_ELEM macro)
173  * for (j=0; j < n; j++) {
174  * col_j = BAND_COL(Jac,j);
175  * for (k=-mupper; k <= mlower; k++) {
176  * generate J_ij = the (i,j)th Jacobian element, i=j+k
177  * col_j[k] = J_ij;
178  * }
179  * }
180  *
181  * (3) (without macros - explicit data structure references)
182  * offset = Jac->smu;
183  * for (j=0; j < n; j++) {
184  * col_j = ((Jac->data)[j])+offset;
185  * for (k=-mupper; k <= mlower; k++) {
186  * generate J_ij = the (i,j)th Jacobian element, i=j+k
187  * col_j[k] = J_ij;
188  * }
189  * }
190  * Caution: Jac->smu is generally NOT the same as mupper.
191  *
192  * The BAND_ELEM(A,i,j) macro is appropriate for use in small
193  * problems in which efficiency of access is NOT a major concern.
194  *
195  * -----------------------------------------------------------------
196  */
197
