225 void messageText(int type, int stream, int indentNext, char *msg, int subline, const int *indexes)
226 {
227  int i;
228  int len;
229 
230  printf("%-17s | ", (subline || (lastStream == stream && level[stream] > 0)) ? "|" : LOG_STREAM_NAME[stream]);
231  printf("%-7s | ", (subline || (lastStream == stream && lastType[stream] == type && level[stream] > 0)) ? "|" : LOG_TYPE_DESC[type]);
232  lastType[stream] = type;
233  lastStream = stream;
234 
235  for(i=0; i<level[stream]; ++i)
236  printf("| ");
237 
238  for(i=0; msg[i]; i++)
239  {
240  if(msg[i] == '\n')
241  {
242  msg[i] = '\0';
243  printf("%s\n", msg);
244  if (msg[i+1]) {
245  messageText(type, stream, 0, &msg[i+1], 1, indexes);
246  }
247  return;
248  }
249  }
250 
251  len = strlen(msg);
252  if (len>0 && msg[len-1]=='\n') {
253  printf("%s", msg);
254  } else {
255  printf("%s\n", msg);
256  }
257  fflush(NULL);
258  if (indentNext) level[stream]++;
259 }

408 static inline jmp_buf* getBestJumpBuffer(threadData_t *threadData)
409 {
410  switch (threadData->currentErrorStage) {
411  case ERROR_EVENTSEARCH:
412  case ERROR_SIMULATION:
413  case ERROR_NONLINEARSOLVER:
414  case ERROR_INTEGRATOR:
415  case ERROR_OPTIMIZE:
416 #ifndef OMC_EMCC
417  if (threadData->simulationJumpBuffer) {
418  return threadData->simulationJumpBuffer;
419  }
420  fprintf(stderr, "getBestJumpBuffer got simulationJumpBuffer=%p\n", threadData->simulationJumpBuffer);
421  abort();
422 #endif
423  case ERROR_EVENTHANDLING:
424  default:
425  if (threadData->globalJumpBuffer) {
426  return threadData->globalJumpBuffer;
427  }
428  if (threadData->mmc_jumper) {
429  return threadData->mmc_jumper;
430  }
431  fprintf(stderr, "getBestJumpBuffer got mmc_jumper=%p, globalJumpBuffer=%p\n", threadData->globalJumpBuffer, threadData->mmc_jumper);
432  abort();
433  }
434 }
