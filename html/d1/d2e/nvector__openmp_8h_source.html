87 /*
88  * -----------------------------------------------------------------
89  * PART II: macros NV_CONTENT_OMP, NV_DATA_OMP, NV_OWN_DATA_OMP,
90  * NV_LENGTH_OMP, and NV_Ith_OMP
91  * -----------------------------------------------------------------
92  * In the descriptions below, the following user declarations
93  * are assumed:
94  *
95  * N_Vector v;
96  * long int i;
97  *
98  * (1) NV_CONTENT_OMP
99  *
100  * This routines gives access to the contents of the OpenMP
101  * vector N_Vector.
102  *
103  * The assignment v_cont = NV_CONTENT_OMP(v) sets v_cont to be
104  * a pointer to the OpenMP N_Vector content structure.
105  *
106  * (2) NV_DATA_OMP NV_OWN_DATA_OMP and NV_LENGTH_OMP
107  *
108  * These routines give access to the individual parts of
109  * the content structure of a OpenMP N_Vector.
110  *
111  * The assignment v_data = NV_DATA_OMP(v) sets v_data to be
112  * a pointer to the first component of v. The assignment
113  * NV_DATA_OMP(v) = data_V sets the component array of v to
114  * be data_v by storing the pointer data_v.
115  *
116  * The assignment v_len = NV_LENGTH_OMP(v) sets v_len to be
117  * the length of v. The call NV_LENGTH_OMP(v) = len_v sets
118  * the length of v to be len_v.
119  *
120  * (3) NV_Ith_OMP
121  *
122  * In the following description, the components of an
123  * N_Vector are numbered 0..n-1, where n is the length of v.
124  *
125  * The assignment r = NV_Ith_OMP(v,i) sets r to be the value of
126  * the ith component of v. The assignment NV_Ith_OMP(v,i) = r
127  * sets the value of the ith component of v to be r.
128  *
129  * Note: When looping over the components of an N_Vector v, it is
130  * more efficient to first obtain the component array via
131  * v_data = NV_DATA_OMP(v) and then access v_data[i] within the
132  * loop than it is to use NV_Ith_OMP(v,i) within the loop.
133  * -----------------------------------------------------------------
134  */
135 
136 #define NV_CONTENT_OMP(v) ( (N_VectorContent_OpenMP)(v->content) )
137 
138 #define NV_LENGTH_OMP(v) ( NV_CONTENT_OMP(v)->length )
139 
140 #define NV_NUM_THREADS_OMP(v) ( NV_CONTENT_OMP(v)->num_threads )
141 
142 #define NV_OWN_DATA_OMP(v) ( NV_CONTENT_OMP(v)->own_data )
143 
144 #define NV_DATA_OMP(v) ( NV_CONTENT_OMP(v)->data )
145 
146 #define NV_Ith_OMP(v,i) ( NV_DATA_OMP(v)[i] )
