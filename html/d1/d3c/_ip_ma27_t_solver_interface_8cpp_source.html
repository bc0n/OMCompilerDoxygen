146  bool Ma27TSolverInterface::InitializeImpl(const OptionsList& options,
147  const std::string& prefix)
148  {
149  options.GetNumericValue("ma27_pivtol", pivtol_, prefix);
150  if (options.GetNumericValue("ma27_pivtolmax", pivtolmax_, prefix)) {
151  ASSERT_EXCEPTION(pivtolmax_>=pivtol_, OPTION_INVALID,
152  "Option \"ma27_pivtolmax\": This value must be between "
153  "ma27_pivtol and 1.");
154  }
155  else {
156  pivtolmax_ = Max(pivtolmax_, pivtol_);
157  }
158 
159  options.GetNumericValue("ma27_liw_init_factor", liw_init_factor_, prefix);
160  options.GetNumericValue("ma27_la_init_factor", la_init_factor_, prefix);
161  options.GetNumericValue("ma27_meminc_factor", meminc_factor_, prefix);
162  options.GetBoolValue("ma27_skip_inertia_check",
163  skip_inertia_check_, prefix);
164  options.GetBoolValue("ma27_ignore_singularity",
165  ignore_singularity_, prefix);
166  // The following option is registered by OrigIpoptNLP
167  options.GetBoolValue("warm_start_same_structure",
168  warm_start_same_structure_, prefix);
169 
170  /* Set the default options for MA27 */
171  F77_FUNC(ma27id,MA27ID)(icntl_, cntl_);
172 #if COIN_IPOPT_VERBOSITY == 0
173 
174  icntl_[0] = 0; // Suppress error messages
175  icntl_[1] = 0; // Suppress diagnostic messages
176 #endif
177 
178  // Reset all private data
179  initialized_=false;
180  pivtol_changed_ = false;
181  refactorize_ = false;
182 
183  la_increase_=false;
184  liw_increase_=false;
185 
186  if (!warm_start_same_structure_) {
187  dim_=0;
188  nonzeros_=0;
189  }
190  else {
191  ASSERT_EXCEPTION(dim_>0 && nonzeros_>0, INVALID_WARMSTART,
192  "Ma27TSolverInterface called with warm_start_same_structure, but the problem is solved for the first time.");
193  }
194 
195  return true;
196  }

198  ESymSolverStatus Ma27TSolverInterface::MultiSolve(bool new_matrix,
199  const Index* airn,
200  const Index* ajcn,
201  Index nrhs,
202  double* rhs_vals,
203  bool check_NegEVals,
204  Index numberOfNegEVals)
205  {
206  DBG_START_METH("Ma27TSolverInterface::MultiSolve",dbg_verbosity);
207  DBG_ASSERT(!check_NegEVals || ProvidesInertia());
208  DBG_ASSERT(initialized_);
209  DBG_ASSERT(la_!=0);
210 
211  if (pivtol_changed_) {
212  DBG_PRINT((1,"Pivot tolerance has changed.\n"));
213  pivtol_changed_ = false;
214  // If the pivot tolerance has been changed but the matrix is not
215  // new, we have to request the values for the matrix again to do
216  // the factorization again.
217  if (!new_matrix) {
218  DBG_PRINT((1,"Ask caller to call again.\n"));
219  refactorize_ = true;
220  return SYMSOLVER_CALL_AGAIN;
221  }
222  }
223 
224  // check if a factorization has to be done
225  DBG_PRINT((1, "new_matrix = %d\n", new_matrix));
226  if (new_matrix || refactorize_) {
227  // perform the factorization
228  ESymSolverStatus retval;
229  retval = Factorization(airn, ajcn, check_NegEVals, numberOfNegEVals);
230  if (retval!=SYMSOLVER_SUCCESS) {
231  DBG_PRINT((1, "FACTORIZATION FAILED!\n"));
232  return retval; // Matrix singular or error occurred
233  }
234  refactorize_ = false;
235  }
236 
237  // do the backsolve
238  return Backsolve(nrhs, rhs_vals);
239  }

266

288  ESymSolverStatus Ma27TSolverInterface::SymbolicFactorization(const Index* airn,
289  const Index* ajcn)
290  {
291  DBG_START_METH("Ma27TSolverInterface::SymbolicFactorization",dbg_verbosity);
292 
293  if (HaveIpData()) {
294  IpData().TimingStats().LinearSystemSymbolicFactorization().Start();
295  }
296 
297  // Get memory for the IW workspace
298  delete [] iw_;
299  iw_ = NULL;
300 
301  // Overestimation factor for LIW (20% recommended in MA27 documentation)
302  const double LiwFact = 2.0; // This is 100% overestimation
303  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
304  "In Ma27TSolverInterface::InitializeStructure: Using overestimation factor LiwFact = %e\n",
305  LiwFact);
306  liw_ = (ipfint)(LiwFact*(double(2*nonzeros_+3*dim_+1)));
307  iw_ = new ipfint[liw_];
308 
309  // Get memory for IKEEP
310  delete [] ikeep_;
311  ikeep_ = NULL;
312  ikeep_ = new ipfint[3*dim_];
313 
314  if (Jnlst().ProduceOutput(J_MOREMATRIX, J_LINEAR_ALGEBRA)) {
315  Jnlst().Printf(J_MOREMATRIX, J_LINEAR_ALGEBRA,
316  "\nMatrix structure given to MA27 with dimension %d and %d nonzero entries:\n", dim_, nonzeros_);
317  for (Index i=0; i<nonzeros_; i++) {
318  Jnlst().Printf(J_MOREMATRIX, J_LINEAR_ALGEBRA, "A[%5d,%5d]\n",
319  airn[i], ajcn[i]);
320  }
321  }
322 
323  // Call MA27AD (cast to ipfint for Index types)
324  ipfint N = dim_;
325  ipfint NZ = nonzeros_;
326  ipfint IFLAG = 0;
327  double OPS;
328  ipfint INFO[20];
329  ipfint* IW1 = new ipfint[2*dim_]; // Get memory for IW1 (only local)
330  F77_FUNC(ma27ad,MA27AD)(&N, &NZ, airn, ajcn, iw_, &liw_, ikeep_,
331  IW1, &nsteps_, &IFLAG, icntl_, cntl_,
332  INFO, &OPS);
333  delete [] IW1; // No longer required
334 
335  // Receive several information
336  const ipfint &iflag = INFO[0]; // Information flag
337  const ipfint &ierror = INFO[1]; // Error flag
338  const ipfint &nrlnec = INFO[4]; // recommended value for la
339  const ipfint &nirnec = INFO[5]; // recommended value for liw
340 
341  Jnlst().Printf(J_MOREDETAILED, J_LINEAR_ALGEBRA,
342  "Return values from MA27AD: IFLAG = %d, IERROR = %d\n",
343  iflag, ierror);
344 
345  // Check if error occurred
346  if (iflag!=0) {
347  Jnlst().Printf(J_ERROR, J_LINEAR_ALGEBRA,
348  "*** Error from MA27AD *** IFLAG = %d IERROR = %d\n", iflag, ierror);
349  if (iflag==1) {
350  Jnlst().Printf(J_ERROR, J_LINEAR_ALGEBRA,
351  "The index of a matrix is out of range.\nPlease check your implementation of the Jacobian and Hessian matrices.\n");
352  }
353  if (HaveIpData()) {
354  IpData().TimingStats().LinearSystemSymbolicFactorization().End();
355  }
356  return SYMSOLVER_FATAL_ERROR;
357  }
358 
359  // ToDo: try and catch
360  // Reserve memory for iw_ for later calls, based on suggested size
361  delete [] iw_;
362  iw_ = NULL;
363  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
364  "Size of integer work space recommended by MA27 is %d\n",
365  nirnec);
366  liw_ = (ipfint)(liw_init_factor_ * (double)(nirnec));
367  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
368  "Setting integer work space size to %d\n", liw_);
369  iw_ = new ipfint[liw_];
370 
371  // Reserve memory for a_
372  delete [] a_;
373  a_ = NULL;
374  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
375  "Size of doublespace recommended by MA27 is %d\n",
376  nrlnec);
377  la_ = Max(nonzeros_,(ipfint)(la_init_factor_ * (double)(nrlnec)));
378  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
379  "Setting double work space size to %d\n", la_);
380  a_ = new double[la_];
381 
382  if (HaveIpData()) {
383  IpData().TimingStats().LinearSystemSymbolicFactorization().End();
384  }
385 
386  return SYMSOLVER_SUCCESS;
387  }

389  ESymSolverStatus
390  Ma27TSolverInterface::Factorization(const Index* airn,
391  const Index* ajcn,
392  bool check_NegEVals,
393  Index numberOfNegEVals)
394  {
395  DBG_START_METH("Ma27TSolverInterface::Factorization",dbg_verbosity);
396  // Check if la should be increased
397  if (HaveIpData()) {
398  IpData().TimingStats().LinearSystemFactorization().Start();
399  }
400  if (la_increase_) {
401  double* a_old = a_;
402  ipfint la_old = la_;
403  la_ = (ipfint)(meminc_factor_ * (double)(la_));
404  a_ = new double[la_];
405  for (Index i=0; i<nonzeros_; i++) {
406  a_[i] = a_old[i];
407  }
408  delete [] a_old;
409  la_increase_ = false;
410  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
411  "In Ma27TSolverInterface::Factorization: Increasing la from %d to %d\n",
412  la_old, la_);
413  }
414 
415  // Check if liw should be increased
416  if (liw_increase_) {
417  delete [] iw_;
418  iw_ = NULL;
419  ipfint liw_old = liw_;
420  liw_ = (ipfint)(meminc_factor_ * (double)(liw_));
421  iw_ = new ipfint[liw_];
422  liw_increase_ = false;
423  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
424  "In Ma27TSolverInterface::Factorization: Increasing liw from %d to %d\n",
425  liw_old, liw_);
426  }
427 
428  ipfint iflag; // Information flag
429  ipfint ncmpbr; // Number of double precision compressions
430  ipfint ncmpbi; // Number of integer compressions
431 
432  // Call MA27BD; possibly repeatedly if workspaces are too small
433  ipfint N=dim_;
434  ipfint NZ=nonzeros_;
435  ipfint* IW1 = new ipfint[2*dim_];
436  ipfint INFO[20];
437  cntl_[0] = pivtol_; // Set pivot tolerance
438 
439  F77_FUNC(ma27bd,MA27BD)(&N, &NZ, airn, ajcn, a_,
440  &la_, iw_, &liw_, ikeep_, &nsteps_,
441  &maxfrt_, IW1, icntl_, cntl_, INFO);
442  delete [] IW1;
443 
444  // Receive information about the factorization
445  iflag = INFO[0]; // Information flag
446  const ipfint &ierror = INFO[1]; // Error flag
447  ncmpbr = INFO[11]; // Number of double compressions
448  ncmpbi = INFO[12]; // Number of integer compressions
449  negevals_ = INFO[14]; // Number of negative eigenvalues
450 
451  Jnlst().Printf(J_MOREDETAILED, J_LINEAR_ALGEBRA,
452  "Return values from MA27BD: IFLAG = %d, IERROR = %d\n",
453  iflag, ierror);
454 
455  DBG_PRINT((1,"Return from MA27BD iflag = %d and ierror = %d\n",
456  iflag, ierror));
457 
458  // Check if factorization failed due to insufficient memory space
459  // iflag==-3 if LIW too small (recommended value in ierror)
460  // iflag==-4 if LA too small (recommended value in ierror)
461  if (iflag==-3 || iflag==-4) {
462  // Increase size of both LIW and LA
463  delete [] iw_;
464  iw_ = NULL;
465  delete [] a_;
466  a_ = NULL;
467  ipfint liw_old = liw_;
468  ipfint la_old = la_;
469  if (iflag==-3) {
470  liw_ = (ipfint)(meminc_factor_ * (double)(ierror));
471  la_ = (ipfint)(meminc_factor_ * (double)(la_));
472  }
473  else {
474  liw_ = (ipfint)(meminc_factor_ * (double)(liw_));
475  la_ = (ipfint)(meminc_factor_ * (double)(ierror));
476  }
477  iw_ = new ipfint[liw_];
478  a_ = new double[la_];
479  Jnlst().Printf(J_WARNING, J_LINEAR_ALGEBRA,
480  "MA27BD returned iflag=%d and requires more memory.\n Increase liw from %d to %d and la from %d to %d and factorize again.\n",
481  iflag, liw_old, liw_, la_old, la_);
482  if (HaveIpData()) {
483  IpData().TimingStats().LinearSystemFactorization().End();
484  }
485  return SYMSOLVER_CALL_AGAIN;
486  }
487 
488  // Check if the system is singular, and if some other error occurred
489  if (iflag==-5 || (!ignore_singularity_ && iflag==3)) {
490  if (HaveIpData()) {
491  IpData().TimingStats().LinearSystemFactorization().End();
492  }
493  return SYMSOLVER_SINGULAR;
494  }
495  else if (iflag==3) {
496  Index missing_rank = dim_-INFO[1];
497  Jnlst().Printf(J_WARNING, J_LINEAR_ALGEBRA,
498  "MA27BD returned iflag=%d and detected rank deficiency of degree %d.\n",
499  iflag, missing_rank);
500  // We correct the number of negative eigenvalues here to include
501  // the zero eigenvalues, since otherwise we indicate the wrong
502  // inertia.
503  negevals_ += missing_rank;
504  }
505  else if (iflag != 0) {
506  // There is some error
507  if (HaveIpData()) {
508  IpData().TimingStats().LinearSystemFactorization().End();
509  }
510  return SYMSOLVER_FATAL_ERROR;
511  }
512 
513  // Check if it might be more efficient to use more memory next time
514  // (if there were too many compressions for this factorization)
515  if (ncmpbr>=10) {
516  la_increase_ = true;
517  Jnlst().Printf(J_WARNING, J_LINEAR_ALGEBRA,
518  "MA27BD returned ncmpbr=%d. Increase la before the next factorization.\n",
519  ncmpbr);
520  }
521  if (ncmpbi>=10) {
522  liw_increase_ = true;
523  Jnlst().Printf(J_WARNING, J_LINEAR_ALGEBRA,
524  "MA27BD returned ncmpbi=%d. Increase liw before the next factorization.\n",
525  ncmpbr);
526  }
527 
528  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
529  "Number of doubles for MA27 to hold factorization (INFO(9)) = %d\n",
530  INFO[8]);
531  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
532  "Number of integers for MA27 to hold factorization (INFO(10)) = %d\n",
533  INFO[9]);
534 
535  // Check whether the number of negative eigenvalues matches the requested
536  // count
537  if (HaveIpData()) {
538  IpData().TimingStats().LinearSystemFactorization().End();
539  }
540  if (!skip_inertia_check_ && check_NegEVals && (numberOfNegEVals!=negevals_)) {
541  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
542  "In Ma27TSolverInterface::Factorization: negevals_ = %d, but numberOfNegEVals = %d\n",
543  negevals_, numberOfNegEVals);
544  return SYMSOLVER_WRONG_INERTIA;
545  }
546 
547  return SYMSOLVER_SUCCESS;
548  }

550  ESymSolverStatus Ma27TSolverInterface::Backsolve(Index nrhs,
551  double *rhs_vals)
552  {
553  DBG_START_METH("Ma27TSolverInterface::Backsolve",dbg_verbosity);
554  if (HaveIpData()) {
555  IpData().TimingStats().LinearSystemBackSolve().Start();
556  }
557 
558  ipfint N=dim_;
559  double* W = new double[maxfrt_];
560  ipfint* IW1 = new ipfint[nsteps_];
561 
562  // For each right hand side, call MA27CD
563  for (Index irhs=0; irhs<nrhs; irhs++) {
564  if (DBG_VERBOSITY()>=2) {
565  for (Index i=0; i<dim_; i++) {
566  DBG_PRINT((2, "rhs[%5d] = %23.15e\n", i, rhs_vals[irhs*dim_+i]));
567  }
568  }
569 
570  F77_FUNC(ma27cd,MA27CD)(&N, a_, &la_, iw_, &liw_, W, &maxfrt_,
571  &rhs_vals[irhs*dim_], IW1, &nsteps_,
572  icntl_, cntl_);
573 
574  if (DBG_VERBOSITY()>=2) {
575  for (Index i=0; i<dim_; i++) {
576  DBG_PRINT((2, "sol[%5d] = %23.15e\n", i, rhs_vals[irhs*dim_+i]));
577  }
578  }
579  }
580  delete [] W;
581  delete [] IW1;
582 
583  if (HaveIpData()) {
584  IpData().TimingStats().LinearSystemBackSolve().End();
585  }
586  return SYMSOLVER_SUCCESS;
587  }
