45 SpgmrMem SpgmrMalloc(int l_max, N_Vector vec_tmpl)
46 {
47  SpgmrMem mem;
48  N_Vector *V, xcor, vtemp;
49  realtype **Hes, *givens, *yg;
50  int k, i;
51 
52  /* Check the input parameters. */
53 
54  if (l_max <= 0) return(NULL);
55 
56  /* Get memory for the Krylov basis vectors V[0], ..., V[l_max]. */
57 
58  V = N_VCloneVectorArray(l_max+1, vec_tmpl);
59  if (V == NULL) return(NULL);
60 
61  /* Get memory for the Hessenberg matrix Hes. */
62 
63  Hes = NULL;
64  Hes = (realtype **) malloc((l_max+1)*sizeof(realtype *));
65  if (Hes == NULL) {
66  N_VDestroyVectorArray(V, l_max+1);
67  return(NULL);
68  }
69 
70  for (k = 0; k <= l_max; k++) {
71  Hes[k] = NULL;
72  Hes[k] = (realtype *) malloc(l_max*sizeof(realtype));
73  if (Hes[k] == NULL) {
74  for (i = 0; i < k; i++) {free(Hes[i]); Hes[i] = NULL;}
75  free(Hes); Hes = NULL;
76  N_VDestroyVectorArray(V, l_max+1);
77  return(NULL);
78  }
79  }
80 
81  /* Get memory for Givens rotation components. */
82 
83  givens = NULL;
84  givens = (realtype *) malloc(2*l_max*sizeof(realtype));
85  if (givens == NULL) {
86  for (i = 0; i <= l_max; i++) {free(Hes[i]); Hes[i] = NULL;}
87  free(Hes); Hes = NULL;
88  N_VDestroyVectorArray(V, l_max+1);
89  return(NULL);
90  }
91 
92  /* Get memory to hold the correction to z_tilde. */
93 
94  xcor = N_VClone(vec_tmpl);
95  if (xcor == NULL) {
96  free(givens); givens = NULL;
97  for (i = 0; i <= l_max; i++) {free(Hes[i]); Hes[i] = NULL;}
98  free(Hes); Hes = NULL;
99  N_VDestroyVectorArray(V, l_max+1);
100  return(NULL);
101  }
102 
103  /* Get memory to hold SPGMR y and g vectors. */
104 
105  yg = NULL;
106  yg = (realtype *) malloc((l_max+1)*sizeof(realtype));
107  if (yg == NULL) {
108  N_VDestroy(xcor);
109  free(givens); givens = NULL;
110  for (i = 0; i <= l_max; i++) {free(Hes[i]); Hes[i] = NULL;}
111  free(Hes); Hes = NULL;
112  N_VDestroyVectorArray(V, l_max+1);
113  return(NULL);
114  }
115 
116  /* Get an array to hold a temporary vector. */
117 
118  vtemp = N_VClone(vec_tmpl);
119  if (vtemp == NULL) {
120  free(yg); yg = NULL;
121  N_VDestroy(xcor);
122  free(givens); givens = NULL;
123  for (i = 0; i <= l_max; i++) {free(Hes[i]); Hes[i] = NULL;}
124  free(Hes); Hes = NULL;
125  N_VDestroyVectorArray(V, l_max+1);
126  return(NULL);
127  }
128 
129  /* Get memory for an SpgmrMemRec containing SPGMR matrices and vectors. */
130 
131  mem = NULL;
132  mem = (SpgmrMem) malloc(sizeof(SpgmrMemRec));
133  if (mem == NULL) {
134  N_VDestroy(vtemp);
135  free(yg); yg = NULL;
136  N_VDestroy(xcor);
137  free(givens); givens = NULL;
138  for (i = 0; i <= l_max; i++) {free(Hes[i]); Hes[i] = NULL;}
139  free(Hes); Hes = NULL;
140  N_VDestroyVectorArray(V, l_max+1);
141  return(NULL);
142  }
143 
144  /* Set the fields of mem. */
145 
146  mem->l_max = l_max;
147  mem->V = V;
148  mem->Hes = Hes;
149  mem->givens = givens;
150  mem->xcor = xcor;
151  mem->yg = yg;
152  mem->vtemp = vtemp;
153 
154  /* Return the pointer to SPGMR memory. */
155 
156  return(mem);
157 }

165 int SpgmrSolve(SpgmrMem mem, void *A_data, N_Vector x, N_Vector b,
166  int pretype, int gstype, realtype delta, int max_restarts,
167  void *P_data, N_Vector s1, N_Vector s2, ATimesFn atimes,
168  PSolveFn psolve, realtype *res_norm, int *nli, int *nps)
169 {
170  N_Vector *V, xcor, vtemp;
171  realtype **Hes, *givens, *yg;
172  realtype beta, rotation_product, r_norm, s_product, rho;
173  booleantype preOnLeft, preOnRight, scale2, scale1, converged;
174  int i, j, k, l, l_plus_1, l_max, krydim, ier, ntries;
175 
176  if (mem == NULL) return(SPGMR_MEM_NULL);
177 
178  /* Initialize some variables */
179 
180  l_plus_1 = 0;
181  krydim = 0;
182 
183  /* Make local copies of mem variables. */
184 
185  l_max = mem->l_max;
186  V = mem->V;
187  Hes = mem->Hes;
188  givens = mem->givens;
189  xcor = mem->xcor;
190  yg = mem->yg;
191  vtemp = mem->vtemp;
192 
193  *nli = *nps = 0; /* Initialize counters */
194  converged = FALSE; /* Initialize converged flag */
195 
196  if (max_restarts < 0) max_restarts = 0;
197 
198  if ((pretype != PREC_LEFT) && (pretype != PREC_RIGHT) && (pretype != PREC_BOTH))
199  pretype = PREC_NONE;
200 
201  preOnLeft = ((pretype == PREC_LEFT) || (pretype == PREC_BOTH));
202  preOnRight = ((pretype == PREC_RIGHT) || (pretype == PREC_BOTH));
203  scale1 = (s1 != NULL);
204  scale2 = (s2 != NULL);
205 
206  /* Set vtemp and V[0] to initial (unscaled) residual r_0 = b - A*x_0. */
207 
208  if (N_VDotProd(x, x) == ZERO) {
209  N_VScale(ONE, b, vtemp);
210  } else {
211  ier = atimes(A_data, x, vtemp);
212  if (ier != 0)
213  return((ier < 0) ? SPGMR_ATIMES_FAIL_UNREC : SPGMR_ATIMES_FAIL_REC);
214  N_VLinearSum(ONE, b, -ONE, vtemp, vtemp);
215  }
216  N_VScale(ONE, vtemp, V[0]);
217 
218  /* Apply left preconditioner and left scaling to V[0] = r_0. */
219 
220  if (preOnLeft) {
221  ier = psolve(P_data, V[0], vtemp, PREC_LEFT);
222  (*nps)++;
223  if (ier != 0)
224  return((ier < 0) ? SPGMR_PSOLVE_FAIL_UNREC : SPGMR_PSOLVE_FAIL_REC);
225  } else {
226  N_VScale(ONE, V[0], vtemp);
227  }
228 
229  if (scale1) {
230  N_VProd(s1, vtemp, V[0]);
231  } else {
232  N_VScale(ONE, vtemp, V[0]);
233  }
234 
235  /* Set r_norm = beta to L2 norm of V[0] = s1 P1_inv r_0, and
236  return if small. */
237 
238  *res_norm = r_norm = beta = SUNRsqrt(N_VDotProd(V[0], V[0]));
239  if (r_norm <= delta)
240  return(SPGMR_SUCCESS);
241 
242  /* Initialize rho to avoid compiler warning message */
243 
244  rho = beta;
245 
246  /* Set xcor = 0. */
247 
248  N_VConst(ZERO, xcor);
249 
250 
251  /* Begin outer iterations: up to (max_restarts + 1) attempts. */
252 
253  for (ntries = 0; ntries <= max_restarts; ntries++) {
254 
255  /* Initialize the Hessenberg matrix Hes and Givens rotation
256  product. Normalize the initial vector V[0]. */
257 
258  for (i = 0; i <= l_max; i++)
259  for (j = 0; j < l_max; j++)
260  Hes[i][j] = ZERO;
261 
262  rotation_product = ONE;
263 
264  N_VScale(ONE/r_norm, V[0], V[0]);
265 
266  /* Inner loop: generate Krylov sequence and Arnoldi basis. */
267 
268  for (l = 0; l < l_max; l++) {
269 
270  (*nli)++;
271 
272  krydim = l_plus_1 = l + 1;
273 
274  /* Generate A-tilde V[l], where A-tilde = s1 P1_inv A P2_inv s2_inv. */
275 
276  /* Apply right scaling: vtemp = s2_inv V[l]. */
277 
278  if (scale2) N_VDiv(V[l], s2, vtemp);
279  else N_VScale(ONE, V[l], vtemp);
280 
281  /* Apply right preconditioner: vtemp = P2_inv s2_inv V[l]. */
282 
283  if (preOnRight) {
284  N_VScale(ONE, vtemp, V[l_plus_1]);
285  ier = psolve(P_data, V[l_plus_1], vtemp, PREC_RIGHT);
286  (*nps)++;
287  if (ier != 0)
288  return((ier < 0) ? SPGMR_PSOLVE_FAIL_UNREC : SPGMR_PSOLVE_FAIL_REC);
289  }
290 
291  /* Apply A: V[l+1] = A P2_inv s2_inv V[l]. */
292 
293  ier = atimes(A_data, vtemp, V[l_plus_1] );
294  if (ier != 0)
295  return((ier < 0) ? SPGMR_ATIMES_FAIL_UNREC : SPGMR_ATIMES_FAIL_REC);
296 
297  /* Apply left preconditioning: vtemp = P1_inv A P2_inv s2_inv V[l]. */
298 
299  if (preOnLeft) {
300  ier = psolve(P_data, V[l_plus_1], vtemp, PREC_LEFT);
301  (*nps)++;
302  if (ier != 0)
303  return((ier < 0) ? SPGMR_PSOLVE_FAIL_UNREC : SPGMR_PSOLVE_FAIL_REC);
304  } else {
305  N_VScale(ONE, V[l_plus_1], vtemp);
306  }
307 
308  /* Apply left scaling: V[l+1] = s1 P1_inv A P2_inv s2_inv V[l]. */
309 
310  if (scale1) {
311  N_VProd(s1, vtemp, V[l_plus_1]);
312  } else {
313  N_VScale(ONE, vtemp, V[l_plus_1]);
314  }
315 
316  /* Orthogonalize V[l+1] against previous V[i]: V[l+1] = w_tilde. */
317 
318  if (gstype == CLASSICAL_GS) {
319  if (ClassicalGS(V, Hes, l_plus_1, l_max, &(Hes[l_plus_1][l]),
320  vtemp, yg) != 0)
321  return(SPGMR_GS_FAIL);
322  } else {
323  if (ModifiedGS(V, Hes, l_plus_1, l_max, &(Hes[l_plus_1][l])) != 0)
324  return(SPGMR_GS_FAIL);
325  }
326 
327  /* Update the QR factorization of Hes. */
328 
329  if(QRfact(krydim, Hes, givens, l) != 0 )
330  return(SPGMR_QRFACT_FAIL);
331 
332  /* Update residual norm estimate; break if convergence test passes. */
333 
334  rotation_product *= givens[2*l+1];
335  *res_norm = rho = SUNRabs(rotation_product*r_norm);
336 
337  if (rho <= delta) { converged = TRUE; break; }
338 
339  /* Normalize V[l+1] with norm value from the Gram-Schmidt routine. */
340 
341  N_VScale(ONE/Hes[l_plus_1][l], V[l_plus_1], V[l_plus_1]);
342  }
343 
344  /* Inner loop is done. Compute the new correction vector xcor. */
345 
346  /* Construct g, then solve for y. */
347 
348  yg[0] = r_norm;
349  for (i = 1; i <= krydim; i++) yg[i]=ZERO;
350  if (QRsol(krydim, Hes, givens, yg) != 0)
351  return(SPGMR_QRSOL_FAIL);
352 
353  /* Add correction vector V_l y to xcor. */
354 
355  for (k = 0; k < krydim; k++)
356  N_VLinearSum(yg[k], V[k], ONE, xcor, xcor);
357 
358  /* If converged, construct the final solution vector x and return. */
359 
360  if (converged) {
361 
362  /* Apply right scaling and right precond.: vtemp = P2_inv s2_inv xcor. */
363 
364  if (scale2) N_VDiv(xcor, s2, xcor);
365  if (preOnRight) {
366  ier = psolve(P_data, xcor, vtemp, PREC_RIGHT);
367  (*nps)++;
368  if (ier != 0)
369  return((ier < 0) ? SPGMR_PSOLVE_FAIL_UNREC : SPGMR_PSOLVE_FAIL_REC);
370  } else {
371  N_VScale(ONE, xcor, vtemp);
372  }
373 
374  /* Add vtemp to initial x to get final solution x, and return */
375 
376  N_VLinearSum(ONE, x, ONE, vtemp, x);
377 
378  return(SPGMR_SUCCESS);
379  }
380 
381  /* Not yet converged; if allowed, prepare for restart. */
382 
383  if (ntries == max_restarts) break;
384 
385  /* Construct last column of Q in yg. */
386 
387  s_product = ONE;
388  for (i = krydim; i > 0; i--) {
389  yg[i] = s_product*givens[2*i-2];
390  s_product *= givens[2*i-1];
391  }
392  yg[0] = s_product;
393 
394  /* Scale r_norm and yg. */
395  r_norm *= s_product;
396  for (i = 0; i <= krydim; i++)
397  yg[i] *= r_norm;
398  r_norm = SUNRabs(r_norm);
399 
400  /* Multiply yg by V_(krydim+1) to get last residual vector; restart. */
401  N_VScale(yg[0], V[0], V[0]);
402  for (k = 1; k <= krydim; k++)
403  N_VLinearSum(yg[k], V[k], ONE, V[0], V[0]);
404 
405  }
406 
407  /* Failed to converge, even after allowed restarts.
408  If the residual norm was reduced below its initial value, compute
409  and return x anyway. Otherwise return failure flag. */
410 
411  if (rho < beta) {
412 
413  /* Apply right scaling and right precond.: vtemp = P2_inv s2_inv xcor. */
414 
415  if (scale2) N_VDiv(xcor, s2, xcor);
416  if (preOnRight) {
417  ier = psolve(P_data, xcor, vtemp, PREC_RIGHT);
418  (*nps)++;
419  if (ier != 0)
420  return((ier < 0) ? SPGMR_PSOLVE_FAIL_UNREC : SPGMR_PSOLVE_FAIL_REC);
421  } else {
422  N_VScale(ONE, xcor, vtemp);
423  }
424 
425  /* Add vtemp to initial x to get final solution x, and return. */
426 
427  N_VLinearSum(ONE, x, ONE, vtemp, x);
428 
429  return(SPGMR_RES_REDUCED);
430  }
431 
432  return(SPGMR_CONV_FAIL);
433 }
