164 protected function handleCommand2
165  input Option<GlobalScript.Statements> inStatements;
166  input Option<Absyn.Program> inProgram;
167  input String inCommand;
168  output String outResult;
169 algorithm
170  outResult := matchcontinue(inStatements, inProgram)
171  local
172  GlobalScript.Statements stmts;
173  Absyn.Program prog, prog2, ast;
174  String result;
175  list<GlobalScript.Variable> vars;
176  SymbolTable table;
177 
178  // Interactively evaluate an algorithm statement or expression.
179  case (SOME(stmts), NONE())
180  equation
181  result = Interactive.evaluate(stmts, false);
182  then result;
183 
184  // Add a class or function to the interactive symbol table.
185  case (NONE(), SOME(prog))
186  equation
187  table = SymbolTable.get();
188  ast = table.ast;
189  vars = table.vars;
190  prog2 = Interactive.addScope(prog, vars);
191  prog2 = Interactive.updateProgram(prog2, ast);
192  if Flags.isSet(Flags.DUMP) then
193  Debug.trace("\n--------------- Parsed program ---------------\n");
194  Dump.dump(prog2);
195  end if;
196  if Flags.isSet(Flags.DUMP_GRAPHVIZ) then
197  DumpGraphviz.dump(prog2);
198  end if;
199  result = makeClassDefResult(prog) "Return vector of toplevel classnames.";
200  SymbolTable.setAbsyn(prog2);
201  then result;
202 
203  // A parser error occured in parseCommand, display the error message. This
204  // is handled here instead of in parseCommand, since parseCommand does not
205  // return a result string.
206  case (NONE(), NONE())
207  equation
208  Print.printBuf("Error occurred building AST\n");
209  result = Print.getString();
210  result = stringAppend(result, "Syntax Error\n");
211  result = stringAppend(result, Error.printMessagesStr(false));
212  then result;
213 
214  // A non-parser error occured, display the error message.
215  case (_, _)
216  equation
217  true = Util.isSome(inStatements) or Util.isSome(inProgram);
218  result = Error.printMessagesStr(false);
219  then result;
220 
221  else
222  equation
223  true = Util.isSome(inStatements) or Util.isSome(inProgram);
224  Error.addMessage(Error.STACK_OVERFLOW, {inCommand});
225  then "";
226 
227  end matchcontinue;
228 end handleCommand2;

384 protected function translateFile
385 "This function invokes the translator on a source file. The argument should be
386  a list with a single file name, with the rest of the list being an optional
387  list of libraries and .mo-files if the file is a .mo-file"
388  input list<String> inStringLst;
389 algorithm
390  _ := matchcontinue (inStringLst)
391  local
392  Absyn.Program p, pLibs;
393  DAE.DAElist d;
394  String s,str,f;
395  list<String> libs;
396  Absyn.Path cname;
397  Boolean silent,notsilent;
398  GlobalScript.Statements stmts;
399  FCore.Cache cache;
400  FCore.Graph env;
401  DAE.FunctionTree funcs;
402  list<Absyn.Class> cls;
403 
404  // A .mo-file, followed by an optional list of extra .mo-files and libraries.
405  // The last class in the first file will be instantiated.
406  case (libs)
407  algorithm
408  //print("Class to instantiate: " + Config.classToInstantiate() + "\n");
409  isEmptyOrFirstIsModelicaFile(libs);
410  execStatReset();
411  // Parse libraries and extra mo-files that might have been given at the command line.
412  for lib in libs loop
413  loadLib(lib);
414  end for;
415  // Show any errors that occured during parsing.
416  showErrors(Print.getErrorString(), ErrorExt.printMessagesStr(false));
417 
418  if Flags.isSet(Flags.DUMP) then
419  Debug.trace("\n--------------- Parsed program ---------------\n");
420  Dump.dump(SymbolTable.getAbsyn());
421  print(Print.getString());
422  end if;
423  if Flags.isSet(Flags.DUMP_GRAPHVIZ) then
424  DumpGraphviz.dump(SymbolTable.getAbsyn());
425  end if;
426 
427  execStat("Parsed file");
428 
429  // Instantiate the program.
430  (cache, env, d, cname) := instantiate();
431  p := SymbolTable.getAbsyn();
432 
433  d := if Flags.isSet(Flags.TRANSFORMS_BEFORE_DUMP) then DAEUtil.transformationsBeforeBackend(cache,env,d) else d;
434 
435  funcs := FCore.getFunctionTree(cache);
436 
437  Print.clearBuf();
438  execStat("Transformations before Dump");
439  s := if Config.silent() then "" else DAEDump.dumpStr(d, funcs);
440  execStat("DAEDump done");
441  Print.printBuf(s);
442  if Flags.isSet(Flags.DAE_DUMP_GRAPHV) then
443  DAEDump.dumpGraphviz(d);
444  end if;
445  execStat("Misc Dump");
446 
447  // Do any transformations required before going into code generation, e.g. if-equations to expressions.
448  d := if boolNot(Flags.isSet(Flags.TRANSFORMS_BEFORE_DUMP)) then DAEUtil.transformationsBeforeBackend(cache,env,d) else d;
449 
450  if not Config.silent() then
451  print(Print.getString());
452  end if;
453  execStat("Transformations before backend");
454 
455  // Run the backend.
456  optimizeDae(cache, env, d, p, cname);
457  // Show any errors or warnings if there are any!
458  showErrors(Print.getErrorString(), ErrorExt.printMessagesStr(false));
459  then ();
460 
461  /* Modelica script file .mos */
462  case (f::libs)
463  algorithm
464  isModelicaScriptFile(f);
465  // loading possible libraries given at the command line
466  for lib in libs loop
467  loadLib(lib);
468  end for;
469 
470  //System.startTimer();
471  //print("\nParseExp");
472  // parse our algorithm given in the script
473  stmts := Parser.parseexp(f);
474  //System.stopTimer();
475  //print("\nParseExp: " + realString(System.getTimerIntervalTime()));
476 
477  // are there any errors?
478  // show errors if there are any
479  showErrors(Print.getErrorString(), ErrorExt.printMessagesStr(false));
480  // evaluate statements and print the result to stdout directly
481  Interactive.evaluateToStdOut(stmts, true);
482  then
483  ();
484 
485  case {f} /* A template file .tpl (in the Susan language)*/
486  equation
487  isCodegenTemplateFile(f);
488  TplMain.main(f);
489  then
490  ();
491 
492  // deal with problems
493  case (f::_)
494  algorithm
495  if System.regularFileExists(f) then
496  print("Error processing file: ");
497  else
498  print(System.gettext("File does not exist: "));
499  end if;
500 
501  print(f); print("\n");
502  // show errors if there are any
503  showErrors(Print.getErrorString(), ErrorExt.printMessagesStr(false));
504  then
505  fail();
506 
507  end matchcontinue;
508 end translateFile;
