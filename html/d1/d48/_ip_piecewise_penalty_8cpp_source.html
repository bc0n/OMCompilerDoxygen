29  bool PiecewisePenalty::Acceptable(Number Fzconst, Number Fzlin)
30  {
31  DBG_START_METH("PiebcewisePenalty::Acceptable", dbg_verbosity);
32  DBG_ASSERT(!IsPiecewisePenaltyListEmpty());
33  bool acceptable = false;
34  std::vector<PiecewisePenEntry>::iterator iter;
35  // Avoid the entries of the piecewise penalty list becoming too many.
36  // Here, we require the entry number is less than or equal to max_piece_number,
37  // unless the entries are accepted by the regular Armijo conditions.
38  Index size = (Index)PiecewisePenalty_list_.size();
39  if (size >= max_piece_number_) {
40  Number trial_inf = Fzlin;
41  Number trial_barrier = Fzconst;
42  // First check the starting entry of the list.
43  iter = PiecewisePenalty_list_.begin();
44  Number value = iter->barrier_obj + iter->pen_r * iter->infeasi
45  - trial_barrier - iter->pen_r * trial_inf;
46  if (value >= 0.) {
47  iter++;
48  value = iter->barrier_obj + iter->pen_r * iter->infeasi
49  - trial_barrier - iter->pen_r * trial_inf;
50  if (value <= 0.) {
51  return false;
52  }
53  }
54  // Then check the ending entry of the list.
55  iter = PiecewisePenalty_list_.end();
56  iter--;
57  value = iter->barrier_obj + iter->pen_r * iter->infeasi
58  - trial_barrier - iter->pen_r * trial_inf;
59  if (value <= 0. && trial_inf <= iter->infeasi) {
60  return false;
61  }
62  // Check the next to the ending entry.
63  if (value >= 0. && trial_inf >= iter->infeasi) {
64  iter-=1;
65  value = iter->barrier_obj + iter->pen_r * iter->infeasi
66  - trial_barrier - iter->pen_r * trial_inf;
67  if (value <= 0.) {
68  return false;
69  }
70  }
71  // Finally, check the middle entries of the list.
72  Number value_left, value_mid, value_right;
73  for (iter = PiecewisePenalty_list_.begin() +1; iter != PiecewisePenalty_list_.end();
74  iter++) {
75  value_mid = iter->barrier_obj + iter->pen_r * iter->infeasi
76  - trial_barrier - iter->pen_r * trial_inf;
77  iter++;
78  value_right = iter->barrier_obj + iter->pen_r * iter->infeasi
79  - trial_barrier - iter->pen_r * trial_inf;
80  iter-=2;
81  value_left = iter->barrier_obj + iter->pen_r * iter->infeasi
82  - trial_barrier - iter->pen_r * trial_inf;
83  iter++;
84  if (value_left <= 0. && value_mid >= 0. && value_right <= 0.) {
85  return false;
86  }
87  }
88  }
89  // Check if the trial point is acceptable to the piecewise list
90  Number Fz;
91  for (iter = PiecewisePenalty_list_.begin(); iter != PiecewisePenalty_list_.end();
92  iter++) {
93  Fz = Fzconst + iter->pen_r * (Fzlin - iter->infeasi) - iter->barrier_obj ;
94  if (Fz < 0.) {
95  acceptable = true;
96  break;
97  }
98  }
99  iter = PiecewisePenalty_list_.end() -1;
100  if (acceptable == false && Fzlin < iter->infeasi) {
101  acceptable = true;
102  }
103  return acceptable;
104  }

120  void PiecewisePenalty::UpdateEntry(Number barrier_obj, Number infeasi)
121  {
122 
123  DBG_START_METH("PiecewisePenalty::UpdateEntry", dbg_verbosity);
124  DBG_ASSERT(!IsPiecewisePenaltyListEmpty());
125 
126  Number Gzi1, Gzi2;
127  Number epsM = 0.0; //1e-20;
128  Number TmpPen = 0.0;
129  // construt a temp list, copy current list to the temp list
130  std::vector<PiecewisePenEntry> TmpList(PiecewisePenalty_list_);
131  // Erases the elements of current list
132  PiecewisePenalty_list_.clear();
133  std::vector<PiecewisePenEntry>::iterator iter = TmpList.begin(), iter2;
134  Gzi1 = barrier_obj + iter->pen_r * ( infeasi - iter->infeasi) - iter->barrier_obj;
135  for (; iter <= TmpList.end()-1; iter++) {
136  // Be careful about this
137  if ( TmpList.size() > 1 && iter <= TmpList.end()-2 ) {
138  iter2 = iter+1;
139  Gzi2 = barrier_obj + iter2->pen_r * ( infeasi - iter2->infeasi) - iter2->barrier_obj;
140  }
141  else {
142  Gzi2 = infeasi - iter->infeasi;
143  }
144  if ( Gzi1 < -epsM && Gzi2 >= epsM ) {
145  if ( IsPiecewisePenaltyListEmpty() ) {
146  AddEntry(TmpPen, barrier_obj, infeasi);
147  }
148  if (Gzi2 > epsM) {
149  TmpPen = (iter->barrier_obj - barrier_obj)/( infeasi - iter->infeasi );
150  AddEntry(TmpPen, iter->barrier_obj, iter->infeasi);
151  }
152  }
153  if (Gzi1 >= epsM && Gzi2 < -epsM) {
154  if (Gzi1 > epsM) {
155  AddEntry(iter->pen_r, iter->barrier_obj, iter->infeasi);
156  }
157  TmpPen = (iter->barrier_obj - barrier_obj)/(infeasi - iter->infeasi);
158  AddEntry(TmpPen, barrier_obj, infeasi);
159  }
160  if (Gzi1 >= epsM && Gzi2 >= epsM) {
161  AddEntry(iter->pen_r, iter->barrier_obj, iter->infeasi);
162  }
163  // handle the last point
164  if ( iter == TmpList.end()-1 ) {
165  if ( Gzi1 < - epsM && Gzi2 < - epsM ) {
166  if ( IsPiecewisePenaltyListEmpty() ) {
167  AddEntry(0.0, barrier_obj, infeasi);
168  }
169  }
170  }
171  Gzi1 = Gzi2;
172  }
173  dim_ = (Index)PiecewisePenalty_list_.size();
174  }
