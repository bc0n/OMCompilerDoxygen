60  // Test functor, that fills the given 1-D vector in parallel. The
61  // functor is like APL's iota, except that it starts with an initial
62  // constant instead of with zero. That makes it more "interesting"
63  // so that tests are more likely to catch bugs.
64  template<class Device>

66  public:
67  // Functors need a execution_space typedef.
68  typedef Device execution_space;

70  // Constructor accepts a View of a 1-D array.
71  FillFunctor (const Kokkos::View<double*, Device>& vec) : vec_ (vec) {}
72 
73  // Initialize the array.
74  KOKKOS_INLINE_FUNCTION
75  void operator () (int i) const {
76  vec_[i] = 42.0 + static_cast<double> (i);
77  }
78 
79  private:
80  Kokkos::View<double*, Device> vec_;
81  };

83  // Custom deallocator for Teuchos::ArrayRCP. It doesn't actually
84  // deallocate the ArrayRCP's data (which belongs to the View passed
85  // into Deallocator's constructor, not to the ArrayRCP). Instead,
86  // it just keeps a reference to the View. That way, the data won't
87  // go away until the View's reference count goes to zero.
88  template<class Scalar, class ViewType>
89  class Deallocator {
90  public:
91  // Constructor takes the View that owns the memory.
92  Deallocator (const ViewType& view) : view_ (view) {}
93 
94  // "Deallocation function" doesn't actually deallocate its input
95  // pointer; the View is responsible for deallocation of its
96  // memory.
97  void free (Scalar*) {}
98  private:
99  ViewType view_; // View that owns the memory
100  };

157 // Get a Kokkos::View of a Teuchos::ArrayView, and make sure that
158 // it points to the same data. Thanks to Christian Trott for
159 // implementing the necessary functionality (View constructor for
160 // certain View specializations, that takes a raw pointer and
161 // dimensions) in Kokkos Array.
162 //
163 // This example will be useful for implementing the
164 // Tpetra::MultiVector methods get1dCopy and get2dCopy.
165 TEUCHOS_UNIT_TEST( LinkTeuchosAndKokkos, ViewOfArrayView ) {
166  TestDevice::initialize();
167  typedef Teuchos::Array<double>::size_type size_type;
168  typedef Kokkos::View<double*, Kokkos::LayoutLeft, TestDevice, Kokkos::MemoryUnmanaged> ka_view_type;
169  typedef Kokkos::View<const double*, Kokkos::LayoutLeft, TestDevice, Kokkos::MemoryUnmanaged> ka_const_view_type;
170 
171  const size_type numElts = 10;
172  Teuchos::Array<double> x (numElts);
173  for (size_type k = 0; k < numElts; ++k) {
174  x[k] = 42.0 + static_cast<double> (k);
175  }
176  // You can make an (unmanaged) View of a raw array with left or
177  // right (Fortran or C) layout on the HostSpace device, just by passing
178  // the array and stride(s) into the constructor. If you want the
179  // dimensions to differ from the strides, you'll have to create a
180  // subview with the desired dimensions.
181  ka_view_type x_view (x.getRawPtr (), x.size ());
182 
183  TEST_EQUALITY( x.size(), static_cast<size_type>(x_view.dimension_0()) );
184  for (size_type k = 0; k < x.size (); ++k) {
185  TEST_EQUALITY( x_view[k], x[k] );
186  }
187 
188  // This ensures that conversions from double* to const double* work correctly.
189  // x.getRawPtr() returns double*.
190  ka_const_view_type x_view_const ( (const double*) x.getRawPtr (), x.size ());
191 
192  TEST_EQUALITY( x.size(), static_cast<size_type>(x_view_const.dimension_0()) );
193  for (size_type k = 0; k < x.size (); ++k) {
194  TEST_EQUALITY( x_view_const[k], x[k] );
195  }
196 
197  TestDevice::finalize();
198 }

201 // Create a 2-D View (LayoutLeft, in HostSpace memory), and then create an
202 // ArrayRCP which owns that View (using a custom destructor). This
203 // will be useful for implementing Tpetra::MultiVector's getData,
204 // getDataNonConst, get1dView, and get1dViewNonConst methods.
205 TEUCHOS_UNIT_TEST( LinkTeuchosAndKokkos, ArrayRCP1D_of_2DView ) {
206  TestDevice::initialize();
207 
208  typedef Kokkos::View<double**, Kokkos::LayoutLeft, TestDevice> ka_view_type;
209 
210  const size_t numRows = 75;
211  const size_t numCols = 5;
212  const size_t stride = 100;
213  const size_t ZERO = static_cast<size_t> (0);
214 
215  ka_view_type X ("X", stride, numCols);
216  ka_view_type X_view = Kokkos::subview (X, std::make_pair (ZERO, numRows), std::make_pair (ZERO, numCols));
217  TEST_EQUALITY(X_view.dimension_0(), numRows);
218  TEST_EQUALITY(X_view.dimension_1(), numCols);
219 
220  // Test that the strides of X_view are correct, for int. Kokkos
221  // Array templates the stride() method on the integer type of the
222  // input. I just want to make sure that this works for different
223  // integer types; I'll test int here and size_t.
224  {
225  int strides[3] = { 0 , 0 , 0 };
226  X_view.stride (strides);
227  TEST_EQUALITY_CONST(strides[0], 1); // stride between X_view(i,j) and X_view(i+1,j)
228  // The stride must be at least as given, but can be greater (due to possible padding for alignment).
229  TEST_ASSERT(static_cast<size_t>(strides[1]) >= stride); // stride between X_view(i,j) and X_view(i,j+1)
230  }
231 
232  // Test that the strides of X_view are correct, for size_t.
233  {
234  size_t strides[3] = { 0 , 0 , 0 };
235  X_view.stride (strides);
236  TEST_EQUALITY_CONST(strides[0], static_cast<size_t>(1)); // stride between X_view(i,j) and X_view(i+1,j)
237  // The stride must be at least as given, but can be greater (due to possible padding for alignment).
238  TEST_ASSERT(strides[1] >= stride); // stride between X_view(i,j) and X_view(i,j+1)
239  }
240 
241  // Create a nonowning "view" of X's data. The ArrayRCP has a custom
242  // destructor which simply holds a reference to the View and doesn't
243  // actually deallocate memory. That way, the ArrayRCP can use the
244  // View's raw pointer, but still defers to the View for memory
245  // management.
246 
247  Teuchos::ArrayRCP<double> Y_values (X.ptr_on_device (), 0, stride*numCols, Deallocator<double, ka_view_type> (X), true);
248  TEST_EQUALITY(Y_values.getRawPtr(), X.ptr_on_device());
249  TEST_EQUALITY(Y_values.getRawPtr(), X_view.ptr_on_device());
250  TEST_EQUALITY(Y_values.size(), stride*numCols);
251 
252  TestDevice::finalize();
253 }

256 // Create a 2-D View (LayoutLeft, in HostSpace memory), and then create an
257 // ArrayRCP<ArrayRCP<double> >, each entry of which owns the
258 // corresponding column of that View (using a custom destructor).
259 // This will be useful for implementing Tpetra::MultiVector's
260 // get2dView and get2dViewNonConst methods.
261 TEUCHOS_UNIT_TEST( LinkTeuchosAndKokkos, ArrayRCP2D_of_2DView ) {
262  // View<double*, LayoutLeft, ...> and View<double*, LayoutRight,
263  // ...> always have unit stride. Furthermore, subview(X,j) returns
264  // a view of a column for LayoutLeft, and a view of a row for
265  // LayoutRight. Thus, to support both layouts, each column of the
266  // multivector still needs to be a View<double**, ...>, and we have
267  // to use the subview() overload that takes two ranges of row and
268  // column indices.
269  typedef Kokkos::View<double**, Kokkos::LayoutLeft, TestDevice> ka_view_type;
270  TestDevice::initialize();
271 
272  const size_t numRows = 75;
273  const size_t numCols = 5;
274  const size_t stride = 100;
275  const size_t ZERO = static_cast<size_t> (0);
276 
277  ka_view_type X ("X", stride, numCols);
278  ka_view_type X_view = Kokkos::subview (X, std::make_pair (ZERO, numRows), std::make_pair (ZERO, numCols));
279  TEST_EQUALITY( & X(0,0) , & X_view(0,0) );
280  TEST_EQUALITY(X_view.dimension_0(), numRows);
281  TEST_EQUALITY(X_view.dimension_1(), numCols);
282 
283  // Test that the strides of X_view are correct, for size_t.
284  {
285  size_t strides[3]; // stride[rank] is the maximum span
286  strides[0] = static_cast<size_t> (0);
287  strides[1] = static_cast<size_t> (0);
288  strides[2] = static_cast<size_t> (0);
289  X_view.stride (strides);
290  TEST_EQUALITY_CONST(strides[0], static_cast<size_t>(1)); // stride between X_view(i,j) and X_view(i+1,j)
291  // The stride must be at least as given, but can be greater (due to possible padding for alignment).
292  TEST_ASSERT(strides[1] >= stride); // stride between X_view(i,j) and X_view(i,j+1)
293  }
294 
295  // Make a 2-D "view" (array of arrays) of X_view. This is how we
296  // will implement Tpetra::MultiVector methods like get2dView.
297  Teuchos::ArrayRCP<Teuchos::ArrayRCP<double> > Y_2D (X_view.dimension_1 ());
298  for (size_t j = 0; j < static_cast<size_t> (X_view.dimension_1 ()); ++j) {
299  ka_view_type X_j = Kokkos::subview (X_view, std::make_pair (ZERO, numRows), std::make_pair (j, j+1));
300  TEST_EQUALITY( & X_view(0,j) , & X_j(0,0) );
301  TEST_EQUALITY(static_cast<size_t>(X_j.dimension_0()), numRows);
302  TEST_EQUALITY_CONST(X_j.dimension_1(), 1);
303 
304  // Test that the strides of X_j are correct.
305  {
306  size_t strides[3]; // stride[rank] is the maximum span
307  strides[0] = static_cast<size_t> (0);
308  strides[1] = static_cast<size_t> (0);
309  strides[2] = static_cast<size_t> (0);
310  X_j.stride (strides);
311  TEST_EQUALITY_CONST(strides[0], static_cast<size_t>(1)); // stride between X_j(i,j) and X_j(i+1,j)
312  // Stride between X_j(i,j) and X_j(i,j+1), even though X_j only
313  // has one column. The stride must be at least as given, but
314  // can be greater (due to possible padding for alignment).
315  TEST_ASSERT(strides[1] >= stride);
316  }
317 
318  // Create a nonowning "view" of X_j's data. The ArrayRCP has a
319  // custom destructor which simply holds a reference to the View
320  // X_j, and doesn't actually deallocate memory. That way, the
321  // ArrayRCP can use the View's raw pointer, but still defers to
322  // the View for memory management.
323  Teuchos::ArrayRCP<double> Y_j (X_j.ptr_on_device (), 0, numRows, Deallocator<double, ka_view_type> (X_j), true);
324  Y_2D[j] = Y_j;
325  }
326  TestDevice::finalize();
327 }
