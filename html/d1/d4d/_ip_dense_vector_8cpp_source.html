526  void DenseVector::AddTwoVectorsImpl(Number a, const Vector& v1,
527  Number b, const Vector& v2, Number c)
528  {
529  const Number* values_v1=NULL;
530  bool homogeneous_v1=false;
531  Number scalar_v1 = 0;
532  if (a!=0.) {
533  const DenseVector* dense_v1 = static_cast<const DenseVector*>(&v1);
534  DBG_ASSERT(dynamic_cast<const DenseVector*>(&v1));
535 
536  DBG_ASSERT(dense_v1->initialized_);
537  DBG_ASSERT(Dim() == dense_v1->Dim());
538  values_v1=dense_v1->values_;
539  homogeneous_v1=dense_v1->homogeneous_;
540  if (homogeneous_v1)
541  scalar_v1 = dense_v1->scalar_;
542  }
543  const Number* values_v2=NULL;
544  bool homogeneous_v2=false;
545  Number scalar_v2 = 0;
546  if (b!=0.) {
547  const DenseVector* dense_v2 = static_cast<const DenseVector*>(&v2);
548  DBG_ASSERT(dynamic_cast<const DenseVector*>(&v2));
549 
550  DBG_ASSERT(dense_v2->initialized_);
551  DBG_ASSERT(Dim() == dense_v2->Dim());
552  values_v2=dense_v2->values_;
553  homogeneous_v2=dense_v2->homogeneous_;
554  if (homogeneous_v2)
555  scalar_v2 = dense_v2->scalar_;
556  }
557  DBG_ASSERT(c==0. || initialized_);
558  if ((c==0. || homogeneous_) && homogeneous_v1 && homogeneous_v2 ) {
559  homogeneous_ = true;
560  Number val = 0;
561  if (c!=0.) {
562  val = c*scalar_;
563  }
564  scalar_ = val + a*scalar_v1 + b*scalar_v2;
565  initialized_ = true;
566  return;
567  }
568  if (c==0.) {
569  // make sure we have memory allocated for this vector
570  values_allocated();
571  homogeneous_ = false;
572  }
573 
574  // If any of the vectors is homogeneous, call the default implementation
575  if ( homogeneous_ || homogeneous_v1 || homogeneous_v2) {
576  // ToDo:Should we implement specialized methods here too?
577  Vector::AddTwoVectorsImpl(a, v1, b, v2, c);
578  return;
579  }

582  // capture all cases for a, b, and c separately...
583  if (c==0 ) {
584  if (a==1.) {
585  if (b==0.) {
586  IpBlasDcopy(Dim(), values_v1, 1, values_, 1);
587  }
588  else if (b==1.) {
589  for (Index i=0; i<Dim(); i++) {
590  values_[i] = values_v1[i] + values_v2[i];
591  }
592  }
593  else if (b==-1.) {
594  for (Index i=0; i<Dim(); i++) {
595  values_[i] = values_v1[i] - values_v2[i];
596  }
597  }
598  else {
599  for (Index i=0; i<Dim(); i++) {
600  values_[i] = values_v1[i] + b*values_v2[i];
601  }
602  }
603  }
604  else if (a==-1.) {
605  if (b==0.) {
606  for (Index i=0; i<Dim(); i++) {
607  values_[i] = -values_v1[i];
608  }
609  }
610  else if (b==1.) {
611  for (Index i=0; i<Dim(); i++) {
612  values_[i] = -values_v1[i] + values_v2[i];
613  }
614  }
615  else if (b==-1.) {
616  for (Index i=0; i<Dim(); i++) {
617  values_[i] = -values_v1[i] - values_v2[i];
618  }
619  }
620  else {
621  for (Index i=0; i<Dim(); i++) {
622  values_[i] = -values_v1[i] + b*values_v2[i];
623  }
624  }
625  }

627  if (b==0.) {
628  Number zero = 0.;

630  }

632  for (Index i=0; i<Dim(); i++) {
633  IpBlasDcopy(Dim(), values_v2, 1, values_, 1);
634  }
635  }
636  else if (b==-1.) {
637  for (Index i=0; i<Dim(); i++) {
638  values_[i] = -values_v2[i];
639  }

644  }

696  for (Index i=0; i<Dim(); i++) {

731  else if (b==1.) {
732  for (Index i=0; i<Dim(); i++) {
733  values_[i] += a*values_v1[i] + values_v2[i];
