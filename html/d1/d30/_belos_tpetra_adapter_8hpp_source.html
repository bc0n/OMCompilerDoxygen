129  static Teuchos::RCP<MV> CloneCopy (const MV& X)
130  {
131  // Make a deep copy of X. The one-argument copy constructor
132  // does a shallow copy by default; the second argument tells it
133  // to do a deep copy.
134  Teuchos::RCP<MV> X_copy (new MV (X, Teuchos::Copy));
135  // Make Tpetra::MultiVector use the new view semantics. This is
136  // a no-op for the Kokkos refactor version of Tpetra; it only
137  // does something for the "classic" version of Tpetra. This
138  // shouldn't matter because Belos only handles MV through RCP
139  // and through this interface anyway, but it doesn't hurt to set
140  // it and make sure that it works.
141  X_copy->setCopyOrView (Teuchos::View);

161  const size_t inNumVecs = mv.getNumVectors ();
162  TEUCHOS_TEST_FOR_EXCEPTION(
163  index.size () > 0 && *std::min_element (index.begin (), index.end ()) < 0,

358  static void
359  MvTimesMatAddMv (Scalar alpha,
360  const MV& A,
361  const Teuchos::SerialDenseMatrix<int, Scalar>& B,
362  Scalar beta,
363  MV& mv)
364  {
365  using Teuchos::ArrayView;
366  using Teuchos::Comm;
367  using Teuchos::rcpFromRef;
368  typedef ::Tpetra::Map<LO, GO, Node> map_type;
369 
370 #ifdef HAVE_BELOS_TPETRA_TIMERS
371  const std::string timerName ("Belos::MVT::MvTimesMatAddMv");
372  Teuchos::RCP<Teuchos::Time> timer =
373  Teuchos::TimeMonitor::lookupCounter (timerName);
374  if (timer.is_null ()) {
375  timer = Teuchos::TimeMonitor::getNewCounter (timerName);
376  }
377  TEUCHOS_TEST_FOR_EXCEPTION(
378  timer.is_null (), std::logic_error,
379  "Belos::MultiVecTraits::MvTimesMatAddMv: "
380  "Failed to look up timer \"" << timerName << "\". "
381  "Please report this bug to the Belos developers.");
382 
383  // This starts the timer. It will be stopped on scope exit.
384  Teuchos::TimeMonitor timeMon (*timer);
385 #endif // HAVE_BELOS_TPETRA_TIMERS
386 
387  // Check if B is 1-by-1, in which case we can just call update()
388  if (B.numRows () == 1 && B.numCols () == 1) {
389  mv.update (alpha*B(0,0), A, beta);
390  return;
391  }
392 
393  // Create local map
394  Teuchos::SerialComm<int> serialComm;
395  map_type LocalMap (B.numRows (), A.getMap ()->getIndexBase (),
396  rcpFromRef<const Comm<int> > (serialComm),
397  ::Tpetra::LocallyReplicated, A.getMap ()->getNode ());
398  // encapsulate Teuchos::SerialDenseMatrix data in ArrayView
399  ArrayView<const Scalar> Bvalues (B.values (), B.stride () * B.numCols ());
400  // create locally replicated MultiVector with a copy of this data
401  MV B_mv (rcpFromRef (LocalMap), Bvalues, B.stride (), B.numCols ());
402  mv.multiply (Teuchos::NO_TRANS, Teuchos::NO_TRANS, alpha, A, B_mv, beta);
403  }

430  static void
431  MvTransMv (const Scalar alpha,
432  const MV& A,
433  const MV& B,
434  Teuchos::SerialDenseMatrix<int,Scalar>& C)
435  {
436  using ::Tpetra::LocallyReplicated;
437  using Teuchos::Comm;
438  using Teuchos::RCP;
439  using Teuchos::rcp;
440  using Teuchos::TimeMonitor;
441  typedef ::Tpetra::Map<LO,GO,Node> map_type;
442 
443 #ifdef HAVE_BELOS_TPETRA_TIMERS
444  const std::string timerName ("Belos::MVT::MvTransMv");
445  RCP<Teuchos::Time> timer = TimeMonitor::lookupCounter (timerName);
446  if (timer.is_null ()) {
447  timer = TimeMonitor::getNewCounter (timerName);
448  }
449  TEUCHOS_TEST_FOR_EXCEPTION(
450  timer.is_null (), std::logic_error, "Belos::MvTransMv: "
451  "Failed to look up timer \"" << timerName << "\". "
452  "Please report this bug to the Belos developers.");
453 
454  // This starts the timer. It will be stopped on scope exit.
455  TimeMonitor timeMon (*timer);
456 #endif // HAVE_BELOS_TPETRA_TIMERS
457 
458  // Form alpha * A^H * B, then copy into the SerialDenseMatrix.
459  // We will create a multivector C_mv from a a local map. This
460  // map has a serial comm, the purpose being to short-circuit the
461  // MultiVector::reduce() call at the end of
462  // MultiVector::multiply(). Otherwise, the reduced multivector
463  // data would be copied back to the GPU, only to turn around and
464  // have to get it back here. This saves us a round trip for
465  // this data.
466  const int numRowsC = C.numRows ();
467  const int numColsC = C.numCols ();
468  const int strideC = C.stride ();
469 
470  // Check if numRowsC == numColsC == 1, in which case we can call dot()
471  if (numRowsC == 1 && numColsC == 1) {
472  if (alpha == Teuchos::ScalarTraits<Scalar>::zero ()) {
473  // Short-circuit, as required by BLAS semantics.
474  C(0,0) = alpha;
475  return;
476  }
477  A.dot (B, Teuchos::ArrayView<Scalar> (C.values (), 1));
478  if (alpha != Teuchos::ScalarTraits<Scalar>::one ()) {
479  C(0,0) *= alpha;
480  }
481  return;
482  }
483 
484  // get comm
485  RCP<const Comm<int> > pcomm = A.getMap ()->getComm ();
486 
487  // create local map with comm
488  RCP<const map_type> LocalMap =
489  rcp (new map_type (numRowsC, 0, pcomm, LocallyReplicated,
490  A.getMap ()->getNode ()));
491  // create local multivector to hold the result
492  const bool INIT_TO_ZERO = true;
493  MV C_mv (LocalMap, numColsC, INIT_TO_ZERO);
494 
495  // multiply result into local multivector
496  C_mv.multiply (Teuchos::CONJ_TRANS, Teuchos::NO_TRANS, alpha, A, B,
497  Teuchos::ScalarTraits<Scalar>::zero ());
498 
499  // create arrayview encapsulating the Teuchos::SerialDenseMatrix
500  Teuchos::ArrayView<Scalar> C_view (C.values (), strideC * numColsC);
501 
502  // No accumulation to do (since Tpetra has already done it);
503  // simply extract the multivector data into C.
504  // Extract a copy of the result into the array view
505  // (and therefore, the SerialDenseMatrix).
506  C_mv.get1dCopy (C_view, strideC);
507  }

534  static void
535  MvNorm (const MV& mv,
536  std::vector<typename Teuchos::ScalarTraits<Scalar>::magnitudeType>& normvec,
537  NormType type=TwoNorm)
538  {
539  typedef typename Teuchos::ScalarTraits<Scalar>::magnitudeType magnitude_type;
540 #ifdef HAVE_TPETRA_DEBUG
541  TEUCHOS_TEST_FOR_EXCEPTION(
542  normvec.size () < static_cast<std::vector<int>::size_type> (mv.getNumVectors ()),
543  std::invalid_argument,
544  "Belos::MultiVecTraits::MvNorm(mv,normvec): The normvec output "
545  "argument must have at least as many entries as the number of vectors "
546  "(columns) in the MultiVector mv. normvec.size() = " << normvec.size ()
547  << " < mv.getNumVectors() = " << mv.getNumVectors () << ".");
548 #endif // HAVE_TPETRA_DEBUG
549  Teuchos::ArrayView<magnitude_type> av (normvec);
550  switch (type) {
551  case OneNorm:
552  mv.norm1 (av (0, mv.getNumVectors ()));
553  break;
554  case TwoNorm:
555  mv.norm2 (av (0, mv.getNumVectors ()));
556  break;
557  case InfNorm:
558  mv.normInf (av (0,mv.getNumVectors ()));
559  break;
560  default:
561  // Throw logic_error rather than invalid_argument, because if
562  // we get here, it's probably the fault of a Belos solver,
563  // rather than a user giving Belos an invalid input.
564  TEUCHOS_TEST_FOR_EXCEPTION(
565  true, std::logic_error,
566  "Belos::MultiVecTraits::MvNorm: Invalid NormType value " << type
567  << ". Valid values are OneNorm=" << OneNorm << ", TwoNorm="
568  << TwoNorm <<", and InfNorm=" << InfNorm << ". If you are a Belos "
569  "user and have not modified Belos in any way, and you get this "
570  "message, then this is probably a bug in the Belos solver you were "
571  "using. Please report this to the Belos developers.");
572  }

598  static void
599  SetBlock (const MV& A, const Teuchos::Range1D& index, MV& mv)
600  {
601  // Range1D bounds are signed; size_t is unsigned.
602  // Assignment of Tpetra::MultiVector is a deep copy.
603 
604  // Tpetra::MultiVector::getNumVectors() returns size_t. It's
605  // fair to assume that the number of vectors won't overflow int,
606  // since the typical use case of multivectors involves few
607  // columns, but it's friendly to check just in case.
608  const size_t maxInt =
609  static_cast<size_t> (Teuchos::OrdinalTraits<int>::max ());
610  const bool overflow =
611  maxInt < A.getNumVectors () && maxInt < mv.getNumVectors ();
612  if (overflow) {
613  std::ostringstream os;
614  os << "Belos::MultiVecTraits::SetBlock(A, index=[" << index.lbound ()
615  << ", " << index.ubound () << "], mv): ";
616  TEUCHOS_TEST_FOR_EXCEPTION(
617  maxInt < A.getNumVectors (), std::range_error, os.str () << "Number "
618  "of columns (size_t) in the input MultiVector 'A' overflows int.");
619  TEUCHOS_TEST_FOR_EXCEPTION(
620  maxInt < mv.getNumVectors (), std::range_error, os.str () << "Number "
621  "of columns (size_t) in the output MultiVector 'mv' overflows int.");
622  }
623  // We've already validated the static casts above.
624  const int numColsA = static_cast<int> (A.getNumVectors ());
625  const int numColsMv = static_cast<int> (mv.getNumVectors ());
626  // 'index' indexes into mv; it's the index set of the target.
627  const bool validIndex =
628  index.lbound () >= 0 && index.ubound () < numColsMv;
629  // We can't take more columns out of A than A has.
630  const bool validSource = index.size () <= numColsA;
631 
632  if (! validIndex || ! validSource) {
633  std::ostringstream os;
634  os << "Belos::MultiVecTraits::SetBlock(A, index=[" << index.lbound ()
635  << ", " << index.ubound () << "], mv): ";
636  TEUCHOS_TEST_FOR_EXCEPTION(
637  index.lbound() < 0, std::invalid_argument,
638  os.str() << "Range lower bound must be nonnegative.");
639  TEUCHOS_TEST_FOR_EXCEPTION(
640  index.ubound() >= numColsMv, std::invalid_argument,
641  os.str() << "Range upper bound must be less than the number of "
642  "columns " << numColsA << " in the 'mv' output argument.");
643  TEUCHOS_TEST_FOR_EXCEPTION(
644  index.size() > numColsA, std::invalid_argument,
645  os.str() << "Range must have no more elements than the number of "
646  "columns " << numColsA << " in the 'A' input argument.");
647  TEUCHOS_TEST_FOR_EXCEPTION(
648  true, std::logic_error, "Should never get here!");
649  }
650 
651  // View of the relevant column(s) of the target multivector mv.
652  // We avoid view creation overhead by only creating a view if
653  // the index range is different than [0, (# columns in mv) - 1].
654  Teuchos::RCP<MV> mv_view;
655  if (index.lbound () == 0 && index.ubound () + 1 == numColsMv) {
656  mv_view = Teuchos::rcpFromRef (mv); // Non-const, non-owning RCP
657  } else {
658  mv_view = CloneViewNonConst (mv, index);
659  }
660 
661  // View of the relevant column(s) of the source multivector A.
662  // If A has fewer columns than mv_view, then create a view of
663  // the first index.size() columns of A.
664  Teuchos::RCP<const MV> A_view;
665  if (index.size () == numColsA) {
666  A_view = Teuchos::rcpFromRef (A); // Const, non-owning RCP
667  } else {
668  A_view = CloneView (A, Teuchos::Range1D (0, index.size () - 1));
669  }
670 
671  ::Tpetra::deep_copy (*mv_view, *A_view);
672  }

674  static void Assign (const MV& A, MV& mv)
675  {
676  const char errPrefix[] = "Belos::MultiVecTraits::Assign(A, mv): ";
677 
678  // Range1D bounds are signed; size_t is unsigned.
679  // Assignment of Tpetra::MultiVector is a deep copy.
680 
681  // Tpetra::MultiVector::getNumVectors() returns size_t. It's
682  // fair to assume that the number of vectors won't overflow int,
683  // since the typical use case of multivectors involves few
684  // columns, but it's friendly to check just in case.
685  const size_t maxInt =
686  static_cast<size_t> (Teuchos::OrdinalTraits<int>::max ());
687  const bool overflow =
688  maxInt < A.getNumVectors () && maxInt < mv.getNumVectors ();
689  if (overflow) {
690  TEUCHOS_TEST_FOR_EXCEPTION(
691  maxInt < A.getNumVectors(), std::range_error,
692  errPrefix << "Number of columns in the input multivector 'A' "
693  "(a size_t) overflows int.");
694  TEUCHOS_TEST_FOR_EXCEPTION(
695  maxInt < mv.getNumVectors(), std::range_error,
696  errPrefix << "Number of columns in the output multivector 'mv' "
697  "(a size_t) overflows int.");
698  TEUCHOS_TEST_FOR_EXCEPTION(
699  true, std::logic_error, "Should never get here!");
700  }
701  // We've already validated the static casts above.
702  const int numColsA = static_cast<int> (A.getNumVectors ());
703  const int numColsMv = static_cast<int> (mv.getNumVectors ());
704  if (numColsA > numColsMv) {
705  TEUCHOS_TEST_FOR_EXCEPTION(
706  numColsA > numColsMv, std::invalid_argument,
707  errPrefix << "Input multivector 'A' has " << numColsA << " columns, "
708  "but output multivector 'mv' has only " << numColsMv << " columns.");
709  TEUCHOS_TEST_FOR_EXCEPTION(true, std::logic_error, "Should never get here!");
710  }
711  if (numColsA == numColsMv) {
712  ::Tpetra::deep_copy (mv, A);
713  } else {
714  Teuchos::RCP<MV> mv_view =
715  CloneViewNonConst (mv, Teuchos::Range1D (0, numColsA - 1));
716  ::Tpetra::deep_copy (*mv_view, A);
717  }
718  }
