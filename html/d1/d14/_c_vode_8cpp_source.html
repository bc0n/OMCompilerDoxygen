329 void Cvode::solve(const SOLVERCALL action)
330 {
331  bool writeEventOutput = (_settings->getGlobalSettings()->getOutputPointType() == OPT_ALL);
332  bool writeOutput = !(_settings->getGlobalSettings()->getOutputPointType() == OPT_NONE);
333 
334 #ifdef RUNTIME_PROFILING
335  MEASURETIME_REGION_DEFINE(cvodeSolveFunctionHandler, "solve");
336  if (MeasureTime::getInstance() != NULL)
337  {
338  MEASURETIME_START(solveFunctionStartValues, cvodeSolveFunctionHandler, "solve");
339  }
340 #endif
341 
342  if (_cvodesettings && _system)
343  {
344  // Solver und System fÃ¼r Integration vorbereiten
345  if ((action & RECORDCALL) && (action & FIRST_CALL))
346  {
347 #ifdef RUNTIME_PROFILING
348  MEASURETIME_REGION_DEFINE(cvodeInitializeHandler, "CVodeInitialize");
349  if (MeasureTime::getInstance() != NULL)
350  {
351  MEASURETIME_START(measuredFunctionStartValues, cvodeInitializeHandler, "CVodeInitialize");
352  }
353 #endif
354 
355  initialize();
356 
357 #ifdef RUNTIME_PROFILING
358  if (MeasureTime::getInstance() != NULL)
359  {
360  MEASURETIME_END(measuredFunctionStartValues, measuredFunctionEndValues, (*measureTimeFunctionsArray)[4], cvodeInitializeHandler);
361  }
362 #endif
363 
364  if (writeOutput)
365  writeToFile(0, _tCurrent, _h);
366  _tLastWrite = 0;
367 
368  return;
369  }
370 
371  if ((action & RECORDCALL) && !(action & FIRST_CALL))
372  {
373  writeToFile(_accStps, _tCurrent, _h);
374  return;
375  }
376 
377  // Nach einem TimeEvent wird der neue Zustand recorded
378  if (action & RECALL)
379  {
380  _firstStep = true;
381  if (writeEventOutput)
382  writeToFile(0, _tCurrent, _h);
383  if (writeOutput)
384  writeCVodeOutput(_tCurrent, _h, _locStps);
385  _continuous_system->getContinuousStates(_z);
386  }
387 
388  // Solver soll fortfahren
389  _solverStatus = ISolver::CONTINUE;
390 
391  while ((_solverStatus & ISolver::CONTINUE) && !_interrupt)
392  {
393  // Zuvor wurde initialize aufgerufen und hat funktioniert => RESET IDID
394  if (_idid == 5000)
395  _idid = 0;
396 
397  // Solveraufruf
398  if (_idid == 0)
399  {
400  // ZÃ¤hler zurÃ¼cksetzen
401  _accStps = 0;
402  _locStps = 0;
403 
404  // Solverstart
405  CVodeCore();
406 
407  }
408 
409  // Integration war nicht erfolgreich und wurde auch nicht vom User unterbrochen
410  if (_idid != 0 && _idid != 1)
411  {
412  _solverStatus = ISolver::SOLVERERROR;
413  //throw ModelicaSimulationError(SOLVER,_idid,_tCurrent,"CVode::solve()");
414  throw ModelicaSimulationError(SOLVER, "CVode::solve()");
415  }
416 
417  // Abbruchkriterium (erreichen der Endzeit)
418  else if ((_tEnd - _tCurrent) <= dynamic_cast<ISolverSettings*>(_cvodesettings)->getEndTimeTol())
419  _solverStatus = DONE;
420  }
421 
422  _firstCall = false;
423 
424  }
425  else
426  {
427 
428  throw ModelicaSimulationError(SOLVER, "CVode::solve()");
429  }
430 
431 #ifdef RUNTIME_PROFILING
432  if (MeasureTime::getInstance() != NULL)
433  {
434  MEASURETIME_END(solveFunctionStartValues, solveFunctionEndValues, (*measureTimeFunctionsArray)[1], cvodeSolveFunctionHandler);
435 
436  long int nst, nfe, nsetups, netf, nni, ncfn;
437  int qlast, qcur;
438  realtype h0u, hlast, hcur, tcur;
439 
440  int flag;
441 
442  flag = CVodeGetIntegratorStats(_cvodeMem, &nst, &nfe, &nsetups, &netf, &qlast, &qcur, &h0u, &hlast, &hcur, &tcur);
443  flag = CVodeGetNonlinSolvStats(_cvodeMem, &nni, &ncfn);
444 
445  MeasureTimeValuesSolver solverVals = MeasureTimeValuesSolver(nfe, netf);
446  (*measureTimeFunctionsArray)[6]->_sumMeasuredValues->_numCalcs += nst;
447  (*measureTimeFunctionsArray)[6]->_sumMeasuredValues->add(&solverVals);
448  }
449 #endif

463 void Cvode::CVodeCore()
464 {
465  _idid = CVodeReInit(_cvodeMem, _tCurrent, _CV_y);
466  _idid = CVodeSetStopTime(_cvodeMem, _tEnd);
467  _idid = CVodeSetInitStep(_cvodeMem, 1e-12);
468  if (_idid < 0)
469  throw ModelicaSimulationError(SOLVER, "CVode::ReInit");
470 
471  bool writeEventOutput = (_settings->getGlobalSettings()->getOutputPointType() == OPT_ALL);
472  bool writeOutput = !(_settings->getGlobalSettings()->getOutputPointType() == OPT_NONE);
473 
474  while ((_solverStatus & ISolver::CONTINUE) && !_interrupt)
475  {
476  _cv_rt = CVode(_cvodeMem, _tEnd, _CV_y, &_tCurrent, CV_ONE_STEP);
477 
478  _idid = CVodeGetNumSteps(_cvodeMem, &_locStps);
479  if (_idid != CV_SUCCESS)
480  throw ModelicaSimulationError(SOLVER, "CVodeGetNumSteps failed. The cvode mem pointer is NULL");
481 
482  _idid = CVodeGetLastStep(_cvodeMem, &_h);
483  if (_idid != CV_SUCCESS)
484  throw ModelicaSimulationError(SOLVER, "CVodeGetLastStep failed. The cvode mem pointer is NULL");
485 
486  //set completed step to system and check if terminate was called
487  if (_continuous_system->stepCompleted(_tCurrent))
488  _solverStatus = DONE;
489 
490  //Check if there was at least one output-point within the last solver interval
491  // -> Write output if true
492  if (writeOutput)
493  {
494 
495  try
496  {
497  writeCVodeOutput(_tCurrent, _h, _locStps);
498  }
499  catch (std::exception& ex)
500  {
501 
502  if (!(_cv_rt == CV_ROOT_RETURN)) // if a zero crossing was dected before the event iteration was called and writeoutput throws and error
503  // for this time step the event iteration evaluates the system with corrected values.
504  {
505  string error_msg = string("CVode write output failed") + ex.what();
506  throw ModelicaSimulationError(SOLVER, error_msg);
507  }
508  }
509  }
510 
511 #ifdef RUNTIME_PROFILING
512  MEASURETIME_REGION_DEFINE(cvodeStepCompletedHandler, "CVodeStepCompleted");
513  if (MeasureTime::getInstance() != NULL)
514  {
515  MEASURETIME_START(measuredFunctionStartValues, cvodeStepCompletedHandler, "CVodeStepCompleted");
516  }
517 #endif
518 
519 
520 
521 #ifdef RUNTIME_PROFILING
522  if (MeasureTime::getInstance() != NULL)
523  {
524  MEASURETIME_END(measuredFunctionStartValues, measuredFunctionEndValues, (*measureTimeFunctionsArray)[5], cvodeStepCompletedHandler);
525  }
526 #endif
527 
528  // Perform state selection
529  bool state_selection = stateSelection();
530  if (state_selection)
531  _continuous_system->getContinuousStates(_z);
532 
533  _zeroFound = false;
534 
535  // Check if step was successful
536  if (check_flag(&_cv_rt, "CVode", 1))
537  {
538  _solverStatus = ISolver::SOLVERERROR;
539  break;
540  }
541 
542  // A root was found
543  if ((_cv_rt == CV_ROOT_RETURN) && !isInterrupted())
544  {
545  // CVode is setting _tCurrent to the time where the first event occurred
546  double _abs = fabs(_tLastEvent - _tCurrent);
547  _zeroFound = true;
548 
549  if ((_abs < 1e-3) && _event_n == 0)
550  {
551  _tLastEvent = _tCurrent;
552  _event_n++;
553  }
554  else if ((_abs < 1e-3) && (_event_n >= 1 && _event_n < 500))
555  {
556  _event_n++;
557  }
558  else if ((_abs >= 1e-3))
559  {
560  //restart event counter
561  _tLastEvent = _tCurrent;
562  _event_n = 0;
563  }
564  else{
565  std::stringstream zeros;
566  _idid = CVodeGetRootInfo(_cvodeMem, _zeroSign);
567  for (int i = 0; i < _dimZeroFunc; i++){
568  if (_zeroSign[i] != 0)
569  zeros << i << " ";
570  }
571  throw ModelicaSimulationError(EVENT_HANDLING, "Number of events of zero function(s) " + zeros.str() + "exceeded in time interval " + to_string(_abs) + " at time " + to_string(_tCurrent));
572  }
573 
574  // CVode has interpolated the states at time 'tCurrent'
575  _time_system->setTime(_tCurrent);
576 
577  // To get steep steps in the result file, two value points (P1 and P2) must be added
578  //
579  // Y | (P2) X...........
580  // | :
581  // | :
582  // |........X (P1)
583  // |---------------------------------->
584  // | ^ t
585  // _tCurrent
586 
587  // Write the values of (P1)
588  if (writeEventOutput)
589  {
590 
591  try
592  {
593  _continuous_system->evaluateAll(IContinuous::CONTINUOUS);
594  }
595  catch (std::exception& ex)
596  {
597 
598  // if a zero crossing was dected before the event iteration was called and evalutateAll throws and error
599  // for this time step the event iteration evaluates the system with corrected values.
600 
601  }
602 
603  writeToFile(0, _tCurrent, _h);
604  }
605 
606  _idid = CVodeGetRootInfo(_cvodeMem, _zeroSign);
607 
608  for (int i = 0; i < _dimZeroFunc; i++)
609  _events[i] = bool(_zeroSign[i]);
610 
611  if (_mixed_system->handleSystemEvents(_events))
612  {
613  // State variables were reinitialized, thus we have to give these values to the cvode-solver
614  // Take care about the memory regions, _z is the same like _CV_y
615  _continuous_system->getContinuousStates(_z);
616  }
617  }
618 
619  if ((_zeroFound || state_selection) && !isInterrupted())
620  {
621  // Write the values of (P2)
622  if (writeEventOutput)
623  {
624  // If we want to write the event-results, we should evaluate the whole system again
625  _continuous_system->evaluateAll(IContinuous::CONTINUOUS);
626  writeToFile(0, _tCurrent, _h);
627  }
628 
629  _idid = CVodeReInit(_cvodeMem, _tCurrent, _CV_y);
630  if (_idid < 0)
631  throw ModelicaSimulationError(SOLVER, "CVode::ReInit()");
632 
633  // Der Eventzeitpunkt kann auf der Endzeit liegen (Time-Events). In diesem Fall wird der Solver beendet, da CVode sonst eine interne Warnung schmeißt
634  if (_tCurrent == _tEnd)
635  _cv_rt = CV_TSTOP_RETURN;
636  if (_continuous_system->stepCompleted(_tCurrent))
637  _solverStatus = DONE;
638  }
639 
640  // ZÃ¤hler fÃ¼r die Anzahl der ausgegebenen Schritte erhÃ¶hen
641  ++_outStps;
642  _tLastSuccess = _tCurrent;
643 
644  if (_cv_rt == CV_TSTOP_RETURN)
645  {
646  _time_system->setTime(_tEnd);
647  //Solver has finished calculation - calculate the final values
648  _continuous_system->setContinuousStates(NV_DATA_S(_CV_y));
649  _continuous_system->evaluateAll(IContinuous::CONTINUOUS);
650  if (writeOutput)
651  writeToFile(0, _tEnd, _h);
652 
653  _accStps += _locStps;
654  _solverStatus = DONE;
655  }
656  }
