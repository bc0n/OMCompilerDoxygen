268 protected function elabXInStateOps "
269 Author: BTH
270  Transform ticksInState() and timeInState() operators to data-flow equations
271 "
272  input FlatSmSemantics inFlatSmSemantics;
273  input Option<DAE.ComponentRef> inEnclosingStateCrefOption "Cref of state that encloses the flat state machiene (NONE() if at top hierarchy)";
274  output FlatSmSemantics outFlatSmSemantics;
275 protected
276  Integer i;
277  Boolean found;
278  DAE.Exp c2, c3, c4, conditionNew, substTickExp, substTimeExp;
279  DAE.ComponentRef stateRef;
280  Transition t2;
281  list<Transition> tElab = {} "Elaborated transitions";
282  list<DAE.Exp> cElab = {} "Elaborated conditions";
283  list<DAE.Element> smeqsElab = {} "Elaborated smeqs";
284  // FLAT_SM_SEMANTICS
285  DAE.Ident ident;
286  array<DAE.Element> smComps "First element is the initial state";
287  list<Transition> t "List/Array of transition data sorted in priority";
288  list<DAE.Exp> c "Transition conditions sorted in priority";
289  list<DAE.Element> smvars "SMS veriables";
290  list<DAE.Element> smknowns "SMS constants/parameters";
291  list<DAE.Element> smeqs "SMS equations";
292  list<DAE.Element> pvars = {} "Propagation related variables";
293  list<DAE.Element> peqs = {} "Propagation equations";
294  Option<DAE.ComponentRef> enclosingStateOption "Cref to enclosing state if any"; // FIXME needed?
295  // TRANSITION
296  Integer from;
297  Integer to;
298  DAE.Exp condition;
299  Boolean immediate;
300  Boolean reset;
301  Boolean synchronize;
302  Integer priority;
303 algorithm
304  FLAT_SM_SEMANTICS(ident, smComps, t, c, smvars, smknowns, smeqs, pvars, peqs, enclosingStateOption) := inFlatSmSemantics;
305 
306  // We have some redundancy here (t[:].condition == c[:]) and thus need to update both
307  i := 0;
308  for tc in List.threadTuple(t,c) loop
309  i := i + 1;
310  (t2, c2) := tc;
311  TRANSITION(from, to, condition, immediate, reset, synchronize, priority) := t2;
312 
313  // Need to access decorations attached to 'from' state
314  DAE.SM_COMP(componentRef=stateRef) := arrayGet(smComps, from);
315 
316  // == Search whether condition contains a subexpression 'ticksInState()', if so, substitute them by 'smComps[from].$ticksInState' ==
317  substTickExp := DAE.CREF(qCref("$ticksInState", DAE.T_INTEGER_DEFAULT, {}, stateRef), DAE.T_INTEGER_DEFAULT);
318  (c3, (_, _, found)) := Expression.traverseExpTopDown(c2, traversingSubsXInState, ("ticksInState", substTickExp, false));
319  if found and isSome(inEnclosingStateCrefOption) then
320  // MLS 3.3 17.1: "can only be used in transition conditions of state machines not present in states of hierarchical state machines" violated
321  Error.addCompilerError("Found 'ticksInState()' within a state of an hierarchical state machine.");
322  fail();
323  end if;
324  // if a transition was updated we also need to update the semantic equation containing that transition's logic
325  smeqsElab := if found then List.map5(smeqs, smeqsSubsXInState, arrayGet(smComps, 1), i, listLength(t), substTickExp, "ticksInState") else smeqs;
326  smeqs := smeqsElab; // use updated smeqs
327 
328  // == Search whether condition contains a subexpression 'timeInState()', if so, substitute them by 'smComps[from].$timeInState' ==
329  substTimeExp := DAE.CREF(qCref("$timeInState", DAE.T_REAL_DEFAULT, {}, stateRef), DAE.T_REAL_DEFAULT);
330  (c4, (_, _, found)) := Expression.traverseExpTopDown(c2, traversingSubsXInState, ("timeInState", substTimeExp, false));
331  if found and isSome(inEnclosingStateCrefOption) then
332  // MLS 3.3 17.1: "can only be used in transition conditions of state machines not present in states of hierarchical state machines" violated
333  Error.addCompilerError("Found 'timeInState()' within a state of an hierarchical state machine.");
334  fail();
335  end if;
336  // if a transition was updated we also need to update the semantic equation containing that transition's logic
337  smeqsElab := if found then List.map5(smeqs, smeqsSubsXInState, arrayGet(smComps, 1), i, listLength(t), substTimeExp, "timeInState") else smeqs;
338  smeqs := smeqsElab; // use updated smeqs
339 
340  tElab := TRANSITION(from, to, c4, immediate, reset, synchronize, priority) :: tElab;
341  cElab := c4 :: cElab;
342  end for;
343 
344  outFlatSmSemantics := FLAT_SM_SEMANTICS(ident, smComps, listReverse(tElab), listReverse(cElab), smvars, smknowns, smeqsElab, pvars, peqs, enclosingStateOption);
345 end elabXInStateOps;
