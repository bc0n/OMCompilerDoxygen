84 public function loadClass
85 "This function takes a \'Path\' and the $OPENMODELICALIBRARY as a string
86  and tries to load the class from the path.
87  If the classname is qualified, the complete package is loaded.
88  E.g. load_class(Modelica.SIunits.Voltage) -> whole Modelica package loaded."
89  input Absyn.Path inPath;
90  input list<String> priorityList;
91  input String modelicaPath;
92  input Option<String> encoding;
93  input Boolean requireExactVersion = false;
94  input Boolean encrypted = false;
95  output Absyn.Program outProgram;
96 algorithm
97  outProgram := matchcontinue (inPath,priorityList,modelicaPath,encoding)
98  local
99  String gd,classname,mp,pack;
100  list<String> mps;
101  Absyn.Program p;
102  Absyn.Path rest;
103  /* Simple names: Just load the file if it can be found in $OPENMODELICALIBRARY */
104  case (Absyn.IDENT(name = classname),_,mp,_)
105  equation
106  gd = System.groupDelimiter();
107  mps = System.strtok(mp, gd);
108  p = loadClassFromMps(classname, priorityList, mps, encoding, requireExactVersion, encrypted);
109  checkOnLoadMessage(p);
110  then
111  p;
112  /* Qualified names: First check if it is defined in a file pack.mo */
113  case (Absyn.QUALIFIED(name = pack),_,mp,_)
114  equation
115  gd = System.groupDelimiter();
116  mps = System.strtok(mp, gd);
117  p = loadClassFromMps(pack, priorityList, mps, encoding, requireExactVersion, encrypted);
118  checkOnLoadMessage(p);
119  then
120  p;
121  /* failure */
122  else
123  equation
124  true = Flags.isSet(Flags.FAILTRACE);
125  Debug.trace("ClassLoader.loadClass failed\n");
126  then
127  fail();
128  end matchcontinue;
129 end loadClass;

131 protected function loadClassFromMps
132 "Loads a class or classes from a set of paths in OPENMODELICALIBRARY"
133  input String id;
134  input list<String> prios;
135  input list<String> mps;
136  input Option<String> encoding;
137  input Boolean requireExactVersion = false;
138  input Boolean encrypted = false;
139  output Absyn.Program outProgram;
140 protected
141  String mp, name, pwd, cmd, version, userLibraries;
142  Boolean isDir, impactOK;
143  Absyn.Class cl;
144 algorithm
145  try
146  (mp,name,isDir) := System.getLoadModelPath(id,prios,mps,requireExactVersion);
147  else
148  pwd := System.pwd();
149  userLibraries := Settings.getHomeDir(Config.getRunningTestsuite()) + "/.openmodelica/libraries/";
150  true := System.directoryExists(userLibraries);
151  true := listMember(userLibraries, mps);
152  System.cd(userLibraries);
153  version := match prios
154  case version::_ guard version <> "default" then version;
155  else "";
156  end match;
157  cmd := "impact install \"" + id + (if version<>"" then "#" + version else "") + "\"";
158  impactOK := 0==System.systemCall(cmd, "/dev/null");
159  System.cd(pwd);
160  if impactOK then
161  Error.addMessage(Error.NOTIFY_IMPACT_FOUND, {id, (if version <> "" then (" "+version) else ""), userLibraries});
162  (mp,name,isDir) := System.getLoadModelPath(id,prios,mps,true);
163  else
164  fail();
165  end if;
166  end try;
167  // print("System.getLoadModelPath: " + id + " {" + stringDelimitList(prios,",") + "} " + stringDelimitList(mps,",") + " => " + mp + " " + name + " " + boolString(isDir));
168  Config.setLanguageStandardFromMSL(name);
169  cl := loadClassFromMp(id, mp, name, isDir, encoding, encrypted);
170  outProgram := Absyn.PROGRAM({cl},Absyn.TOP());
171 end loadClassFromMps;

173 protected function loadClassFromMp
174  input String id "the actual class name";
175  input String path;
176  input String name;
177  input Boolean isDir;
178  input Option<String> optEncoding;
179  input Boolean encrypted = false;
180  output Absyn.Class outClass;
181 algorithm
182  outClass := match (id,path,name,isDir,optEncoding)
183  local
184  String pd,encoding,encodingfile;
185  Absyn.Class cl;
186  list<String> filenames;
187  LoadFileStrategy strategy;
188  Boolean lveStarted;
189  Option<Integer> lveInstance;
190 
191  case (_,_,_,false,_)
192  equation
193  pd = System.pathDelimiter();
194  /* Check for path/package.encoding; OpenModelica extension */
195  encodingfile = stringAppendList({path,pd,"package.encoding"});
196  encoding = System.trimChar(System.trimChar(if System.regularFileExists(encodingfile) then System.readFile(encodingfile) else Util.getOptionOrDefault(optEncoding,"UTF-8"),"\n")," ");
197  strategy = STRATEGY_ON_DEMAND(encoding);
198  cl = parsePackageFile(path + pd + name, strategy, false, Absyn.TOP(), id);
199  then
200  cl;
201 
202  case (_,_,_,true,_)
203  equation
204  /* Check for path/package.encoding; OpenModelica extension */
205  pd = System.pathDelimiter();
206  encodingfile = stringAppendList({path,pd,name,pd,"package.encoding"});
207  encoding = System.trimChar(System.trimChar(if System.regularFileExists(encodingfile) then System.readFile(encodingfile) else Util.getOptionOrDefault(optEncoding,"UTF-8"),"\n")," ");
208 
209  if encrypted then
210  (lveStarted, lveInstance) = Parser.startLibraryVendorExecutable(path + pd + name);
211  if not lveStarted then
212  fail();
213  end if;
214  end if;
215 
216  if (Config.getRunningTestsuite() or Config.noProc()==1) and not encrypted then
217  strategy = STRATEGY_ON_DEMAND(encoding);
218  else
219  filenames = getAllFilesFromDirectory(path + pd + name, encrypted);
220  // print("Files load in parallel:\n" + stringDelimitList(filenames, "\n") + "\n");
221  strategy = STRATEGY_HASHTABLE(Parser.parallelParseFiles(filenames, encoding, Config.noProc(), path + pd + name, lveInstance));
222  end if;
223  cl = loadCompletePackageFromMp(id, name, path, strategy, Absyn.TOP(), Error.getNumErrorMessages(), encrypted);
224  if (encrypted and lveStarted) then
225  Parser.stopLibraryVendorExecutable(lveInstance);
226  end if;
227  then
228  cl;
229  end match;
230 end loadClassFromMp;

250 protected function loadCompletePackageFromMp
251 "Loads a whole package from the ModelicaPaths defined in OPENMODELICALIBRARY"
252  input String id "actual class identifier";
253  input Absyn.Ident inIdent;
254  input String inString;
255  input LoadFileStrategy strategy;
256  input Absyn.Within inWithin;
257  input Integer numError;
258  input Boolean encrypted = false;
259  output Absyn.Class cl;
260 algorithm
261  cl := matchcontinue (id,inIdent,inString,inWithin)
262  local
263  String pd,mp_1,packagefile,orderfile,pack,mp,name,str;
264  Absyn.Within within_;
265  list<String> tv;
266  Boolean pp,fp,ep;
267  Absyn.Restriction r;
268  list<Absyn.NamedArg> ca;
269  list<Absyn.ClassPart> cp;
270  Option<String> cmt;
271  SourceInfo info;
272  Absyn.Path path;
273  Absyn.Within w2;
274  list<PackageOrder> reverseOrder;
275  list<Absyn.Annotation> ann;
276  case (_,pack,mp,within_)
277  equation
278  pd = System.pathDelimiter();
279  mp_1 = stringAppendList({mp,pd,pack});
280  packagefile = stringAppendList({mp_1,pd,if encrypted then "package.moc" else "package.mo"});
281  orderfile = stringAppendList({mp_1,pd,"package.order"});
282  if not System.regularFileExists(packagefile) then
283  Error.addInternalError("Expected file " + packagefile + " to exist", sourceInfo());
284  fail();
285  end if;
286  // print("Look for " + packagefile + "\n");
287  (cl as Absyn.CLASS(name,pp,fp,ep,r,Absyn.PARTS(tv,ca,cp,ann,cmt),info)) = parsePackageFile(packagefile, strategy, true, within_, id);
288  // print("Got " + packagefile + "\n");
289  reverseOrder = getPackageContentNames(cl, orderfile, mp_1, Error.getNumErrorMessages(), encrypted);
290  path = Absyn.joinWithinPath(within_,Absyn.IDENT(id));
291  w2 = Absyn.WITHIN(path);
292  cp = List.fold4(reverseOrder, loadCompletePackageFromMp2, mp_1, strategy, w2, encrypted, {});
293  then Absyn.CLASS(name,pp,fp,ep,r,Absyn.PARTS(tv,ca,cp,ann,cmt),info);
294  case (_,pack,mp,_)
295  equation
296  true = numError == Error.getNumErrorMessages();
297  Error.addInternalError("loadCompletePackageFromMp failed for unknown reason: mp=" + mp + " pack=" + pack, sourceInfo());
298  then fail();
299  end matchcontinue;
300 end loadCompletePackageFromMp;

323 protected function loadCompletePackageFromMp2
324 "Loads a whole package from the ModelicaPaths defined in OPENMODELICALIBRARY"
325  input PackageOrder po "mo-file or directory";
326  input String mp;
327  input LoadFileStrategy strategy;
328  input Absyn.Within w1 "With the parent added";
329  input Boolean encrypted = false;
330  input list<Absyn.ClassPart> acc;
331  output list<Absyn.ClassPart> cps;
332 algorithm
333  cps := match po
334  local
335  Absyn.ElementItem ei;
336  String pd,file,id;
337  Absyn.ClassPart cp;
338  Absyn.Class cl;
339  Boolean bDirectoryAndFileExists;
340 
341  case CLASSPART(cp)
342  equation
343  cps = mergeBefore(cp,acc);
344  then cps;
345 
346  case ELEMENT(ei,true)
347  equation
348  cps = mergeBefore(Absyn.PUBLIC({ei}),acc);
349  then cps;
350 
351  case ELEMENT(ei,false)
352  equation
353  cps = mergeBefore(Absyn.PROTECTED({ei}),acc);
354  then cps;
355 
356  case CLASSLOAD(id)
357  equation
358  pd = System.pathDelimiter();
359  file = mp + pd + id + (if encrypted then "/package.moc" else "/package.mo");
360  bDirectoryAndFileExists = System.directoryExists(mp + pd + id) and System.regularFileExists(file);
361  if bDirectoryAndFileExists then
362  cl = loadCompletePackageFromMp(id,id,mp,strategy,w1,Error.getNumErrorMessages(),encrypted);
363  ei = Absyn.makeClassElement(cl);
364  cps = mergeBefore(Absyn.PUBLIC({ei}),acc);
365  else
366  file = mp + pd + id + (if encrypted then ".moc" else ".mo");
367  if not System.regularFileExists(file) then
368  Error.addInternalError("Expected file " + file + " to exist", sourceInfo());
369  fail();
370  end if;
371  cl = parsePackageFile(file, strategy, false, w1, id);
372  ei = Absyn.makeClassElement(cl);
373  cps = mergeBefore(Absyn.PUBLIC({ei}),acc);
374  end if;
375  then cps;
376 
377  end match;
378 end loadCompletePackageFromMp2;

380 public function parsePackageFile
381  "Parses a file containing a single class that matches the within"
382  input String name;
383  input LoadFileStrategy strategy;
384  input Boolean expectPackage;
385  input Absyn.Within w1 "Expected within of the package";
386  input String pack "Expected name of the package";
387  output Absyn.Class cl;
388 protected
389  list<Absyn.Class> cs;
390  Absyn.Within w2;
391  list<String> classNames;
392  SourceInfo info;
393  String str,s1,s2,cname;
394  Absyn.ClassDef body;
395 algorithm
396  Absyn.PROGRAM(cs,w2) := getProgramFromStrategy(name, strategy);
397  classNames := List.map(cs, Absyn.getClassName);
398  str := stringDelimitList(classNames,", ");
399  if not listLength(cs)==1 then
400  Error.addSourceMessage(Error.LIBRARY_ONE_PACKAGE_PER_FILE, {str}, SOURCEINFO(name,true,0,0,0,0,0.0));
401  fail();
402  end if;
403  (cl as Absyn.CLASS(name=cname,body=body,info=info))::{} := cs;
404  if not stringEqual(cname,pack) then
405  Error.addSourceMessage(Error.LIBRARY_UNEXPECTED_NAME, {pack,cname}, info);
406  fail();
407  end if;
408  s1 := Absyn.withinString(w1);
409  s2 := Absyn.withinString(w2);
410  if not (Absyn.withinEqual(w1,w2) or Config.languageStandardAtMost(Config.LanguageStandard.'2.x')) then
411  Error.addSourceMessage(Error.LIBRARY_UNEXPECTED_WITHIN, {s1,s2}, info);
412  fail();
413  elseif expectPackage and not Absyn.isParts(body) then
414  Error.addSourceMessage(Error.LIBRARY_EXPECTED_PARTS, {pack}, info);
415  fail();
416  end if;
417 end parsePackageFile;

427 protected function getPackageContentNames
428  "Gets the names of packages to load before the package.mo, and the ones to load after"
429  input Absyn.Class cl;
430  input String filename;
431  input String mp;
432  input Integer numError;
433  input Boolean encrypted = false;
434  output list<PackageOrder> po "reverse";
435 algorithm
436  (po) := matchcontinue (cl,filename,mp,numError)
437  local
438  String contents, duplicatesStr, differencesStr, classFilename;
439  list<String> duplicates, namesToFind, mofiles, subdirs, differences, intersection;
440  list<Absyn.ClassPart> cp;
441  SourceInfo info;
442  list<PackageOrder> po1, po2;
443 
444  case (Absyn.CLASS(body=Absyn.PARTS(classParts=cp),info=info),_,_,_)
445  equation
446  if (System.regularFileExists(filename)) then
447  contents = System.readFile(filename);
448  namesToFind = System.strtok(contents, "\n");
449  namesToFind = List.removeOnTrue("",stringEqual,List.map(namesToFind,System.trimWhitespace));
450  duplicates = List.sortedDuplicates(List.sort(namesToFind,Util.strcmpBool),stringEq);
451  duplicatesStr = stringDelimitList(duplicates, ", ");
452  Error.assertionOrAddSourceMessage(listEmpty(duplicates),Error.PACKAGE_ORDER_DUPLICATES,{duplicatesStr},SOURCEINFO(filename,true,0,0,0,0,0.0));
453 
454  if encrypted then
455  // get all the .moc files in the directory!
456  mofiles = List.map(System.mocFiles(mp), Util.removeLast4Char);
457  else
458  // get all the .mo files in the directory!
459  mofiles = List.map(System.moFiles(mp), Util.removeLast3Char);
460  end if;
461  // get all the subdirs
462  subdirs = System.subDirectories(mp);
463  subdirs = List.filter2OnTrue(subdirs, existPackage, mp, encrypted);
464  // build a list
465  intersection = List.intersectionOnTrue(subdirs,mofiles,stringEq);
466  differencesStr = stringDelimitList(List.map1(intersection, getBothPackageAndFilename, mp), ", ");
467  Error.assertionOrAddSourceMessage(listEmpty(intersection),Error.PACKAGE_DUPLICATE_CHILDREN,{differencesStr},SOURCEINFO(filename,true,0,0,0,0,0.0));
468  mofiles = listAppend(subdirs,mofiles);
469  // check if all are present in the package.order
470  differences = List.setDifference(mofiles, namesToFind);
471  // issue a warning if not all are present
472  differencesStr = stringDelimitList(differences, "\n\t");
473  Error.assertionOrAddSourceMessage(listEmpty(differences),Error.PACKAGE_ORDER_FILE_NOT_COMPLETE,{differencesStr},SOURCEINFO(filename,true,0,0,0,0,0.0));
474  po1 = getPackageContentNamesinParts(namesToFind,cp,{});
475  List.map3_0(po1,checkPackageOrderFilesExist,mp,info,encrypted);
476 
477  po2 = List.map(differences, makeClassLoad);
478 
479  po = listAppend(po2, po1);
480  else // file not found
481  mofiles = List.map(System.moFiles(mp), Util.removeLast3Char) "Here .mo files in same directory as package.mo should be loaded as sub-packages";
482  subdirs = System.subDirectories(mp);
483  subdirs = List.filter2OnTrue(subdirs, existPackage, mp, encrypted);
484  mofiles = List.sort(listAppend(subdirs,mofiles), Util.strcmpBool);
485  // Look for duplicates
486  intersection = List.sortedDuplicates(mofiles,stringEq);
487  differencesStr = stringDelimitList(List.map1(intersection, getBothPackageAndFilename, mp), ", ");
488  Error.assertionOrAddSourceMessage(listEmpty(intersection),Error.PACKAGE_DUPLICATE_CHILDREN,{differencesStr},info);
489 
490  po = listAppend(List.map(cp, makeClassPart),List.map(mofiles, makeClassLoad));
491  end if;
492  then
493  po;
494 
495  case (Absyn.CLASS(info=info),_,_,_)
496  equation
497  true = numError == Error.getNumErrorMessages();
498  Error.addSourceMessage(Error.INTERNAL_ERROR,{"getPackageContentNames failed for unknown reason"},info);
499  then fail();
500 
501  end matchcontinue;
502 end getPackageContentNames;
