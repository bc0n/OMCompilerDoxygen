58 void zmq::xpub_t::xattach_pipe (pipe_t *pipe_, bool subscribe_to_all_)
59 {
60  zmq_assert (pipe_);
61  dist.attach (pipe_);
62 
63  // If subscribe_to_all_ is specified, the caller would like to subscribe
64  // to all data on this pipe, implicitly.
65  if (subscribe_to_all_)
66  subscriptions.add (NULL, 0, pipe_);
67 
68  // if welcome message exists, send a copy of it
69  if (welcome_msg.size () > 0)
70  {
71  msg_t copy;
72  copy.init ();
73  int rc = copy.copy (welcome_msg);
74  errno_assert (rc == 0);
75  bool ok = pipe_->write (&copy);
76  zmq_assert (ok);
77  pipe_->flush ();
78  }
79 
80  // The pipe is active when attached. Let's read the subscriptions from
81  // it, if any.
82  xread_activated (pipe_);
83 }

85 void zmq::xpub_t::xread_activated (pipe_t *pipe_)
86 {
87  // There are some subscriptions waiting. Let's process them.
88  msg_t sub;
89  while (pipe_->read (&sub)) {
90  // Apply the subscription to the trie
91  unsigned char *const data = (unsigned char *) sub.data ();
92  const size_t size = sub.size ();
93  metadata_t* metadata = sub.metadata();
94  if (size > 0 && (*data == 0 || *data == 1)) {
95  if (manual)
96  {
97  // Store manual subscription to use on termination
98  if (*data == 0)
99  manual_subscriptions.rm (data + 1, size - 1, pipe_);
100  else
101  manual_subscriptions.add (data + 1, size - 1, pipe_);
102 
103  pending_pipes.push_back (pipe_);
104  pending_data.push_back (blob_t (data, size));
105  if (metadata)
106  metadata->add_ref ();
107  pending_metadata.push_back (metadata);
108  pending_flags.push_back (0);
109  }
110  else
111  {
112  bool unique;
113  if (*data == 0)
114  unique = subscriptions.rm (data + 1, size - 1, pipe_);
115  else
116  unique = subscriptions.add (data + 1, size - 1, pipe_);
117 
118  // If the (un)subscription is not a duplicate store it so that it can be
119  // passed to the user on next recv call unless verbose mode is enabled
120  // which makes to pass always these messages.
121  if (options.type == ZMQ_XPUB && (unique || (*data == 1 && verbose_subs) ||
122  (*data == 0 && verbose_unsubs && verbose_subs))) {
123  pending_data.push_back (blob_t(data, size));
124  if (metadata)
125  metadata->add_ref ();
126  pending_metadata.push_back (metadata);
127  pending_flags.push_back (0);
128  }
129  }
130  }
131  else {
132  // Process user message coming upstream from xsub socket
133  pending_data.push_back (blob_t (data, size));
134  if (metadata)
135  metadata->add_ref ();
136  pending_metadata.push_back (metadata);
137  pending_flags.push_back (sub.flags ());
138  }
139  sub.close ();
140  }
141 }

148 int zmq::xpub_t::xsetsockopt (int option_, const void *optval_,
149  size_t optvallen_)
150 {
151  if (option_ == ZMQ_XPUB_VERBOSE
152  || option_ == ZMQ_XPUB_VERBOSER
153  || option_ == ZMQ_XPUB_NODROP
154  || option_ == ZMQ_XPUB_MANUAL) {
155  if (optvallen_ != sizeof (int) || *static_cast <const int*> (optval_) < 0) {
156  errno = EINVAL;
157  return -1;
158  }
159  if (option_ == ZMQ_XPUB_VERBOSE) {
160  verbose_subs = (*static_cast <const int*> (optval_) != 0);
161  verbose_unsubs = 0;
162  }
163  else
164  if (option_ == ZMQ_XPUB_VERBOSER) {
165  verbose_subs = (*static_cast <const int*> (optval_) != 0);
166  verbose_unsubs = verbose_subs;
167  }
168  else
169  if (option_ == ZMQ_XPUB_NODROP)
170  lossy = (*static_cast <const int*> (optval_) == 0);
171  else
172  if (option_ == ZMQ_XPUB_MANUAL)
173  manual = (*static_cast <const int*> (optval_) != 0);
174  }
175  else
176  if (option_ == ZMQ_SUBSCRIBE && manual) {
177  if (last_pipe != NULL)
178  subscriptions.add ((unsigned char *) optval_, optvallen_, last_pipe);
179  }
180  else
181  if (option_ == ZMQ_UNSUBSCRIBE && manual) {
182  if (last_pipe != NULL)
183  subscriptions.rm ((unsigned char *) optval_, optvallen_, last_pipe);
184  }
185  else
186  if (option_ == ZMQ_XPUB_WELCOME_MSG) {
187  welcome_msg.close ();
188 
189  if (optvallen_ > 0) {
190  int rc = welcome_msg.init_size (optvallen_);
191  errno_assert(rc == 0);
192 
193  unsigned char *data = (unsigned char*) welcome_msg.data ();
194  memcpy (data, optval_, optvallen_);
195  }
196  else
197  welcome_msg.init ();
198  }
199  else {
200  errno = EINVAL;
201  return -1;
202  }
203  return 0;
204 }

206 void zmq::xpub_t::xpipe_terminated (pipe_t *pipe_)
207 {
208  if (manual)
209  {
210  // Remove the pipe from the trie and send corresponding manual
211  // unsubscriptions upstream.
212  manual_subscriptions.rm (pipe_, send_unsubscription, this, false);
213  }
214  else
215  {
216  // Remove the pipe from the trie. If there are topics that nobody
217  // is interested in anymore, send corresponding unsubscriptions
218  // upstream.
219  subscriptions.rm (pipe_, send_unsubscription, this, !verbose_unsubs);
220  }
221 
222  dist.pipe_terminated (pipe_);
223 }

231 int zmq::xpub_t::xsend (msg_t *msg_)
232 {
233  bool msg_more = msg_->flags () & msg_t::more ? true : false;
234 
235  // For the first part of multi-part message, find the matching pipes.
236  if (!more) {
237  subscriptions.match ((unsigned char*) msg_->data (), msg_->size (),
238  mark_as_matching, this);
239  // If inverted matching is used, reverse the selection now
240  if (options.invert_matching) {
241  dist.reverse_match();
242  }
243  }
244 
245  int rc = -1; // Assume we fail
246  if (lossy || dist.check_hwm ()) {
247  if (dist.send_to_matching (msg_) == 0) {
248  // If we are at the end of multi-part message we can mark
249  // all the pipes as non-matching.
250  if (!msg_more)
251  dist.unmatch ();
252  more = msg_more;
253  rc = 0; // Yay, sent successfully
254  }
255  }
256  else
257  errno = EAGAIN;
258  return rc;
259 }

266 int zmq::xpub_t::xrecv (msg_t *msg_)
267 {
268  // If there is at least one
269  if (pending_data.empty ()) {
270  errno = EAGAIN;
271  return -1;
272  }
273 
274  // User is reading a message, set last_pipe and remove it from the deque
275  if (manual && !pending_pipes.empty ()) {
276  last_pipe = pending_pipes.front ();
277  pending_pipes.pop_front ();
278  }
279 
280  int rc = msg_->close ();
281  errno_assert (rc == 0);
282  rc = msg_->init_size (pending_data.front ().size ());
283  errno_assert (rc == 0);
284  memcpy (msg_->data (),
285  pending_data.front ().data (),
286  pending_data.front ().size ());
287 
288  // set metadata only if there is some
289  if (metadata_t* metadata = pending_metadata.front ()) {
290  msg_->set_metadata (metadata);
291  // Remove ref corresponding to vector placement
292  metadata->drop_ref();
293  }
294 
295  msg_->set_flags (pending_flags.front ());
296  pending_data.pop_front ();
297  pending_metadata.pop_front ();
298  pending_flags.pop_front ();
299  return 0;
300 }

307 void zmq::xpub_t::send_unsubscription (unsigned char *data_, size_t size_,
308  void *arg_)
309 {
310  xpub_t *self = (xpub_t*) arg_;
311 
312  if (self->options.type != ZMQ_PUB) {
313  // Place the unsubscription to the queue of pending (un)subscriptions
314  // to be retrieved by the user later on.
315  blob_t unsub (size_ + 1, 0);
316  unsub [0] = 0;
317  if (size_ > 0)
318  memcpy (&unsub [1], data_, size_);
319  self->pending_data.push_back (unsub);
320  self->pending_metadata.push_back (NULL);
321  self->pending_flags.push_back (0);
322 
323  if (self->manual) {
324  self->last_pipe = NULL;
325  self->pending_pipes.push_back (NULL);
326  }
327  }
