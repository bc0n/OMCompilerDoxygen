66 void zmq::poller_base_t::cancel_timer (i_poll_events *sink_, int id_)
67 {
68  // Complexity of this operation is O(n). We assume it is rarely used.
69  for (timers_t::iterator it = timers.begin (); it != timers.end (); ++it)
70  if (it->second.sink == sink_ && it->second.id == id_) {
71  timers.erase (it);
72  return;
73  }
74 
75  // Timer not found.
76  zmq_assert (false);
77 }

79 uint64_t zmq::poller_base_t::execute_timers ()
80 {
81  // Fast track.
82  if (timers.empty ())
83  return 0;
84 
85  // Get the current time.
86  uint64_t current = clock.now_ms ();
87 
88  // Execute the timers that are already due.
89  timers_t::iterator it = timers.begin ();
90  while (it != timers.end ()) {
91 
92  // If we have to wait to execute the item, same will be true about
93  // all the following items (multimap is sorted). Thus we can stop
94  // checking the subsequent timers and return the time to wait for
95  // the next timer (at least 1ms).
96  if (it->first > current)
97  return it->first - current;
98 
99  // Trigger the timer.
100  it->second.sink->timer_event (it->second.id);
101 
102  // Remove it from the list of active timers.
103  timers_t::iterator o = it;
104  ++it;
105  timers.erase (o);
106  }
107 
108  // There are no more timers.
109  return 0;
110 }
