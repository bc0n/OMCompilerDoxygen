73 Bool evalfG(Index n, double * vopt, Bool new_x, int m, Number *g, void * useData){
74  OptData *optData = (OptData*)useData;
75 
76  const int nx = optData->dim.nx;
77  const int nv = optData->dim.nv;
78  const int nc = optData->dim.nc;
79  const int ncf = optData->dim.ncf;
80  const int nsi = optData->dim.nsi;
81  const int np = optData->dim.np;
82  const int index_con = optData->dim.index_con;
83  const int index_conf = optData->dim.index_conf;
84 
85  modelica_real ***v;
86  long double a[5][5];
87  long double *sdt;
88  double * vv[np+1];
89  int i, j, k, shift;
90 
91 
92  if(new_x){
93  optData2ModelData(optData, vopt, optData->index);
94  }
95 
96  v = optData->v;
97  memcpy(a ,optData->rk.a, sizeof(optData->rk.a));
98 
99  vv[0] = optData->sv0;
100  vv[1] = vopt;
101  for(j = 1; j < np; ++j){
102  vv[j + 1] = vv[j] + nv;
103  }
104 
105  if(np == 3){
106  for(i = 0, shift = 0; i <nsi; ++i){
107  sdt = optData->bounds.scaldt[i];
108  /*1*/
109  for(k=0; k< nx; ++k){
110  g[shift++] = (a[0][0]*vv[0][k] + a[0][3]*vv[3][k] + sdt[k]*v[i][0][k+nx])
111  -(a[0][1]*vv[1][k] + a[0][2]*vv[2][k]);
112  }
113  memcpy(g + shift, &v[i][0][index_con], nc*sizeof(double));
114  shift += nc;
115 
116  /*2*/
117  for(k=0; k< nx; ++k){
118  g[shift++] = (a[1][1]*vv[1][k] + sdt[k]*v[i][1][k+nx])
119  - (a[1][0]*vv[0][k] + a[1][2]*vv[2][k] + a[1][3]*vv[3][k]);
120  }
121  memcpy(g + shift, &v[i][1][index_con], nc*sizeof(double));
122  shift +=nc;
123 
124  /*3*/
125  for(k=0; k< nx; ++k){
126  g[shift++] = (a[2][0]*vv[0][k] + a[2][2]*vv[2][k] + sdt[k]*v[i][2][nx+k])
127  -(a[2][1]*vv[1][k] + a[2][3]*vv[3][k]);
128  }
129  memcpy(g + shift, &v[i][2][index_con], nc*sizeof(double));
130  shift +=nc;
131  vv[0] = vv[np];
132  for(j = 0; j < np; ++j)
133  vv[j + 1] = vv[j] + nv;
134  }
135  /*terminal constraint(s)*/
136  memcpy(g + shift, &v[nsi-1][2][index_conf], ncf*sizeof(double));
137 
138  }else if(np == 1){
139  for(i = 0, shift = 0; i <nsi; ++i){
140  sdt = optData->bounds.scaldt[i];
141  for(k = 0; k < nx; ++k)
142  g[shift++] = vv[0][k] + (sdt[k]*v[i][0][k+nx] - vv[1][k]);
143 
144  memcpy(g + shift, &v[i][0][index_con], nc*sizeof(double));
145  shift += nc;
146  vv[0] = vv[np];
147  for(j = 0; j < np; ++j)
148  vv[j + 1] = vv[j] + nv;
149  }
150  /*terminal constraint(s)*/
151  memcpy(g + m - ncf, &v[nsi-1][0][index_conf], ncf*sizeof(double));
152  }
153  if(ACTIVE_STREAM(LOG_IPOPT_ERROR)){
154  const int nJ = optData->dim.nJ;
155  printMaxError(g, m, nx, nJ, optData->time.t, np ,nsi ,optData->data, optData);
156  }
157  return TRUE;
158 }
