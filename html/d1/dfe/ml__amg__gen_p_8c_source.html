32 int ML_Gen_MGHierarchy_UsingAMG(ML *ml, int start,
33  int increment_or_decrement, ML_AMG *amg)
34 {
35  int i, j, level, idata=0, nrows, blksize;
36  double dnnz = 1.;
37  ML_AMG *ml_amg;
38 #ifdef ML_TIMING
39  double t0;
40 #endif
41 
42  /* ----------------------------------------------------------------- */
43  /* if user does not provide a ML_AMG object, create a default */
44  /* ----------------------------------------------------------------- */
45 
46  if ( amg == NULL ) ML_AMG_Create( &ml_amg );
47  else ml_amg = amg;
48  ML_AMG_Set_MaxLevels( ml_amg, ml->ML_num_levels);
49  ML_AMG_Set_StartLevel( ml_amg, start );
50 
51  /* ----------------------------------------------------------------- */
52  /* if system AMG is requested, set appropriate parameters */
53  /* ----------------------------------------------------------------- */
54 
55  blksize = ml_amg->num_PDE_eqns;
56  if ( blksize > 1 && ml_amg->amg_scheme == ML_AMG_SYSTEM_UNKNOWN )
57  {
58  nrows = ml->Amat[start].outvec_leng;
59  if ( nrows % blksize != 0 )
60  {
61  printf("Gen_AMG ERROR : local nrows not divisible by blksize\n");
62  exit(1);
63  }
64  ML_memory_alloc((void **) &(ml_amg->blk_info), nrows*sizeof(int), "AM1");
65  for ( i = 0; i < nrows; i+= blksize )
66  for ( j = 0; j < blksize; j++ ) ml_amg->blk_info[i+j] = j;
67  }
68 
69  /* ----------------------------------------------------------------- */
70  /* create multilevel hierarchy */
71  /* ----------------------------------------------------------------- */
72 
73  idata = ML_gmax_int(idata, ml->comm);
74  if ( ml->comm->ML_mypid == 0 && ml_amg->print_flag < ML_Get_PrintLevel())
75  ML_AMG_Print(ml_amg);
76 #ifdef ML_TIMING
77  t0 = GetClock();
78 #endif
79  idata = ML_gmax_int(idata, ml->comm);
80 
81  if (increment_or_decrement == ML_INCREASING)
82  {
83  level = ML_AMG_Gen_MGHierarchy(ml, start, ML_AMG_Increment_Level,
84  ML_AMG_Gen_Prolongator, NULL, ml_amg);
85  }
86  else if (increment_or_decrement == ML_DECREASING)
87  {
88  level = ML_AMG_Gen_MGHierarchy(ml, start, ML_AMG_Decrement_Level,
89  ML_AMG_Gen_Prolongator, NULL, ml_amg);
90  }
91  else
92  {
93  if ( ml->comm->ML_mypid == 0 )
94  printf("ML_Gen_MGHierarchy_UsingAMG : unknown inc_or_dec choice\n");
95  exit(1);
96  }
97 #ifdef ML_TIMING
98  t0 = GetClock() - t0;
99  if ( ml->comm->ML_mypid == 0 && ML_Get_PrintLevel() > 10 )
100  printf("AMG total setup time = %e\n", t0);
101 #endif
102 
103  /* ----------------------------------------------------------------- */
104  /* compute operator complexity */
105  /* ----------------------------------------------------------------- */
106 
107  if (increment_or_decrement == ML_INCREASING)
108  dnnz = (double) ml->Amat[level-start-1].N_nonzeros;
109  else if (increment_or_decrement == ML_DECREASING)
110  dnnz = (double) ml->Amat[start+1-level].N_nonzeros;
111  dnnz = ML_gsum_double( dnnz, ml->comm );
112  ml_amg->operator_complexity += dnnz;
113 
114  idata = ML_gmax_int(idata, ml->comm);
115  if ( ml->comm->ML_mypid == 0 && ml_amg->print_flag < ML_Get_PrintLevel())
116  ML_AMG_Print_Complexity(ml_amg);
117  idata = ML_gmax_int(idata, ml->comm);
118 
119  if ( amg == NULL ) ML_AMG_Destroy( &ml_amg );
120  return(level);
121 }

188 int ML_AMG_Gen_Prolongator(ML *ml,int level, int clevel, void *data,
189  ML_AMG *amg)
190 {
191  int Ncoarse, Nfine, gNfine, gNcoarse;
192  ML_Operator *Amat, *Pmatrix, *AMGIdentity;
193 
194 #ifdef ML_TIMING
195  double t0;
196  t0 = GetClock();
197 #endif
198 
199  Amat = (ML_Operator *) data;
200  Nfine = Amat->outvec_leng;
201  gNfine = ML_Comm_GsumInt( ml->comm, Nfine);
202  ML_AMG_Set_CurrentLevel( amg, level );
203  if ( ml->comm->ML_mypid == 0 && amg->print_flag < ML_Get_PrintLevel())
204  printf("+++++++++++++++++++++++++++++++++++++++++++++++\n");
205  Pmatrix = ML_Operator_Create(ml->comm);
206  Ncoarse = ML_AMG_Coarsen(amg, Amat, &Pmatrix, ml->comm);
207  gNcoarse = ML_Comm_GsumInt( ml->comm, Ncoarse);
208  if ( ml->comm->ML_mypid == 0 && amg->print_flag < ML_Get_PrintLevel())
209  printf("AMG at level %2d = %d\n", level, gNcoarse);
210  if ( gNcoarse == 0 || (1.0*gNfine)/(1.0*gNcoarse+0.1) < 1.05 )
211  {
212  ML_Operator_Destroy(&Pmatrix);
213  return -1;
214  }
215  AMGIdentity = ML_Operator_Create(ml->comm);
216  ML_Operator_Set_ApplyFuncData(AMGIdentity, Amat->invec_leng,
217  Amat->outvec_leng, (void*) Amat,
218  Amat->matvec->Nrows, NULL, 0);
219  ML_Operator_Set_Getrow(AMGIdentity, Amat->getrow->Nrows,
220  ML_AMG_Identity_Getrows);
221  ML_CommInfoOP_Clone(&(AMGIdentity->getrow->pre_comm),Amat->getrow->pre_comm);
222  ML_2matmult(AMGIdentity, Pmatrix, &(ml->Pmat[clevel]), ML_CSR_MATRIX );
223  ML_Operator_Destroy(&AMGIdentity);
224  ML_Operator_Destroy(&Pmatrix);
225  ML_Operator_Set_1Levels(&(ml->Pmat[clevel]),
226  &(ml->SingleLevel[clevel]), &(ml->SingleLevel[level]));
227 
228 #ifdef ML_TIMING
229  ml->Pmat[clevel].build_time = GetClock() - t0;
230  ml->timing->total_build_time += ml->Pmat[clevel].build_time;
231 #endif
232  return 0;
233 }

240 int ML_AMG_Increment_Level(ML *ml, int current_level, ML_Operator *Amat,
241  ML_AMG *amg)
242 {
243  int total_size, temp;
244 
245  if (current_level == ml->ML_num_levels-1) return(-1);
246 
247  total_size = Amat->invec_leng;
248  ML_gsum_scalar_int(&total_size, &temp,ml->comm);
249  if ( total_size <= amg->max_coarse_size ) return(-1);
250 
251  return(current_level+1);
252 }
