48 zmq::fd_t zmq::open_socket (int domain_, int type_, int protocol_)
49 {
50  int rc;
51 
52  // Setting this option result in sane behaviour when exec() functions
53  // are used. Old sockets are closed and don't block TCP ports etc.
54 #if defined ZMQ_HAVE_SOCK_CLOEXEC
55  type_ |= SOCK_CLOEXEC;
56 #endif
57 
58  fd_t s = socket (domain_, type_, protocol_);
59 #ifdef ZMQ_HAVE_WINDOWS
60  if (s == INVALID_SOCKET)
61  return INVALID_SOCKET;
62 #else
63  if (s == -1)
64  return -1;
65 #endif
66 
67  // If there's no SOCK_CLOEXEC, let's try the second best option. Note that
68  // race condition can cause socket not to be closed (if fork happens
69  // between socket creation and this point).
70 #if !defined ZMQ_HAVE_SOCK_CLOEXEC && defined FD_CLOEXEC
71  rc = fcntl (s, F_SETFD, FD_CLOEXEC);
72  errno_assert (rc != -1);
73 #endif
74 
75  // On Windows, preventing sockets to be inherited by child processes.
76 #if defined ZMQ_HAVE_WINDOWS && defined HANDLE_FLAG_INHERIT
77  BOOL brc = SetHandleInformation ((HANDLE) s, HANDLE_FLAG_INHERIT, 0);
78  win_assert (brc);
79 #endif
80 
81  // Socket is not yet connected so EINVAL is not a valid networking error
82  rc = zmq::set_nosigpipe (s);
83  errno_assert (rc == 0);
84 
85  return s;
86 }

88 void zmq::unblock_socket (fd_t s_)
89 {
90 #if defined ZMQ_HAVE_WINDOWS
91  u_long nonblock = 1;
92  int rc = ioctlsocket (s_, FIONBIO, &nonblock);
93  wsa_assert (rc != SOCKET_ERROR);
94 #elif defined ZMQ_HAVE_OPENVMS
95  int nonblock = 1;
96  int rc = ioctl (s_, FIONBIO, &nonblock);
97  errno_assert (rc != -1);
98 #else
99  int flags = fcntl (s_, F_GETFL, 0);
100  if (flags == -1)
101  flags = 0;
102  int rc = fcntl (s_, F_SETFL, flags | O_NONBLOCK);
103  errno_assert (rc != -1);
104 #endif
105 }

127 int zmq::get_peer_ip_address (fd_t sockfd_, std::string &ip_addr_)
128 {
129  int rc;
130  struct sockaddr_storage ss;
131 
132 #if defined ZMQ_HAVE_HPUX || defined ZMQ_HAVE_WINDOWS
133  int addrlen = static_cast <int> (sizeof ss);
134 #else
135  socklen_t addrlen = sizeof ss;
136 #endif
137  rc = getpeername (sockfd_, (struct sockaddr*) &ss, &addrlen);
138 #ifdef ZMQ_HAVE_WINDOWS
139  if (rc == SOCKET_ERROR) {
140  const int last_error = WSAGetLastError();
141  wsa_assert (last_error != WSANOTINITIALISED &&
142  last_error != WSAEFAULT &&
143  last_error != WSAEINPROGRESS &&
144  last_error != WSAENOTSOCK);
145  return 0;
146  }
147 #else
148  if (rc == -1) {
149  errno_assert (errno != EBADF &&
150  errno != EFAULT &&
151  errno != ENOTSOCK);
152  return 0;
153  }
154 #endif
155 
156  char host [NI_MAXHOST];
157  rc = getnameinfo ((struct sockaddr*) &ss, addrlen, host, sizeof host,
158  NULL, 0, NI_NUMERICHOST);
159  if (rc != 0)
160  return 0;
161 
162  ip_addr_ = host;
163 
164  union {
165  struct sockaddr sa;
166  struct sockaddr_storage sa_stor;
167  } u;
168 
169  u.sa_stor = ss;
170  return (int) u.sa.sa_family;
171 }

201 int zmq::set_nosigpipe (fd_t s_)
202 {
203 #ifdef SO_NOSIGPIPE
204  // Make sure that SIGPIPE signal is not generated when writing to a
205  // connection that was already closed by the peer.
206  // As per POSIX spec, EINVAL will be returned if the socket was valid but
207  // the connection has been reset by the peer. Return an error so that the
208  // socket can be closed and the connection retried if necessary.
209  int set = 1;
210  int rc = setsockopt (s_, SOL_SOCKET, SO_NOSIGPIPE, &set, sizeof (int));
211  if (rc != 0 && errno == EINVAL)
212  return -1;
213  errno_assert (rc == 0);
214 #else
215  LIBZMQ_UNUSED (s_);
216 #endif
217 
218  return 0;
219 }
