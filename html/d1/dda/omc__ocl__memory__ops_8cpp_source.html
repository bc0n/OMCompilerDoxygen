125 void ocl_create_execution_memory_buffer(device_buffer* d_buff){
126 
127  cl_int err;
128  cl_ulong mem;
129  cl_ulong mem2;
130  cl_mem tmp;
131  cl_ulong size;
132 
133  clGetDeviceInfo(ocl_device, CL_DEVICE_GLOBAL_MEM_SIZE, sizeof(cl_ulong), &mem, NULL);
134  clGetDeviceInfo(ocl_device, CL_DEVICE_MAX_MEM_ALLOC_SIZE, sizeof(cl_ulong), &mem2, NULL);
135  size = mem/OCL_BUFFER_SIZE_FRACTION;
136  size = (size > mem2 ? mem2 : size);
137  size = 30 * 1024 * 1024 * sizeof(modelica_integer);
138  //printf("Allocating : %llu MB for execution memory buffer \n", size/1024/1024);
139  tmp = clCreateBuffer(device_context, CL_MEM_READ_WRITE,
140  size, NULL, &err);
141 
142  ocl_error_check(OCL_CREATE_BUFFER, err);
143  if(err){
144  printf("Error: Allocating execution memory buffer");
145  exit(1);
146  }
147 
148  d_buff->buffer = tmp;
149  d_buff->size = size;
150 }
