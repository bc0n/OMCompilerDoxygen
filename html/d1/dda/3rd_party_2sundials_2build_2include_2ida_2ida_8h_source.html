100 /*
101  * ----------------------------------------------------------------
102  * Type : IDAResFn
103  * ----------------------------------------------------------------
104  * The F function which defines the DAE system F(t,y,y')=0
105  * must have type IDAResFn.
106  * Symbols are as follows:
107  * t <-> t y <-> yy
108  * y' <-> yp F <-> rr
109  * A IDAResFn takes as input the independent variable value t,
110  * the dependent variable vector yy, and the derivative (with
111  * respect to t) of the yy vector, yp. It stores the result of
112  * F(t,y,y') in the vector rr. The yy, yp, and rr arguments are of
113  * type N_Vector. The user_data parameter is the pointer user_data
114  * passed by the user to the IDASetUserData routine. This user-supplied
115  * pointer is passed to the user's res function every time it is called,
116  * to provide access in res to user data.
117  *
118  * A IDAResFn res should return a value of 0 if successful, a positive
119  * value if a recoverable error occured (e.g. yy has an illegal value),
120  * or a negative value if a nonrecoverable error occured. In the latter
121  * case, the program halts. If a recoverable error occured, the integrator
122  * will attempt to correct and retry.
123  * ----------------------------------------------------------------
124  */
125 
126 typedef int (*IDAResFn)(realtype tt, N_Vector yy, N_Vector yp,
127  N_Vector rr, void *user_data);
128 
129 /*
130  * -----------------------------------------------------------------
131  * Type : IDARootFn
132  * -----------------------------------------------------------------
133  * A function g, which defines a set of functions g_i(t,y,y') whose
134  * roots are sought during the integration, must have type IDARootFn.
135  * The function g takes as input the independent variable value t,
136  * the dependent variable vector y, and its t-derivative yp (= y').
137  * It stores the nrtfn values g_i(t,y,y') in the realtype array gout.
138  * (Allocation of memory for gout is handled within IDA.)
139  * The user_data parameter is the same as that passed by the user
140  * to the IDASetUserData routine. This user-supplied pointer is
141  * passed to the user's g function every time it is called.
142  *
143  * An IDARootFn should return 0 if successful or a non-zero value
144  * if an error occured (in which case the integration will be halted).
145  * -----------------------------------------------------------------
146  */
147 
148 typedef int (*IDARootFn)(realtype t, N_Vector y, N_Vector yp,
149  realtype *gout, void *user_data);

151 /*
152  * -----------------------------------------------------------------
153  * Type : IDAEwtFn
154  * -----------------------------------------------------------------
155  * A function e, which sets the error weight vector ewt, must have
156  * type IDAEwtFn.
157  * The function e takes as input the current dependent variable y.
158  * It must set the vector of error weights used in the WRMS norm:
159  *
160  * ||y||_WRMS = sqrt [ 1/N * sum ( ewt_i * y_i)^2 ]
161  *
162  * Typically, the vector ewt has components:
163  *
164  * ewt_i = 1 / (reltol * |y_i| + abstol_i)
165  *
166  * The user_data parameter is the same as that passed by the user
167  * to the IDASetUserData routine. This user-supplied pointer is
168  * passed to the user's e function every time it is called.
169  * An IDAEwtFn e must return 0 if the error weight vector has been
170  * successfuly set and a non-zero value otherwise.
171  * -----------------------------------------------------------------
172  */
173 
174 typedef int (*IDAEwtFn)(N_Vector y, N_Vector ewt, void *user_data);

390 /*
391  * ----------------------------------------------------------------
392  * Function : IDAReInit
393  * ----------------------------------------------------------------
394  * IDAReInit re-initializes IDA for the solution of a problem,
395  * where a prior call to IDAInit has been made.
396  * IDAReInit performs the same input checking and initializations
397  * that IDAInit does.
398  * But it does no memory allocation, assuming that the existing
399  * internal memory is sufficient for the new problem.
400  *
401  * The use of IDAReInit requires that the maximum method order,
402  * maxord, is no larger for the new problem than for the problem
403  * specified in the last call to IDAInit. This condition is
404  * automatically fulfilled if the default value for maxord is
405  * specified.
406  *
407  * Following the call to IDAReInit, a call to the linear solver
408  * specification routine is necessary if a different linear solver
409  * is chosen, but may not be otherwise. If the same linear solver
410  * is chosen, and there are no changes in its input parameters,
411  * then no call to that routine is needed.
412  *
413  * The first argument to IDAReInit is:
414  *
415  * ida_mem = pointer to IDA memory returned by IDACreate.
416  *
417  * All the remaining arguments to IDAReInit have names and
418  * meanings identical to those of IDAInit.
419  *
420  * The return value of IDAReInit is equal to SUCCESS = 0 if there
421  * were no errors; otherwise it is a negative int equal to:
422  * IDA_MEM_NULL indicating ida_mem was NULL, or
423  * IDA_NO_MALLOC indicating that ida_mem was not allocated.
424  * IDA_ILL_INPUT indicating an input argument was illegal
425  * (including an attempt to increase maxord).
426  * In case of an error return, an error message is also printed.
427  * ----------------------------------------------------------------
428  */
429

433 /*
434  * -----------------------------------------------------------------
435  * Functions : IDASStolerances
436  * IDASVtolerances
437  * IDAWFtolerances
438  * -----------------------------------------------------------------
439  *
440  * These functions specify the integration tolerances. One of them
441  * MUST be called before the first call to IDA.
442  *
443  * IDASStolerances specifies scalar relative and absolute tolerances.
444  * IDASVtolerances specifies scalar relative tolerance and a vector
445  * absolute tolerance (a potentially different absolute tolerance
446  * for each vector component).
447  * IDAWFtolerances specifies a user-provides function (of type IDAEwtFn)
448  * which will be called to set the error weight vector.
449  *
450  * The tolerances reltol and abstol define a vector of error weights,
451  * ewt, with components
452  * ewt[i] = 1/(reltol*abs(y[i]) + abstol) (in the SS case), or
453  * ewt[i] = 1/(reltol*abs(y[i]) + abstol[i]) (in the SV case).
454  * This vector is used in all error and convergence tests, which
455  * use a weighted RMS norm on all error-like vectors v:
456  * WRMSnorm(v) = sqrt( (1/N) sum(i=1..N) (v[i]*ewt[i])^2 ),
457  * where N is the problem dimension.
458  *
459  * The return value of these functions is equal to IDA_SUCCESS = 0 if
460  * there were no errors; otherwise it is a negative int equal to:
461  * IDa_MEM_NULL indicating ida_mem was NULL (i.e.,
462  * IDACreate has not been called).
463  * IDA_NO_MALLOC indicating that ida_mem has not been
464  * allocated (i.e., IDAInit has not been
465  * called).
466  * IDA_ILL_INPUT indicating an input argument was illegal
467  * (e.g. a negative tolerance)
468  * In case of an error return, an error message is also printed.
469  * -----------------------------------------------------------------
470  */
471 
472 SUNDIALS_EXPORT int IDASStolerances(void *ida_mem, realtype reltol, realtype abstol);

570 /*
571  * ----------------------------------------------------------------
572  * Function : IDACalcIC
573  * ----------------------------------------------------------------
574  * IDACalcIC calculates corrected initial conditions for the DAE
575  * system for a class of index-one problems of semi-implicit form.
576  * It uses Newton iteration combined with a Linesearch algorithm.
577  * Calling IDACalcIC is optional. It is only necessary when the
578  * initial conditions do not solve the given system. I.e., if
579  * y0 and yp0 are known to satisfy F(t0, y0, yp0) = 0, then
580  * a call to IDACalcIC is NOT necessary (for index-one problems).
581  *
582  * A call to IDACalcIC must be preceded by a successful call to
583  * IDAInit or IDAReInit for the given DAE problem, and by a
584  * successful call to the linear system solver specification
585  * routine.
586  *
587  * The call to IDACalcIC should precede the call(s) to IDASolve
588  * for the given problem.
589  *
590  * The arguments to IDACalcIC are as follows:
591  *
592  * ida_mem is the pointer to IDA memory returned by IDACreate.
593  *
594  * icopt is the option of IDACalcIC to be used.
595  * icopt = IDA_YA_YDP_INIT directs IDACalcIC to compute
596  * the algebraic components of y and differential
597  * components of y', given the differential
598  * components of y. This option requires that the
599  * N_Vector id was set through a call to IDASetId
600  * specifying the differential and algebraic
601  * components.
602  * icopt = IDA_Y_INIT directs IDACalcIC to compute all
603  * components of y, given y'. id is not required.
604  *
605  * tout1 is the first value of t at which a soluton will be
606  * requested (from IDASolve). (This is needed here to
607  * determine the direction of integration and rough scale
608  * in the independent variable t.)
609  *
610  *
611  * IDACalcIC returns an int flag. Its symbolic values and their
612  * meanings are as follows. (The numerical return values are set
613  * above in this file.) All unsuccessful returns give a negative
614  * return value. If IFACalcIC failed, y0 and yp0 contain
615  * (possibly) altered values, computed during the attempt.

664 /*
665  * ----------------------------------------------------------------
666  * Function : IDASolve
667  * ----------------------------------------------------------------
668  * IDASolve integrates the DAE over an interval in t, the
669  * independent variable. If itask is IDA_NORMAL, then the solver
670  * integrates from its current internal t value to a point at or
671  * beyond tout, then interpolates to t = tout and returns y(tret)
672  * in the user-allocated vector yret. In general, tret = tout.
673  * If itask is IDA_ONE_STEP, then the solver takes one internal
674  * step of the independent variable and returns in yret the value
675  * of y at the new internal independent variable value. In this
676  * case, tout is used only during the first call to IDASolve to
677  * determine the direction of integration and the rough scale of
678  * the problem. If tstop is enabled (through a call to IDASetStopTime),
679  * then IDASolve returns the solution at tstop. Once the integrator
680  * returns at a tstop time, any future testing for tstop is disabled
681  * (and can be reenabled only though a new call to IDASetStopTime).
682  * The time reached by the solver is placed in (*tret). The
683  * user is responsible for allocating the memory for this value.
684  *
685  * ida_mem is the pointer (void) to IDA memory returned by
686  * IDACreate.
687  *
688  * tout is the next independent variable value at which a
689  * computed solution is desired.
690  *
691  * tret is a pointer to a real location. IDASolve sets (*tret)
692  * to the actual t value reached, corresponding to the
693  * solution vector yret. In IDA_NORMAL mode, with no
694  * errors and no roots found, (*tret) = tout.
695  *
696  * yret is the computed solution vector. With no errors,
697  * yret = y(tret).
698  *
699  * ypret is the derivative of the computed solution at t = tret.
700  *
701  * Note: yret and ypret may be the same N_Vectors as y0 and yp0
702  * in the call to IDAInit or IDAReInit.
703  *
704  * itask is IDA_NORMAL or IDA_ONE_STEP. These two modes are described above.
705  *
706  *
707  * The return values for IDASolve are described below.
708  * (The numerical return values are defined above in this file.)
709  * All unsuccessful returns give a negative return value.
710  *
711  * IDA_SUCCESS
712  * IDASolve succeeded and no roots were found.
713  *
714  * IDA_ROOT_RETURN: IDASolve succeeded, and found one or more roots.
715  * If nrtfn > 1, call IDAGetRootInfo to see which g_i were found
716  * to have a root at (*tret).
717  *
718  * IDA_TSTOP_RETURN:
719  * IDASolve returns computed results for the independent variable
720  * value tstop. That is, tstop was reached.
721  *
722  * IDA_MEM_NULL:
723  * The IDA_mem argument was NULL.
724  *
725  * IDA_ILL_INPUT:
726  * One of the inputs to IDASolve is illegal. This includes the
727  * situation when a component of the error weight vectors
728  * becomes < 0 during internal stepping. It also includes the
729  * situation where a root of one of the root functions was found
730  * both at t0 and very near t0. The ILL_INPUT flag
731  * will also be returned if the linear solver function IDA---
732  * (called by the user after calling IDACreate) failed to set one
733  * of the linear solver-related fields in ida_mem or if the linear
734  * solver's init routine failed. In any case, the user should see
735  * the printed error message for more details.
736  *
737  * IDA_TOO_MUCH_WORK:
738  * The solver took mxstep internal steps but could not reach tout.
739  * The default value for mxstep is MXSTEP_DEFAULT = 500.
740  *
741  * IDA_TOO_MUCH_ACC:
742  * The solver could not satisfy the accuracy demanded by the user
743  * for some internal step.
744  *
745  * IDA_ERR_FAIL:
746  * Error test failures occurred too many times (=MXETF = 10) during
747  * one internal step.
748  *
749  * IDA_CONV_FAIL:
750  * Convergence test failures occurred too many times (= MXNCF = 10)
751  * during one internal step.
752  *
753  * IDA_LSETUP_FAIL:
754  * The linear solver's setup routine failed
755  * in an unrecoverable manner.
756  *
757  * IDA_LSOLVE_FAIL:
758  * The linear solver's solve routine failed
759  * in an unrecoverable manner.
760  *
761  * IDA_CONSTR_FAIL:

763  * and the solver was unable to recover.
764  *
765  * IDA_REP_RES_ERR:

768  *

771  * flag.
