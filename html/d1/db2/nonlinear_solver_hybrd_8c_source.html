212 static int getNumericalJacobian(struct dataAndSys* dataAndSysNum, double* jac, const double* x, double* f)
213 {
214  struct dataAndSys *dataSys = (struct dataAndSys*) dataAndSysNum;
215  NONLINEAR_SYSTEM_DATA* systemData = &(dataSys->data->simulationInfo->nonlinearSystemData[dataSys->sysNumber]);
216  DATA_HYBRD* solverData = (DATA_HYBRD*)(systemData->solverData);
217 
218  double delta_h = sqrt(solverData->epsfcn);
219  double delta_hh, delta_hhh, deltaInv;
220  integer iflag = 1;
221  int i, j, l;
222 
223  memcpy(solverData->xSave, x, solverData->n*sizeof(double));
224 
225  for(i = 0; i < solverData->n ; ++i)
226  {
227  delta_hhh = solverData->epsfcn * f[i];
228  delta_hh = fmax(delta_h * fmax(fabs(x[i]), fabs(delta_hhh)), delta_h);
229  delta_hh = ((f[i] >= 0) ? delta_hh : -delta_hh);
230  delta_hh = x[i] + delta_hh - x[i];
231  deltaInv = 1. / delta_hh;
232  solverData->xSave[i] = x[i] + delta_hh;
233 
234  infoStreamPrint(LOG_NLS_JAC, 0, "%d. %s = %f (delta_hh = %f)", i+1, modelInfoGetEquation(&dataSys->data->modelData->modelDataXml,systemData->equationIndex).vars[i], solverData->xSave[i], delta_hh);
235  wrapper_fvec_hybrj(&solverData->n, (const double*) solverData->xSave, solverData->fvecSave, solverData->fjacobian, &solverData->ldfjac, &iflag, dataSys);
236 
237  for(j = 0; j < solverData->n; ++j)
238  {
239  l = i*solverData->n+j;
240  solverData->fjacobian[l] = jac[l] = (solverData->fvecSave[j] - f[j]) * deltaInv;
241  }
242  solverData->xSave[i] = x[i];
243  }
244 
245  return 0;
246 }

258 static int getAnalyticalJacobian(struct dataAndSys* dataSys, double* jac)
259 {
260  int i, j, k, l, ii;
261  DATA *data = (dataSys->data);
262  threadData_t *threadData = dataSys->threadData;
263  NONLINEAR_SYSTEM_DATA* systemData = &(data->simulationInfo->nonlinearSystemData[dataSys->sysNumber]);
264  DATA_HYBRD* solverData = (DATA_HYBRD*)(systemData->solverData);
265  const int index = systemData->jacobianIndex;
266  ANALYTIC_JACOBIAN* jacobian = &(data->simulationInfo->analyticJacobians[systemData->jacobianIndex]);
267 
268  memset(jac, 0, (solverData->n)*(solverData->n)*sizeof(double));
269  memset(solverData->fjacobian, 0, (solverData->n)*(solverData->n)*sizeof(double));
270 
271  for(i=0; i < jacobian->sparsePattern.maxColors; i++)
272  {
273  /* activate seed variable for the corresponding color */
274  for(ii=0; ii < jacobian->sizeCols; ii++)
275  if(jacobian->sparsePattern.colorCols[ii]-1 == i)
276  jacobian->seedVars[ii] = 1;
277 
278  ((systemData->analyticalJacobianColumn))(data, threadData, jacobian, NULL);
279 
280  for(j=0; j<jacobian->sizeCols; j++)
281  {
282  if(jacobian->seedVars[j] == 1)
283  {
284  ii = jacobian->sparsePattern.leadindex[j];
285  while(ii < jacobian->sparsePattern.leadindex[j+1])
286  {
287  l = jacobian->sparsePattern.index[ii];
288  k = j*jacobian->sizeRows + l;
289  solverData->fjacobian[k] = jac[k] = jacobian->resultVars[l];
290  ii++;
291  };
292  }
293  /* de-activate seed variable for the corresponding color */
294  if(jacobian->sparsePattern.colorCols[j]-1 == i)
295  jacobian->seedVars[j] = 0;
296  }
297 
298  }
299 
300  return 0;
301 }

308 static void wrapper_fvec_hybrj(const integer* n, const double* x, double* f, double* fjac, const integer* ldjac, const integer* iflag, void* dataAndSysNum)
309 {
310  int i,j;
311  struct dataAndSys *dataSys = (struct dataAndSys*) dataAndSysNum;
312  DATA *data = (dataSys->data);
313  void *dataAndThreadData[2] = {data, dataSys->threadData};
314  NONLINEAR_SYSTEM_DATA* systemData = &(data->simulationInfo->nonlinearSystemData[dataSys->sysNumber]);
315  DATA_HYBRD* solverData = (DATA_HYBRD*)(systemData->solverData);
316  int continuous = data->simulationInfo->solveContinuous;
317 
318  switch(*iflag)
319  {
320  case 1:
321  /* re-scaling x vector */
322  if(solverData->useXScaling)
323  for(i=0; i<*n; i++)
324  solverData->xScaled[i] = x[i]*solverData->xScalefactors[i];
325 
326  /* debug output */
327  if(ACTIVE_STREAM(LOG_NLS_RES)) {
328  infoStreamPrint(LOG_NLS_RES, 0, "-- residual function call %d -- scaling = %d", (int)solverData->nfev, solverData->useXScaling);
329  printVector(x, n, LOG_NLS_RES, "x vector (scaled)");
330  printVector(solverData->xScaled, n, LOG_NLS_RES, "x vector");
331  }
332 
333  /* call residual function */
334  if(solverData->useXScaling){
335  (systemData->residualFunc)(dataAndThreadData, (const double*) solverData->xScaled, f, (const int*)iflag);
336  } else {
337  (systemData->residualFunc)(dataAndThreadData, x, f, (const int*)iflag);
338  }
339 
340  /* debug output */
341  if(ACTIVE_STREAM(LOG_NLS_RES)) {
342  printVector(f, n, LOG_NLS_RES, "residuals");
343  infoStreamPrint(LOG_NLS_RES, 0, "-- end of residual function call %d --", (int)solverData->nfev);
344  }
345 
346  solverData->numberOfFunctionEvaluations++;
347  break;
348  case 2:
349  /* set residual function continuous for jacobian calculation */
350  if(continuous)
351  data->simulationInfo->solveContinuous = 0;
352 
353  if(ACTIVE_STREAM(LOG_NLS_RES))
354  infoStreamPrint(LOG_NLS_RES, 0, "-- begin calculating jacobian --");
355 
356  /* performance measurement */
357  rt_ext_tp_tick(&systemData->jacobianTimeClock);
358 
359  /* call apropreated jacobian function */
360  if(systemData->jacobianIndex != -1){
361  integer iflagtmp = 1;
362  wrapper_fvec_hybrj(n, x, f, fjac, ldjac, &iflagtmp, dataSys);
363 
364  getAnalyticalJacobian(dataSys, fjac);
365  }
366  else{
367  getNumericalJacobian(dataSys, fjac, x, f);
368  }
369 
370  /* debug output */
371  if (ACTIVE_STREAM(LOG_NLS_RES)) {
372  infoStreamPrint(LOG_NLS_RES, 0, "-- end calculating jacobian --");
373 
374  if(ACTIVE_STREAM(LOG_NLS_JAC))
375  {
376  char *buffer = (char*)malloc(sizeof(char)*(*n)*25);
377  infoStreamPrint(LOG_NLS_JAC, 1, "jacobian matrix [%dx%d]", (int)*n, (int)*n);
378  for(i=0; i<*n; i++)
379  {
380  buffer[0] = 0;
381  for(j=0; j<*n; j++)
382  sprintf(buffer, "%s%20.12g ", buffer, fjac[i*solverData->n+j]);
383  infoStreamPrint(LOG_NLS_JAC, 0, "%s", buffer);
384  }
385  messageClose(LOG_NLS_JAC);
386  free(buffer);
387  }
388  }
389  /* reset residual function again */
390  if(continuous)
391  data->simulationInfo->solveContinuous = 1;
392 
393  /* performance measurement and statistics */
394  systemData->jacobianTime += rt_ext_tp_tock(&(systemData->jacobianTimeClock));
395  systemData->numberOfJEval++;
396 
397  break;
398 
399  default:
400  throwStreamPrint(NULL, "Well, this is embarrasing. The non-linear solver should never call this case.%d", (int)*iflag);
401  break;
402  }
403 }

412 int solveHybrd(DATA *data, threadData_t *threadData, int sysNumber)
413 {
414  NONLINEAR_SYSTEM_DATA* systemData = &(data->simulationInfo->nonlinearSystemData[sysNumber]);
415  DATA_HYBRD* solverData = (DATA_HYBRD*)systemData->solverData;
416  /*
417  * We are given the number of the non-linear system.
418  * We want to look it up among all equations.
419  */
420  int eqSystemNumber = systemData->equationIndex;
421 
422  int i, j;
423  integer iflag = 1;
424  double xerror, xerror_scaled;
425  int success = 0;
426  double local_tol = 1e-12;
427  double initial_factor = solverData->factor;
428  int nfunc_evals = 0;
429  int continuous = 1;
430  int nonContinuousCase = 0;
431 
432  int giveUp = 0;
433  int retries = 0;
434  int retries2 = 0;
435  int retries3 = 0;
436  int assertCalled = 0;
437  int assertRetries = 0;
438  int assertMessage = 0;
439 
440  modelica_boolean* relationsPreBackup;
441 
442  struct dataAndSys dataAndSysNumber = {data, threadData, sysNumber};
443 
444  relationsPreBackup = (modelica_boolean*) malloc(data->modelData->nRelations*sizeof(modelica_boolean));
445 
446  solverData->numberOfFunctionEvaluations = 0;
447 
448  // Initialize lambda variable
449  if (data->simulationInfo->nonlinearSystemData[sysNumber].homotopySupport) {
450  solverData->x[solverData->n] = 1.0;
451  solverData->xSave[solverData->n] = 1.0;
452  solverData->xScaled[solverData->n] = 1.0;
453  }
454  else {
455  solverData->x[solverData->n] = 0.0;
456  solverData->xSave[solverData->n] = 0.0;
457  solverData->xScaled[solverData->n] = 0.0;
458  }
459 
460  /* debug output */
461  if(ACTIVE_STREAM(LOG_NLS_V))
462  {
463  int indexes[2] = {1,eqSystemNumber};
464  infoStreamPrintWithEquationIndexes(LOG_NLS_V, 1, indexes, "Start solving non-linear system >>%d<< using Hybrd solver at time %g", eqSystemNumber, data->localData[0]->timeValue);
465  for(i=0; i<solverData->n; i++)
466  {
467  infoStreamPrint(LOG_NLS_V, 1, "%d. %s = %f", i+1, modelInfoGetEquation(&data->modelData->modelDataXml,eqSystemNumber).vars[i], systemData->nlsx[i]);
468  infoStreamPrint(LOG_NLS_V, 0, " nominal = %f\nold = %f\nextrapolated = %f",
469  systemData->nominal[i], systemData->nlsxOld[i], systemData->nlsxExtrapolation[i]);
470  messageClose(LOG_NLS_V);
471  }
472  messageClose(LOG_NLS_V);
473  }
474 
475  /* set x vector */
476  if(data->simulationInfo->discreteCall)
477  memcpy(solverData->x, systemData->nlsx, solverData->n*(sizeof(double)));
478  else
479  memcpy(solverData->x, systemData->nlsxExtrapolation, solverData->n*(sizeof(double)));
480 
481  for(i=0; i<solverData->n; i++){
482  solverData->xScalefactors[i] = fmax(fabs(solverData->x[i]), systemData->nominal[i]);
483  }
484 
485  /* start solving loop */
486  while(!giveUp && !success)
487  {
488  for(i=0; i<solverData->n; i++)
489  solverData->xScalefactors[i] = fmax(fabs(solverData->x[i]), systemData->nominal[i]);
490 
491  /* debug output */
492  if(ACTIVE_STREAM(LOG_NLS_V)) {
493  printVector(solverData->xScalefactors, &(solverData->n), LOG_NLS_V, "scaling factors x vector");
494  printVector(solverData->x, &(solverData->n), LOG_NLS_V, "Iteration variable values");
495  }
496 
497  /* Scaling x vector */
498  if(solverData->useXScaling) {
499  for(i=0; i<solverData->n; i++) {
500  solverData->x[i] = (1.0/solverData->xScalefactors[i]) * solverData->x[i];
501  }
502  }
503 
504  /* debug output */
505  if(ACTIVE_STREAM(LOG_NLS_V))
506  {
507  printVector(solverData->x, &solverData->n, LOG_NLS_V, "Iteration variable values (scaled)");
508  }
509 
510  /* set residual function continuous
511  */
512  if(continuous) {
513  ((DATA*)data)->simulationInfo->solveContinuous = 1;
514  } else {
515  ((DATA*)data)->simulationInfo->solveContinuous = 0;
516  }
517 
518  giveUp = 1;
519 
520  /* try */
521  {
522  int success = 0;
523 #ifndef OMC_EMCC
524  MMC_TRY_INTERNAL(simulationJumpBuffer)
525 #endif
526  hybrj_(wrapper_fvec_hybrj, &solverData->n, solverData->x,
527  solverData->fvec, solverData->fjac, &solverData->ldfjac, &solverData->xtol,
528  &solverData->maxfev, solverData->diag, &solverData->mode, &solverData->factor,
529  &solverData->nprint, &solverData->info, &solverData->nfev, &solverData->njev, solverData->r__,
530  &solverData->lr, solverData->qtf, solverData->wa1, solverData->wa2,
531  solverData->wa3, solverData->wa4, (void*) &dataAndSysNumber);
532 
533  success = 1;
534  if(assertCalled)
535  {
536  infoStreamPrint(LOG_NLS_V, 0, "After assertions failed, found a solution for which assertions did not fail.");
537  /* re-scaling x vector */
538  for(i=0; i<solverData->n; i++){
539  if(solverData->useXScaling)
540  systemData->nlsxOld[i] = solverData->x[i]*solverData->xScalefactors[i];
541  else
542  systemData->nlsxOld[i] = solverData->x[i];
543  }
544  }
545  assertRetries = 0;
546  assertCalled = 0;
547  success = 1;
548 #ifndef OMC_EMCC
549  MMC_CATCH_INTERNAL(simulationJumpBuffer)
550 #endif
551  /* catch */
552  if (!success)
553  {
554  if (!assertMessage)
555  {
556  if (ACTIVE_WARNING_STREAM(LOG_STDOUT))
557  {
558  if(data->simulationInfo->initial)
559  warningStreamPrint(LOG_STDOUT, 1, "While solving non-linear system an assertion failed during initialization.");
560  else
561  warningStreamPrint(LOG_STDOUT, 1, "While solving non-linear system an assertion failed at time %g.", data->localData[0]->timeValue);
562  warningStreamPrint(LOG_STDOUT, 0, "The non-linear solver tries to solve the problem that could take some time.");
563  warningStreamPrint(LOG_STDOUT, 0, "It could help to provide better start-values for the iteration variables.");
564  if (!ACTIVE_STREAM(LOG_NLS_V))
565  warningStreamPrint(LOG_STDOUT, 0, "For more information simulate with -lv LOG_NLS_V");
566  messageClose(LOG_STDOUT);
567  }
568  assertMessage = 1;
569  }
570 
571  solverData->info = -1;
572  xerror_scaled = 1;
573  xerror = 1;
574  assertCalled = 1;
575  }
576  }
577 
578  /* set residual function continuous */
579  if(continuous)
580  {
581  ((DATA*)data)->simulationInfo->solveContinuous = 0;
582  }
583  else
584  {
585  ((DATA*)data)->simulationInfo->solveContinuous = 1;
586  }
587 
588  /* re-scaling x vector */
589  if(solverData->useXScaling)
590  for(i=0; i<solverData->n; i++)
591  solverData->x[i] = solverData->x[i]*solverData->xScalefactors[i];
592 
593  /* check for proper inputs */
594  if(solverData->info == 0) {
595  printErrorEqSyst(IMPROPER_INPUT, modelInfoGetEquation(&data->modelData->modelDataXml, eqSystemNumber),
596  data->localData[0]->timeValue);
597  }
598 
599  if(solverData->info != -1)
600  {
601  /* evaluate with discontinuities */
602  if(data->simulationInfo->discreteCall){
603  int scaling = solverData->useXScaling;
604  int success = 0;
605  if(scaling)
606  solverData->useXScaling = 0;
607 
608  ((DATA*)data)->simulationInfo->solveContinuous = 0;
609 
610  /* try */
611 #ifndef OMC_EMCC
612  MMC_TRY_INTERNAL(simulationJumpBuffer)
613 #endif
614  wrapper_fvec_hybrj(&solverData->n, solverData->x, solverData->fvec, solverData->fjac, &solverData->ldfjac, &iflag, (void*) &dataAndSysNumber);
615  success = 1;
616 #ifndef OMC_EMCC
617  MMC_CATCH_INTERNAL(simulationJumpBuffer)
618 #endif
619  /* catch */
620  if (!success)
621  {
622  warningStreamPrint(LOG_STDOUT, 0, "Non-Linear Solver try to handle a problem with a called assert.");
623 
624  solverData->info = -1;
625  xerror_scaled = 1;
626  xerror = 1;
627  assertCalled = 1;
