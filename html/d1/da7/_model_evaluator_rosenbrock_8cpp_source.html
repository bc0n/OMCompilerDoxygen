64  d_(0.0),
65  epetra_comm_(comm)

67  using Teuchos::rcp;

70

72  x_space_ = Thyra::create_VectorSpace(map_x_);
73 
74  f_epetra_map_ = map_x_;
75  f_space_ = x_space_;
76 
77  x0_ = rcp(new Epetra_Vector(*map_x_));
78  (*x0_)[0] = -1.2;

80

82  (*x_analytic_)[0] = 1.0;
83  (*x_analytic_)[1] = 1.0;
84 
85  // Initialize the graph for W CrsMatrix object
86  W_graph_ = rcp(new Epetra_CrsGraph(::Copy,*map_x_,nx));
87  {
88  int indices[nx] = { 0, 1 };
89  for( int i = 0; i < nx; ++i )
90  W_graph_->InsertGlobalIndices(i,nx,indices);
91  }
92  W_graph_->FillComplete();
93 
94 }

179 void
180 RosenbrockModelEvaluator::
181 evalModelImpl(const Thyra::ModelEvaluatorBase::InArgs<double>& inArgs,
182  const Thyra::ModelEvaluatorBase::OutArgs<double>& outArgs) const
183 {
184  using Teuchos::dyn_cast;
185  using Teuchos::rcp_dynamic_cast;
186  //
187  // Get the input arguments
188  //
189 
190  TEUCHOS_ASSERT(nonnull(inArgs.get_x()));
191  const Thyra::ConstDetachedVectorView<double> x(inArgs.get_x());
192 
193  Teuchos::RCP<const Thyra::VectorBase<double> > x_dot_rcp = inArgs.get_x_dot();
194  double alpha = inArgs.get_alpha();
195  double beta = inArgs.get_beta();
196  //double time = inArgs.get_t();
197 
198  //
199  // Get the output arguments
200  //
201  const Teuchos::RCP< Thyra::VectorBase<double> > f_out = outArgs.get_f();
202  const Teuchos::RCP< Thyra::LinearOpBase< double > > W_out = outArgs.get_W_op();
203 
204  //
205  // Compute the functions
206  //
207  if(nonnull(f_out)) {
208  if (nonnull(x_dot_rcp)) {
209  NOX_FUNC_TIME_MONITOR("RosenbrockModelEvaluator::eval f_out");
210  Thyra::DetachedVectorView<double> f(f_out);
211  const Thyra::ConstDetachedVectorView<double> x_dot(x_dot_rcp);
212  f[0] = x_dot[0] + 10.0 * ( x[1] - x[0] * x[0] );
213  f[1] = x_dot[1] + 1.0 - x[0];
214  std::cout << "Residual Evaluation: Transient" << std::endl;
215  }
216  else {
217  NOX_FUNC_TIME_MONITOR("RosenbrockModelEvaluator::eval f_out");
218  const Thyra::DetachedVectorView<double> f(f_out);
219  f[0] = 10.0 * ( x[1] - x[0] * x[0] );
220  f[1] = 1.0 - x[0];
221  std::cout << "Residual Evaluation: Steady-State" << std::endl;
222  }
223  }
224  if(nonnull(W_out)) {
225  Teuchos::RCP<Epetra_Operator> W_epetra= Thyra::get_Epetra_Operator(*W_out);
226  Teuchos::RCP<Epetra_CrsMatrix> W_epetracrs = rcp_dynamic_cast<Epetra_CrsMatrix>(W_epetra);
227  TEUCHOS_ASSERT(nonnull(W_epetracrs));
228  Epetra_CrsMatrix& DfDx = *W_epetracrs;
229 
230  DfDx.PutScalar(0.0);
231 
232  int indexes[2];
233  indexes[0] = 0;
234  indexes[1] = 1;
235 
236  double values[2];
237 
238  if (nonnull(x_dot_rcp)) {
239  // Augment Jacobian for Pseudo-transient, use identity for V
240  // Row [0]
241  values[0] = alpha * 1.0 + beta * ( -20.0 * x[0] );
242  values[1] = beta * 10.0;
243  DfDx.SumIntoGlobalValues( 0, 2, values, indexes );
244  // Row [1]
245  values[0] = beta * (-1.0);
246  values[1] = alpha * 1.0 + beta * 0.0;
247  DfDx.SumIntoGlobalValues( 1, 2, values, indexes );
248  std::cout << "Jacobian Evaluation: Transient, alpha=" << alpha << ", beta=" << beta << std::endl;
249  }
250  else {
251  // Row [0]
252  values[0] = -20.0 * x[0];
253  values[1] = 10.0;
254  DfDx.SumIntoGlobalValues( 0, 2, values, indexes );
255  // Row [1]
256  values[0] = -1.0;
257  values[1] = 0.0;
258  DfDx.SumIntoGlobalValues( 1, 2, values, indexes );
259  std::cout << "Jacobian Evaluation: Steady-State, alpha=" << alpha << ", beta=" << beta << std::endl;
260  }
261 
262  }
