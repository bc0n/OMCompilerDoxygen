54 /***************************************
55  * Subspace Iteration *
56  ***************************************
57  QR-factorize VR = Z for the starting matrix Z
58  for k=1,2,...
59  if Power Iteration
60  R = A * V
61  if Inverse Iteration
62  R = (A - lshift I)^-1 * V
63  if Approximate Inverse Iteration
64  R = (M - lshift I)^-1 * V
65  if Rayleigh Quotient Iteration
66  R = (A - mu I)^-1 * V
67  R = V*R
68  resid = ||Z - VR||_2
69  QR-factorize VR = Z
70  ***************************************/

94 LIS_INT lis_esi_malloc_work(LIS_ESOLVER esolver)
95 {
96  LIS_VECTOR *work;
97  LIS_INT i,j,worklen,err,ss;
98 
99  LIS_DEBUG_FUNC_IN;
100 
101  ss = esolver->options[LIS_EOPTIONS_SUBSPACE];
102 
103  worklen = NWORK + ss;
104  work = (LIS_VECTOR *)lis_malloc( worklen*sizeof(LIS_VECTOR),"lis_esi_malloc_work::work" );
105  if( work==NULL )
106  {
107  LIS_SETERR_MEM(worklen*sizeof(LIS_VECTOR));
108  return LIS_ERR_OUT_OF_MEMORY;
109  }
110  if( esolver->eprecision==LIS_PRECISION_DEFAULT )
111  {
112  for(i=0;i<worklen;i++)
113  {
114  err = lis_vector_duplicate(esolver->A,&work[i]);
115  if( err ) break;
116  }
117  }
118  else
119  {
120  for(i=0;i<worklen;i++)
121  {
122  err = lis_vector_duplicateex(LIS_PRECISION_QUAD,esolver->A,&work[i]);
123  if( err ) break;
124  }
125  }
126  if( i<worklen )
127  {
128  for(j=0;j<i;j++) lis_vector_destroy(work[j]);
129  lis_free(work);
130  return err;
131  }
132  esolver->worklen = worklen;
133  esolver->work = work;
134 
135  LIS_DEBUG_FUNC_OUT;
136  return LIS_SUCCESS;
137 }

139 #undef __FUNC__
140 #define __FUNC__ "lis_esi"
141 LIS_INT lis_esi(LIS_ESOLVER esolver)
142 {
143  LIS_MATRIX A;
144  LIS_VECTOR x, Ax;
145  LIS_SCALAR xAx, xx, mu, lshift;
146  LIS_INT ss;
147  LIS_INT emaxiter;
148  LIS_REAL tol;
149  LIS_INT i,j,k;
150  LIS_SCALAR evalue,dotvr;
151  LIS_INT iter,giter,output,niesolver;
152  LIS_INT nprocs,my_rank;
153  LIS_REAL nrm2,dot,resid,resid0;
154  LIS_VECTOR *v,r,q;
155  LIS_SOLVER solver;
156  LIS_PRECON precon;
157  double times,itimes,ptimes,p_c_times,p_i_times;
158  LIS_INT err;
159  LIS_INT nsol, precon_type;
160  char solvername[128], preconname[128], esolvername[128];
161 
162  LIS_DEBUG_FUNC_IN;
163 
164  A = esolver->A;
165  x = esolver->x;
166 
167  ss = esolver->options[LIS_EOPTIONS_SUBSPACE];
168  emaxiter = esolver->options[LIS_EOPTIONS_MAXITER];
169  tol = esolver->params[LIS_EPARAMS_RESID - LIS_EOPTIONS_LEN];
170  lshift = esolver->lshift;
171  output = esolver->options[LIS_EOPTIONS_OUTPUT];
172  niesolver = esolver->options[LIS_EOPTIONS_INNER_ESOLVER];
173 
174  r = esolver->work[0];
175  q = esolver->work[1];
176  v = &esolver->work[2];
177  Ax = esolver->work[3];
178  lis_vector_set_all(1.0,r);
179  lis_vector_nrm2(r, &nrm2);
180  lis_vector_scale(1/nrm2,r);
181 
182  lis_esolver_get_esolvername(niesolver, esolvername);
183 #ifdef _LONGLONG
184  if( A->my_rank==0 ) printf("iesolver : %s %lld\n", esolvername, niesolver);
185 #else
186  if( A->my_rank==0 ) printf("iesolver : %s %d\n", esolvername, niesolver);
187 #endif
188 
189  switch ( niesolver )
190  {
191  case LIS_ESOLVER_II:
192  lis_solver_create(&solver);
193  lis_solver_set_option("-i cg -p ilu",solver);
194  lis_solver_set_optionC(solver);
195  lis_solver_get_solver(solver, &nsol);
196  lis_solver_get_precon(solver, &precon_type);
197  lis_solver_get_solvername(nsol, solvername);
198  lis_solver_get_preconname(precon_type, preconname);
199 #ifdef _LONGLONG
200  if( A->my_rank==0 ) printf("solver : %s %lld\n", solvername, nsol);
201  if( A->my_rank==0 ) printf("precon : %s %lld\n", preconname, precon_type);
202 #else
203  if( A->my_rank==0 ) printf("solver : %s %d\n", solvername, nsol);
204  if( A->my_rank==0 ) printf("precon : %s %d\n", preconname, precon_type);
205 #endif
206 #ifdef _LONG__DOUBLE
207  if( A->my_rank==0 ) printf("local shift = %Le\n", lshift);
208 #else
209  if( A->my_rank==0 ) printf("local shift = %e\n", lshift);
210 #endif
211  if (lshift != 0) lis_matrix_shift_diagonal(A, lshift);
212  break;
213  case LIS_ESOLVER_AII:
214  lis_solver_create(&solver);
215  lis_solver_set_option("-i cg -p ilu",solver);
216  lis_solver_set_optionC(solver);
217  lis_solver_get_solver(solver, &nsol);
218  lis_solver_get_precon(solver, &precon_type);
219  lis_solver_get_solvername(nsol, solvername);
220  lis_solver_get_preconname(precon_type, preconname);
221 #ifdef _LONGLONG
222  if( A->my_rank==0 ) printf("solver : %s %lld\n", solvername, nsol);
223  if( A->my_rank==0 ) printf("precon : %s %lld\n", preconname, precon_type);
224 #else
225  if( A->my_rank==0 ) printf("solver : %s %d\n", solvername, nsol);
226  if( A->my_rank==0 ) printf("precon : %s %d\n", preconname, precon_type);
227 #endif
228 #ifdef _LONG__DOUBLE
229  if( A->my_rank==0 ) printf("local shift = %Le\n", lshift);
230 #else
231  if( A->my_rank==0 ) printf("local shift = %e\n", lshift);
232 #endif
233  if (lshift != 0) lis_matrix_shift_diagonal(A, lshift);
234  lis_vector_set_all(1.0,q);
235  lis_solve(A, q, x, solver);
236  lis_precon_create(solver, &precon);
237  solver->precon = precon;
238  break;
239  case LIS_ESOLVER_RQI:
240  lis_solver_create(&solver);
241  lis_solver_set_option("-p ilu -maxiter 10",solver);
242  lis_solver_set_optionC(solver);
243  lis_solver_get_solver(solver, &nsol);
244  lis_solver_get_precon(solver, &precon_type);
245  lis_solver_get_solvername(nsol, solvername);
246  lis_solver_get_preconname(precon_type, preconname);
247 #ifdef _LONGLONG
248  if( A->my_rank==0 ) printf("solver : %s %lld\n", solvername, nsol);
249  if( A->my_rank==0 ) printf("precon : %s %lld\n", preconname, precon_type);
250 #else
251  if( A->my_rank==0 ) printf("solver : %s %d\n", solvername, nsol);
252  if( A->my_rank==0 ) printf("precon : %s %d\n", preconname, precon_type);
253 #endif
254 #ifdef _LONG__DOUBLE
255  if( A->my_rank==0 ) printf("local shift = %Le\n", lshift);
256 #else
257  if( A->my_rank==0 ) printf("local shift = %e\n", lshift);
258 #endif
259  if (lshift != 0) lis_matrix_shift_diagonal(A, lshift);
260  break;
261  }
262 
263  giter=0;
264  j=0;
265  while (j<ss)
266  {
267  lis_vector_duplicate(A,&esolver->evector[j]);
268  j = j+1;
269  lis_vector_copy(r, v[j]);
270 
271  if (niesolver==LIS_ESOLVER_II || niesolver==LIS_ESOLVER_RQI)
272  {
273  /* create preconditioner */
274  solver->A = A;
275  err = lis_precon_create(solver, &precon);
276  if( err )
277  {
278  lis_solver_work_destroy(solver);
279  solver->retcode = err;
280  return err;
281  }
282  }
283 
284  if (niesolver==LIS_ESOLVER_RQI)
285  {
286  lis_vector_nrm2(x, &nrm2);
287  lis_vector_scale(1/nrm2, x);
288  lis_matvec(A, x, Ax);
289  lis_vector_dot(x, Ax, &xAx);
290  lis_vector_dot(x, x, &xx);
291  mu = xAx / xx;
292  }
293 
294  iter = 0;
295  while (iter<emaxiter)
296  {
297  iter = iter+1;
298  giter = giter+1;
299 
300  /* QR-factorize VR = Z for starting vector Z */
301  for (k=1;k<j;k++)
302  {
303  lis_vector_dot(v[j], v[k], &dot);
304  lis_vector_axpy(-dot, v[k], v[j]);
305  }
306 
307  switch( niesolver )
308  {
309  case LIS_ESOLVER_PI:
310 
311  /* R = A * V */
312  lis_matvec(A,v[j],r);
313 
314  break;
315 
316  case LIS_ESOLVER_II:
317 
318  /* R = (A - lshift I)^-1 * V */
319  lis_solve_kernel(A, v[j], r, solver, precon);
320 
321  break;
322 
323  case LIS_ESOLVER_AII:
324 
325  /* R = (M - lshift I)^-1 * V */
326  lis_psolve(solver, v[j], r);
327 
328  break;
329 
330  case LIS_ESOLVER_RQI:
331 
332  /* R = (A - mu I)^-1 * V */
333  lis_vector_nrm2(v[j], &nrm2);
334  lis_vector_scale(1/nrm2, v[j]);
335  lis_matrix_shift_diagonal(A, -mu);
336  lis_solve_kernel(A, v[j], r, solver, precon);
337  lis_matrix_shift_diagonal(A, mu);
338  break;
339  }
340 
341  if ( j==1 && ( niesolver==LIS_ESOLVER_II || niesolver==LIS_ESOLVER_AII || niesolver==LIS_ESOLVER_RQI ))
342  {
343  lis_solver_get_timeex(solver,&times,&itimes,&ptimes,&p_c_times,&p_i_times);
344  esolver->ptimes += solver->ptimes;
345  esolver->itimes += solver->itimes;
346  esolver->p_c_times += solver->p_c_times;
347  esolver->p_i_times += solver->p_i_times;
348  }
349 
350  /* R = V*R */
351  lis_vector_nrm2(r, &nrm2);
352  lis_vector_dot(v[j],r,&dotvr);
353  mu = mu + 1/dotvr;
354 
355  /* resid = ||Z - VR||_2 */
356  lis_vector_axpyz(-dotvr,v[j],r,q);
357  lis_vector_nrm2(q, &resid);
358  resid = fabs(resid / dotvr);
359 
360  lis_vector_scale(1/nrm2,r);
361  lis_vector_copy(r, v[j]);
362 
363  /* convergence check */
364  if ( j==1 )
365  {
366  if( output & LIS_PRINT_MEM ) esolver->residual[iter] = resid;
367  esolver->iter = iter;
368  esolver->resid = resid;
369  }
370 
371 
372  if( output & LIS_PRINT_OUT && A->my_rank==0 ) lis_print_rhistory(iter,resid);
373  if (tol>resid) break;
374  }
375 
376  if (niesolver==LIS_ESOLVER_II || niesolver==LIS_ESOLVER_RQI)
377  {
378  lis_precon_destroy(precon);
379  }
380 
381  switch ( niesolver )
382  {
383  case LIS_ESOLVER_PI:
384  esolver->evalue[j-1] = dotvr;
385  break;
386  case LIS_ESOLVER_II:
387  esolver->evalue[j-1] = 1/dotvr;
388  break;
389  case LIS_ESOLVER_AII:
390  esolver->evalue[j-1] = 1/dotvr;
391  break;
392  case LIS_ESOLVER_RQI:
393  esolver->evalue[j-1] = mu;
394  break;
395  }
396  lis_vector_copy(v[j], esolver->evector[j-1]);
397 
398  if (A->my_rank==0 && ss>1)
399  {
400 #ifdef _LONGLONG
401  if( output & LIS_PRINT_OUT ) printf("Subspace: mode number = %lld\n", j-1);
402 #else
403  if( output & LIS_PRINT_OUT ) printf("Subspace: mode number = %d\n", j-1);
404 #endif
405 #ifdef _LONG__DOUBLE
406  if( output & LIS_PRINT_OUT ) printf("Subspace: eigenvalue = %Le\n", esolver->evalue[j-1]);
407 #else
408  if( output & LIS_PRINT_OUT ) printf("Subspace: eigenvalue = %e\n", esolver->evalue[j-1]);
409 #endif
410 #ifdef _LONGLONG
411  if( output & LIS_PRINT_OUT ) printf("Subspace: number of iterations = %lld\n",iter);
412 #else
413  if( output & LIS_PRINT_OUT ) printf("Subspace: number of iterations = %d\n",iter);
414 #endif
415 #ifdef _LONG__DOUBLE
416  if( output & LIS_PRINT_OUT ) printf("Subspace: relative residual 2-norm = %Le\n",resid);
417 #else
418  if( output & LIS_PRINT_OUT ) printf("Subspace: relative residual 2-norm = %e\n",resid);
419 #endif
420  }
421  }
422 
423  lis_vector_copy(esolver->evector[esolver->options[LIS_EOPTIONS_MODE]], esolver->x);
424 
425  switch ( niesolver )
426  {
427  case LIS_ESOLVER_II:
428  if (lshift != 0) lis_matrix_shift_diagonal(A, -lshift);
429  lis_solver_destroy(solver);
430  break;
431  case LIS_ESOLVER_AII:
432  if (lshift != 0) lis_matrix_shift_diagonal(A, -lshift);
433  lis_precon_destroy(precon);
434  lis_solver_destroy(solver);
435  break;
436  case LIS_ESOLVER_RQI:
437  if (lshift != 0) lis_matrix_shift_diagonal(A, -lshift);
438  lis_solver_destroy(solver);
439  break;
440  }
441 
442  LIS_DEBUG_FUNC_OUT;
443  return LIS_SUCCESS;
444 }
