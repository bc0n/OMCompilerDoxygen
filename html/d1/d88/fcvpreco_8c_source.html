73 /* C function FCVPSet to interface between CVODE and a Fortran subroutine
74  FCVPSET for setup of a Krylov preconditioner.
75  Addresses of t, y, fy, jok, gamma, h, vtemp1, vtemp2, vtemp3, and the
76  address jcurPtr are passed to FCVPSET, using the routine
77  N_VGetArrayPointer from NVECTOR.
78  A return flag ier from FCVPSET is returned by FCVPSet.
79  Auxiliary data is assumed to be communicated by common blocks. */
80 
81 int FCVPSet(realtype t, N_Vector y, N_Vector fy, booleantype jok,
82  booleantype *jcurPtr, realtype gamma,
83  void *user_data,
84  N_Vector vtemp1, N_Vector vtemp2, N_Vector vtemp3)
85 {
86  int ier = 0;

88  realtype h;

92 
93  ydata = N_VGetArrayPointer(y);
