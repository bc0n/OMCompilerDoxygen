111  int OperatorRangePoints = OperatorRangeMap().NumGlobalPoints();

126  if ((*InvDiagonal_)[i] != 0.0)
127  (*InvDiagonal_)[i] = 1.0 / (*InvDiagonal_)[i];
128  double nrm;

134  if(SmootherSweeps)
135  ML_CHK_ERR(SetupSmoother());
136 
137 
138  if(MaxLevels > 0) {
139  /* Build the prolongator */
140  ML_CHK_ERR(BuildProlongator());

150 
151  /* Setup Preconditioner on Coarse Matrix */
152  CoarsePC = new MultiLevelPreconditioner(*CoarseMatrix,ListCoarse);
153  if(!CoarsePC) ML_CHK_ERR(-2);

179 int ML_Epetra::FaceMatrixFreePreconditioner::BuildNullspace(Epetra_MultiVector *& nullspace){
180  int Nf=FaceRangeMap_->NumMyElements();
181 
182  /* Pull the coordinates from Teuchos */
183  double * xcoord=List_.get("x-coordinates",(double*)0);
184  double * ycoord=List_.get("y-coordinates",(double*)0);
185  double * zcoord=List_.get("z-coordinates",(double*)0);
186  dim=(xcoord!=0) + (ycoord!=0) + (zcoord!=0);
187 
188  // Sanity check
189  if(dim!=3){if(!Comm_->MyPID()) printf("ERROR: FaceMatrixFreePreconditioner only works in 3D"); ML_CHK_ERR(-1);}
190 
191  // Build the (unimported) coordinate multivector
192  double **d_coords=new double* [dim];
193  d_coords[0]=xcoord; d_coords[1]=ycoord;
194  if(dim==3) d_coords[2]=zcoord;
195  Epetra_MultiVector n_coords_domain(View,*NodeDomainMap_,d_coords,dim);
196  Epetra_MultiVector *n_coords;
197 
198  // Import coordinate info
199  if(FaceNode_Matrix_->Importer()){
200  n_coords=new Epetra_MultiVector(FaceNode_Matrix_->ColMap(),dim);
201  n_coords->PutScalar(0.0);
202  n_coords->Import(n_coords_domain,*FaceNode_Matrix_->Importer(),Add);
203  }
204  else n_coords=&n_coords_domain;
205 
206  // Sanity HAQ - Only works on Hexes
207  if(FaceNode_Matrix_->GlobalMaxNumEntries()!=4)
208  {if(!Comm_->MyPID()) printf("ERROR: FaceMatrixFreePreconditioner only works on Hexes"); ML_CHK_ERR(-2);}
209 
210  // Allocate vector
211  nullspace=new Epetra_MultiVector(*FaceDomainMap_,3);
212 
213  // Fill matrix - NTS this will *NOT* do periodic BCs correctly.
214  double *a=new double[dim];
215  double *b=new double[dim];
216  double *c=new double[dim];
217  for(int i=0;i<Nf;i++){
218  int Ni,*indices;
219  double *values;
220  FaceNode_Matrix_->ExtractMyRowView(i,Ni,values,indices);
221  if(Ni != 4){
222  printf("ERROR: Face %d has only %d nodes\n",i,Ni);
223  ML_free(a);
224  ML_free(b);
225  ML_free(c);
226  ML_CHK_ERR(-1);
227  }
228 
229  a[0] = (*n_coords)[0][indices[1]] - (*n_coords)[0][indices[0]];
230  a[1] = (*n_coords)[1][indices[1]] - (*n_coords)[1][indices[0]];
231  a[2] = (*n_coords)[2][indices[1]] - (*n_coords)[2][indices[0]];
232 
233  b[0] = (*n_coords)[0][indices[2]] - (*n_coords)[0][indices[0]];
234  b[1] = (*n_coords)[1][indices[2]] - (*n_coords)[1][indices[0]];
235  b[2] = (*n_coords)[2][indices[2]] - (*n_coords)[2][indices[0]];
236 
237  cross_product(a,b,c);
238 
239  // HAQ - Hardwiring for hexes
240  // HAQ - Absolute value, presuming all hexes are actually pointed the same way. This is a HAQ!!!
241  (*nullspace)[0][i]=ABS(c[0])/6.0;
242  (*nullspace)[1][i]=ABS(c[1])/6.0;
243  (*nullspace)[2][i]=ABS(c[2])/6.0;
244  }
245 
246  /* Cleanup */
247  if(FaceNode_Matrix_->Importer()) delete n_coords;
248  delete [] a; delete [] b; delete [] c;
249  delete [] d_coords;
250  return 0;
251 }

282 int ML_Epetra::FaceMatrixFreePreconditioner::BuildProlongator()
283 {
284 
285  /* Wrap TMT_Matrix in a ML_Operator */
286  ML_Operator* TMT_ML = ML_Operator_Create(ml_comm_);
287  ML_Operator_WrapEpetraCrsMatrix(const_cast<Epetra_CrsMatrix*>(&*TMT_Matrix_),TMT_ML);
288 
289  /* Nodal Aggregation */
290  ML_Aggregate_Struct *MLAggr=0;
291  ML_Operator *P=0;
292  int NumAggregates;
293  int rv=ML_Epetra::RefMaxwell_Aggregate_Nodes(*TMT_Matrix_,List_,ml_comm_,std::string("FMFP (level 0) :"),MLAggr,P,NumAggregates);
294  if(rv || !P) {if(!Comm_->MyPID()) printf("ERROR: Building nodal P\n");ML_CHK_ERR(-1);}
295 
296  /* Build 1-unknown sparsity of prolongator */
297  Epetra_CrsMatrix *Psparse=0;
298  PBuildSparsity(P,Psparse);
299  if(!Psparse) {if(!Comm_->MyPID()) printf("ERROR: Building Psparse\n");ML_CHK_ERR(-2);}
300 
301  /* Build the "nullspace" */
302  Epetra_MultiVector *nullspace;
303  BuildNullspace(nullspace);
304  if(!nullspace) {if(!Comm_->MyPID()) printf("ERROR: Building Nullspace\n");ML_CHK_ERR(-3);}
305 
306 
307  /* Build the DomainMap of the new operator*/
308  const Epetra_Map & FineColMap = Psparse->ColMap();
309  CoarseMap_=new Epetra_Map(-1,NumAggregates*dim,0,*Comm_);
310 
311  /* Allocate the Prolongator_ */
312  int max_nz_per_row=Psparse->MaxNumEntries();
313  Prolongator_=new Epetra_CrsMatrix(Copy,*FaceRangeMap_,0);
314  int ne1, *idx1, *idx2;
315  idx2=new int [dim*max_nz_per_row];
316  double *vals1, *vals2;
317  vals2=new double[dim*max_nz_per_row];
318  int nonzeros;
319 
320  for(int i=0;i<Prolongator_->NumMyRows();i++){
321  Psparse->ExtractMyRowView(i,ne1,vals1,idx1);
322  nonzeros=0;
323  for(int j=0;j<ne1;j++) nonzeros+=ABS(vals1[j])>0;
324 
325  for(int j=0;j<ne1;j++){
326  for(int k=0;k<dim;k++) {
327  idx2[j*dim+k]=FineColMap.GID(idx1[j])*dim+k;
328  //FIX: This works only because there's an implicit linear mapping which
329  //we're exploiting.
330  if(idx2[j*dim+k]==-1) printf("[%d] ERROR: idx1[j]=%d / idx1[j]*dim+k=%d does not have a GID!\n",Comm_->MyPID(),idx1[j],idx1[j]*dim+k);
331  if(vals1[j]==0 ) vals2[j*dim+k]=0;
332  else vals2[j*dim+k]=(*nullspace)[k][i] / nonzeros;
333  }/*end for*/
334  }/*end for*/
335  Prolongator_->InsertGlobalValues(FaceRangeMap_->GID(i),dim*ne1,vals2,idx2);
336  }/*end for*/
337 
338 
339  /* FillComplete / OptimizeStorage for Prolongator*/
340  Prolongator_->FillComplete(*CoarseMap_,*FaceRangeMap_);
341  Prolongator_->OptimizeStorage();
342 
343 #ifndef NO_OUTPUT
344  /* DEBUG: Dump aggregates */
345  Epetra_IntVector AGG(View,*NodeDomainMap_,MLAggr->aggr_info[0]);
346  IVOUT(AGG,"agg.dat");
347  EpetraExt::RowMatrixToMatlabFile("fmfp_psparse.dat",*Psparse);
348  EpetraExt::RowMatrixToMatlabFile("fmfp_prolongator.dat",*Prolongator_);
349  EpetraExt::VectorToMatrixMarketFile("fmfp_null0.dat",*(*nullspace)(0));
350  EpetraExt::VectorToMatrixMarketFile("fmfp_null1.dat",*(*nullspace)(1));
351  EpetraExt::VectorToMatrixMarketFile("fmfp_null2.dat",*(*nullspace)(2));
352 
353 #endif
354 
355  /* EXPERIMENTAL: Normalize Prolongator Columns */
356  bool normalize_prolongator=List_.get("face matrix free: normalize prolongator",false);
357  if(normalize_prolongator){
358  Epetra_Vector n_vector(*CoarseMap_,false);
359  Prolongator_->InvColSums(n_vector);
360  Prolongator_->RightScale(n_vector);
361  }/*end if*/
362 
363  /* Post-wrapping to convert to ML indexing */
364 #ifdef HAVE_ML_EPETRAEXT
365  Prolongator_ = dynamic_cast<Epetra_CrsMatrix*>(ModifyEpetraMatrixColMap(*Prolongator_,ProlongatorColMapTrans_,"Prolongator",(verbose_&&!Comm_->MyPID())));
366 #endif
367 
368  /* Cleanup */
369  ML_qr_fix_Destroy();
370  ML_Aggregate_Destroy(&MLAggr);
371  ML_Operator_Destroy(&TMT_ML);
372  ML_Operator_Destroy(&P);
373 
374  delete nullspace;
375  delete Psparse;
376  delete [] idx2;
377  delete [] vals2;
378  return 0;
379 }/*end BuildProlongator_*/

387 {

391  ML_Operator *R= ML_Operator_Create(ml_comm_);

393

395  ML_CHK_ERR(ML_Operator_WrapEpetraCrsMatrix(Prolongator_,P,verbose_));
396  P->num_rigid=P->num_PDEs=dim;
397 
398  //NTS: ML_CHK_ERR won't work on this: it returns 1
399  ML_Operator_Transpose_byrow(P, R);
400 
401  /* OPTION: Disable the addon */
402  const Epetra_CrsMatrix *Op11crs = dynamic_cast<const Epetra_CrsMatrix*>(&*Operator_);
403  const Epetra_Operator_With_MatMat *Op11mm = dynamic_cast<const Epetra_Operator_With_MatMat*>(&*Operator_);
404

406  if(Op11crs){
407  if(verbose_ && !Comm_->MyPID()) printf("FMFP: Running *without* addon\n");
408  ML_Operator *SM_ML = ML_Operator_Create(ml_comm_);
409  Temp_ML = ML_Operator_Create(ml_comm_);
410  ML_Operator_WrapEpetraCrsMatrix((Epetra_CrsMatrix*)Op11crs,SM_ML,verbose_);
411  ML_2matmult(SM_ML,P,Temp_ML,ML_CSR_MATRIX);
412  ML_Operator_Destroy(&SM_ML);
413  }

417  }

422  }
423

427

429  int nnz=100;
430  double time;
431  ML_Operator2EpetraCrsMatrix(CoarseMat_ML,CoarseMatrix,nnz,true,time,0,verbose_);
432  // NTS: This is a hack to get around the sticking ones on the diagonal issue;
433 
434  /* Cleanup */
435  ML_Operator_Destroy(&P);
436  ML_Operator_Destroy(&R);
437  ML_Operator_Destroy(&Temp_ML);
438  ML_Operator_Destroy(&CoarseMat_ML);CoarseMat_ML=0;//HAX
439  return 0;
440 }/*end FormCoarseMatrix*/

482 // ================================================ ====== ==== ==== == =
484 int ML_Epetra::FaceMatrixFreePreconditioner::ApplyInverse(const Epetra_MultiVector& B_, Epetra_MultiVector& X) const{
485  const Epetra_MultiVector *B;
486  Epetra_MultiVector *Bcopy=0;
487 
488  /* Sanity Checks */
489  int NumVectors=B_.NumVectors();
490  if (!B_.Map().SameAs(*FaceDomainMap_)) ML_CHK_ERR(-1);
491  if (NumVectors != X.NumVectors()) ML_CHK_ERR(-1);
492 
493  Epetra_MultiVector r_edge(*FaceDomainMap_,NumVectors,false);
494  Epetra_MultiVector e_edge(*FaceDomainMap_,NumVectors,false);
495  Epetra_MultiVector e_node(*CoarseMap_,NumVectors,false);
496  Epetra_MultiVector r_node(*CoarseMap_,NumVectors,false);
497 
498  /* Deal with the B==X case */
499  if (B_.Pointers()[0] == X.Pointers()[0]){
500  Bcopy=new Epetra_MultiVector(B_);
501  B=Bcopy;
502  X.PutScalar(0.0);
503  }
504  else B=&B_;
505 
506 
507  for(int i=0;i<num_cycles;i++){
508  /* Pre-smoothing */
509 #ifdef HAVE_ML_IFPACK
510  if(Smoother_) ML_CHK_ERR(Smoother_->ApplyInverse(*B,X));
511 #endif
512 
513  if(MaxLevels > 0){
514  if(i != 0
515 #ifdef HAVE_ML_IFPACK
516  || Smoother_
517 #endif
518  ){
519  /* Calculate Residual (r_e = b - (S+M+Addon) * x) */
520  ML_CHK_ERR(Operator_->Apply(X,r_edge));
521  ML_CHK_ERR(r_edge.Update(1.0,*B,-1.0));
522 
523  /* Xfer to coarse grid (r_n = P' * r_e) */
524  ML_CHK_ERR(Prolongator_->Multiply(true,r_edge,r_node));
525  }
526  else{
527  /* Xfer to coarse grid (r_n = P' * r_e) */
528  ML_CHK_ERR(Prolongator_->Multiply(true,*B,r_node));
529  }
530 
531  /* AMG on coarse grid (e_n = (CoarseMatrix)^{-1} r_n) */
532  ML_CHK_ERR(CoarsePC->ApplyInverse(r_node,e_node));
