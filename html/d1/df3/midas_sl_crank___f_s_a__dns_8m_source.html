154 function Q = force(yy, data)
155 
156 a = data.a;
157 k = data.params(1);
158 c = data.params(2);
159 l0 = data.l0;
160 F = data.F;
161 
162 q = yy(1); % crank angle
163 x = yy(2); % slider position
164 p = yy(3); % conecting rod angle
165 
166 qd = yy(4); % crank angular velocity
167 xd = yy(5); % slider horizontal velocity
168 pd = yy(6); % conecting rod angular velocity
169 
170 s1 = sin(q);
171 c1 = cos(q);
172 s2 = sin(p);
173 c2 = cos(p);
174 s21 = s2*c1 - c2*s1;
175 c21 = c2*c1 + s2*s1;
176 
177 l2 = x^2 - x*(c2+a*c1) + (1.0 + a^2)/4.0 + a*c21/2.0;
178 l = sqrt(l2);
179 ld = 2.0*x*xd - xd*(c2+a*c1) + x*(s2*pd+a*s1*qd) - a*s21*(pd-qd)/2.0;
180 ld = ld / (2.0*l);
181 
182 f = k*(l-l0) + c*ld;
183 fl = f/l;
184 
185 Q(1) = - fl * a * (s21/2.0 + x*s1) / 2.0;
186 Q(2) = fl * (c2/2.0 - x + a*c1/2.0) + F;
187 Q(3) = - fl * (x*s2 - a*s21/2.0) / 2.0 - F*s2;
188 
189 return

227 res(3) = yp(3) - pd + s2*mu1 - c2*mu2;
228

236 res(8) = -s2 - a*s1;
237 
238 % Velocity constraints
239 res(9) = a*s1*qd + xd + s2*pd;
240 res(10) = -a*c1*qd - c2*pd;
