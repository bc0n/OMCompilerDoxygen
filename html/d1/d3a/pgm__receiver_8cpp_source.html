118 void zmq::pgm_receiver_t::restart_input ()
119 {
120  zmq_assert (session != NULL);
121  zmq_assert (active_tsi != NULL);
122 
123  const peers_t::iterator it = peers.find (*active_tsi);
124  zmq_assert (it != peers.end ());
125  zmq_assert (it->second.joined);
126 
127  // Push the pending message into the session.
128  int rc = session->push_msg (it->second.decoder->msg ());
129  errno_assert (rc == 0);
130 
131  if (insize > 0) {
132  rc = process_input (it->second.decoder);
133  if (rc == -1) {
134  // HWM reached; we will try later.
135  if (errno == EAGAIN) {
136  session->flush ();
137  return;
138  }
139  // Data error. Delete message decoder, mark the
140  // peer as not joined and drop remaining data.
141  it->second.joined = false;
142  LIBZMQ_DELETE(it->second.decoder);
143  insize = 0;
144  }
145  }
146 
147  // Resume polling.
148  set_pollin (pipe_handle);
149  set_pollin (socket_handle);
150 
151  active_tsi = NULL;
152  in_event ();
153 }

155 void zmq::pgm_receiver_t::in_event ()
156 {
157  // Read data from the underlying pgm_socket.
158  const pgm_tsi_t *tsi = NULL;
159 
160  if (has_rx_timer) {
161  cancel_timer (rx_timer_id);
162  has_rx_timer = false;
163  }
164 
165  // TODO: This loop can effectively block other engines in the same I/O
166  // thread in the case of high load.
167  while (true) {
168 
169  // Get new batch of data.
170  // Note the workaround made not to break strict-aliasing rules.
171  void *tmp = NULL;
172  ssize_t received = pgm_socket.receive (&tmp, &tsi);
173  inpos = (unsigned char*) tmp;
174 
175  // No data to process. This may happen if the packet received is
176  // neither ODATA nor ODATA.
177  if (received == 0) {
178  if (errno == ENOMEM || errno == EBUSY) {
179  const long timeout = pgm_socket.get_rx_timeout ();
180  add_timer (timeout, rx_timer_id);
181  has_rx_timer = true;
182  }
183  break;
184  }
185 
186  // Find the peer based on its TSI.
187  peers_t::iterator it = peers.find (*tsi);
188 
189  // Data loss. Delete decoder and mark the peer as disjoint.
190  if (received == -1) {
191  if (it != peers.end ()) {
192  it->second.joined = false;
193  if (it->second.decoder != NULL) {
194  LIBZMQ_DELETE(it->second.decoder);
195  }
196  }
197  break;
198  }
199 
200  // New peer. Add it to the list of know but unjoint peers.
201  if (it == peers.end ()) {
202  peer_info_t peer_info = {false, NULL};
203  it = peers.insert (peers_t::value_type (*tsi, peer_info)).first;
204  }
205 
206  insize = static_cast <size_t> (received);
207 
208  // Read the offset of the fist message in the current packet.
209  zmq_assert (insize >= sizeof (uint16_t));
210  uint16_t offset = get_uint16 (inpos);
211  inpos += sizeof (uint16_t);
212  insize -= sizeof (uint16_t);
213 
214  // Join the stream if needed.
215  if (!it->second.joined) {
216 
217  // There is no beginning of the message in current packet.
218  // Ignore the data.
219  if (offset == 0xffff)
220  continue;
221 
222  zmq_assert (offset <= insize);
223  zmq_assert (it->second.decoder == NULL);
224 
225  // We have to move data to the beginning of the first message.
226  inpos += offset;
227  insize -= offset;
228 
229  // Mark the stream as joined.
230  it->second.joined = true;
231 
232  // Create and connect decoder for the peer.
233  it->second.decoder = new (std::nothrow)
234  v1_decoder_t (0, options.maxmsgsize);
235  alloc_assert (it->second.decoder);
236  }
237 
238  int rc = process_input (it->second.decoder);
239  if (rc == -1) {
240  if (errno == EAGAIN) {
241  active_tsi = tsi;
242 
243  // Stop polling.
244  reset_pollin (pipe_handle);
245  reset_pollin (socket_handle);
246 
247  break;
248  }
249 
250  it->second.joined = false;
251  LIBZMQ_DELETE(it->second.decoder);
252  insize = 0;
253  }
254  }
255 
256  // Flush any messages decoder may have produced.
257  session->flush ();
258 }

260 int zmq::pgm_receiver_t::process_input (v1_decoder_t *decoder)
261 {
262  zmq_assert (session != NULL);
263 
264  while (insize > 0) {
265  size_t n = 0;
266  int rc = decoder->decode (inpos, insize, n);
267  if (rc == -1)
268  return -1;
269  inpos += n;
270  insize -= n;
271  if (rc == 0)
272  break;
273  rc = session->push_msg (decoder->msg ());
274  if (rc == -1) {
275  errno_assert (errno == EAGAIN);
276  return -1;
277  }
278  }
279  return 0;
280 }
