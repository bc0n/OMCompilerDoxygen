331 algorithm
332  nEqns := BackendDAEUtil.systemSize(isyst);
333  nVars := BackendVariable.varsSize(BackendVariable.daeVars(isyst));
334 
335  // over-determined system: nEqns > nVars
336  if intGt(nEqns, nVars) then
337  if Flags.isSet(Flags.INITIALIZATION) then
338  Error.addCompilerWarning("It was not possible to solve the over-determined initial system (" + intString(nEqns) + " equations and " + intString(nVars) + " variables)");
339  BackendDump.dumpEqSystem(isyst, "It was not possible to solve the over-determined initial system (" + intString(nEqns) + " equations and " + intString(nVars) + " variables)");
340  end if;
341 
342  fail();
343  end if;
344 
345  // under-determined system: nEqns < nVars
346  if intLt(nEqns, nVars) then
347  if Flags.isSet(Flags.INITIALIZATION) then
348  Error.addCompilerWarning("It was not possible to solve the under-determined initial system (" + intString(nEqns) + " equations and " + intString(nVars) + " variables)");
349  BackendDump.dumpEqSystem(isyst, "It was not possible to solve the under-determined initial system (" + intString(nEqns) + " equations and " + intString(nVars) + " variables)");
350  end if;
351 
352  fail();
353  end if;
354 end solveInitialSystemEqSystem;

391 protected function inlineWhenForInitializationEquation "author: lochel"
392  input BackendDAE.Equation inEq;
393  input tuple<list<BackendDAE.Equation>, HashSet.HashSet> inTpl;
394  output BackendDAE.Equation outEq = inEq;
395  output tuple<list<BackendDAE.Equation>, HashSet.HashSet> outTpl;
396 protected
397  BackendDAE.EquationAttributes eqAttr;
398  BackendDAE.WhenEquation weqn;
399  DAE.Algorithm alg;
400  DAE.ElementSource source;
401  DAE.Expand crefExpand;
402  HashSet.HashSet leftCrs;
403  Integer size;
404  list<BackendDAE.Equation> eqns;
405  list<BackendDAE.Equation> accEq;
406  list<DAE.Statement> stmts;
407 algorithm
408  (accEq, leftCrs) := inTpl;
409  outTpl := match (inEq)
410  // when equation
411  case BackendDAE.WHEN_EQUATION(whenEquation=weqn, source=source, attr=eqAttr) equation
412  (leftCrs, eqns) = inlineWhenForInitializationWhenEquation(weqn, source, eqAttr, accEq, leftCrs);
413  then (eqns, leftCrs);
414

416  case BackendDAE.ALGORITHM(alg=alg, source=source, expand=crefExpand) equation
417  DAE.ALGORITHM_STMTS(statementLst=stmts) = alg;

419  alg = DAE.ALGORITHM_STMTS(stmts);
420  size = listLength(CheckModel.checkAndGetAlgorithmOutputs(alg, source, crefExpand));
421  eqns = List.consOnTrue(not listEmpty(stmts), BackendDAE.ALGORITHM(size, alg, source, crefExpand, BackendDAE.EQ_ATTR_DEFAULT_DYNAMIC), accEq);
422  then (eqns, leftCrs);
423 
424  else (inEq::accEq, leftCrs);
425  end match;

428 protected function inlineWhenForInitializationWhenEquation "author: lochel"
429  input BackendDAE.WhenEquation inWEqn;
430  input DAE.ElementSource inSource;
431  input BackendDAE.EquationAttributes inEqAttr;
432  input list<BackendDAE.Equation> inEqns;
433  input HashSet.HashSet inLeftCrs;
434  output HashSet.HashSet outLeftCrs = inLeftCrs;
435  output list<BackendDAE.Equation> outEqns = inEqns;
436 protected
437  DAE.Exp lhs, condition, e;

441  DAE.ComponentRef cr;

445 algorithm
446  outEqns := match(inWEqn)

448  active := Expression.containsInitialCall(condition);
449  for stmt in whenStmtLst loop
450  _ := match stmt
451  case BackendDAE.ASSIGN(left = DAE.CREF(componentRef = cr), right = e) equation
452  if active then
453  lhs = Expression.crefExp(cr);
454  eqn = BackendEquation.generateEquation(lhs, e, inSource, inEqAttr);
455  outEqns = eqn::outEqns;
456  else
457  outLeftCrs = List.fold(ComponentReference.expandCref(cr, true), BaseHashSet.add, outLeftCrs);
458  end if;
459  then ();
460 
461  case BackendDAE.ASSIGN(left = lhs as DAE.TUPLE(PR = eLst), right = e) algorithm
462  if active then
463  eqn := BackendEquation.generateEquation(lhs, e, inSource, inEqAttr);
464  outEqns := eqn::outEqns;
465  else
466  crefLst := List.flatten(List.map(eLst,Expression.getAllCrefs));
467  for cr in crefLst loop
468  outLeftCrs := List.fold(ComponentReference.expandCref(cr, true), BaseHashSet.add, outLeftCrs);
469  end for;
470  end if;
471  then ();
472 
473  case BackendDAE.NORETCALL(exp=e, source=source) algorithm
474  if active then
475  //eqn := BackendEquation.generateEquation(DAE.CREF(DAE.emptyCref, DAE.T_UNKNOWN_DEFAULT), e, inSource, inEqAttr);
476  eqn := BackendDAE.ALGORITHM(0, DAE.ALGORITHM_STMTS({DAE.STMT_NORETCALL(e, source)}), inSource, DAE.EXPAND(), BackendDAE.EQ_ATTR_DEFAULT_INITIAL);
477  outEqns := eqn::outEqns;
478  end if;
479  then ();
480 
481  // Everything else doesn't have to be handled
482  else ();
483  end match;
484  end for;
485  then outEqns;
486  else outEqns;
487  end match;
488 end inlineWhenForInitializationWhenEquation;
