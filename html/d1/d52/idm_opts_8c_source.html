59 int get_IntgrOptions(const mxArray *options, idmPbData thisPb, booleantype fwd,
60  int *maxord,
61  long int *mxsteps,
62  int *itol, realtype *reltol, double *Sabstol, double **Vabstol,
63  double *hin, double *hmax,
64  double *tstop,
65  booleantype *suppress,
66  booleantype *errmsg,
67  double **id, double **cnstr,
68  booleantype *res_s)
69 {
70  mxArray *opt;
71  char *bufval;
72  int buflen, status, q;
73  long int i, m, n;
74  double *tmp;
75  char *fctName;
76  char *fwd_fctName = "IDAInit/IDAReInit";
77  char *bck_fctName = "IDAInitB/IDAReInitB";
78 
79  if (fwd) fctName = fwd_fctName;
80  else fctName = bck_fctName;
81 
82  /* Set default values */
83 
84  *maxord = 5;
85 
86  *mxsteps = 0;
87 
88  *itol = IDA_SS;
89  *reltol = 1.0e-3;
90  *Sabstol = 1.0e-6;
91  *Vabstol = NULL;
92 
93  *hin = 0.0;
94  *hmax = 0.0;
95 
96  *res_s = FALSE;
97 
98  *suppress = FALSE;
99 
100  *id = NULL;
101  *cnstr = NULL;
102 
103  Ng = 0;
104  tstopSet = FALSE;
105  mon = FALSE;
106 
107  *errmsg = TRUE;
108 
109  /* Return now if options was empty */
110 
111  if (mxIsEmpty(options)) return(0);
112 
113  /* User data */
114 
115  opt = mxGetField(options,0,"UserData");
116  if ( !mxIsEmpty(opt) ) {
117  mxDestroyArray(mtlb_data);
118  mtlb_data = mxDuplicateArray(opt);
119  }
120 
121  /* Tolerances */
122 
123  opt = mxGetField(options,0,"RelTol");
124  if ( !mxIsEmpty(opt) ) {
125  *reltol = *mxGetPr(opt);
126  if (*reltol < 0.0 ) {
127  idmErrHandler(-999, "IDAS", fctName, "RelTol is negative.", NULL);
128  return(-1);
129  }
130  }
131 
132  opt = mxGetField(options,0,"AbsTol");
133  if ( !mxIsEmpty(opt) ) {
134  m = mxGetM(opt);
135  n = mxGetN(opt);
136  if ( (n != 1) && (m != 1) ) {
137  idmErrHandler(-999, "IDAS", fctName, "AbsTol is not a scalar or a vector.", NULL);
138  return(-1);
139  }
140  if ( m > n ) n = m;
141  tmp = mxGetPr(opt);
142  if (n == 1) {
143  *itol = IDA_SS;
144  *Sabstol = *tmp;
145  if (*Sabstol < 0.0) {
146  idmErrHandler(-999, "IDAS", fctName, "AbsTol is negative.", NULL);
147  return(-1);
148  }
149  } else if (n == N) {
150  *itol = IDA_SV;
151  *Vabstol = (double *) malloc(N*sizeof(double));
152  for(i=0;i<N;i++) {
153  (*Vabstol)[i] = tmp[i];
154  if (tmp[i] < 0.0) {
155  idmErrHandler(-999, "IDAS", fctName, "AbsTol has a negative component.", NULL);
156  return(-1);
157  }
158  }
159  } else {
160  idmErrHandler(-999, "IDAS", fctName, "AbsTol does not contain N elements.", NULL);
161  return(-1);
162  }
163  }
164 
165  /* Maximum number of steps */
166 
167  opt = mxGetField(options,0,"MaxNumSteps");
168  if ( !mxIsEmpty(opt) ) {
169  *mxsteps = (int)*mxGetPr(opt);
170  if (*mxsteps < 0) {
171  idmErrHandler(-999, "IDAS", fctName, "MaxNumSteps is negative.", NULL);
172  return(-1);
173  }
174  }
175 
176  /* Maximum order */
177 
178  opt = mxGetField(options,0,"MaxOrder");
179  if ( !mxIsEmpty(opt) ) {
180  q = (int)*mxGetPr(opt);
181  if (q <= 0) {
182  idmErrHandler(-999, "IDAS", fctName, "MaxOrder must be positive.", NULL);
183  return(-1);
184  }
185  if (q > *maxord) {
186  idmErrHandler(-999, "IDAS", fctName, "MaxOrder is too large for BDF.", NULL);
187  return(-1);
188  }
189  *maxord = q;
190  }
191 
192  /* Initial step size */
193 
194  opt = mxGetField(options,0,"InitialStep");
195  if ( !mxIsEmpty(opt) ) {
196  *hin = *mxGetPr(opt);
197  }
198 
199  /* Maximum step size */
200 
201  opt = mxGetField(options,0,"MaxStep");
202  if ( !mxIsEmpty(opt) ) {
203  tmp = mxGetPr(opt);
204  if (*tmp < 0.0) {
205  idmErrHandler(-999, "IDAS", fctName, "MaxStep is negative.", NULL);
206  return(-1);
207  }
208  if ( mxIsInf(*tmp) )
209  *hmax = 0.0;
210  else
211  *hmax = *tmp;
212  }
213 
214  /* ID vector */
215 
216  opt = mxGetField(options,0,"VariableTypes");
217  if ( !mxIsEmpty(opt) ) {
218  m = mxGetM(opt);
219  n = mxGetN(opt);
220  if ( (n != 1) && (m != 1) ) {
221  idmErrHandler(-999, "IDAS", fctName, "VariableTypes is not a vector.", NULL);
222  return(-1);
223  }
224  if ( m > n ) n = m;
225  if (n == N) {
226  tmp = mxGetPr(opt);
227  *id = (double *)malloc(N*sizeof(double));
228  for(i=0;i<N;i++)
229  (*id)[i] = tmp[i];
230  } else {
231  idmErrHandler(-999, "IDAS", fctName, "VariableTypes has wrong dimension." , NULL);
232  return(-1);
233  }
234  }
235 
236  /* Suppress algebraic variables? */
237 
238  opt = mxGetField(options,0,"SuppressAlgVars");
239  if ( !mxIsEmpty(opt) ) {
240  buflen = mxGetM(opt) * mxGetN(opt) + 1;
241  bufval = mxCalloc(buflen, sizeof(char));
242  status = mxGetString(opt, bufval, buflen);
243  if(status != 0) {
244  idmErrHandler(-999, "IDAS", fctName, "Canot parse SuppressAlgVars.", NULL);
245  return(-1);
246  }
247  if(!strcmp(bufval,"on")) {
248  *suppress = TRUE;
249  } else if(!strcmp(bufval,"off")) {
250  *suppress = FALSE;
251  } else {
252  idmErrHandler(-999, "IDAS", fctName, "SuppressAlgVars has an illegal value.", NULL);
253  return(-1);
254  }
255  }
256 
257  /* Monitor? */
258 
259  opt = mxGetField(options,0,"MonitorFn");
260  if ( !mxIsEmpty(opt) ) {
261  mon = TRUE;
262  mxDestroyArray(mtlb_MONfct);
263  mtlb_MONfct = mxDuplicateArray(opt);
264  opt = mxGetField(options,0,"MonitorData");
265  if ( !mxIsEmpty(opt) ) {
266  mxDestroyArray(mtlb_MONdata);
267  mtlb_MONdata = mxDuplicateArray(opt);
268  }
269  }
270 
271  /* The remaining options are interpreted either for
272  * forward problems only or backward problems only */
273 
274  if (fwd) { /* FORWARD PROBLEM ONLY */
275 
276 
277  /* Disable error/warning messages? */
278  opt = mxGetField(options,0,"ErrorMessages");
279  if ( !mxIsEmpty(opt) ) {
280  if (!mxIsLogical(opt)) {
281  idmErrHandler(-999, "IDAS", fctName, "ErrorMessages is not a logical scalar.", NULL);
282  return(-1);
283  }
284  if (mxIsLogicalScalarTrue(opt)) *errmsg = TRUE;
285  else *errmsg = FALSE;
286  }
287 
288 
289  /* Stopping time */
290  opt = mxGetField(options,0,"StopTime");
291  if ( !mxIsEmpty(opt) ) {
292  *tstop = *mxGetPr(opt);
293  tstopSet = TRUE;
294  }
295 
296  /* Number of root functions */
297  opt = mxGetField(options,0,"NumRoots");
298  if ( !mxIsEmpty(opt) ) {
299 
300  Ng = (int)*mxGetPr(opt);
301  if (Ng < 0) {
302  idmErrHandler(-999, "IDAS", fctName, "NumRoots is negative.", NULL);
303  return(-1);
304  }
305  if (Ng > 0) {
306  /* Roots function */
307  opt = mxGetField(options,0,"RootsFn");
308  if ( !mxIsEmpty(opt) ) {
309  mxDestroyArray(mtlb_Gfct);
310  mtlb_Gfct = mxDuplicateArray(opt);
311  } else {
312  idmErrHandler(-999, "IDAS", fctName, "RootsFn required for NumRoots > 0", NULL);
313  return(-1);
314  }
315  }
316 
317  }
318 
319  /* Constraints vector */
320  opt = mxGetField(options,0,"ConstraintTypes");
321  if ( !mxIsEmpty(opt) ) {
322  m = mxGetM(opt);
323  n = mxGetN(opt);
324  if ( (n != 1) && (m != 1) ) {
325  idmErrHandler(-999, "IDAS", fctName, "ConstraintTypes is not a vector.", NULL);
326  return(-1);
327  }
328  if ( m > n ) n = m;
329  if (n == N) {
330  tmp = mxGetPr(opt);
331  *cnstr = (double *)malloc(N*sizeof(double));
332  for(i=0;i<N;i++)
333  (*cnstr)[i] = tmp[i];
334  } else {
335  idmErrHandler(-999, "IDAS", fctName, "ConstraintTypes has wrong dimension.", NULL);
336  return(-1);
337  }
338  }
339 
340  } else { /* BACKWARD PROBLEM ONLY */
341 
342  /* Dependency on forward sensitivities */
343 
344  opt = mxGetField(options,0,"SensDependent");
345  if ( !mxIsEmpty(opt) ) {
346  if (!mxIsLogical(opt)) {
347  idmErrHandler(-999, "IDAS", fctName, "SensDependent is not a logical scalar.", NULL);
348  return(-1);
349  }
350  if (mxIsLogicalScalarTrue(opt)) *res_s = TRUE;
351  else *res_s = FALSE;
352  }
353 
354  }
355 
356  /* We made it here without problems */
357 
358  return(0);
359 }

362 int get_LinSolvOptions(const mxArray *options, idmPbData thisPb, booleantype fwd,
363  long int *mupper, long int *mlower,
364  long int *mudq, long int *mldq, double *dqrely,
365  int *gstype, int *maxl)
366 {
367  mxArray *opt;
368  char *bufval;
369  int buflen, status;
370  char *fctName;
371  char *fwd_fctName = "IDAInit/IDAReInit";
372  char *bck_fctName = "IDAInitB/IDAReInitB";
373 
374  if (fwd) fctName = fwd_fctName;
375  else fctName = bck_fctName;
376 
377  *mupper = 0;
378  *mlower = 0;
379 
380  *mudq = 0;
381  *mldq = 0;
382  *dqrely = 0.0;
383 
384  *gstype = MODIFIED_GS;
385  *maxl = 0;
386 
387  /* Return now if options was empty */
388 
389  if (mxIsEmpty(options)) return(0);
390 
391  /* Linear solver type */
392 
393  opt = mxGetField(options,0,"LinearSolver");
394  if ( !mxIsEmpty(opt) ) {
395  buflen = mxGetM(opt) * mxGetN(opt) + 1;
396  bufval = mxCalloc(buflen, sizeof(char));
397  status = mxGetString(opt, bufval, buflen);
398  if(status != 0) {
399  idmErrHandler(-999, "IDAS", fctName,
400  "Cannot parse LinearSolver.", NULL);
401  return(-1);
402  }
403  if(!strcmp(bufval,"Band")) ls = LS_BAND;
404  else if(!strcmp(bufval,"GMRES")) ls = LS_SPGMR;
405  else if(!strcmp(bufval,"BiCGStab")) ls = LS_SPBCG;
406  else if(!strcmp(bufval,"TFQMR")) ls = LS_SPTFQMR;
407  else if(!strcmp(bufval,"Dense")) ls = LS_DENSE;
408  else {
409  idmErrHandler(-999, "IDAS", fctName, "LinearSolver has an illegal value.", NULL);
410  return(-1);
411  }
412  }
413 
414  /* Jacobian function */
415 
416  opt = mxGetField(options,0,"JacobianFn");
417  if ( !mxIsEmpty(opt) ) {
418  mxDestroyArray(mtlb_JACfct);
419  mtlb_JACfct = mxDuplicateArray(opt);
420  }
421 
422  /* Band linear solver */
423 
424  if (ls==LS_BAND) {
425 
426  opt = mxGetField(options,0,"UpperBwidth");
427  if ( !mxIsEmpty(opt) )
428  *mupper = (long int)*mxGetPr(opt);
429 
430  opt = mxGetField(options,0,"LowerBwidth");
431  if ( !mxIsEmpty(opt) )
432  *mlower = (long int)*mxGetPr(opt);
433 
434  }
435 
436  /* SPGMR linear solver options */
437 
438  if (ls==LS_SPGMR) {
439 
440  /* Type of Gram-Schmidt procedure */
441 
442  opt = mxGetField(options,0,"GramSchmidtType");
443  if ( !mxIsEmpty(opt) ) {
444  buflen = mxGetM(opt) * mxGetN(opt) + 1;
445  bufval = mxCalloc(buflen, sizeof(char));
446  status = mxGetString(opt, bufval, buflen);
447  if(status != 0) {
448  idmErrHandler(-999, "IDAS", fctName, "Cannot parse GramSchmidtType.", NULL);
449  return(-1);
450  }
451  if(!strcmp(bufval,"Classical")) *gstype = CLASSICAL_GS;
452  else if(!strcmp(bufval,"Modified")) *gstype = MODIFIED_GS;
453  else {
454  idmErrHandler(-999, "IDAS", fctName, "GramSchmidtType has an illegal value.", NULL);
455  return(-1);
456  }
457  }
458 
459  }
460 
461  /* SPILS linear solver options */
462 
463  if ( (ls==LS_SPGMR) || (ls==LS_SPBCG) || (ls==LS_SPTFQMR) ) {
464 
465  /* Max. dimension of Krylov subspace */
466 
467  opt = mxGetField(options,0,"KrylovMaxDim");
468  if ( !mxIsEmpty(opt) ) {
469  *maxl = (int)*mxGetPr(opt);
470  if (*maxl < 0) {
471  idmErrHandler(-999, "IDAS", fctName, "KrylovMaxDim is negative.", NULL);
472  return(-1);
473  }
474  }
475 
476  /* User defined precoditioning */
477 
478  opt = mxGetField(options,0,"PrecSetupFn");
479  if ( !mxIsEmpty(opt) ) {
480  mxDestroyArray(mtlb_PSETfct);
481  mtlb_PSETfct = mxDuplicateArray(opt);
482  }
483 
484  opt = mxGetField(options,0,"PrecSolveFn");
485  if ( !mxIsEmpty(opt) ) {
486  mxDestroyArray(mtlb_PSOLfct);
487  mtlb_PSOLfct = mxDuplicateArray(opt);
488  }
489 
490  /* Preconditioner module */
491 
492  opt = mxGetField(options,0,"PrecModule");
493  if ( !mxIsEmpty(opt) ) {
494  buflen = mxGetM(opt) * mxGetN(opt) + 1;
495  bufval = mxCalloc(buflen, sizeof(char));
496  status = mxGetString(opt, bufval, buflen);
497  if(status != 0) {
498  idmErrHandler(-999, "IDAS", fctName, "Cannot parse PrecModule.", NULL);
499  return(-1);
500  }
501  if(!strcmp(bufval,"BBDPre")) pm = PM_BBDPRE;
502  else if(!strcmp(bufval,"UserDefined")) pm = PM_NONE;
503  else {
504  idmErrHandler(-999, "IDAS", fctName, "PrecModule has an illegal value.", NULL);
505  return(-1);
506  }
507  }
508 
509  if (pm == PM_BBDPRE) {
510 
511  opt = mxGetField(options,0,"UpperBwidth");
512  if ( !mxIsEmpty(opt) )
513  *mupper = (long int)*mxGetPr(opt);
514 
515  opt = mxGetField(options,0,"LowerBwidth");
516  if ( !mxIsEmpty(opt) )
517  *mlower = (long int)*mxGetPr(opt);
518 
519  opt = mxGetField(options,0,"UpperBwidthDQ");
520  if ( !mxIsEmpty(opt) )
521  *mudq = (long int)*mxGetPr(opt);
522 
523  opt = mxGetField(options,0,"LowerBwidthDQ");
524  if ( !mxIsEmpty(opt) )
525  *mldq = (long int)*mxGetPr(opt);
526 
527  opt = mxGetField(options,0,"GlocalFn");
528  if ( !mxIsEmpty(opt) ) {
529  mxDestroyArray(mtlb_GLOCfct);
530  mtlb_GLOCfct = mxDuplicateArray(opt);
531  } else {
532  idmErrHandler(-999, "IDAS", fctName, "GlocalFn required for BBD preconditioner.", NULL);
533  return(-1);
534  }
535 
536  opt = mxGetField(options,0,"GcommFn");
537  if ( !mxIsEmpty(opt) ) {
538  mxDestroyArray(mtlb_GCOMfct);
539  mtlb_GCOMfct = mxDuplicateArray(opt);
540  }
541 
542  }
543 
544  }
545 
546 
547  /* We made it here without problems */
548 
549  return(0);
550 
551 }

661 int get_FSAOptions(const mxArray *options, idmPbData thisPb,

664  int *dqtype, double *rho,
