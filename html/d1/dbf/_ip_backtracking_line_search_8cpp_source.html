203  bool BacktrackingLineSearch::InitializeImpl(const OptionsList& options,
204  const std::string& prefix)
205  {
206  options.GetNumericValue("alpha_red_factor", alpha_red_factor_, prefix);
207  options.GetBoolValue("magic_steps", magic_steps_, prefix);
208  options.GetBoolValue("accept_every_trial_step", accept_every_trial_step_, prefix);
209  options.GetIntegerValue("accept_after_max_steps", accept_after_max_steps_, prefix);
210  Index enum_int;
211  bool is_default = !options.GetEnumValue("alpha_for_y", enum_int, prefix);
212  alpha_for_y_ = AlphaForYEnum(enum_int);
213  if (is_default && acceptor_->HasComputeAlphaForY()) {
214  alpha_for_y_ = LSACCEPTOR_ALPHA_FOR_Y;
215  }
216  options.GetNumericValue("alpha_for_y_tol", alpha_for_y_tol_, prefix);
217  options.GetNumericValue("expect_infeasible_problem_ctol", expect_infeasible_problem_ctol_, prefix);
218  options.GetNumericValue("expect_infeasible_problem_ytol", expect_infeasible_problem_ytol_, prefix);
219  options.GetBoolValue("expect_infeasible_problem", expect_infeasible_problem_, prefix);
220 
221  options.GetBoolValue("start_with_resto", start_with_resto_, prefix);
222 
223  options.GetNumericValue("tiny_step_tol", tiny_step_tol_, prefix);
224  options.GetNumericValue("tiny_step_y_tol", tiny_step_y_tol_, prefix);
225  options.GetIntegerValue("watchdog_trial_iter_max", watchdog_trial_iter_max_, prefix);
226  options.GetIntegerValue("watchdog_shortened_iter_trigger", watchdog_shortened_iter_trigger_, prefix);
227  options.GetNumericValue("soft_resto_pderror_reduction_factor",
228  soft_resto_pderror_reduction_factor_, prefix);
229  options.GetIntegerValue("max_soft_resto_iters", max_soft_resto_iters_,
230  prefix);
231 
232  bool retvalue = true;
233  if (IsValid(resto_phase_)) {
234  if (!resto_phase_->Initialize(Jnlst(), IpNLP(), IpData(), IpCq(),
235  options, prefix)) {
236  return false;
237  }
238  }
239  if (!acceptor_->Initialize(Jnlst(), IpNLP(), IpData(), IpCq(),
240  options, prefix)) {
241  return false;
242  }
243 
244  rigorous_ = true;
245  skipped_line_search_ = false;
246  tiny_step_last_iteration_ = false;
247  fallback_activated_ = false;
248 
249  Reset();
250 
251  count_successive_shortened_steps_ = 0;
252 
253  acceptable_iterate_ = NULL;
254  acceptable_iteration_number_ = -1;
255 
256  last_mu_ = -1.;
257 
258  return retvalue;

261  void BacktrackingLineSearch::FindAcceptableTrialPoint()
262  {
263  DBG_START_METH("BacktrackingLineSearch::FindAcceptableTrialPoint",
264  dbg_verbosity);
265  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
266  "--> Starting line search in iteration %d <--\n",
267  IpData().iter_count());
268 
269  Number curr_mu = IpData().curr_mu();
270  if (last_mu_!=curr_mu) {
271  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
272  "Mu has changed in line search - resetting watchdog counters.\n");
273  // Inactivate the watchdog and release all stored data
274  in_watchdog_ = false;
275  watchdog_iterate_ = NULL;
276  watchdog_delta_ = NULL;
277  watchdog_shortened_iter_ = 0;
278  last_mu_ = curr_mu;
279  }
280 
281  // If the problem is square, we want to enable the
282  // expect_infeasible_problem option automatically so that the
283  // restoration phase is entered soon. This can be over-written
284  // by the Acceptor.
285  if (!acceptor_->NeverRestorationPhase() && IpCq().IsSquareProblem()) {
286  //expect_infeasible_problem_ = true;
287  expect_infeasible_problem_ctol_ = 0.;
288  }
289 
290  // Store current iterate if the optimality error is on acceptable
291  // level to restored if things fail later
292  if (CurrentIsAcceptable()) {
293  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
294  "Storing current iterate as backup acceptable point.\n");
295  StoreAcceptablePoint();
296  }
297 
298  // First assume that line search will find an acceptable trial point
299  skipped_line_search_ = false;
300 
301  // Get the search directions (this will store the actual search
302  // direction, possibly including higher order corrections)
303  SmartPtr<IteratesVector> actual_delta;
304 
305  bool goto_resto = false;
306  if (fallback_activated_) {
307  // In this case, the algorithm had trouble to continue and wants
308  // to call the restoration phase immediately if available
309  if (IsValid(resto_phase_)) {
310  goto_resto = true;
311  }
312  else {
313  if (acceptor_->DoFallback()) {
314  in_watchdog_ = false;
315  watchdog_iterate_ = NULL;
316  watchdog_delta_ = NULL;
317  count_successive_shortened_steps_ = 0;
318  watchdog_shortened_iter_ = 0;
319  IpData().Set_info_alpha_primal_char('X');
320  fallback_activated_ = false;
321  return;
322  }
323  else {
324  THROW_EXCEPTION(RESTORATION_FAILED,
325  "We are in an emergency mode, but not restoration phase or other fall back is available.");
326  }
327  }
328  fallback_activated_ = false; // reset the flag
329  }
330  else {
331  // Initialize the acceptor for this backtracking line search
332  acceptor_->InitThisLineSearch(in_watchdog_);
333  actual_delta = IpData().delta()->MakeNewContainer();
334  }
335 
336  if (start_with_resto_) {
337  // If the user requested to start with the restoration phase,
338  // skip the line search and do exactly that. Reset the flag so
339  // that this happens only once.
340  goto_resto = true;
341  start_with_resto_= false;
342  }
343 
344  if (expect_infeasible_problem_ &&
345  Max(IpData().curr()->y_c()->Amax(),
346  IpData().curr()->y_d()->Amax()) > expect_infeasible_problem_ytol_) {
347  goto_resto = true;
348  }
349 
350  bool accept = false;
351  bool corr_taken = false;
352  bool soc_taken = false;
353  Index n_steps = 0;
354  Number alpha_primal = 0.;
355 
356  // Check if search direction becomes too small
357  bool tiny_step = (!goto_resto && DetectTinyStep());
358 
359  if (in_watchdog_ && (goto_resto || tiny_step)) {
360  // If the step could not be computed or is too small and the
361  // watchdog is active, stop the watch dog and resume everything
362  // from reference point
363  StopWatchDog(actual_delta);
364  goto_resto = false;
365  tiny_step = false;
366  }
367 
368  // Check if we want to wake up the watchdog
369  if (watchdog_shortened_iter_trigger_ > 0 &&
370  !in_watchdog_ && !goto_resto && !tiny_step &&
371  !in_soft_resto_phase_ && !expect_infeasible_problem_ &&
372  watchdog_shortened_iter_ >= watchdog_shortened_iter_trigger_) {
373  StartWatchDog();
374  }
375 
376  // Handle the situation of a tiny step
377  if (tiny_step) {
378  alpha_primal =
379  IpCq().curr_primal_frac_to_the_bound(IpData().curr_tau());
380  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
381  "Tiny step detected. Use step size alpha = %e unchecked\n",
382  alpha_primal);
383  IpData().SetTrialPrimalVariablesFromStep(alpha_primal, *IpData().delta()->x(), *IpData().delta()->s());
384 
385  // Evaluate functions at trial point - if that fails, don't use
386  // the tiny step and continue with regular line search
387  try {
388  IpCq().trial_barrier_obj();
389  IpCq().trial_constraint_violation();
390  }
391  catch (IpoptNLP::Eval_Error& e) {
392  e.ReportException(Jnlst(), J_DETAILED);
393  tiny_step = false;
394  }
395 
396  if (tiny_step) {
397  IpData().Set_info_ls_count(0);
398 
399  if (tiny_step_last_iteration_) {
400  IpData().Set_info_alpha_primal_char('T');
401  IpData().Set_tiny_step_flag(true);
402  }
403  }
404  else {
405  IpData().Set_info_alpha_primal_char('t');
406  }
407 
408  // If the step in the dual variables is also small, we remember
409  // that we just did a tiny step so that next time we might
410  // decide to quit
411  Number delta_y_norm = Max(IpData().delta()->y_c()->Amax(),
412  IpData().delta()->y_d()->Amax());
413  if (delta_y_norm < tiny_step_y_tol_) {
414  tiny_step_last_iteration_ = true;
415  }
416  else {
417  tiny_step_last_iteration_ = false;
418  }
419  accept = true;
420  }
421  else {
422  tiny_step_last_iteration_ = false;
423  }
424 
425  if (!goto_resto && !tiny_step) {
426 
427  if (in_soft_resto_phase_) {
428  soft_resto_counter_++;
429  if (soft_resto_counter_ > max_soft_resto_iters_) {
430  accept = false;
431  }
432  else {
433  // If we are currently in the soft restoration phase, continue
434  // that way, and switch back if enough progress is made to the
435  // original criterion (e.g., the filter)
436  bool satisfies_original_criterion = false;
437  // ToDo use tiny_step in TrySoftRestoStep?
438  accept = TrySoftRestoStep(actual_delta,
439  satisfies_original_criterion);
440  if (accept) {
441  IpData().Set_info_alpha_primal_char('s');
442  if (satisfies_original_criterion) {
443  in_soft_resto_phase_ = false;
444  soft_resto_counter_ = 0;
445  IpData().Set_info_alpha_primal_char('S');
446  }
447  }
448  }
449  }
450  else {
451  // Start the backtracking line search
452  bool done = false;
453  bool skip_first_trial_point = false;
454  bool evaluation_error;
455  while (!done) {
456  accept = DoBacktrackingLineSearch(skip_first_trial_point,
457  alpha_primal,
458  corr_taken,
459  soc_taken,
460  n_steps,
461  evaluation_error,
462  actual_delta);
463  DBG_PRINT((1, "evaluation_error = %d\n", evaluation_error));
464  if (in_watchdog_) {
465  if (accept) {
466  in_watchdog_ = false;
467  IpData().Append_info_string("W");
468  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
469  "Watch dog procedure successful!\n");
470  done = true;
471  }
472  else {
473  watchdog_trial_iter_++;
474  if (evaluation_error ||
475  watchdog_trial_iter_ > watchdog_trial_iter_max_) {
476  StopWatchDog(actual_delta);
477  skip_first_trial_point = true;
478  }
479  else {
480  done = true;
481  accept = true;
482  }
483  }
484  }
485  else {
486  done = true;
487  }
488  }
489  } /* else: if (in_soft_resto_phase_) { */
490  } /* if (!goto_resto && !tiny_step) { */
491 
492  // If line search has been aborted because the step size becomes
493  // too small, go to the restoration phase or continue with soft
494  // restoration phase
495  if (!accept) {
496  // If we are not asked to do a rigorous line search, do no call
497  // the restoration phase.
498  if (!rigorous_) {
499  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH, "Skipping call of restoration phase...\n");
500  skipped_line_search_=true;
501  }
502  else {
503  // Check if we should start the soft restoration phase
504  if (!in_soft_resto_phase_
505  && !goto_resto && !expect_infeasible_problem_) {
506  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
507  "--> Starting soft restoration phase <--\n");
508  // Prepare the restoration phase, e.g., augment the filter
509  // with the current point.
510  acceptor_->PrepareRestoPhaseStart();
511 
512  // Try the current search direction for the soft restoration phase
513  bool satisfies_original_criterion;
514  accept = TrySoftRestoStep(actual_delta,
515  satisfies_original_criterion);
516  // If it has been accepted: If the original criterion is also
517  // satisfied, we can just take that step and continue with
518  // the regular algorithm, otherwise we stay in the soft
519  // restoration phase
520  if (accept) {
521  if (satisfies_original_criterion) {
522  IpData().Set_info_alpha_primal_char('S');
523  }
524  else {
525  in_soft_resto_phase_ = true;
526  IpData().Set_info_alpha_primal_char('s');
527  }
528  }
529  }
530 
531  if (!accept) {
532  // Go to the restoration phase
533  if (!in_soft_resto_phase_) {
534  // Prepare the restoration phase, e.g., augment the filter
535  // with the current point. If we are already in the soft
536  // restoration phase, this has been done earlier
537  acceptor_->PrepareRestoPhaseStart();
538  }
539  if (CurrentIsAcceptable()) {
540  THROW_EXCEPTION(ACCEPTABLE_POINT_REACHED,
541  "Restoration phase called at acceptable point.");
542  }
543 
544  if (!IsValid(resto_phase_)) {
545  THROW_EXCEPTION(IpoptException, "No Restoration Phase given to this Backtracking Line Search Object!");
546  }
547  // ToDo make the 1e-2 below a parameter?
548  if (IpCq().curr_constraint_violation()<=
549  1e-2*IpData().tol()) {
550  bool found_acceptable = RestoreAcceptablePoint();
551  if (found_acceptable) {
552  Jnlst().Printf(J_WARNING, J_LINE_SEARCH,
553  "Restoration phase is called at almost feasible point,\n but acceptable point from iteration %d could be restored.\n", acceptable_iteration_number_);
554  THROW_EXCEPTION(ACCEPTABLE_POINT_REACHED,
555  "Restoration phase called at almost feasible point, but acceptable point could be restored.\n");
556  }
557  else {
558  // ToDo does that happen too often?
559  Jnlst().Printf(J_STRONGWARNING, J_LINE_SEARCH,
560  "Restoration phase is called at point that is almost feasible,\n with constraint violation %e. Abort.\n", IpCq().curr_constraint_violation());
561  THROW_EXCEPTION(RESTORATION_FAILED,
562  "Restoration phase called, but point is almost feasible.");
563  }
564  }
565 
566  // Set the info fields for the first output line in the
567  // restoration phase which reflects why the restoration phase
568  // was called
569  IpData().Set_info_alpha_primal(alpha_primal);
570  IpData().Set_info_alpha_dual(0.);
571  IpData().Set_info_alpha_primal_char('R');
572  IpData().Set_info_ls_count(n_steps+1);
573 
574  accept = resto_phase_->PerformRestoration();
575  if (!accept) {
576  bool found_acceptable = RestoreAcceptablePoint();
577  if (found_acceptable) {
578  THROW_EXCEPTION(ACCEPTABLE_POINT_REACHED,
579  "Restoration phase failed, but acceptable point could be restore.\n");
580  }
581  else {
582  THROW_EXCEPTION(RESTORATION_FAILED,
583  "Failed restoration phase!!!");
584  }
585  }
586  count_successive_shortened_steps_ = 0;
587  if (expect_infeasible_problem_) {
588  expect_infeasible_problem_ = false;
589  }
590  in_soft_resto_phase_ = false;
591  soft_resto_counter_ = 0;
592  watchdog_shortened_iter_ = 0;
593  }
594  }
595  }
596  else if (!in_soft_resto_phase_ || tiny_step) {
597  // we didn't do the restoration phase and are now updating the
598  // dual variables of the trial point
599  Number alpha_dual_max =
600  IpCq().dual_frac_to_the_bound(IpData().curr_tau(),
601  *actual_delta->z_L(), *actual_delta->z_U(),
602  *actual_delta->v_L(), *actual_delta->v_U());
603 
604  PerformDualStep(alpha_primal, alpha_dual_max, actual_delta);
605 
606  if (n_steps==0) {
607  // accepted this if a full step was
608  // taken
609  count_successive_shortened_steps_ = 0;
610  watchdog_shortened_iter_ = 0;
611  }
612  else {
613  count_successive_shortened_steps_++;
614  watchdog_shortened_iter_++;
615  }
616 
617  if (expect_infeasible_problem_ &&
618  IpCq().curr_constraint_violation() <= expect_infeasible_problem_ctol_) {

620  "Constraint violation is with %e less than expect_infeasible_problem_ctol.\nDisable expect_infeasible_problem_heuristic.\n", IpCq().curr_constraint_violation());

622  }
623

625  // case we skip the usual ending stuff
626  if (acceptor_->RestoredIterate()) {
627  in_watchdog_ = false;
628  watchdog_iterate_ = NULL;
629  watchdog_delta_ = NULL;
630  count_successive_shortened_steps_ = 0;
631  watchdog_shortened_iter_ = 0;
632  IpData().Set_info_alpha_primal_char('r');
633  }

637  bool BacktrackingLineSearch::DoBacktrackingLineSearch(bool skip_first_trial_point,
638  Number& alpha_primal,
639  bool& corr_taken,
640  bool& soc_taken,
641  Index& n_steps,
642  bool& evaluation_error,
643  SmartPtr<IteratesVector>& actual_delta)
644  {
645  evaluation_error = false;
646  bool accept = false;
647 
648  DBG_START_METH("BacktrackingLineSearch::DoBacktrackingLineSearch",
649  dbg_verbosity);
650 
651  // Compute primal fraction-to-the-boundary value
652  Number alpha_primal_max =
653  IpCq().primal_frac_to_the_bound(IpData().curr_tau(),
654  *actual_delta->x(),
655  *actual_delta->s());
656 
657  // Compute smallest step size allowed
658  Number alpha_min = alpha_primal_max;
659  if (!in_watchdog_) {
660  alpha_min = acceptor_->CalculateAlphaMin();
661  }
662  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
663  "minimal step size ALPHA_MIN = %E\n", alpha_min);
664 
665  // Start line search from maximal step size
666  alpha_primal = alpha_primal_max;
667 
668  // Step size used in ftype and armijo tests
669  Number alpha_primal_test = alpha_primal;
670  if (in_watchdog_) {
671  alpha_primal_test = watchdog_alpha_primal_test_;
672  }
673 
674  if (skip_first_trial_point) {
675  alpha_primal *= alpha_red_factor_;
676  }
677 
678  if (!skip_first_trial_point) {
679  // Before we do the actual backtracking line search for the
680  // regular primal-dual search direction, let's see if a step
681  // including a higher-order correctior is already acceptable
682  accept = acceptor_->TryCorrector(alpha_primal_test,
683  alpha_primal,
684  actual_delta);
685  }
686  if (accept) {
687  corr_taken = true;
688  }
689 
690  if (!accept) {
691  // Loop over decreaseing step sizes until acceptable point is
692  // found or until step size becomes too small
693 
694  while (alpha_primal>alpha_min ||
695  n_steps == 0 ) {
696  // always allow the "full" step if it is
697  // acceptable (even if alpha_primal<=alpha_min)
698  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
699  "Starting checks for alpha (primal) = %8.2e\n",
700  alpha_primal);
701 
702  try {
703  // Compute the primal trial point
704  IpData().SetTrialPrimalVariablesFromStep(alpha_primal, *actual_delta->x(), *actual_delta->s());
705 
706  if (magic_steps_) {
707  PerformMagicStep();
708  }
709 
710  // If it is acceptable, stop the search
711  alpha_primal_test = alpha_primal;
712  if (accept_every_trial_step_ ||
713  (accept_after_max_steps_!=-1 &&
714  n_steps >= accept_after_max_steps_)) {
715  // We call the evaluation at the trial point here, so that an
716  // exception will the thrown if there are problem during the
717  // evaluation of the functions (in that case, we want to further
718  // reduce the step size
719  IpCq().trial_barrier_obj();
720  IpCq().trial_constraint_violation();
721  IpData().Append_info_string("MaxS");
722  Reset();
723  accept = true;
724  }
725  else {
726  accept = acceptor_->CheckAcceptabilityOfTrialPoint(alpha_primal_test);
727  }
728  }
729  catch (IpoptNLP::Eval_Error& e) {
730  e.ReportException(Jnlst(), J_DETAILED);
731  Jnlst().Printf(J_WARNING, J_LINE_SEARCH,
732  "Warning: Cutting back alpha due to evaluation error\n");
733  IpData().Append_info_string("e");
734  accept = false;
735  evaluation_error = true;
736  }
737 
738  if (accept) {
739  break;
740  }
741 
742  if (in_watchdog_) {
743  break;
744  }
745 
746  // Decide if we want to go to the restoration phase in a
747  // short cut to check if the problem is infeasible
748  if (expect_infeasible_problem_) {
749  if (count_successive_shortened_steps_>=5) {
750  break;
751  }
752  }
753 
754  // try second order correction step if the function could
755  // be evaluated
756  // DoTo: check if we want to do SOC when watchdog is active
757  if (!evaluation_error) {
758  Number theta_curr = IpCq().curr_constraint_violation();
759  Number theta_trial = IpCq().trial_constraint_violation();
760  if (alpha_primal==alpha_primal_max && // i.e. first trial point
761  theta_curr<=theta_trial) {
762  // Try second order correction
763  accept = acceptor_->TrySecondOrderCorrection(alpha_primal_test,
764  alpha_primal,
765  actual_delta);
766  }
767  if (accept) {
768  soc_taken = true;
769  break;
770  }
771  }
772 
773  // Point is not yet acceptable, try a shorter one
774  alpha_primal *= alpha_red_factor_;
775  n_steps++;
776  }
777  } /* if (!accept) */
778 
779  char info_alpha_primal_char='?';
780  if (!accept && in_watchdog_) {
781  info_alpha_primal_char = 'w';
782  }
783  else if (accept) {
784  info_alpha_primal_char =
785  acceptor_->UpdateForNextIteration(alpha_primal_test);
786  }
787  if (soc_taken) {
788  info_alpha_primal_char = (char)toupper(info_alpha_primal_char);
789  }
790  IpData().Set_info_alpha_primal_char(info_alpha_primal_char);
791  IpData().Set_info_ls_count(n_steps+1);
792  if (corr_taken) {
793  IpData().Append_info_string("C");
794  }
795 
796  return accept;
797  }

852  void BacktrackingLineSearch::PerformDualStep(Number alpha_primal,
853  Number alpha_dual,
854  SmartPtr<IteratesVector>& delta)
855  {
856  DBG_START_FUN("BacktrackingLineSearch::PerformDualStep", dbg_verbosity);
857 
858  // set the bound multipliers from the step
859  IpData().SetTrialBoundMultipliersFromStep(alpha_dual, *delta->z_L(), *delta->z_U(), *delta->v_L(), *delta->v_U());
860 
861  Number alpha_y=-1.;
862  switch (alpha_for_y_) {
863  case LSACCEPTOR_ALPHA_FOR_Y:
864  alpha_y = acceptor_->ComputeAlphaForY(alpha_primal, alpha_dual, delta);
865  break;
866  case PRIMAL_ALPHA_FOR_Y:
867  case PRIMAL_AND_FULL_ALPHA_FOR_Y:
868  alpha_y = alpha_primal;
869  if (alpha_for_y_ == PRIMAL_AND_FULL_ALPHA_FOR_Y) {
870  Number dxnorm = Max(delta->x()->Amax(), delta->s()->Amax());
871  if (dxnorm <= alpha_for_y_tol_) {
872  alpha_y = 1.;
873  }
874  }
875  break;
876  case DUAL_ALPHA_FOR_Y:
877  case DUAL_AND_FULL_ALPHA_FOR_Y:
878  alpha_y = alpha_dual;
879  if (alpha_for_y_ == DUAL_AND_FULL_ALPHA_FOR_Y) {
880  Number dxnorm = Max(delta->x()->Amax(), delta->s()->Amax());
881  if (dxnorm <= alpha_for_y_tol_) {
882  alpha_y = 1.;
883  }
884  }
885  break;
886  case MIN_ALPHA_FOR_Y:
887  alpha_y = Min(alpha_dual, alpha_primal);
888  break;
889  case MAX_ALPHA_FOR_Y:
890  alpha_y = Max(alpha_dual, alpha_primal);
891  break;
892  case FULL_STEP_FOR_Y:
893  alpha_y = 1;
894  break;
895  case MIN_DUAL_INFEAS_ALPHA_FOR_Y:
896  case SAFE_MIN_DUAL_INFEAS_ALPHA_FOR_Y:
897  // Here we compute the step size for y so that the dual
898  // infeasibility is minimized along delta_y
899 
900  // compute the dual infeasibility at new point with old y
901  SmartPtr<IteratesVector> temp_trial
902  = IpData().trial()->MakeNewContainer();
903  temp_trial->Set_y_c(*IpData().curr()->y_c());
904  temp_trial->Set_y_d(*IpData().curr()->y_d());
905  IpData().set_trial(temp_trial);
906  SmartPtr<const Vector> dual_inf_x = IpCq().trial_grad_lag_x();
907  SmartPtr<const Vector> dual_inf_s = IpCq().trial_grad_lag_s();
908 
909  SmartPtr<Vector> new_jac_times_delta_y =
910  IpData().curr()->x()->MakeNew();
911  new_jac_times_delta_y->AddTwoVectors(1., *IpCq().trial_jac_cT_times_vec(*delta->y_c()),
912  1., *IpCq().trial_jac_dT_times_vec(*delta->y_d()),
913  0.);
914 
915  Number a = pow(new_jac_times_delta_y->Nrm2(), 2.) +
916  pow(delta->y_d()->Nrm2(), 2.);
917  Number b = dual_inf_x->Dot(*new_jac_times_delta_y) -
918  dual_inf_s->Dot(*delta->y_d());
919 
920  Number alpha = - b/a;
921 
922  if (alpha_for_y_==SAFE_MIN_DUAL_INFEAS_ALPHA_FOR_Y) {
923  alpha_y = Min(Max(alpha_primal, alpha_dual), Max(alpha, Min(alpha_primal, alpha_dual)));
924  }
925  else {
926  alpha_y = Min(1., Max(0., alpha));
927  }
928  break;
929  } //switch (alpha_for_y)
930 
931  // Set the eq multipliers from the step now that alpha_y
932  // has been calculated.
933  DBG_PRINT((1, "alpha_y = %e\n", alpha_y));
934  DBG_PRINT_VECTOR(2, "delta_y_c", *delta->y_c());
935  DBG_PRINT_VECTOR(2, "delta_y_d", *delta->y_d());
936  IpData().SetTrialEqMultipliersFromStep(alpha_y, *delta->y_c(), *delta->y_d());
937 
938  // Set some information for iteration summary output
939  IpData().Set_info_alpha_primal(alpha_primal);
940  IpData().Set_info_alpha_dual(alpha_dual);
941  }

944  BacktrackingLineSearch::PerformMagicStep()

950  IpCq().trial_barrier_obj(),

955  DBG_PRINT_VECTOR(2, "slack_s_U in", *IpCq().trial_slack_s_U());
956 
957  SmartPtr<const Vector> d_L = IpNLP().d_L();
958  SmartPtr<const Matrix> Pd_L = IpNLP().Pd_L();
959  SmartPtr<Vector> delta_s_magic_L = d_L->MakeNew();
960  delta_s_magic_L->Set(0.);
961  SmartPtr<Vector> tmp = d_L->MakeNew();
962  Pd_L->TransMultVector(1., *IpCq().trial_d_minus_s(), 0., *tmp);
963  delta_s_magic_L->ElementWiseMax(*tmp);
964 
965  SmartPtr<const Vector> d_U = IpNLP().d_U();
966  SmartPtr<const Matrix> Pd_U = IpNLP().Pd_U();
967  SmartPtr<Vector> delta_s_magic_U = d_U->MakeNew();
968  delta_s_magic_U->Set(0.);
969  tmp = d_U->MakeNew();
970  Pd_U->TransMultVector(1., *IpCq().trial_d_minus_s(), 0., *tmp);
971  delta_s_magic_U->ElementWiseMin(*tmp);
972 
973  SmartPtr<Vector> delta_s_magic = IpData().trial()->s()->MakeNew();
974  Pd_L->MultVector(1., *delta_s_magic_L, 0., *delta_s_magic);
975  Pd_U->MultVector(1., *delta_s_magic_U, 1., *delta_s_magic);
976  delta_s_magic_L = NULL; // free memory
977  delta_s_magic_U = NULL; // free memory
978 
979  // Now find those entries with both lower and upper bounds, there
980  // the step is too large
981  // ToDo this should only be done if there are inequality
982  // constraints with two bounds
983  // also this can be done in a smaller space (d_L or d_U whichever
984  // is smaller)
985  tmp = delta_s_magic->MakeNew();
986  tmp->Copy(*IpData().trial()->s());
987  Pd_L->MultVector(1., *d_L, -2., *tmp);
988  Pd_U->MultVector(1., *d_U, 1., *tmp);
989  SmartPtr<Vector> tmp2 = tmp->MakeNew();
990  tmp2->Copy(*tmp);
991  tmp2->ElementWiseAbs();
992  tmp->Axpy(-2., *delta_s_magic);
993  tmp->ElementWiseAbs();
994  // now, tmp2 = |d_L + d_u - 2*s| and tmp = |d_L + d_u - 2*(s+Delta s)|
995  // we want to throw out those for which tmp2 > tmp
996  tmp->Axpy(-1., *tmp2);
997  tmp->ElementWiseSgn();
998  tmp2->Set(0.);
999  tmp2->ElementWiseMax(*tmp);
1000  tmp = d_L->MakeNew();
1001  Pd_L->TransMultVector(1., *tmp2, 0., *tmp);
1002  Pd_L->MultVector(1., *tmp, 0., *tmp2);
1003  tmp = d_U->MakeNew();
1004  Pd_U->TransMultVector(1., *tmp2, 0., *tmp);
1005  Pd_U->MultVector(1., *tmp, 0., *tmp2);
1006  DBG_PRINT_VECTOR(2, "tmp indicator", *tmp2)
1007  // tmp2 now is one for those entries with both bounds, for which
1008  // no step should be taken
1009 
1010  tmp = delta_s_magic->MakeNew();
1011  tmp->Copy(*delta_s_magic);
1012  tmp->ElementWiseMultiply(*tmp2);
1013  delta_s_magic->Axpy(-1., *tmp);
1014 
1015  Number delta_s_magic_max = delta_s_magic->Amax();
1016  Number mach_eps = std::numeric_limits<Number>::epsilon();
1017  if (delta_s_magic_max>0.) {
1018  if (delta_s_magic_max > 10*mach_eps*IpData().trial()->s()->Amax()) {
1019  IpData().Append_info_string("M");
1020  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH, "Magic step with max-norm %.6e taken.\n", delta_s_magic->Amax());
1021  delta_s_magic->Print(Jnlst(), J_MOREVECTOR, J_LINE_SEARCH,
1022  "delta_s_magic");
1023  }
1024 
1025  // now finally compute the new overall slacks
1026  delta_s_magic->Axpy(1., *IpData().trial()->s());
1027  SmartPtr<IteratesVector> trial = IpData().trial()->MakeNewContainer();
1028  trial->Set_s(*delta_s_magic);
1029 
1030  // also update the set in the dual variables
1031 
1032 
1033  IpData().set_trial(trial);
1034  }
1035 
1036  DBG_PRINT((1,"Outgoing barr = %e and constrviol %e\n", IpCq().trial_barrier_obj(), IpCq().trial_constraint_violation()));
1037  DBG_PRINT_VECTOR(2, "s out", *IpData().trial()->s());
1038  DBG_PRINT_VECTOR(2, "d minus s out", *IpCq().trial_d_minus_s());
1039  DBG_PRINT_VECTOR(2, "slack_s_L out", *IpCq().trial_slack_s_L());

1043  bool BacktrackingLineSearch::TrySoftRestoStep(SmartPtr<IteratesVector>& actual_delta,
1044  bool &satisfies_original_criterion)
1045  {
1046  DBG_START_FUN("BacktrackingLineSearch::TrySoftRestoStep", dbg_verbosity);
1047 
1048  if (soft_resto_pderror_reduction_factor_==0.) {
1049  return false;
1050  }
1051 
1052  satisfies_original_criterion = false;
1053 
1054  // ToDo: Need to decide if we want to try a corrector step first
1055 
1056  // Compute the maximal step sizes (we use identical step sizes for
1057  // primal and dual variables
1058  Number alpha_primal_max =
1059  IpCq().primal_frac_to_the_bound(IpData().curr_tau(),
1060  *actual_delta->x(),
1061  *actual_delta->s());
1062  Number alpha_dual_max =
1063  IpCq().dual_frac_to_the_bound(IpData().curr_tau(),
1064  *actual_delta->z_L(),
1065  *actual_delta->z_U(),
1066  *actual_delta->v_L(),
1067  *actual_delta->v_U());
1068  Number alpha = Min(alpha_primal_max, alpha_dual_max);
1069 
1070  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
1071  "Trying soft restoration phase step with step length %13.6e\n",
1072  alpha);
1073 
1074  // We allow up to three trials in case there is an evaluation
1075  // error for the functions
1076  bool done=false;
1077  Index count=3;
1078  while (!done && count>0) {
1079  // Set the trial point
1080  IpData().SetTrialPrimalVariablesFromStep(alpha, *actual_delta->x(), *actual_delta->s());
1081  PerformDualStep(alpha, alpha, actual_delta);
1082 
1083  // Check if that point is acceptable with respect to the current
1084  // original filter
1085  try {
1086  IpCq().trial_barrier_obj();
1087  IpCq().trial_constraint_violation();
1088  done=true;
1089  }
1090  catch (IpoptNLP::Eval_Error& e) {
1091  e.ReportException(Jnlst(), J_DETAILED);
1092  Jnlst().Printf(J_WARNING, J_LINE_SEARCH,
1093  "Warning: Evaluation error during soft restoration phase step.\n");
1094  IpData().Append_info_string("e");
1095  count--;
1096  }
1097  }
1098  if (!done) {
1099  return false;
1100  }
1101 
1102  if (acceptor_->CheckAcceptabilityOfTrialPoint(0.)) {
1103  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
1104  " Trial step acceptable with respect to original backtracking globalization.\n");
1105  satisfies_original_criterion = true;
1106  return true;
1107  }
1108 
1109  // Evaluate the optimality error at the new point
1110  Number mu = .0;
1111  if (!IpData().FreeMuMode()) {
1112  mu = IpData().curr_mu();
1113  }
1114  Number trial_pderror;
1115  Number curr_pderror;
1116  try {
1117  trial_pderror = IpCq().trial_primal_dual_system_error(mu);
1118  curr_pderror = IpCq().curr_primal_dual_system_error(mu);
1119  }
1120  catch (IpoptNLP::Eval_Error& e) {
1121  e.ReportException(Jnlst(), J_DETAILED);
1122  Jnlst().Printf(J_WARNING, J_LINE_SEARCH,
1123  "Warning: Evaluation error during soft restoration phase step.\n");
1124  IpData().Append_info_string("e");
1125  return false;
1126  }
1127 
1128  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
1129  " Primal-dual error at current point: %23.16e\n", curr_pderror);
1130  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
1131  " Primal-dual error at trial point : %23.16e\n", trial_pderror);
1132  // Check if there is sufficient reduction in the optimality error
1133  if (trial_pderror <= soft_resto_pderror_reduction_factor_*curr_pderror) {
1134  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
1135  " Trial step accepted.\n");
1136  return true;
1137  }
1138 
1139  Jnlst().Printf(J_DETAILED, J_LINE_SEARCH,
1140  " Trial step rejected.\n");
1141  return false;
1142  }
