58 static inline const char* skipObjectRest(const char* str, int first)
59 {
60  str=skipSpace(str);
61  while (*str != '}') {
62  if (!first) {
63  if (*str != ',') {
64  fprintf(stderr, "JSON object expected ',' or '}', got: %.20s\n", str);
65  abort();
66  }
67  str++;
68  } else {
69  first = 0;
70  }
71  str = skipValue(str);
72  str = skipSpace(str);
73  if (*str++ != ':') {
74  fprintf(stderr, "JSON object expected ':', got: %.20s\n", str);
75  abort();
76  }
77  str = skipValue(str);
78  str = skipSpace(str);
79  }
80  return str+1;
81 }

83 static const char* skipValue(const char* str)
84 {
85  str = skipSpace(str);
86  switch (*str) {
87  case '{':
88  {
89  str = skipObjectRest(str+1,1);
90  return str;
91  }
92  case '[':
93  {
94  int first = 1;
95  str = skipSpace(str+1);
96  while (*str != ']') {
97  if (!first && *str++ != ',') {
98  fprintf(stderr, "JSON array expected ',' or ']', got: %.20s\n", str);
99  abort();
100  }
101  first = 0;
102  str = skipValue(str);
103  str = skipSpace(str);
104  }
105  return str+1;
106  }
107  case '"':
108  str++;
109  do {
110  switch (*str) {
111  case '\0': fprintf(stderr, "Found end of file, expected end of string"); abort();
112  case '\\':
113  if (str+1 == '\0') {
114  fprintf(stderr, "Found end of file, expected end of string"); abort();
115  }
116  str+=2;
117  break;
118  case '"':
119  return str+1;
120  default:
121  str++;
122  }
123  } while (1);
124  abort();
125  case '-':
126  case '0':
127  case '1':
128  case '2':
129  case '3':
130  case '4':
131  case '5':
132  case '6':
133  case '7':
134  case '8':
135  case '9':
136  {
137  char *endptr = NULL;
138  strtod(str,&endptr);
139  if (str == endptr) {
140  fprintf(stderr, "Not a number, got %.20s\n", str);
141  abort();
142  }
143  return endptr;
144  }
145  default:
146  fprintf(stderr, "JSON value expected, got: %.20s\n", str);
147  abort();
148  }
149 }

190 static inline const char *skipFieldIfExist(const char *str,const char *name)
191 {
192  const char *s = str;
193  int len = strlen(name);
194  if (*s != ',') {
195  return str;
196  }
197  s++;
198  if (*s != '\"' || strncmp(s+1,name,len)) {
199  return str;
200  }
201  s += len + 1;
202  if (strncmp("\":", s, 2)) {
203  return str;
204  }
205  s += 2;
206  s = skipSpace(s);
207  s = skipValue(s);
208  s = skipSpace(s);
209  s = skipSpace(s);
210  return s;
211 }

213 static const char* readEquation(const char *str,EQUATION_INFO *xml,int i)
214 {
215  int n=0,j;
216  const char *str2;
217  str=assertChar(str,'{');
218  str=assertStringValue(str,"eqIndex");
219  str=assertChar(str,':');
220  str=assertNumber(str,i);
221  str=skipSpace(str);
222  xml->id = i;
223  str = skipFieldIfExist(str, "parent");
224  str = skipFieldIfExist(str, "section");
225  if ((measure_time_flag & 1) && 0==strncmp(",\"tag\":\"system\"", str, 15)) {
226  xml->profileBlockIndex = -1;
227  str += 15;
228  } else if ((measure_time_flag & 1) && 0==strncmp(",\"tag\":\"tornsystem\"", str, 19)) {
229  xml->profileBlockIndex = -1;
230  str += 19;
231  } else {
232  xml->profileBlockIndex = 0;
233  }
234  str = skipFieldIfExist(str, "tag");
235  str = skipFieldIfExist(str, "display");
236  str = skipFieldIfExist(str, "unknowns");
237  if (strncmp(",\"defines\":[", str, 12)) {
238  xml->numVar = 0;
239  xml->vars = 0;
240  str = skipObjectRest(str,0);
241  return str;
242  }
243  str += 12;
244  str = skipSpace(str);
245  if (*str == ']') {
246  xml->numVar = 0;
247  xml->vars = 0;
248  return skipObjectRest(str-1,0);
249  }
250  str2 = skipSpace(str);
251  while (1) {
252  str=skipValue(str);
253  n++;
254  str=skipSpace(str);
255  if (*str != ',') {
256  break;
257  }
258  str++;
259  };
260  assertChar(str, ']');
261  xml->numVar = n;
262  xml->vars = malloc(sizeof(const char*)*n);
263  str = str2;
264  for (j=0; j<n; j++) {
265  const char *str3 = skipSpace(str);
266  char *tmp;
267  int len=0;
268  str = assertChar(str, '\"');
269  while (*str != '\"' && *str) {
270  len++;
271  str++;
272  }
273  str = assertChar(str, '\"');
274  tmp = malloc(len+1);
275  strncpy(tmp, str3+1, len);
276  tmp[len] = '\0';
277  xml->vars[j] = tmp;
278  if (j != n-1) {
279  str = assertChar(str, ',');
280  }
281  }
282  str = assertChar(skipSpace(str), ']');
283  return skipObjectRest(str,0);
284 }

287 {
