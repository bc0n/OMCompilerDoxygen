54 /*---------------------------------------------------------------
55  ARKBand:
56 
57  This routine initializes the memory record and sets various function
58  fields specific to the band linear solver module. ARKBand first calls
59  the existing lfree routine if this is not NULL. It then sets the
60  ark_linit, ark_lsetup, ark_lsolve, and ark_lfree fields in (*arkode_mem)
61  to be arkBandInit, arkBandSetup, arkBandSolve, and arkBandFree,
62  respectively. It allocates memory for a structure of type
63  ARKDlsMemRec and sets the ark_lmem field in (*arkode_mem) to the
64  address of this structure. It sets setupNonNull in (*arkode_mem) to be
65  TRUE, d_mu to be mupper, d_ml to be mlower, and the d_jac field to be
66  arkDlsBandDQJac. Finally, it allocates memory for M, savedJ, and pivot.
67  The ARKBand return value is SUCCESS = 0, LMEM_FAIL = -1, or
68  LIN_ILL_INPUT = -2.
69 
70  NOTE: The band linear solver assumes a serial implementation
71  of the NVECTOR package. Therefore, ARKBand will first
72  test for compatible a compatible N_Vector internal
73  representation by checking that the function
74  N_VGetArrayPointer exists.
75 ---------------------------------------------------------------*/
76 int ARKBand(void *arkode_mem, long int N, long int mupper, long int mlower)
77 {
78  ARKodeMem ark_mem;
79  ARKDlsMem arkdls_mem;
80 
81  /* Return immediately if arkode_mem is NULL */
82  if (arkode_mem == NULL) {
83  arkProcessError(NULL, ARKDLS_MEM_NULL, "ARKBAND", "ARKBand", MSGD_ARKMEM_NULL);
84  return(ARKDLS_MEM_NULL);
85  }
86  ark_mem = (ARKodeMem) arkode_mem;
87 
88  /* Test if the NVECTOR package is compatible with the BAND solver */
89  if (ark_mem->ark_tempv->ops->nvgetarraypointer == NULL) {
90  arkProcessError(ark_mem, ARKDLS_ILL_INPUT, "ARKBAND", "ARKBand", MSGD_BAD_NVECTOR);
91  return(ARKDLS_ILL_INPUT);
92  }
93 
94  if (ark_mem->ark_lfree != NULL) ark_mem->ark_lfree(ark_mem);
95 
96  /* Set four main function fields in ark_mem */
97  ark_mem->ark_linit = arkBandInit;
98  ark_mem->ark_lsetup = arkBandSetup;
99  ark_mem->ark_lsolve = arkBandSolve;
100  ark_mem->ark_lfree = arkBandFree;
101  ark_mem->ark_lsolve_type = 2;
102 
103  /* Get memory for ARKDlsMemRec */
104  arkdls_mem = NULL;
105  arkdls_mem = (ARKDlsMem) malloc(sizeof(struct ARKDlsMemRec));
106  if (arkdls_mem == NULL) {
107  arkProcessError(ark_mem, ARKDLS_MEM_FAIL, "ARKBAND", "ARKBand", MSGD_MEM_FAIL);
108  return(ARKDLS_MEM_FAIL);
109  }
110 
111  /* Set matrix type */
112  arkdls_mem->d_type = SUNDIALS_BAND;
113 
114  /* Initialize Jacobian-related data */
115  arkdls_mem->d_jacDQ = TRUE;
116  arkdls_mem->d_bjac = NULL;
117  arkdls_mem->d_J_data = NULL;
118  arkdls_mem->d_last_flag = ARKDLS_SUCCESS;
119  ark_mem->ark_setupNonNull = TRUE;
120 
121  /* Initialize counters */
122  arkdls_mem->d_nje = 0;
123  arkdls_mem->d_nfeDQ = 0;
124  arkdls_mem->d_nstlj = 0;
125 
126  /* Load problem dimension */
127  arkdls_mem->d_n = N;
128 
129  /* Load half-bandwiths in arkdls_mem */
130  arkdls_mem->d_ml = mlower;
131  arkdls_mem->d_mu = mupper;
132 
133  /* Test ml and mu for legality */
134  if ((arkdls_mem->d_ml < 0) || (arkdls_mem->d_mu < 0) || (arkdls_mem->d_ml >= N) || (arkdls_mem->d_mu >= N)) {
135  arkProcessError(ark_mem, ARKDLS_ILL_INPUT, "ARKBAND", "ARKBand", MSGD_BAD_SIZES);
136  free(arkdls_mem); arkdls_mem = NULL;
137  return(ARKDLS_ILL_INPUT);
138  }
139 
140  /* Set extended upper half-bandwith for M (required for pivoting) */
141  arkdls_mem->d_smu = SUNMIN(N-1, arkdls_mem->d_mu + arkdls_mem->d_ml);
142 
143  /* Allocate memory for M, savedJ, and pivot arrays */
144  arkdls_mem->d_M = NULL;
145  arkdls_mem->d_M = NewBandMat(N, arkdls_mem->d_mu, arkdls_mem->d_ml, arkdls_mem->d_smu);
146  if (arkdls_mem->d_M == NULL) {
147  arkProcessError(ark_mem, ARKDLS_MEM_FAIL, "ARKBAND", "ARKBand", MSGD_MEM_FAIL);
148  free(arkdls_mem); arkdls_mem = NULL;
149  return(ARKDLS_MEM_FAIL);
150  }
151  arkdls_mem->d_savedJ = NULL;
152  arkdls_mem->d_savedJ = NewBandMat(N, arkdls_mem->d_mu, arkdls_mem->d_ml, arkdls_mem->d_mu);
153  if (arkdls_mem->d_savedJ == NULL) {
154  arkProcessError(ark_mem, ARKDLS_MEM_FAIL, "ARKBAND", "ARKBand", MSGD_MEM_FAIL);
155  DestroyMat(arkdls_mem->d_M);
156  free(arkdls_mem); arkdls_mem = NULL;
157  return(ARKDLS_MEM_FAIL);
158  }
159  arkdls_mem->d_lpivots = NULL;
160  arkdls_mem->d_lpivots = NewLintArray(N);
161  if (arkdls_mem->d_lpivots == NULL) {
162  arkProcessError(ark_mem, ARKDLS_MEM_FAIL, "ARKBAND", "ARKBand", MSGD_MEM_FAIL);
163  DestroyMat(arkdls_mem->d_M);
164  DestroyMat(arkdls_mem->d_savedJ);
165  free(arkdls_mem); arkdls_mem = NULL;
166  return(ARKDLS_MEM_FAIL);
167  }
168 
169  /* Attach linear solver memory to integrator memory */
170  ark_mem->ark_lmem = arkdls_mem;
171 
172  return(ARKDLS_SUCCESS);
173 }

203 /*---------------------------------------------------------------
204  arkBandSetup:
205 
206  This routine does the setup operations for the band linear
207  solver. It makes a decision whether or not to call the Jacobian
208  evaluation routine based on various state variables, and if not
209  it uses the saved copy. In any case, it constructs the Newton
210  matrix A = M - gamma*J, updates counters, and calls the band
211  LU factorization routine.
212 ---------------------------------------------------------------*/
213 static int arkBandSetup(ARKodeMem ark_mem, int convfail,
214  N_Vector ypred, N_Vector fpred,
215  booleantype *jcurPtr, N_Vector vtemp1,
216  N_Vector vtemp2, N_Vector vtemp3)
217 {
218  booleantype jbad, jok;
219  realtype dgamma;
220  long int i, j, ier, ml, mu, N, M, is, ie;
221  DlsMat A, Mass;
222  ARKDlsMem arkdls_mem;
223  ARKDlsMassMem arkdls_mass_mem;
224  int retval;
225 
226  arkdls_mem = (ARKDlsMem) ark_mem->ark_lmem;
227 
228  /* Use nst, gamma/gammap, and convfail to set J eval. flag jok */
229  dgamma = SUNRabs((ark_mem->ark_gamma/ark_mem->ark_gammap) - ONE);
230  jbad = (ark_mem->ark_nst == 0) ||
231  (ark_mem->ark_nst > arkdls_mem->d_nstlj + ARKD_MSBJ) ||
232  ((convfail == ARK_FAIL_BAD_J) && (dgamma < ARKD_DGMAX)) ||
233  (convfail == ARK_FAIL_OTHER);
234  jok = !jbad;
235 
236  if (jok) {
237 
238  /* If jok = TRUE, use saved copy of J */
239  *jcurPtr = FALSE;
240  BandCopy(arkdls_mem->d_savedJ, arkdls_mem->d_M, arkdls_mem->d_mu, arkdls_mem->d_ml);
241 
242  } else {
243 
244  /* If jok = FALSE, call jac routine for new J value */
245  arkdls_mem->d_nje++;
246  arkdls_mem->d_nstlj = ark_mem->ark_nst;
247  *jcurPtr = TRUE;
248  SetToZero(arkdls_mem->d_M);
249 
250  retval = arkdls_mem->d_bjac(arkdls_mem->d_n, arkdls_mem->d_mu, arkdls_mem->d_ml,
251  ark_mem->ark_tn, ypred, fpred, arkdls_mem->d_M,
252  arkdls_mem->d_J_data, vtemp1, vtemp2, vtemp3);
253  if (retval < 0) {
254  arkProcessError(ark_mem, ARKDLS_JACFUNC_UNRECVR, "ARKBAND", "arkBandSetup", MSGD_JACFUNC_FAILED);
255  arkdls_mem->d_last_flag = ARKDLS_JACFUNC_UNRECVR;
256  return(-1);
257  }
258  if (retval > 0) {
259  arkdls_mem->d_last_flag = ARKDLS_JACFUNC_RECVR;
260  return(1);
261  }
262 
263  BandCopy(arkdls_mem->d_M, arkdls_mem->d_savedJ, arkdls_mem->d_mu, arkdls_mem->d_ml);
264 
265  }
266 
267  /* Scale J by -gamma */
268  BandScale(-ark_mem->ark_gamma, arkdls_mem->d_M);
269 
270  /* Add mass matrix to get A = M-gamma*J*/
271  if (ark_mem->ark_mass_matrix) {
272 
273  /* Compute mass matrix */
274  arkdls_mass_mem = (ARKDlsMassMem) ark_mem->ark_mass_mem;
275  SetToZero(arkdls_mass_mem->d_M);
276  retval = arkdls_mass_mem->d_bmass(arkdls_mass_mem->d_n, arkdls_mass_mem->d_mu,
277  arkdls_mass_mem->d_ml, ark_mem->ark_tn,
278  arkdls_mass_mem->d_M, arkdls_mass_mem->d_M_data,
279  vtemp1, vtemp2, vtemp3);
280  arkdls_mass_mem->d_nme++;
281  if (retval < 0) {
282  arkProcessError(ark_mem, ARKDLS_MASSFUNC_UNRECVR, "ARKBAND",
283  "arkBandSetup", MSGD_MASSFUNC_FAILED);
284  arkdls_mem->d_last_flag = ARKDLS_MASSFUNC_UNRECVR;
285  return(-1);
286  }
287  if (retval > 0) {
288  arkdls_mem->d_last_flag = ARKDLS_MASSFUNC_RECVR;
289  return(1);
290  }
291 
292  /* perform matrix sum */
293  ml = arkdls_mem->d_M->ml;
294  mu = arkdls_mem->d_M->mu;
295  N = arkdls_mem->d_M->N;
296  M = arkdls_mem->d_M->M;
297  A = arkdls_mem->d_M;
298  Mass = arkdls_mass_mem->d_M;
299  for (j=0; j<N; j++) { /* loop over columns */
300  is = (0 > j-mu) ? 0 : j-mu; /* colum nonzero bounds */
301  ie = (M-1 < j+ml) ? M-1 : j+ml;
302  for (i=is; i<=ie; i++) { /* loop over rows */
303  BAND_ELEM(A,i,j) += BAND_ELEM(Mass,i,j);
304  }
305  }
306 
307  } else {
308  AddIdentity(arkdls_mem->d_M);
309  }
310 
311  /* Do LU factorization of M */
312  ier = BandGBTRF(arkdls_mem->d_M, arkdls_mem->d_lpivots);
313 
314  /* Return 0 if the LU was complete; otherwise return 1 */
315  if (ier > 0) {
316  arkdls_mem->d_last_flag = ier;
317  return(1);
318  }
319  arkdls_mem->d_last_flag = ARKDLS_SUCCESS;
320  return(0);
321 }

324 /*---------------------------------------------------------------
325  arkBandSolve:
326 
327  This routine handles the solve operation for the band linear solver
328  by calling the band backsolve routine. The return value is 0.
329 ---------------------------------------------------------------*/
330 static int arkBandSolve(ARKodeMem ark_mem, N_Vector b, N_Vector weight,
331  N_Vector ycur, N_Vector fcur)
332 {
333  ARKDlsMem arkdls_mem;
334  realtype *bd;
335 
336  arkdls_mem = (ARKDlsMem) ark_mem->ark_lmem;
337 
338  bd = N_VGetArrayPointer(b);
339 
340  BandGBTRS(arkdls_mem->d_M, arkdls_mem->d_lpivots, bd);
341 
342  /* scale the correction to account for change in gamma */
343  if (ark_mem->ark_gamrat != ONE)
344  N_VScale(TWO/(ONE + ark_mem->ark_gamrat), b, b);
345 
346  arkdls_mem->d_last_flag = ARKDLS_SUCCESS;
347  return(0);
348 }

372 /*---------------------------------------------------------------
373  ARKMassBand:
374 
375  This routine initializes the memory record and sets various
376  function fields specific to the band mass matrix linear solver
377  module. ARKMassBand first calls the existing mfree routine if
378  this is not NULL. It then sets the ark_minit, ark_msetup,
379  ark_msolve, and ark_mfree fields in (*arkode_mem) to be

382  structure of type ARKDlsMassMemRec and sets the ark_mass_mem
383  field in (*arkode_mem) to the address of this structure. It
384  sets MassSetupNonNull in (*arkode_mem) to be TRUE, d_mu to be
385  mupper and d_ml to be mlower. Finally, it allocates memory for
386  M and pivot. The ARKMassBand return value is SUCCESS = 0,
387  LMEM_FAIL = -1, or LIN_ILL_INPUT = -2.
388 
389  NOTE: The band linear solver assumes a serial implementation
390  of the NVECTOR package. Therefore, ARKMassBand will first
391  test for compatible a compatible N_Vector internal
392  representation by checking that the function
393  N_VGetArrayPointer exists.

400 
401  /* Return immediately if arkode_mem is NULL */
402  if (arkode_mem == NULL) {
403  arkProcessError(NULL, ARKDLS_MEM_NULL, "ARKBAND", "ARKMassBand", MSGD_ARKMEM_NULL);
404  return(ARKDLS_MEM_NULL);

411  return(ARKDLS_ILL_INPUT);
412  }
413 
414  if (ark_mem->ark_mfree != NULL) ark_mem->ark_mfree(ark_mem);
415 
416  /* Set four main function fields in ark_mem, enable mass matrix */
417  ark_mem->ark_mass_matrix = TRUE;
418  ark_mem->ark_minit = arkMassBandInit;
419  ark_mem->ark_msetup = arkMassBandSetup;
420  ark_mem->ark_msolve = arkMassBandSolve;
421  ark_mem->ark_mfree = arkMassBandFree;

424  ark_mem->ark_msolve_type = 2;
425 
426  /* Get memory for ARKDlsMassMemRec */

431  return(ARKDLS_MEM_FAIL);
432  }
433 
434  /* Set matrix type */
435  arkdls_mem->d_type = SUNDIALS_BAND;
436 
437  /* Initialize mass-matrix-related data */
438  arkdls_mem->d_nme = 0;
439  arkdls_mem->d_bmass = bmass;
440  arkdls_mem->d_M_data = NULL;
441  arkdls_mem->d_last_flag = ARKDLS_SUCCESS;
442  ark_mem->ark_MassSetupNonNull = TRUE;
443 
444  /* Load problem dimension */
445  arkdls_mem->d_n = N;
446 
447  /* Load half-bandwiths in arkdls_mem */
448  arkdls_mem->d_ml = mlower;
449  arkdls_mem->d_mu = mupper;
450 
451  /* Test ml and mu for legality */

453  arkProcessError(ark_mem, ARKDLS_ILL_INPUT, "ARKBAND", "ARKMassBand", MSGD_BAD_SIZES);
454  free(arkdls_mem); arkdls_mem = NULL;
455  return(ARKDLS_ILL_INPUT);
456  }
457 
458  /* Set extended upper half-bandwith for M (required for pivoting) */
459  arkdls_mem->d_smu = SUNMIN(N-1, arkdls_mem->d_mu + arkdls_mem->d_ml);
460 
461  /* Allocate memory for M, M_lu, and pivot arrays */
462  arkdls_mem->d_M = NULL;
463  arkdls_mem->d_M = NewBandMat(N, arkdls_mem->d_mu, arkdls_mem->d_ml, arkdls_mem->d_mu);
464  if (arkdls_mem->d_M == NULL) {
465  arkProcessError(ark_mem, ARKDLS_MEM_FAIL, "ARKBAND", "ARKMassBand", MSGD_MEM_FAIL);
466  free(arkdls_mem); arkdls_mem = NULL;
467  return(ARKDLS_MEM_FAIL);
468  }
469  arkdls_mem->d_M_lu = NULL;
470  arkdls_mem->d_M_lu = NewBandMat(N, arkdls_mem->d_mu, arkdls_mem->d_ml, arkdls_mem->d_smu);
471  if (arkdls_mem->d_M_lu == NULL) {
472  arkProcessError(ark_mem, ARKDLS_MEM_FAIL, "ARKBAND", "ARKMassBand", MSGD_MEM_FAIL);
473  DestroyMat(arkdls_mem->d_M);
474  free(arkdls_mem); arkdls_mem = NULL;
475  return(ARKDLS_MEM_FAIL);
476  }
477  arkdls_mem->d_lpivots = NULL;
478  arkdls_mem->d_lpivots = NewLintArray(N);
479  if (arkdls_mem->d_lpivots == NULL) {
480  arkProcessError(ark_mem, ARKDLS_MEM_FAIL, "ARKBAND", "ARKMassBand", MSGD_MEM_FAIL);
481  DestroyMat(arkdls_mem->d_M);
482  DestroyMat(arkdls_mem->d_M_lu);
483  free(arkdls_mem); arkdls_mem = NULL;
484  return(ARKDLS_MEM_FAIL);
485  }
486 
487  /* Attach linear solver memory to integrator memory */
488  ark_mem->ark_mass_mem = arkdls_mem;
489 
490  return(ARKDLS_SUCCESS);
491 }

521  N_Vector vtemp2, N_Vector vtemp3)
522 {
523  long int ier;
524  ARKDlsMassMem arkdls_mem;
525  int retval;
526
