140 static double* alloc_real_matrix(int N, int M, void *data)
141 {
142  double *matrix;
143  void *tmp = data;
144  int i, j;
145 
146  matrix = (double*)malloc(N * M * sizeof(double));
147  assert(matrix != NULL);
148 
149  if(data) {
150  for(i = 0; i < N; ++i) {
151  tmp = MMC_CAR(data);
152  for(j = 0; j < M; ++j) {
153  matrix[j * N + i] = mmc_prim_get_real(MMC_CAR(tmp));
154  tmp = MMC_CDR(tmp);
155  }
156  data = MMC_CDR(data);
157  }
158  }
159 
160  return matrix;
161 }

216 static void* mk_rml_real_matrix(int N, int M, double *data)
217 {
218  void *res, *tmp;
219  int i, j;
220 
221  res = mmc_mk_nil();
222  for(i = N - 1; i >= 0; --i) {
223  tmp = mmc_mk_nil();
224  for(j = M - 1; j >= 0; --j) {
225  tmp = mmc_mk_cons(mmc_mk_rcon(data[j * N + i]), tmp);
226  }
227  res = mmc_mk_cons(tmp, res);
228  }
229 
230  return res;
231 }
