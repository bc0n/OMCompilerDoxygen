136  bool
137  implementsApplyTranspose (const Epetra_Operator& Op)
138  {
139  // We must first check Op's current transpose state, so that we
140  // leave Op in the same state as it was given to us.
141  const bool transposed = Op.UseTranspose ();
142 
143  // If Op is already transposed, then obviously it implements
144  // applying the transpose.
145  if (transposed) {
146  return true;
147  }
148 
149  // SetUseTranspose() is a nonconst method, so we have to cast
150  // away const-ness of Op in order to call it. Epetra_Operator
151  // gives us no other way to query whether it knows how to apply
152  // the transpose.
153  const int transposeFailed =
154  const_cast<Epetra_Operator&>(Op).SetUseTranspose (true);
155 
156  // SetUseTranspose() follows the POSIX convention that returning
157  // zero means success. If SetUseTranspose() failed, the best we can
158  // do is assume that the state of Op was not changed. Any
159  // reasonable implementation of SetUseTranspose() should not change
160  // the state of the operator if setting the transpose failed.
161  if (transposeFailed != 0) {
162  // Make sure that the failed call didn't managed to change the
163  // state of Op. It shouldn't have, but if it did, then Op is
164  // broken.
165  const bool currentTransposedState = Op.UseTranspose();
166  TEUCHOS_TEST_FOR_EXCEPTION(currentTransposedState != transposed, std::logic_error,
167  "implementsApplyTranspose: Op.SetUseTranspose(true) "
168  "failed (with error code " << transposeFailed << "), "
169  "but nevertheless managed to change the transposed "
170  "state of Op. On input, Op.UseTranspose() = "
171  << transposed << ", but now, Op.UseTranspose() = "
172  << currentTransposedState << ". This likely indicates"
173  " a bug in the implementation of the operator Op.");
174  return false;
175  }
176 
177  // Now we need to restore the original transpose state. We can
178  // do this unconditionally because if we're here, Op's state is
179  // not transposed.
180  {
181  const int errcode =
182  const_cast<Epetra_Operator&>(Op).SetUseTranspose (false);
183  TEUCHOS_TEST_FOR_EXCEPTION(errcode != 0, std::logic_error,
184  "implementsApplyTranspose: Op.SetUseTranspose(true) "
185  "succeeded, but Op.SetUseTranspose(false) failed with "
186  "error code " << errcode << ". This likely indicates "
187  "a bug in the implementation of the operator Op.");
188  }
189  // If we made it this far, then Op implements transpose.
190  return true;
191  }

200  class EpetraOperatorTransposeScopeGuard {
201  private:
203  const Epetra_Operator& Op_;
204 
206  const bool originalTransposeFlag_;
207 
213  const bool newTransposeFlag_;
214 
215  public:
222  EpetraOperatorTransposeScopeGuard (const Epetra_Operator& Op,
223  const ETrans trans)
224  : Op_ (Op),
225  originalTransposeFlag_ (Op.UseTranspose ()),
226  newTransposeFlag_ (trans != NOTRANS)
227  {
228  // If necessary, set (or unset) the transpose flag to the value
229  // corresponding to 'trans'.
230  if (newTransposeFlag_ != originalTransposeFlag_) {
231 
232 #ifdef BELOS_EPETRA_AGGRESSIVE_DEBUGGING
233  using std::endl;
234  std::ostream& err = *(getErrStream (Op_));
235  err << "---- EpetraOperatorTransposeScopeGuard constructor:" << endl
236  << "------ Toggling transpose flag from " << originalTransposeFlag_
237  << " to " << newTransposeFlag_ << endl;
238 #endif // BELOS_EPETRA_AGGRESSIVE_DEBUGGING
239 
240  // Toggle the transpose flag. The destructor will restore
241  // its original value.
242  const int info =
243  const_cast<Epetra_Operator &>(Op_).SetUseTranspose (newTransposeFlag_);
244  TEUCHOS_TEST_FOR_EXCEPTION(info != 0, EpetraOpFailure,
245  "Toggling the transpose flag of the operator failed, "
246  "returning a nonzero error code of " << info
247  << ". That probably means that the Epetra_Operator "
248  "subclass instance doesn't know how to apply its "
249  "transpose. Are you perhaps using a subclass of "
250  "Epetra_Operator for which applying the transpose "
251  "is not implemented? Anyway, just to help with "
252  "debugging, you specified trans="
253  << etransToString (trans)
254  << ", and the original operator was set "
255  << (originalTransposeFlag_ ? "" : "NOT ")
256  << "to apply the transpose.");
257  }
258  }
259 
260  ~EpetraOperatorTransposeScopeGuard ()
261  {
262  // SetUseTranspose() changes the state of the operator, so if
263  // applicable, we have to change the state back.
264  if (newTransposeFlag_ != originalTransposeFlag_) {
265 
266 #ifdef BELOS_EPETRA_AGGRESSIVE_DEBUGGING
267  using std::endl;
268  std::ostream& err = *(getErrStream (Op_));
269  err << "---- EpetraOperatorTransposeScopeGuard destructor:" << endl
270  << "------ Toggling transpose flag from " << newTransposeFlag_
271  << " to " << originalTransposeFlag_ << endl;
272 #endif // BELOS_EPETRA_AGGRESSIVE_DEBUGGING
273 
274  const int info =
275  const_cast<Epetra_Operator &>(Op_).SetUseTranspose (originalTransposeFlag_);
276  TEUCHOS_TEST_FOR_EXCEPTION(info != 0, EpetraOpFailure,
277  "Resetting the original "
278  "transpose flag value of the Epetra_Operator failed, "
279  "returning a nonzero error code of " << info << ". "
280  "This is an unusual error, since we were able to call "
281  "its SetUseTranspose() method successfully before. "
282  "This suggests a bug in the subclass of Epetra_Operator "
283  "which you are currently using. This is probably not a "
284  "Belos bug.");
285  }
286 
287  // Make sure that the transpose flag has its original value. If
288  // not, we throw std::logic_error instead of EpetraOpFailure, since
289  // that's definitely a bug. It's safe to do this check whether or
290  // not we actually had to toggle the transpose flag. Any reasonable
291  // implementation of Epetra_Operator should make calling
292  // UseTranspose() cheap.
293  //
294  // Note to code maintainers: The reason we capture the value of
295  // UseTranspose() instead of just calling it twice, is that if the
296  // exception test really does trigger, then there is something
297  // fundamentally wrong. If something is that wrong, then we want to
298  // avoid further calls to the operator's methods. For example, it
299  // could be that the UseTranspose() method is erroneously changing
300  // the value of the flag, so if we call that method twice, it might
301  // have the right value on the second call. This would make the
302  // resulting exception message confusing.
303  const bool finalTransposeFlag = Op_.UseTranspose ();
304  TEUCHOS_TEST_FOR_EXCEPTION(originalTransposeFlag_ != finalTransposeFlag,
305  std::logic_error,
306  "Belos::OperatorTraits::Apply: The value of the "
307  "Epetra_Operator's transpose flag changed unexpectedly!"
308  " The original value at the top of this method was "
309  << originalTransposeFlag_ << ", and its new value is "
310  << finalTransposeFlag << ". This suggests either a "
311  "bug in Belos (in the implementation of this routine), "
312  "or a bug in the operator.");
313  }
314  };

593 void
594 EpetraOp::Apply (const MultiVec<double>& x,
595  MultiVec<double>& y,
596  ETrans trans) const
597 {
598 #ifdef BELOS_EPETRA_AGGRESSIVE_DEBUGGING
599  using std::endl;
600 
601  std::ostream& err = *(getErrStream (*Epetra_Op));
602  err << "-- Belos::EpetraOp::Apply:" << endl
603  << "---- Implements Belos::Operator<double>::Apply()" << endl
604  << "---- trans input = " << etransToString (trans) << endl
605  << "---- On input: Epetra_Op->UseTranspose() = "
606  << Epetra_Op->UseTranspose() << endl;
607 #endif // BELOS_EPETRA_AGGRESSIVE_DEBUGGING
608 
609  MultiVec<double> & temp_x = const_cast<MultiVec<double> &>(x);
610  Epetra_MultiVector* vec_x = dynamic_cast<Epetra_MultiVector* >(&temp_x);
611  Epetra_MultiVector* vec_y = dynamic_cast<Epetra_MultiVector* >(&y);
612 
613  TEUCHOS_TEST_FOR_EXCEPTION(vec_x==NULL || vec_y==NULL, EpetraOpFailure,
614  "Belos::EpetraOp::Apply: x and/or y could not be "
615  "dynamic cast to an Epetra_MultiVector.");
616 
617  // Temporarily set the transpose state of Op, if it's not the same
618  // as trans, and restore it on exit of this scope.
619  EpetraOperatorTransposeScopeGuard guard (*Epetra_Op, trans);
620 
621 #ifdef BELOS_EPETRA_AGGRESSIVE_DEBUGGING
622  err << "---- Before calling Apply: Epetra_Op->UseTranspose() = "
623  << Epetra_Op->UseTranspose() << endl;
624 #endif // BELOS_EPETRA_AGGRESSIVE_DEBUGGING
625 
626  // Apply the operator to x and put the result in y.
