73 //
74 // trans_tag will output tags that are compatible with BLAS and LAPACK, either one
75 // of tag::transpose, tag::no_transpose, or tag::conjugate.
76 // It needs an library-orientation (Order) before it can make a decision about whether
77 // a tranpose is in order. Consult the various cases above to see what kind of
78 // tag is being generated in what kind of situation.
79 //
80 template< typename T, typename Order >
81 typename result_of::trans_tag< T, Order >::type trans_tag( const T& t, Order ) {
82  return result_of::trans_tag< T, Order >::type();
83 }
