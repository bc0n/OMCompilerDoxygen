83 // Create a very simple square test matrix. We use the identity
84 // matrix here. The point of this test is NOT to exercise the solver;
85 // it's just to check that Belos' LinearSolverFactory can create
86 // working solvers. Belos has more rigorous tests for its solvers.
87 template<class SC, class LO, class GO, class NT>
88 Teuchos::RCP<Tpetra::CrsMatrix<SC, LO, GO, NT> >
89 createTestMatrix (Teuchos::FancyOStream& out,
90  const Teuchos::RCP<const Teuchos::Comm<int> >& comm,
91  const Tpetra::global_size_t gblNumRows)
92 {
93  using Teuchos::Comm;
94  using Teuchos::RCP;
95  using Teuchos::rcp;
96  using std::endl;
97  typedef Tpetra::CrsMatrix<SC,LO,GO,NT> MAT;
98  typedef Tpetra::Map<LO,GO,NT> map_type;
99  typedef Teuchos::ScalarTraits<SC> STS;
100 
101  Teuchos::OSTab tab0 (out);
102  out << "Create test matrix with " << gblNumRows << " row(s)" << endl;
103  Teuchos::OSTab tab1 (out);
104 
105  TEUCHOS_TEST_FOR_EXCEPTION
106  ( gblNumRows == 0, std::invalid_argument, "gblNumRows = 0");
107 
108  const GO indexBase = 0;
109  RCP<const map_type> rowMap (new map_type (gblNumRows, indexBase, comm));
110  // For this particular row matrix, the row Map and the column Map
111  // are the same. Giving a column Map to CrsMatrix's constructor
112  // lets us use local indices.
113  RCP<const map_type> colMap = rowMap;
114  const size_t maxNumEntPerRow = 1;
115  RCP<MAT> A (new MAT (rowMap, colMap, maxNumEntPerRow, Tpetra::StaticProfile));
116 
117  if (rowMap->getNodeNumElements () != 0) {
118  Teuchos::Array<SC> vals (1);
119  Teuchos::Array<LO> inds (1);
120  for (LO lclRow = rowMap->getMinLocalIndex ();
121  lclRow <= rowMap->getMaxLocalIndex (); ++lclRow) {
122  inds[0] = lclRow;
123  vals[0] = STS::one ();
124  A->insertLocalValues (lclRow, inds (), vals ());
125  }
126  }
127 
128  RCP<const map_type> domMap = rowMap;
129  RCP<const map_type> ranMap = rowMap;
130  A->fillComplete (domMap, ranMap);
131  return A;
132 }

134 // Create a very simple square test linear system (matrix, right-hand
135 // side(s), and exact solution(s). We use the identity matrix here.
136 // The point of this test is NOT to exercise the preconditioner; it's
137 // just to check that its LinearSolverFactory can create working
138 // preconditioners. Belos has more rigorous tests for each of its
139 // preconditioners.
140 template<class SC, class LO, class GO, class NT>
141 void
142 createTestProblem (Teuchos::FancyOStream& out,
143  Teuchos::RCP<Tpetra::MultiVector<SC, LO, GO, NT> >& X,
144  Teuchos::RCP<Tpetra::CrsMatrix<SC, LO, GO, NT> >& A,
145  Teuchos::RCP<Tpetra::MultiVector<SC, LO, GO, NT> >& B,
146  const Teuchos::RCP<const Teuchos::Comm<int> >& comm,
147  const Tpetra::global_size_t gblNumRows,
148  const size_t numVecs)
149 {
150  using Teuchos::Comm;
151  using Teuchos::RCP;
152  using Teuchos::rcp;
153  using std::endl;
154  typedef Tpetra::MultiVector<SC,LO,GO,NT> MV;
155  typedef Teuchos::ScalarTraits<SC> STS;
156 
157  A = createTestMatrix<SC, LO, GO, NT> (out, comm, gblNumRows);
158  X = rcp (new MV (A->getDomainMap (), numVecs));
159  B = rcp (new MV (A->getRangeMap (), numVecs));
160 
161  X->putScalar (STS::one ());
162  A->apply (*X, *B);
163 }

168  bool& success,

180  typedef Tpetra::MultiVector<SC,LO,GO,NT> MV;
181  typedef Teuchos::ScalarTraits<SC> STS;
182  typedef typename MV::mag_type mag_type;
183 
184  Teuchos::OSTab tab0 (out);
185  out << "Test solver \"" << solverName << "\" from Belos package" << endl;
186  Teuchos::OSTab tab1 (out);
187 
188  RCP<Trilinos::Details::LinearSolver<MV, OP, mag_type> > solver;
189  try {
190  solver = Trilinos::Details::getLinearSolver<MV, OP, mag_type> ("Belos", solverName);
191  } catch (std::exception& e) {
192  out << "*** FAILED: getLinearSolver threw an exception: " << e.what () << endl;
193  success = false;
194  return;
195  }
196  TEST_EQUALITY_CONST( solver.is_null (), false );
197  if (solver.is_null ()) {
198  out << "*** FAILED to create solver \"" << solverName
199  << "\" from Belos package" << endl;
200  success = false;
201  return;
202  }
203 
204  out << "Set matrix" << endl;
205  solver->setMatrix (A);
206 
207  out << "Compute symbolic and numeric factorization" << endl;
208  solver->symbolic ();
209  solver->numeric ();
210 
211  out << "Apply solver to \"solve\" AX=B for X. Belos already has solver "
212  "tests; the point is to check that solve() doesn't throw." << endl;
213  X.putScalar (STS::zero ());
214  solver->solve (X, B);
215 }
