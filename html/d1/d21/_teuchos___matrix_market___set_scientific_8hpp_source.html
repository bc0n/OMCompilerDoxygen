87  template<class Scalar>
88  class SetScientificImpl<Scalar, true> {
89  public:
90  typedef Scalar scalar_type;
91 
92  SetScientificImpl (std::ostream& out) :
93  out_ (out), originalFlags_ (out.flags())
94  {
95  typedef Teuchos::ScalarTraits<scalar_type> STS;
96  typedef typename STS::magnitudeType magnitude_type;
97  typedef Teuchos::ScalarTraits<magnitude_type> STM;
98 
99  // Print floating-point values in scientific notation.
100  out << std::scientific;
101 
102  // We're writing decimal digits, so compute the number of
103  // digits we need to get reasonable accuracy when reading
104  // values back in.
105  //
106  // There is actually an algorithm, due to Guy Steele (yes,
107  // Java's Guy Steele) et al., for idempotent printing of
108  // finite-length floating-point values. We should actually
109  // implement that algorithm, but I don't have time for that
110  // now. Currently, I just print no more than (one decimal
111  // digit more than (the number of decimal digits justified
112  // by the precision of magnitude_type)).
113  //
114  // We need to use STM's log10() rather than (say) std::log10
115  // here, because STM::base() returns a magnitude_type, not
116  // one of C++'s standard integer types.
117  const magnitude_type numDecDigits = STM::t() * STM::log10 (STM::base());
118 
119  // Round and add one. The cast to int should not overflow
120  // unless STM::t() is _extremely_ large, so we don't need to
121  // check for that case here.
122  const magnitude_type one = STM::one();
123  const magnitude_type two = one + one;
124  // Cast from magnitude_type to int, since std::ostream's
125  // precision() method expects an int input.
126  const int prec = 1 + Teuchos::as<int> (magnitude_type((two*numDecDigits + one) / two));
127 
128  // Set the number of (decimal) digits after the decimal
129  // point to print.
130  out.precision (prec);
131  }
132 
133  ~SetScientificImpl () {
134  out_.flags (originalFlags_);
135  }
136 
137  private:
139  std::ostream& out_;
140 
142  std::ios_base::fmtflags originalFlags_;
143  };
