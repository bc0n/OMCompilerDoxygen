89 static int simulationUpdate(DATA* data, threadData_t *threadData, SOLVER_INFO* solverInfo)
90 {
91  prefixedName_updateContinuousSystem(data, threadData);
92 
93  if (solverInfo->solverMethod == S_SYM_SOLVER_SSC) {
94  data->simulationInfo->inlineData->dt = solverInfo->solverStepSize;
95  //data->callback->symbolicInlineSystems(data, threadData);
96  }
97 
98  saveZeroCrossings(data, threadData);
99 
100  /***** Event handling *****/
101  if (measure_time_flag) rt_tick(SIM_TIMER_EVENT);
102 
103  int syncRet = handleTimers(data, threadData, solverInfo);
104  int syncRet1;
105  do
106  {
107  int eventType = checkEvents(data, threadData, solverInfo->eventLst, !solverInfo->solverRootFinding, /*out*/ &solverInfo->currentTime);
108  if(eventType > 0 || syncRet == 2) /* event */
109  {
110  threadData->currentErrorStage = ERROR_EVENTHANDLING;
111  infoStreamPrint(LOG_EVENTS, 1, "%s event at time=%.12g", eventType == 1 ? "time" : "state", solverInfo->currentTime);
112  /* prevent emit if noEventEmit flag is used */
113  if (!(omc_flag[FLAG_NOEVENTEMIT])) /* output left limit */
114  sim_result.emit(&sim_result, data, threadData);
115  handleEvents(data, threadData, solverInfo->eventLst, &(solverInfo->currentTime), solverInfo);
116  cleanUpOldValueListAfterEvent(data, solverInfo->currentTime);
117  messageClose(LOG_EVENTS);
118  threadData->currentErrorStage = ERROR_SIMULATION;
119  solverInfo->didEventStep = 1;
120  overwriteOldSimulationData(data);
121  }
122  else /* no event */
123  {
124  solverInfo->laststep = solverInfo->currentTime;
125  solverInfo->didEventStep = 0;
126  }
127 
128  if (measure_time_flag) rt_accumulate(SIM_TIMER_EVENT);
129  /***** End event handling *****/
130 
131 
132  /***** check state selection *****/
133  if (stateSelection(data, threadData, 1, 1))
134  {
135  /* if new set is calculated reinit the solver */
136  solverInfo->didEventStep = 1;
137  overwriteOldSimulationData(data);
138  }
139 
140  /* Check for warning of variables out of range assert(min<x || x>xmax, ...)*/
141  data->callback->checkForAsserts(data, threadData);
142 
143  storePreValues(data);
144  storeOldValues(data);
145 
146  syncRet1 = handleTimers(data, threadData, solverInfo);
147  syncRet = syncRet1 == 0 ? syncRet : syncRet1;
148  } while (syncRet1);
149  return syncRet;
150 }

168 static void fmtInit(DATA* data, MEASURE_TIME* mt)
169 {
170  mt->fmtReal = NULL;
171  mt->fmtInt = NULL;
172  if(measure_time_flag)
173  {
174  const char* fullFileName;
175  if (omc_flag[FLAG_OUTPUT_PATH]) { /* read the output path from the command line (if any) */
176  if (0 > GC_asprintf(&fullFileName, "%s/%s", omc_flagValue[FLAG_OUTPUT_PATH], data->modelData->modelFilePrefix)) {
177  throwStreamPrint(NULL, "perform_simulation.c: Error: can not allocate memory.");
178  }
179  } else {
180  fullFileName = data->modelData->modelFilePrefix;
181  }
182  size_t len = strlen(fullFileName);
183  char* filename = (char*) malloc((len+15) * sizeof(char));
184  strncpy(filename,fullFileName,len);
185  strncpy(&filename[len],"_prof.realdata",15);
186  mt->fmtReal = fopen(filename, "wb");
187  if(!mt->fmtReal)
188  {
189  warningStreamPrint(LOG_STDOUT, 0, "Time measurements output file %s could not be opened: %s", filename, strerror(errno));
190  }
191  strncpy(&filename[len],"_prof.intdata",14);
192  mt->fmtInt = fopen(filename, "wb");
193  if(!mt->fmtInt)
194  {
195  warningStreamPrint(LOG_STDOUT, 0, "Time measurements output file %s could not be opened: %s", filename, strerror(errno));
196  fclose(mt->fmtReal);
197  mt->fmtReal = NULL;
198  }
199  free(filename);
200  }
201 }

203 static void fmtEmitStep(DATA* data, threadData_t *threadData, MEASURE_TIME* mt, SOLVER_INFO* solverInfo)
204 {
205  if(mt->fmtReal)
206  {
207  int i, flag=1;
208  double tmpdbl;
209  unsigned int tmpint;
210  int total = data->modelData->modelDataXml.nFunctions + data->modelData->modelDataXml.nProfileBlocks;
211  rt_tick(SIM_TIMER_OVERHEAD);
212  rt_accumulate(SIM_TIMER_STEP);
213 
214  /* Disable time measurements if we have trouble writing to the file... */
215  flag = flag && 1 == fwrite(&mt->stepNo, sizeof(unsigned int), 1, mt->fmtInt);
216  mt->stepNo++;
217  flag = flag && 1 == fwrite(&(data->localData[0]->timeValue), sizeof(double), 1, mt->fmtReal);
218  tmpdbl = rt_accumulated(SIM_TIMER_STEP);
219  flag = flag && 1 == fwrite(&tmpdbl, sizeof(double), 1, mt->fmtReal);
220  flag = flag && total == fwrite(rt_ncall_arr(SIM_TIMER_FIRST_FUNCTION), sizeof(uint32_t), total, mt->fmtInt);
221  for(i=0; i<data->modelData->modelDataXml.nFunctions + data->modelData->modelDataXml.nProfileBlocks; i++) {
222  tmpdbl = rt_accumulated(i + SIM_TIMER_FIRST_FUNCTION);
223  flag = flag && 1 == fwrite(&tmpdbl, sizeof(double), 1, mt->fmtReal);
224  }
225  rt_accumulate(SIM_TIMER_OVERHEAD);
226 
227  if(!flag)
228  {
229  warningStreamPrint(LOG_SOLVER, 0, "Disabled time measurements because the output file could not be generated: %s", strerror(errno));
230  fclose(mt->fmtInt);
231  fclose(mt->fmtReal);
232  mt->fmtInt = NULL;
233  mt->fmtReal = NULL;
234  }
235  }
236 
237  /* prevent emit if noEventEmit flag is used, if it's an event */
238  if ((omc_flag[FLAG_NOEVENTEMIT] && solverInfo->didEventStep == 0) || !omc_flag[FLAG_NOEVENTEMIT]) {
239  sim_result.emit(&sim_result, data, threadData);
240  }
241 #if !defined(OMC_MINIMAL_RUNTIME)
242  if (embedded_server_update(data->embeddedServerState, data->localData[0]->timeValue)) {
243  solverInfo->didEventStep = 1;
244  overwriteOldSimulationData(data);
245  storePreValues(data); // Maybe??
246  storeOldValues(data); // Maybe??
247  sim_result.emit(&sim_result, data, threadData);
248  }
249  if (data->real_time_sync.enabled) {
250  double time = data->localData[0]->timeValue;
251  int64_t res = rt_ext_tp_sync_nanosec(&data->real_time_sync.clock, (uint64_t) (data->real_time_sync.scaling*(time-data->real_time_sync.time)*1e9));
252  int64_t maxLateNano = data->simulationInfo->stepSize*1e9*0.1*data->real_time_sync.scaling /* Maximum late time: 10% of step size */;
253  if (res > maxLateNano) {
254  int t=0,tMaxLate=0;
255  const char *unit = prettyPrintNanoSec(res, &t);
256  const char *unit2 = prettyPrintNanoSec(maxLateNano, &tMaxLate);
257  errorStreamPrint(LOG_RT, 0, "Missed deadline at time %g; delta was %d %s (maxLate=%d %s)", time, t, unit, tMaxLate, unit2);
258  }
259  if (res > data->real_time_sync.maxLate) {
260  data->real_time_sync.maxLate = res;
261  }
262  }
263 
264  printAllVarsDebug(data, 0, LOG_DEBUG); /* ??? */
265 #endif
266 }

340 int prefixedName_performSimulation(DATA* data, threadData_t *threadData, SOLVER_INFO* solverInfo)
341 {
342  TRACE_PUSH
343 
344  int retValIntegrator=0;
345  int retValue=0;
346  int i, retry=0, steadStateReached=0;
347 
348  unsigned int __currStepNo = 0;
349 
350  SIMULATION_INFO *simInfo = data->simulationInfo;
351  solverInfo->currentTime = simInfo->startTime;
352 
353  MEASURE_TIME fmt;
354  fmtInit(data, &fmt);
355 
356  printAllVarsDebug(data, 0, LOG_DEBUG); /* ??? */
357  if (!compiledInDAEMode)
358  {
359  printSparseStructure(&(data->simulationInfo->analyticJacobians[data->callback->INDEX_JAC_A].sparsePattern),
360  data->simulationInfo->analyticJacobians[data->callback->INDEX_JAC_A].sizeRows,
361  data->simulationInfo->analyticJacobians[data->callback->INDEX_JAC_A].sizeCols,
362  LOG_SOLVER, "ODE sparse pattern");
363  }
364  else
365  {
366  printSparseStructure(data->simulationInfo->daeModeData->sparsePattern,
367  data->simulationInfo->daeModeData->nResidualVars,
368  data->simulationInfo->daeModeData->nResidualVars,
369  LOG_SOLVER, "DAE sparse pattern");
370  }
371 
372  if(terminationTerminate)
373  {
374  printInfo(stdout, TermInfo);
375  fputc('\n', stdout);
376  infoStreamPrint(LOG_STDOUT, 0, "Simulation call terminate() at initialization (time %f)\nMessage : %s", data->localData[0]->timeValue, TermMsg);
377  data->simulationInfo->stopTime = solverInfo->currentTime;
378  } else {
379  modelica_boolean syncStep = 0;
380 
381  /***** Start main simulation loop *****/
382  while(solverInfo->currentTime < simInfo->stopTime || !simInfo->useStopTime)
383  {
384  int success = 0;
385  threadData->currentErrorStage = ERROR_SIMULATION;
386 
387 #ifdef USE_DEBUG_TRACE
388  if(useStream[LOG_TRACE]) {
389  printf("TRACE: push loop step=%u, time=%.12g\n", __currStepNo, solverInfo->currentTime);
390  }
391 #endif
392 
393  /* check for steady state */
394  if (omc_flag[FLAG_STEADY_STATE])
395  {
396  if (0 < data->modelData->nStates)
397  {
398  int i;
399  double maxDer = 0.0;
400  double currDer;
401  for(i=data->modelData->nStates; i<2*data->modelData->nStates; ++i)
402  {
403  currDer = fabs(data->localData[0]->realVars[i] / data->modelData->realVarsData[i].attribute.nominal);
404  if(maxDer < currDer)
405  maxDer = currDer;
406  }
407  if (maxDer < steadyStateTol) {
408  steadStateReached=1;
409  infoStreamPrint(LOG_STDOUT, 0, "steady state reached at time = %g\n * max(|d(x_i)/dt|/nominal(x_i)) = %g\n * relative tolerance = %g", solverInfo->currentTime, maxDer, steadyStateTol);
410  break;
411  }
412  }
413  else
414  throwStreamPrint(threadData, "No states in model. Flag -steadyState can only be used if states are present.");
415  }
416 
417  omc_alloc_interface.collect_a_little();
418 
419  /* try */
420 #if !defined(OMC_EMCC)
421  MMC_TRY_INTERNAL(simulationJumpBuffer)
422 #endif
423  {
424  printAllVars(data, 0, LOG_SOLVER_V);
425 
426  clear_rt_step(data);
427  if (!compiledInDAEMode) /* do not use ringbuffer for daeMode */
428  rotateRingBuffer(data->simulationData, 1, (void**) data->localData);
429 
430  modelica_boolean syncEventStep = solverInfo->didEventStep || syncStep;
431 
432  /***** Calculation next step size *****/
433  if(syncEventStep) {
434  infoStreamPrint(LOG_SOLVER, 0, "offset value for the next step: %.16g", (solverInfo->currentTime - solverInfo->laststep));
435  } else {
436  if (solverInfo->solverNoEquidistantGrid)
437  {
438  if (solverInfo->currentTime >= solverInfo->lastdesiredStep)
439  {
440  do {
441  __currStepNo++;
442  solverInfo->currentStepSize = (double)(__currStepNo*(simInfo->stopTime-simInfo->startTime))/(simInfo->numSteps) + simInfo->startTime - solverInfo->currentTime;
443  } while(solverInfo->currentStepSize <= 0);
444  }
445  } else {
446  __currStepNo++;
447  }
448  }
449 
450  solverInfo->currentStepSize = (double)(__currStepNo*(simInfo->stopTime-simInfo->startTime))/(simInfo->numSteps) + simInfo->startTime - solverInfo->currentTime;
451 
452  solverInfo->lastdesiredStep = solverInfo->currentTime + solverInfo->currentStepSize;
453 
454  /* if retry reduce stepsize */
455  if (0 != retry) {
456  solverInfo->currentStepSize /= 2;
457  }
458  /***** End calculation next step size *****/
459 
460  checkForSynchronous(data, solverInfo);
461  /* check for next time event */
462  checkForSampleEvent(data, solverInfo);
463 
464  /* if regular output point and last time events are almost equals
465  * skip that step and go further */
466  if (solverInfo->currentStepSize < 1e-15 && syncEventStep){
467  __currStepNo++;
468  continue;
469  }
470 
471  /*
472  * integration step determine all states by a integration method
473  * update continuous system
474  */
475  infoStreamPrint(LOG_SOLVER, 1, "call solver from %g to %g (stepSize: %.15g)", solverInfo->currentTime, solverInfo->currentTime + solverInfo->currentStepSize, solverInfo->currentStepSize);
476  retValIntegrator = simulationStep(data, threadData, solverInfo);
477  infoStreamPrint(LOG_SOLVER, 0, "finished solver step %g", solverInfo->currentTime);
478  messageClose(LOG_SOLVER);
479 
480  if (S_OPTIMIZATION == solverInfo->solverMethod) break;
481  syncStep = simulationUpdate(data, threadData, solverInfo);
482  retry = 0; /* reset retry */
483 
484  fmtEmitStep(data, threadData, &fmt, solverInfo);
485  saveIntegratorStats(solverInfo);
486  checkSimulationTerminated(data, solverInfo);
487 
488  /* terminate for some cases:
489  * - integrator fails
490  * - non-linear system failed to solve
491  * - assert was called
492  */
493  if (retValIntegrator) {
494  retValue = -1 + retValIntegrator;
495  infoStreamPrint(LOG_STDOUT, 0, "model terminate | Integrator failed. | Simulation terminated at time %g", solverInfo->currentTime);
496  break;
497  } else if(check_nonlinear_solutions(data, 0)) {
498  retValue = -2;
499  infoStreamPrint(LOG_STDOUT, 0, "model terminate | non-linear system solver failed. | Simulation terminated at time %g", solverInfo->currentTime);
500  break;
501  } else if(check_linear_solutions(data, 0)) {
502  retValue = -3;
503  infoStreamPrint(LOG_STDOUT, 0, "model terminate | linear system solver failed. | Simulation terminated at time %g", solverInfo->currentTime);
504  break;
505  } else if(check_mixed_solutions(data, 0)) {
506  retValue = -4;
507  infoStreamPrint(LOG_STDOUT, 0, "model terminate | mixed system solver failed. | Simulation terminated at time %g", solverInfo->currentTime);
508  break;
509  }
510  success = 1;
511  }
512 #if !defined(OMC_EMCC)
513  MMC_CATCH_INTERNAL(simulationJumpBuffer)
514 #endif
515  if (!success) { /* catch */
516  if(0 == retry) {
517  retrySimulationStep(data, threadData, solverInfo);
518  retry = 1;
519  } else {
520  retValue = -1;
521  infoStreamPrint(LOG_STDOUT, 0, "model terminate | Simulation terminated by an assert at time: %g", data->localData[0]->timeValue);
522  break;
523  }
524  }
525 
526  TRACE_POP /* pop loop */
527  } /* end while solver */
528  } /* end else */
529 
530  fmtClose(&fmt);
531 
532  if (omc_flag[FLAG_STEADY_STATE] && !steadStateReached) {
533  warningStreamPrint(LOG_STDOUT, 0, "Steady state has not been reached.\nThis may be due to too restrictive relative tolerance (%g) or short stopTime (%g).", steadyStateTol, simInfo->stopTime);
534  }
535 
536  TRACE_POP
537  return retValue;
538 }
