61 /*
62  * -----------------------------------------------------------------
63  * CVSuperLUMT
64  * -----------------------------------------------------------------
65  * This routine initializes the memory record and sets various function
66  * fields specific to the CVODE / SuperLUMT linear solver module.
67  * CVSUPERLUMT first calls the existing lfree routine if this is not NULL.
68  * Then it sets the cv_linit, cv_lsetup, cv_lsolve, and
69  * cv_lfree fields in (*cv_mem) to be cvSuperLUMTInit, cvSuperLUMTSetup,
70  * cvSuperLUMTSolve, and cvSuperLUMTFree, respectively.
71  * It allocates memory for a structure of type CVsluMemRec and sets
72  * the cv_lmem field in (*cvode_mem) to the address of this structure.
73  * It sets setupNonNull in (*cvode_mem) to TRUE.
74  * Finally, it allocates memory for SuperLUMT.
75  * The return value is CVSLS_SUCCESS = 0, CVSLS_LMEM_FAIL = -1,
76  * or CVSLS_ILL_INPUT = -2.
77  *
78  * NOTE: The SuperLUMT linear solver assumes a serial implementation
79  * of the NVECTOR package. Therefore, CVSuperLUMT will first
80  * test for a compatible N_Vector internal representation
81  * by checking that the function N_VGetArrayPointer exists.
82  * -----------------------------------------------------------------
83  */
84 
85 int CVSuperLUMT(void *cvode_mem, int num_threads, int n, int nnz)
86 {
87  CVodeMem cv_mem;
88  CVSlsMem cvsls_mem;
89  SLUMTData slumt_data;
90  int *perm_c, *perm_r;
91  int nrhs, panel_size, relax;
92  double *bd;
93  SuperMatrix *B;
94 
95  /* Return immediately if cv_mem is NULL. */
96  if (cvode_mem == NULL) {
97  cvProcessError(NULL, CVSLS_MEM_NULL, "CVSLS", "cvSuperLUMT",
98  MSGSP_CVMEM_NULL);
99  return(CVSLS_MEM_NULL);
100  }
101  cv_mem = (CVodeMem) cvode_mem;
102 
103  /* Test if the NVECTOR package is compatible with the Direct solver */
104  if (cv_mem->cv_tempv->ops->nvgetarraypointer == NULL) {
105  cvProcessError(cv_mem, CVSLS_ILL_INPUT, "CVSLS", "cvSuperLUMT",
106  MSGSP_BAD_NVECTOR);
107  return(CVSLS_ILL_INPUT);
108  }
109 
110  if (cv_mem->cv_lfree != NULL) cv_mem->cv_lfree(cv_mem);
111 
112  /* Set five main function fields in cv_mem. */
113  cv_mem->cv_linit = cvSuperLUMTInit;
114  cv_mem->cv_lsetup = cvSuperLUMTSetup;
115  cv_mem->cv_lsolve = cvSuperLUMTSolve;
116  cv_mem->cv_lfree = cvSuperLUMTFree;
117 
118  /* Get memory for CVSlsMemRec. */
119  cvsls_mem = (CVSlsMem) malloc(sizeof(struct CVSlsMemRec));
120  if (cvsls_mem == NULL) {
121  cvProcessError(cv_mem, CVSLS_MEM_FAIL, "CVSLS", "cvSuperLUMT",
122  MSGSP_MEM_FAIL);
123  return(CVSLS_MEM_FAIL);
124  }
125 
126  /* Get memory for SLUMT_data. */
127  slumt_data = (SLUMTData)malloc(sizeof(struct SLUMTDataRec));
128  if (slumt_data == NULL) {
129  cvProcessError(cv_mem, CVSLS_MEM_FAIL, "CVSLS", "cvSuperLUMT",
130  MSGSP_MEM_FAIL);
131  return(CVSLS_MEM_FAIL);
132  }
133 
134  cv_mem->cv_setupNonNull = TRUE;
135 
136  /* Set default Jacobian routine and Jacobian data */
137  cvsls_mem->s_jaceval = NULL;
138  cvsls_mem->s_jacdata = cv_mem->cv_user_data;
139 
140  /* Allocate memory for the sparse Jacobian */
141  cvsls_mem->s_JacMat = NewSparseMat(n, n, nnz);
142  if (cvsls_mem->s_JacMat == NULL) {
143  cvProcessError(cv_mem, CVSLS_MEM_FAIL, "CVSLS", "cvSuperLUMT",
144  MSGSP_MEM_FAIL);
145  free(cvsls_mem);
146  return(CVSLS_MEM_FAIL);
147  }
148 
149  /* Allocate memory for saved sparse Jacobian */
150  cvsls_mem->s_savedJ = NewSparseMat(n, n, nnz);
151  if (cvsls_mem->s_savedJ == NULL) {
152  cvProcessError(cv_mem, CVSLS_MEM_FAIL, "CVSLS", "cvSuperLUMT",
153  MSGSP_MEM_FAIL);
154  DestroySparseMat(cvsls_mem->s_JacMat);
155  free(cvsls_mem);
156  return(CVSLS_MEM_FAIL);
157  }
158 
159  /* Set up memory for the permutations */
160  perm_r = (int *)malloc(n*sizeof(int));
161  if (perm_r == NULL) {
162  cvProcessError(cv_mem, CVSLS_MEM_FAIL, "CVSLS", "cvSuperLUMT",
163  MSGSP_MEM_FAIL);
164  return(CVSLS_MEM_FAIL);
165  }
166  perm_c = (int *)malloc(n*sizeof(int));
167  if (perm_c == NULL) {
168  cvProcessError(cv_mem, CVSLS_MEM_FAIL, "CVSLS", "cvSuperLUMT",
169  MSGSP_MEM_FAIL);
170  free(perm_r);
171  return(CVSLS_MEM_FAIL);
172  }
173  slumt_data->perm_r = perm_r;
174  slumt_data->perm_c = perm_c;
175 
176  /* Set default parameters for SuperLU */
177  slumt_data->num_threads = num_threads;
178  slumt_data->diag_pivot_thresh = 1.0;
179 
180  /* Allocate structures for SuperLU */
181  slumt_data->Gstat = (Gstat_t *)malloc(sizeof(Gstat_t));
182  slumt_data->s_A = (SuperMatrix *)malloc(sizeof(SuperMatrix));
183  slumt_data->s_AC = (SuperMatrix *)malloc(sizeof(SuperMatrix));
184  slumt_data->s_L = (SuperMatrix *)malloc(sizeof(SuperMatrix));
185  slumt_data->s_U = (SuperMatrix *)malloc(sizeof(SuperMatrix));
186  slumt_data->s_A->Store = NULL;
187  slumt_data->s_AC->Store = NULL;
188  slumt_data->s_L->Store = NULL;
189  slumt_data->s_U->Store = NULL;
190  slumt_data->superlumt_options = (superlumt_options_t *)malloc(sizeof(superlumt_options_t));
191 
192  panel_size = sp_ienv(1);
193  relax = sp_ienv(2);
194  StatAlloc(cvsls_mem->s_JacMat->N, num_threads, panel_size, relax,
195  slumt_data->Gstat);
196 
197  /* Create RHS matrix */
198  nrhs = 1;
199  bd = NULL;
200  B = (SuperMatrix *)malloc(sizeof(SuperMatrix));
201  B->Store = NULL;
202  dCreate_Dense_Matrix(B, n, nrhs, bd, n,
203  SLU_DN, SLU_D, SLU_GE);
204  slumt_data->s_B = B;
205 
206  /* Set ordering to COLAMD as the cvode default use.
207  Users can set a different value with CVSuperLUMTSetOrdering,
208  and the user-set value is loaded before any call to factorize the
209  matrix in cvSuperLUMTSetup. */
210  slumt_data->s_ordering = 3;
211 
212  /* Attach linear solver memory to the integrator memory */
213  cvsls_mem->s_solver_data = (void *) slumt_data;
214  cv_mem->cv_lmem = cvsls_mem;
215 
216  cvsls_mem->s_last_flag = CVSLS_SUCCESS;
217 
218  return(CVSLS_SUCCESS);
219 }

258 /*
259  This routine does the setup operations for the CVSuperLUMT linear
260  solver module. It calls the Jacobian evaluation routine,
261  updates counters, and calls the LU factorization routine.
262  The return value is either
263  CVSLS_SUCCESS = 0 if successful,
264  +1 if the jac routine failed recoverably or the
265  LU factorization failed, or
266  -1 if the jac routine failed unrecoverably.
267 */
268 
269 static int cvSuperLUMTSetup(CVodeMem cv_mem, int convfail, N_Vector ypred,
270  N_Vector fpred, booleantype *jcurPtr,
271  N_Vector vtemp1, N_Vector vtemp2, N_Vector vtemp3)
272 {
273  booleantype jbad, jok;
274  int retval, info;
275  int nprocs, panel_size, relax, permc_spec, lwork;
276  int *perm_r, *perm_c;
277  long int nst, nstlj;
278  realtype tn, gamma, gammap, dgamma;
279  double diag_pivot_thresh, drop_tol;
280  fact_t fact;
281  trans_t trans;
282  yes_no_t refact, usepr;
283  CVSlsMem cvsls_mem;
284  CVSlsSparseJacFn jaceval;
285  SuperMatrix *A, *AC, *L, *U;
286  Gstat_t *Gstat;
287  superlumt_options_t *superlumt_options;
288  SLUMTData slumt_data;
289  SlsMat JacMat, savedJ;
290  void *jacdata;
291  void *work;
292 
293  cvsls_mem = (CVSlsMem) (cv_mem->cv_lmem);
294  tn = cv_mem->cv_tn;
295  gamma = cv_mem->cv_gamma;
296  gammap = cv_mem->cv_gammap;
297  nst = cv_mem->cv_nst;
298 
299  slumt_data = (SLUMTData) cvsls_mem->s_solver_data;
300 
301  jaceval = cvsls_mem->s_jaceval;
302  jacdata = cvsls_mem->s_jacdata;
303  JacMat = cvsls_mem->s_JacMat;
304  savedJ = cvsls_mem->s_savedJ;
305  nstlj = cvsls_mem->s_nstlj;
306 
307  superlumt_options = slumt_data->superlumt_options;
308  A = slumt_data->s_A;
309  AC = slumt_data->s_AC;
310  L = slumt_data->s_L;
311  U = slumt_data->s_U;
312  Gstat = slumt_data->Gstat;
313  perm_r = slumt_data->perm_r;
314  perm_c = slumt_data->perm_c;
315  nprocs = slumt_data->num_threads;
316  diag_pivot_thresh = slumt_data->diag_pivot_thresh;
317 
318  /* Set option values for SuperLU_MT */
319  panel_size = sp_ienv(1);
320  relax = sp_ienv(2);
321  fact = EQUILIBRATE;
322  trans = NOTRANS;
323  usepr = NO;
324  drop_tol = 0.0;
325  lwork = 0;
326  work = NULL;
327 
328  /* Check that Jacobian eval routine is set */
329  if (jaceval == NULL) {
330  cvProcessError(cv_mem, CVSLS_JAC_NOSET, "CVSLS", "cvSuperLUMTSetup",
331  MSGSP_JAC_NOSET);
332  free(cvsls_mem); cvsls_mem = NULL;
333  return(CVSLS_JAC_NOSET);
334  }
335 
336  /* Determine whether Jacobian needs to be recalculated */
337  dgamma = SUNRabs((gamma/gammap) - ONE);
338  jbad = (nst == 0) || (nst > nstlj + CVS_MSBJ) ||
339  ((convfail == CV_FAIL_BAD_J) && (dgamma < CVS_DGMAX)) ||
340  (convfail == CV_FAIL_OTHER);
341  jok = !jbad;
342 
343  if (jok) {
344  /* If jok = TRUE, use saved copy of J */
345  *jcurPtr = FALSE;
346  CopySparseMat(savedJ, JacMat);
347  } else {
348  /* If jok = FALSE, call jac routine for new J value */
349  cvsls_mem->s_nje++;
350  cvsls_mem->s_nstlj = nst;
351  *jcurPtr = TRUE;
352  SlsSetToZero(JacMat);
353  retval = jaceval(tn, ypred, fpred, JacMat, jacdata, vtemp1, vtemp2, vtemp3);
354  if (retval < 0) {
355  cvProcessError(cv_mem, CVSLS_JACFUNC_UNRECVR, "CVSLS", "cvSuperLUMTSetup", MSGSP_JACFUNC_FAILED);
356  cvsls_mem->s_last_flag = CVSLS_JACFUNC_UNRECVR;
357  return(-1);
358  }
359  if (retval > 0) {
360  cvsls_mem->s_last_flag = CVSLS_JACFUNC_RECVR;
361  return(1);
362  }
363 
364  CopySparseMat(JacMat, savedJ);
365  }
366 
367  /* Scale and add I to get M = I - gamma*J */
368  ScaleSparseMat(-gamma, JacMat);
369  AddIdentitySparseMat(JacMat);
370 
371  if (A->Store) {
372  SUPERLU_FREE(A->Store);
373  }
374  dCreate_CompCol_Matrix(A, JacMat->M, JacMat->N, JacMat->NNZ,
375  JacMat->data, JacMat->rowvals, JacMat->colptrs,
376  SLU_NC, SLU_D, SLU_GE);
377 
378  if (cvsls_mem->s_first_factorize) {
379  /* ------------------------------------------------------------
380  Get column permutation vector perm_c[], according to permc_spec:
381  permc_spec = 3: approximate minimum degree for unsymmetric matrices
382  ------------------------------------------------------------*/
383  permc_spec = slumt_data->s_ordering;
384  get_perm_c(permc_spec, A, perm_c);
385 
386  refact= NO;
387  cvsls_mem->s_first_factorize = 0;
388  }
389  else {
390  /* ------------------------------------------------------------
391  Re-initialize statistics variables
392  ------------------------------------------------------------*/
393  StatInit(JacMat->N, nprocs, Gstat);
394  Destroy_CompCol_Permuted(AC);
395  refact= YES;
396  }
397 
398  /* ------------------------------------------------------------
399  Initialize the option structure superlumt_options using the
400  user-input parameters; Subsequent calls will re-initialize
401  options.
402  Apply perm_c to the columns of original A to form AC.
403  ------------------------------------------------------------*/
404  pdgstrf_init(nprocs, fact, trans, refact, panel_size, relax,
405  diag_pivot_thresh, usepr, drop_tol, perm_c, perm_r,
406  work, lwork, A, AC, superlumt_options, Gstat);
407  /* ------------------------------------------------------------
408  Compute the LU factorization of A.
409  The following routine will create nprocs threads.
410  ------------------------------------------------------------*/
411  pdgstrf(superlumt_options, AC, perm_r, L, U, Gstat, &info);
412 
413  if (info != 0) {
414  cvsls_mem->s_last_flag = info;
415  return(+1);
416  }
417 
418  cvsls_mem->s_last_flag = CVSLS_SUCCESS;
419 
420  return(0);
421 }

424  This routine handles the solve operation for the CVSuperLUMT linear

429 static int cvSuperLUMTSolve(CVodeMem cv_mem, N_Vector b, N_Vector weight,
430  N_Vector ycur, N_Vector fcur)
431 {
432  int info, trans, lmm;
433  int *perm_r, *perm_c;
434  realtype gamrat;
435  CVSlsMem cvsls_mem;
436  SuperMatrix *L, *U, *B;
437  Gstat_t *Gstat;
438  DNformat *Bstore;
439  SLUMTData slumt_data;
440  realtype *bd;
441 
442  gamrat = cv_mem->cv_gamrat;
443  lmm = cv_mem->cv_lmm;
444 
445  cvsls_mem = (CVSlsMem) cv_mem->cv_lmem;
446  slumt_data = (SLUMTData) cvsls_mem->s_solver_data;
447 
448  L = slumt_data->s_L;
449  U = slumt_data->s_U;
450  perm_r = slumt_data->perm_r;
451  perm_c = slumt_data->perm_c;
452  Gstat = slumt_data->Gstat;
453  B = slumt_data->s_B;
454 
455  bd = N_VGetArrayPointer(b);
456  Bstore = (DNformat *) (B->Store);
457  Bstore->nzval = bd;
458 
459  /* Call SuperLUMT to solve the linear system using L and U */
460  trans = NOTRANS;
461  dgstrs(trans, L, U, perm_r, perm_c, B, Gstat, &info);
462 
463  /* Scale the correction to account for change in gamma. */
464  if ((lmm == CV_BDF) && (gamrat != ONE)) {
465  N_VScale(TWO/(ONE + gamrat), b, b);
466  }
467 
468  Bstore->nzval = NULL;
469 
470  cvsls_mem->s_last_flag = CVSLS_SUCCESS;
471  return(CVSLS_SUCCESS);
472 }

501  cvsls_mem->s_JacMat = NULL;

506  }
507 
508  free(slumt_data->s_B);
509  free(slumt_data->s_A);
510  free(slumt_data->s_AC);
511  free(slumt_data->s_L);
512  free(slumt_data->s_U);
513 
514  free(slumt_data);
515  slumt_data = NULL;
516  free(cv_mem->cv_lmem);
517 
518  return;

578 
579 int CVSuperLUMTB(void *cvode_mem, int which, int num_threads,
580  int n, int nnz)
581 {
582  CVodeMem cv_mem;
583  CVadjMem ca_mem;
584  CVodeBMem cvB_mem;
585  void *cvodeB_mem;
586  CVSlsMemB cvslsB_mem;
587  int flag;
588 
589  /* Check if cvode_mem exists */
590  if (cvode_mem == NULL) {
591  cvProcessError(NULL, CVSLS_MEM_NULL, "CVSSuperLUMT", "CVSuperLUMTB", MSGSP_CVMEM_NULL);
592  return(CVSLS_MEM_NULL);
593  }
594  cv_mem = (CVodeMem) cvode_mem;
595 
596  /* Was ASA initialized? */
597  if (cv_mem->cv_adjMallocDone == FALSE) {
598  cvProcessError(cv_mem, CVSLS_NO_ADJ, "CVSSuperLUMT", "CVSuperLUMTB", MSGSP_NO_ADJ);
599  return(CVSLS_NO_ADJ);
600  }
