121 void *zmq_ctx_new (void)
122 {
123 #if defined ZMQ_HAVE_OPENPGM
124 
125  // Init PGM transport. Ensure threading and timer are enabled. Find PGM
126  // protocol ID. Note that if you want to use gettimeofday and sleep for
127  // openPGM timing, set environment variables PGM_TIMER to "GTOD" and
128  // PGM_SLEEP to "USLEEP".
129  pgm_error_t *pgm_error = NULL;
130  const bool ok = pgm_init (&pgm_error);
131  if (ok != TRUE) {
132 
133  // Invalid parameters don't set pgm_error_t
134  zmq_assert (pgm_error != NULL);
135  if (pgm_error->domain == PGM_ERROR_DOMAIN_TIME && (
136  pgm_error->code == PGM_ERROR_FAILED)) {
137 
138  // Failed to access RTC or HPET device.
139  pgm_error_free (pgm_error);
140  errno = EINVAL;
141  return NULL;
142  }
143 
144  // PGM_ERROR_DOMAIN_ENGINE: WSAStartup errors or missing WSARecvMsg.
145  zmq_assert (false);
146  }
147 #endif
148 
149 #ifdef ZMQ_HAVE_WINDOWS
150  // Intialise Windows sockets. Note that WSAStartup can be called multiple
151  // times given that WSACleanup will be called for each WSAStartup.
152  // We do this before the ctx constructor since its embedded mailbox_t
153  // object needs Winsock to be up and running.
154  WORD version_requested = MAKEWORD (2, 2);
155  WSADATA wsa_data;
156  int rc = WSAStartup (version_requested, &wsa_data);
157  zmq_assert (rc == 0);
158  zmq_assert (LOBYTE (wsa_data.wVersion) == 2 &&
159  HIBYTE (wsa_data.wVersion) == 2);
160 #endif
161 
162  // Create 0MQ context.
163  zmq::ctx_t *ctx = new (std::nothrow) zmq::ctx_t;
164  alloc_assert (ctx);
165  return ctx;
166 }

168 int zmq_ctx_term (void *ctx_)
169 {
170  if (!ctx_ || !((zmq::ctx_t *) ctx_)->check_tag ()) {
171  errno = EFAULT;
172  return -1;
173  }
174 
175  int rc = ((zmq::ctx_t *) ctx_)->terminate ();
176  int en = errno;
177 
178  // Shut down only if termination was not interrupted by a signal.
179  if (!rc || en != EINTR) {
180 #ifdef ZMQ_HAVE_WINDOWS
181  // On Windows, uninitialise socket layer.
182  rc = WSACleanup ();
183  wsa_assert (rc != SOCKET_ERROR);
184 #endif
185 
186 #if defined ZMQ_HAVE_OPENPGM
187  // Shut down the OpenPGM library.
188  if (pgm_shutdown () != TRUE)
189  zmq_assert (false);
190 #endif
191  }
192 
193  errno = en;
194  return rc;
195 }

393 int zmq_send (void *s_, const void *buf_, size_t len_, int flags_)
394 {
395  if (!s_ || !((zmq::socket_base_t*) s_)->check_tag ()) {
396  errno = ENOTSOCK;
397  return -1;
398  }
399  zmq_msg_t msg;
400  if (zmq_msg_init_size (&msg, len_))
401  return -1;
402 
403  // We explicitly allow a send from NULL, size zero
404  if (len_) {
405  assert (buf_);
406  memcpy (zmq_msg_data (&msg), buf_, len_);
407  }
408  zmq::socket_base_t *s = (zmq::socket_base_t *) s_;
409  int rc = s_sendmsg (s, &msg, flags_);
410  if (unlikely (rc < 0)) {
411  int err = errno;
412  int rc2 = zmq_msg_close (&msg);
413  errno_assert (rc2 == 0);
414  errno = err;
415  return -1;
416  }
417  // Note the optimisation here. We don't close the msg object as it is
418  // empty anyway. This may change when implementation of zmq_msg_t changes.
419  return rc;

513 int zmq_recv (void *s_, void *buf_, size_t len_, int flags_)
514 {
515  if (!s_ || !((zmq::socket_base_t*) s_)->check_tag ()) {
516  errno = ENOTSOCK;
517  return -1;
518  }
519  zmq_msg_t msg;
520  int rc = zmq_msg_init (&msg);
521  errno_assert (rc == 0);
522 
523  zmq::socket_base_t *s = (zmq::socket_base_t *) s_;
524  int nbytes = s_recvmsg (s, &msg, flags_);
525  if (unlikely (nbytes < 0)) {
526  int err = errno;
527  rc = zmq_msg_close (&msg);
528  errno_assert (rc == 0);
529  errno = err;
530  return -1;
531  }

533  // An oversized message is silently truncated.
534  size_t to_copy = size_t (nbytes) < len_ ? size_t (nbytes) : len_;
535 
536  // We explicitly allow a null buffer argument if len is zero
537  if (to_copy) {
538  assert (buf_);
539  memcpy (buf_, zmq_msg_data (&msg), to_copy);
540  }
541  rc = zmq_msg_close (&msg);
542  errno_assert (rc == 0);
543 
544  return nbytes;

547 // Receive a multi-part message
548 //
549 // Receives up to *count_ parts of a multi-part message.
550 // Sets *count_ to the actual number of parts read.
551 // ZMQ_RCVMORE is set to indicate if a complete multi-part message was read.
552 // Returns number of message parts read, or -1 on error.
553 //
554 // Note: even if -1 is returned, some parts of the message
555 // may have been read. Therefore the client must consult
556 // *count_ to retrieve message parts successfully read,
557 // even if -1 is returned.
558 //
559 // The iov_base* buffers of each iovec *a_ filled in by this
560 // function may be freed using free().
561 // TODO: this function has no man page
562 //
563 int zmq_recviov (void *s_, iovec *a_, size_t *count_, int flags_)
564 {
565  if (!s_ || !((zmq::socket_base_t*) s_)->check_tag ()) {
566  errno = ENOTSOCK;
567  return -1;
568  }
569  if (unlikely (!count_ || *count_ <= 0 || !a_)) {
570  errno = EINVAL;
571  return -1;
572  }
573 
574  zmq::socket_base_t *s = (zmq::socket_base_t *) s_;
575 
576  size_t count = *count_;
577  int nread = 0;
578  bool recvmore = true;
579 
580  *count_ = 0;
581 
582  for (size_t i = 0; recvmore && i < count; ++i) {
583 
584  zmq_msg_t msg;
585  int rc = zmq_msg_init (&msg);
586  errno_assert (rc == 0);
587 
588  int nbytes = s_recvmsg (s, &msg, flags_);
589  if (unlikely (nbytes < 0)) {
590  int err = errno;
591  rc = zmq_msg_close (&msg);
592  errno_assert (rc == 0);
593  errno = err;
594  nread = -1;
595  break;
596  }
597 
598  a_[i].iov_len = zmq_msg_size (&msg);
599  a_[i].iov_base = static_cast<char *> (malloc(a_[i].iov_len));
600  if (unlikely (!a_[i].iov_base)) {
601  errno = ENOMEM;
602  return -1;
603  }
604  memcpy(a_[i].iov_base,static_cast<char *> (zmq_msg_data (&msg)),
605  a_[i].iov_len);
606  // Assume zmq_socket ZMQ_RVCMORE is properly set.
607  zmq::msg_t* p_msg = reinterpret_cast<zmq::msg_t*>(&msg);
608  recvmore = p_msg->flags() & zmq::msg_t::more;
609  rc = zmq_msg_close(&msg);
610  errno_assert (rc == 0);
611  ++*count_;
612  ++nread;
613  }
614  return nread;
615 }

754 #if defined ZMQ_HAVE_POLLER
755 inline int zmq_poller_poll (zmq_pollitem_t *items_, int nitems_, long timeout_)
756 {
757  // implement zmq_poll on top of zmq_poller
758  int rc;
759  zmq_poller_event_t *events;
760  zmq::socket_poller_t poller;
761  events = new (std::nothrow) zmq_poller_event_t[nitems_];
762  alloc_assert(events);
763 
764  bool repeat_items = false;
765  // Register sockets with poller
766  for (int i = 0; i < nitems_; i++) {
767  items_[i].revents = 0;
768 
769  bool modify = false;
770  short e = items_[i].events;
771  if (items_[i].socket) {
772  // Poll item is a 0MQ socket.
773  for (int j = 0; j < i; ++j) {
774  // Check for repeat entries
775  if (items_[j].socket == items_[i].socket) {
776  repeat_items = true;
777  modify = true;
778  e |= items_[j].events;
779  }
780  }
781  if (modify) {
782  rc = zmq_poller_modify (&poller, items_[i].socket, e);
783  } else {
784  rc = zmq_poller_add (&poller, items_[i].socket, NULL, e);
785  }
786  if (rc < 0) {
787  delete [] events;
788  return rc;
789  }
790  } else {
791  // Poll item is a raw file descriptor.
792  for (int j = 0; j < i; ++j) {
793  // Check for repeat entries
794  if (!items_[j].socket && items_[j].fd == items_[i].fd) {
795  repeat_items = true;
796  modify = true;
797  e |= items_[j].events;
798  }
799  }
800  if (modify) {
801  rc = zmq_poller_modify_fd (&poller, items_[i].fd, e);
802  } else {
803  rc = zmq_poller_add_fd (&poller, items_[i].fd, NULL, e);
804  }
805  if (rc < 0) {
806  delete [] events;
807  return rc;
808  }
809  }
810  }
811 
812  // Wait for events
813  rc = zmq_poller_wait_all (&poller, events, nitems_, timeout_);
814  if (rc < 0) {
815  delete [] events;
816  if (zmq_errno() == ETIMEDOUT) {
817  return 0;
818  }
819  return rc;
820  }
821 
822  // Transform poller events into zmq_pollitem events.
823  // items_ contains all items, while events only contains fired events.
824  // If no sockets are repeated (likely), the two are still co-ordered, so step through the items
825  // checking for matches only on the first event.
826  // If there are repeat items, they cannot be assumed to be co-ordered,
827  // so each pollitem must check fired events from the beginning.
828  int j_start = 0, found_events = rc;
829  for (int i = 0; i < nitems_; i++) {
830  for (int j = j_start; j < found_events; ++j) {
831  if (
832  (items_[i].socket && items_[i].socket == events[j].socket) ||
833  (!(items_[i].socket || events[j].socket) && items_[i].fd == events[j].fd)
834  ) {
835  items_[i].revents = events[j].events & items_[i].events;
836  if (!repeat_items) {
837  // no repeats, we can ignore events we've already seen
838  j_start++;
839  }
840  break;
841  }
842  if (!repeat_items) {
843  // no repeats, never have to look at j > j_start
844  break;
845  }
846  }
847  }
848 
849  // Cleanup
850  delete [] events;
851  return rc;
852 }

855 int zmq_poll (zmq_pollitem_t *items_, int nitems_, long timeout_)
856 {
857  // TODO: the function implementation can just call zmq_pollfd_poll with
858  // pollfd as NULL, however pollfd is not yet stable.
859 #if defined ZMQ_HAVE_POLLER
860  // if poller is present, use that.
861  return zmq_poller_poll(items_, nitems_, timeout_);
862 #else
863 #if defined ZMQ_POLL_BASED_ON_POLL
864  if (unlikely (nitems_ < 0)) {
865  errno = EINVAL;
866  return -1;
867  }
868  if (unlikely (nitems_ == 0)) {
869  if (timeout_ == 0)
870  return 0;
871 #if defined ZMQ_HAVE_WINDOWS
872  Sleep (timeout_ > 0 ? timeout_ : INFINITE);
873  return 0;
874 #elif defined ZMQ_HAVE_ANDROID
875  usleep (timeout_ * 1000);
876  return 0;
877 #else
878  return usleep (timeout_ * 1000);
879 #endif
880  }
881 
882  if (!items_) {
883  errno = EFAULT;
884  return -1;
885  }
886 
887  zmq::clock_t clock;
888  uint64_t now = 0;
889  uint64_t end = 0;
890  pollfd spollfds[ZMQ_POLLITEMS_DFLT];
891  pollfd *pollfds = spollfds;
892 
893  if (nitems_ > ZMQ_POLLITEMS_DFLT) {
894  pollfds = (pollfd*) malloc (nitems_ * sizeof (pollfd));
895  alloc_assert (pollfds);
896  }
897 
898  // Build pollset for poll () system call.
899  for (int i = 0; i != nitems_; i++) {
900 
901  // If the poll item is a 0MQ socket, we poll on the file descriptor
902  // retrieved by the ZMQ_FD socket option.
903  if (items_ [i].socket) {
904  size_t zmq_fd_size = sizeof (zmq::fd_t);
905  if (zmq_getsockopt (items_ [i].socket, ZMQ_FD, &pollfds [i].fd,
906  &zmq_fd_size) == -1) {
907  if (pollfds != spollfds)
908  free (pollfds);
909  return -1;
910  }
911  pollfds [i].events = items_ [i].events ? POLLIN : 0;
912  }
913  // Else, the poll item is a raw file descriptor. Just convert the
914  // events to normal POLLIN/POLLOUT for poll ().
915  else {
916  pollfds [i].fd = items_ [i].fd;
917  pollfds [i].events =
918  (items_ [i].events & ZMQ_POLLIN ? POLLIN : 0) |
919  (items_ [i].events & ZMQ_POLLOUT ? POLLOUT : 0) |
920  (items_ [i].events & ZMQ_POLLPRI ? POLLPRI : 0);
921  }
922  }
923 
924  bool first_pass = true;
925  int nevents = 0;
926 
927  while (true) {
928  // Compute the timeout for the subsequent poll.
929  int timeout;
930  if (first_pass)
931  timeout = 0;
932  else
933  if (timeout_ < 0)
934  timeout = -1;
935  else
936  timeout = end - now;
937 
938  // Wait for events.
939  {
940  int rc = poll (pollfds, nitems_, timeout);
941  if (rc == -1 && errno == EINTR) {
942  if (pollfds != spollfds)
943  free (pollfds);
944  return -1;
945  }
946  errno_assert (rc >= 0);
947  }
948  // Check for the events.
949  for (int i = 0; i != nitems_; i++) {
950 
951  items_ [i].revents = 0;
952 
953  // The poll item is a 0MQ socket. Retrieve pending events
954  // using the ZMQ_EVENTS socket option.
955  if (items_ [i].socket) {
956  size_t zmq_events_size = sizeof (uint32_t);
957  uint32_t zmq_events;
958  if (zmq_getsockopt (items_ [i].socket, ZMQ_EVENTS, &zmq_events,
959  &zmq_events_size) == -1) {
960  if (pollfds != spollfds)
961  free (pollfds);
962  return -1;
963  }
964  if ((items_ [i].events & ZMQ_POLLOUT) &&
965  (zmq_events & ZMQ_POLLOUT))
966  items_ [i].revents |= ZMQ_POLLOUT;
967  if ((items_ [i].events & ZMQ_POLLIN) &&
968  (zmq_events & ZMQ_POLLIN))
969  items_ [i].revents |= ZMQ_POLLIN;
970  }
971  // Else, the poll item is a raw file descriptor, simply convert
972  // the events to zmq_pollitem_t-style format.
973  else {
974  if (pollfds [i].revents & POLLIN)
975  items_ [i].revents |= ZMQ_POLLIN;
976  if (pollfds [i].revents & POLLOUT)
977  items_ [i].revents |= ZMQ_POLLOUT;
978  if (pollfds [i].revents & POLLPRI)
979  items_ [i].revents |= ZMQ_POLLPRI;
980  if (pollfds [i].revents & ~(POLLIN | POLLOUT | POLLPRI))
981  items_ [i].revents |= ZMQ_POLLERR;
982  }
983 
984  if (items_ [i].revents)
985  nevents++;
986  }
987 
988  // If timeout is zero, exit immediately whether there are events or not.
989  if (timeout_ == 0)
990  break;
991 
992  // If there are events to return, we can exit immediately.
993  if (nevents)
994  break;
995 
996  // At this point we are meant to wait for events but there are none.
997  // If timeout is infinite we can just loop until we get some events.
998  if (timeout_ < 0) {
999  if (first_pass)
1000  first_pass = false;
1001  continue;
1002  }
1003 
1004  // The timeout is finite and there are no events. In the first pass
1005  // we get a timestamp of when the polling have begun. (We assume that
1006  // first pass have taken negligible time). We also compute the time
1007  // when the polling should time out.
1008  if (first_pass) {
1009  now = clock.now_ms ();
1010  end = now + timeout_;
1011  if (now == end)
1012  break;
1013  first_pass = false;
1014  continue;
1015  }
1016 
1017  // Find out whether timeout have expired.
1018  now = clock.now_ms ();
1019  if (now >= end)
1020  break;
1021  }
1022 
1023  if (pollfds != spollfds)
1024  free (pollfds);
1025  return nevents;
1026 
1027 #elif defined ZMQ_POLL_BASED_ON_SELECT
1028 
1029  if (unlikely (nitems_ < 0)) {
1030  errno = EINVAL;
1031  return -1;
1032  }
1033  if (unlikely (nitems_ == 0)) {
1034  if (timeout_ == 0)
1035  return 0;
1036 #if defined ZMQ_HAVE_WINDOWS
1037  Sleep (timeout_ > 0 ? timeout_ : INFINITE);
1038  return 0;
1039 #else
1040  return usleep (timeout_ * 1000);
1041 #endif
1042  }
1043  zmq::clock_t clock;
1044  uint64_t now = 0;
1045  uint64_t end = 0;
1046 
1047  // Ensure we do not attempt to select () on more than FD_SETSIZE
1048  // file descriptors.
1049  zmq_assert (nitems_ <= FD_SETSIZE);
1050 
1051  fd_set pollset_in;
1052  FD_ZERO (&pollset_in);
1053  fd_set pollset_out;
1054  FD_ZERO (&pollset_out);
1055  fd_set pollset_err;
1056  FD_ZERO (&pollset_err);
1057 
1058  zmq::fd_t maxfd = 0;
1059 
1060  // Build the fd_sets for passing to select ().
1061  for (int i = 0; i != nitems_; i++) {
1062 
1063  // If the poll item is a 0MQ socket we are interested in input on the
1064  // notification file descriptor retrieved by the ZMQ_FD socket option.
1065  if (items_ [i].socket) {
1066  size_t zmq_fd_size = sizeof (zmq::fd_t);
1067  zmq::fd_t notify_fd;
1068  if (zmq_getsockopt (items_ [i].socket, ZMQ_FD, &notify_fd,
1069  &zmq_fd_size) == -1)
1070  return -1;
1071  if (items_ [i].events) {
1072  FD_SET (notify_fd, &pollset_in);
1073  if (maxfd < notify_fd)
1074  maxfd = notify_fd;
1075  }
1076  }
1077  // Else, the poll item is a raw file descriptor. Convert the poll item
1078  // events to the appropriate fd_sets.
1079  else {
1080  if (items_ [i].events & ZMQ_POLLIN)
1081  FD_SET (items_ [i].fd, &pollset_in);
1082  if (items_ [i].events & ZMQ_POLLOUT)
1083  FD_SET (items_ [i].fd, &pollset_out);
1084  if (items_ [i].events & ZMQ_POLLERR)
1085  FD_SET (items_ [i].fd, &pollset_err);
1086  if (maxfd < items_ [i].fd)
1087  maxfd = items_ [i].fd;
1088  }
1089  }
1090 
1091  bool first_pass = true;
1092  int nevents = 0;
1093  fd_set inset, outset, errset;
1094 
1095  while (true) {
1096 
1097  // Compute the timeout for the subsequent poll.
1098  timeval timeout;
1099  timeval *ptimeout;
1100  if (first_pass) {
1101  timeout.tv_sec = 0;
1102  timeout.tv_usec = 0;
1103  ptimeout = &timeout;
1104  }
1105  else
1106  if (timeout_ < 0)
1107  ptimeout = NULL;
1108  else {
1109  timeout.tv_sec = (long) ((end - now) / 1000);
1110  timeout.tv_usec = (long) ((end - now) % 1000 * 1000);
1111  ptimeout = &timeout;
1112  }
1113 
1114  // Wait for events. Ignore interrupts if there's infinite timeout.
1115  while (true) {
1116 #if defined ZMQ_HAVE_WINDOWS
1117  // On Windows we don't need to copy the whole fd_set.
1118  // SOCKETS are continuous from the beginning of fd_array in fd_set.
1119  // We just need to copy fd_count elements of fd_array.
1120  // We gain huge memcpy() improvement if number of used SOCKETs is much lower than FD_SETSIZE.
1121  memcpy (&inset, &pollset_in , (char *) (pollset_in.fd_array + pollset_in.fd_count ) - (char *) &pollset_in );
1122  memcpy (&outset, &pollset_out, (char *) (pollset_out.fd_array + pollset_out.fd_count) - (char *) &pollset_out);
1123  memcpy (&errset, &pollset_err, (char *) (pollset_err.fd_array + pollset_err.fd_count) - (char *) &pollset_err);
1124  int rc = select (0, &inset, &outset, &errset, ptimeout);
1125  if (unlikely (rc == SOCKET_ERROR)) {
1126  errno = zmq::wsa_error_to_errno (WSAGetLastError ());
1127  wsa_assert (errno == ENOTSOCK);
1128  return -1;
1129  }
1130 #else
1131  memcpy (&inset, &pollset_in, sizeof (fd_set));
1132  memcpy (&outset, &pollset_out, sizeof (fd_set));
1133  memcpy (&errset, &pollset_err, sizeof (fd_set));
1134  int rc = select (maxfd + 1, &inset, &outset, &errset, ptimeout);
1135  if (unlikely (rc == -1)) {
1136  errno_assert (errno == EINTR || errno == EBADF);
1137  return -1;
1138  }
1139 #endif
1140  break;
1141  }
1142 
1143  // Check for the events.
1144  for (int i = 0; i != nitems_; i++) {
1145 
1146  items_ [i].revents = 0;
1147 
1148  // The poll item is a 0MQ socket. Retrieve pending events
1149  // using the ZMQ_EVENTS socket option.
1150  if (items_ [i].socket) {
1151  size_t zmq_events_size = sizeof (uint32_t);
1152  uint32_t zmq_events;
1153  if (zmq_getsockopt (items_ [i].socket, ZMQ_EVENTS, &zmq_events,
1154  &zmq_events_size) == -1)
1155  return -1;
1156  if ((items_ [i].events & ZMQ_POLLOUT) &&
1157  (zmq_events & ZMQ_POLLOUT))
1158  items_ [i].revents |= ZMQ_POLLOUT;
1159  if ((items_ [i].events & ZMQ_POLLIN) &&
1160  (zmq_events & ZMQ_POLLIN))
1161  items_ [i].revents |= ZMQ_POLLIN;
1162  }
1163  // Else, the poll item is a raw file descriptor, simply convert
1164  // the events to zmq_pollitem_t-style format.
1165  else {
1166  if (FD_ISSET (items_ [i].fd, &inset))
1167  items_ [i].revents |= ZMQ_POLLIN;
1168  if (FD_ISSET (items_ [i].fd, &outset))
1169  items_ [i].revents |= ZMQ_POLLOUT;
1170  if (FD_ISSET (items_ [i].fd, &errset))
1171  items_ [i].revents |= ZMQ_POLLERR;
1172  }
1173 
1174  if (items_ [i].revents)
1175  nevents++;
1176  }
1177 
1178  // If timeout is zero, exit immediately whether there are events or not.
1179  if (timeout_ == 0)
1180  break;
1181 
1182  // If there are events to return, we can exit immediately.
1183  if (nevents)
1184  break;
1185 
1186  // At this point we are meant to wait for events but there are none.
1187  // If timeout is infinite we can just loop until we get some events.
1188  if (timeout_ < 0) {
1189  if (first_pass)
1190  first_pass = false;
1191  continue;
1192  }
1193 
1194  // The timeout is finite and there are no events. In the first pass
1195  // we get a timestamp of when the polling have begun. (We assume that
1196  // first pass have taken negligible time). We also compute the time
1197  // when the polling should time out.
1198  if (first_pass) {
1199  now = clock.now_ms ();
1200  end = now + timeout_;
1201  if (now == end)
1202  break;
1203  first_pass = false;
1204  continue;
1205  }
1206 
1207  // Find out whether timeout have expired.
1208  now = clock.now_ms ();
1209  if (now >= end)
1210  break;
1211  }
1212 
1213  return nevents;
1214 
1215 #else
1216  // Exotic platforms that support neither poll() nor select().
1217  errno = ENOTSUP;
1218  return -1;
1219 #endif
1220 #endif // ZMQ_HAVE_POLLER
1221 }
