321  *
322  * CVodeCreate creates an internal memory block for a problem to

324  * If successful, CVodeCreate returns a pointer to the problem memory.

438 int CVodeInit(void *cvode_mem, CVRhsFn f, realtype t0, N_Vector y0)
439 {
440  CVodeMem cv_mem;
441  booleantype nvectorOK, allocOK;
442  long int lrw1, liw1;
443  int i,k;
444 
445  /* Check cvode_mem */
446 
447  if (cvode_mem==NULL) {
448  cvProcessError(NULL, CV_MEM_NULL, "CVODE", "CVodeInit", MSGCV_NO_MEM);
449  return(CV_MEM_NULL);
450  }
451  cv_mem = (CVodeMem) cvode_mem;
452 
453  /* Check for legal input parameters */
454 
455  if (y0==NULL) {
456  cvProcessError(cv_mem, CV_ILL_INPUT, "CVODE", "CVodeInit", MSGCV_NULL_Y0);
457  return(CV_ILL_INPUT);
458  }
459 
460  if (f == NULL) {
461  cvProcessError(cv_mem, CV_ILL_INPUT, "CVODE", "CVodeInit", MSGCV_NULL_F);
462  return(CV_ILL_INPUT);
463  }
464 
465  /* Test if all required vector operations are implemented */
466 
467  nvectorOK = cvCheckNvector(y0);
468  if(!nvectorOK) {
469  cvProcessError(cv_mem, CV_ILL_INPUT, "CVODE", "CVodeInit", MSGCV_BAD_NVECTOR);
470  return(CV_ILL_INPUT);
471  }
472 
473  /* Set space requirements for one N_Vector */
474 
475  if (y0->ops->nvspace != NULL) {
476  N_VSpace(y0, &lrw1, &liw1);
477  } else {
478  lrw1 = 0;
479  liw1 = 0;
480  }
481  cv_mem->cv_lrw1 = lrw1;
482  cv_mem->cv_liw1 = liw1;
483 
484  /* Allocate the vectors (using y0 as a template) */
485 
486  allocOK = cvAllocVectors(cv_mem, y0);
487  if (!allocOK) {
488  cvProcessError(cv_mem, CV_MEM_FAIL, "CVODE", "CVodeInit", MSGCV_MEM_FAIL);
489  return(CV_MEM_FAIL);
490  }
491 
492  /* All error checking is complete at this point */
493 
494  /* Copy the input parameters into CVODE state */
495 
496  cv_mem->cv_f = f;
497  cv_mem->cv_tn = t0;
498 
499  /* Set step parameters */
500 
501  cv_mem->cv_q = 1;
502  cv_mem->cv_L = 2;
503  cv_mem->cv_qwait = cv_mem->cv_L;
504  cv_mem->cv_etamax = ETAMX1;
505 
506  cv_mem->cv_qu = 0;
507  cv_mem->cv_hu = ZERO;
508  cv_mem->cv_tolsf = ONE;
509 
510  /* Set the linear solver addresses to NULL.
511  (We check != NULL later, in CVode, if using CV_NEWTON.) */
512 
513  cv_mem->cv_linit = NULL;
514  cv_mem->cv_lsetup = NULL;
515  cv_mem->cv_lsolve = NULL;
516  cv_mem->cv_lfree = NULL;
517  cv_mem->cv_lmem = NULL;
518 
519  /* Initialize zn[0] in the history array */
520 
521  N_VScale(ONE, y0, cv_mem->cv_zn[0]);
522 
523  /* Initialize all the counters */
524 
525  cv_mem->cv_nst = 0;
526  cv_mem->cv_nfe = 0;
527  cv_mem->cv_ncfn = 0;
528  cv_mem->cv_netf = 0;
529  cv_mem->cv_nni = 0;
530  cv_mem->cv_nsetups = 0;
531  cv_mem->cv_nhnil = 0;
532  cv_mem->cv_nstlp = 0;
533  cv_mem->cv_nscon = 0;
534  cv_mem->cv_nge = 0;
535 
536  cv_mem->cv_irfnd = 0;
537 
538  /* Initialize other integrator optional outputs */
539 
540  cv_mem->cv_h0u = ZERO;
541  cv_mem->cv_next_h = ZERO;
542  cv_mem->cv_next_q = 0;
543 
544  /* Initialize Stablilty Limit Detection data */
545  /* NOTE: We do this even if stab lim det was not
546  turned on yet. This way, the user can turn it
547  on at any time */
548 
549  cv_mem->cv_nor = 0;
550  for (i = 1; i <= 5; i++)
551  for (k = 1; k <= 3; k++)
552  cv_mem->cv_ssdat[i-1][k-1] = ZERO;
553 
554  /* Problem has been successfully initialized */
555 
556  cv_mem->cv_MallocDone = TRUE;
557 
558  return(CV_SUCCESS);

1058  * CVode
1059  *
1060  * This routine is the main driver of the CVODE package.
1061  *
1062  * It integrates over a time interval defined by the user, by calling
1063  * cvStep to do internal time steps.
1064  *
1065  * The first time that CVode is called for a successfully initialized
1066  * problem, it computes a tentative initial step size h.
1067  *
1068  * CVode supports two modes, specified by itask: CV_NORMAL, CV_ONE_STEP.
1069  * In the CV_NORMAL mode, the solver steps until it reaches or passes tout
1070  * and then interpolates to obtain y(tout).
1071  * In the CV_ONE_STEP mode, it takes one internal step and returns.
1072  */
1073 
1074 int CVode(void *cvode_mem, realtype tout, N_Vector yout,
1075  realtype *tret, int itask)
1076 {
1077  CVodeMem cv_mem;
1078  long int nstloc;
1079  int retval, hflag, kflag, istate, ir, ier, irfndp;
1080  int ewtsetOK;
1081  realtype troundoff, tout_hin, rh, nrm;
1082  booleantype inactive_roots;
1083 
1084  /*
1085  * -------------------------------------
1086  * 1. Check and process inputs
1087  * -------------------------------------
1088  */
1089 
1090  /* Check if cvode_mem exists */
1091  if (cvode_mem == NULL) {
1092  cvProcessError(NULL, CV_MEM_NULL, "CVODE", "CVode", MSGCV_NO_MEM);
1093  return(CV_MEM_NULL);
1094  }
1095  cv_mem = (CVodeMem) cvode_mem;
1096 
1097  /* Check if cvode_mem was allocated */
1098  if (cv_mem->cv_MallocDone == FALSE) {
1099  cvProcessError(cv_mem, CV_NO_MALLOC, "CVODE", "CVode", MSGCV_NO_MALLOC);
1100  return(CV_NO_MALLOC);
1101  }
1102 
1103  /* Check for yout != NULL */
1104  if ((y = yout) == NULL) {
1105  cvProcessError(cv_mem, CV_ILL_INPUT, "CVODE", "CVode", MSGCV_YOUT_NULL);
1106  return(CV_ILL_INPUT);
1107  }
1108 
1109  /* Check for tret != NULL */
1110  if (tret == NULL) {
1111  cvProcessError(cv_mem, CV_ILL_INPUT, "CVODE", "CVode", MSGCV_TRET_NULL);
1112  return(CV_ILL_INPUT);
1113  }
1114 
1115  /* Check for valid itask */
1116  if ( (itask != CV_NORMAL) && (itask != CV_ONE_STEP) ) {
1117  cvProcessError(cv_mem, CV_ILL_INPUT, "CVODE", "CVode", MSGCV_BAD_ITASK);
1118  return(CV_ILL_INPUT);
1119  }
1120 
1121  if (itask == CV_NORMAL) toutc = tout;
1122  taskc = itask;
1123 
1124  /*
1125  * ----------------------------------------
1126  * 2. Initializations performed only at
1127  * the first step (nst=0):
1128  * - initial setup
1129  * - initialize Nordsieck history array
1130  * - compute initial step size
1131  * - check for approach to tstop
1132  * - check for approach to a root
1133  * ----------------------------------------
1134  */
1135 
1136  if (nst == 0) {
1137 
1138  tretlast = *tret = tn;
1139 
1140  ier = cvInitialSetup(cv_mem);
1141  if (ier!= CV_SUCCESS) return(ier);
1142 
1143  /* Call f at (t0,y0), set zn[1] = y'(t0),
1144  set initial h (from H0 or cvHin), and scale zn[1] by h.
1145  Also check for zeros of root function g at and near t0. */
1146 
1147  retval = f(tn, zn[0], zn[1], user_data);
1148  nfe++;
1149  if (retval < 0) {
1150  cvProcessError(cv_mem, CV_RHSFUNC_FAIL, "CVODE", "CVode", MSGCV_RHSFUNC_FAILED, tn);
1151  return(CV_RHSFUNC_FAIL);
1152  }
1153  if (retval > 0) {
1154  cvProcessError(cv_mem, CV_FIRST_RHSFUNC_ERR, "CVODE", "CVode", MSGCV_RHSFUNC_FIRST);
1155  return(CV_FIRST_RHSFUNC_ERR);
1156  }
1157 
1158  /* Test input tstop for legality. */
1159 
1160  if (tstopset) {
1161  if ( (tstop - tn)*(tout - tn) <= ZERO ) {
1162  cvProcessError(cv_mem, CV_ILL_INPUT, "CVODE", "CVode", MSGCV_BAD_TSTOP, tstop, tn);
1163  return(CV_ILL_INPUT);
1164  }
1165  }
1166 
1167  /* Set initial h (from H0 or cvHin). */
1168

1170  if ( (h != ZERO) && ((tout-tn)*h < ZERO) ) {
1171  cvProcessError(cv_mem, CV_ILL_INPUT, "CVODE", "CVode", MSGCV_BAD_H0);
1172  return(CV_ILL_INPUT);
1173  }
1174  if (h == ZERO) {
1175  tout_hin = tout;
1176  if ( tstopset && (tout-tn)*(tout-tstop) > ZERO ) tout_hin = tstop;
1177  hflag = cvHin(cv_mem, tout_hin);
1178  if (hflag != CV_SUCCESS) {
1179  istate = cvHandleFailure(cv_mem, hflag);
1180  return(istate);
1181  }
1182  }
1183  rh = SUNRabs(h)*hmax_inv;
1184  if (rh > ONE) h /= rh;
1185  if (SUNRabs(h) < hmin) h *= hmin/SUNRabs(h);
1186 
1187  /* Check for approach to tstop */
1188 
1189  if (tstopset) {
1190  if ( (tn + h - tstop)*h > ZERO )
1191  h = (tstop - tn)*(ONE-FOUR*uround);
1192  }

1195 
1196  hscale = h;
1197  h0u = h;
1198  hprime = h;
1199 
1200  N_VScale(h, zn[1], zn[1]);
1201 
1202  /* Check for zeros of root function g at and near t0. */
1203

1205 
1206  retval = cvRcheck1(cv_mem);
1207

1209  cvProcessError(cv_mem, CV_RTFUNC_FAIL, "CVODE", "cvRcheck1", MSGCV_RTFUNC_FAILED, tn);
1210  return(CV_RTFUNC_FAIL);
1211  }
1212 
1213  }

1217  /*
1218  * ------------------------------------------------------
1219  * 3. At following steps, perform stop tests:
1220  * - check for root in last step
1221  * - check if we passed tstop
1222  * - check if we passed tout (NORMAL mode)
1223  * - check if current tn was returned (ONE_STEP mode)
1224  * - check if we are close to tstop
1225  * (adjust step size if needed)
1226  * -------------------------------------------------------
1227  */
1228 
1229  if (nst > 0) {
1230 
1231  /* Estimate an infinitesimal time interval to be used as
1232  a roundoff for time quantities (based on current time
1233  and step size) */
1234  troundoff = FUZZ_FACTOR*uround*(SUNRabs(tn) + SUNRabs(h));
1235 
1236  /* First, check for a root in the last step taken, other than the
1237  last root found, if any. If itask = CV_ONE_STEP and y(tn) was not
1238  returned because of an intervening root, return y(tn) now. */
1239  if (nrtfn > 0) {
1240 
1241  irfndp = irfnd;
1242 
1243  retval = cvRcheck2(cv_mem);
1244 
1245  if (retval == CLOSERT) {
1246  cvProcessError(cv_mem, CV_ILL_INPUT, "CVODE", "cvRcheck2", MSGCV_CLOSE_ROOTS, tlo);
1247  return(CV_ILL_INPUT);
1248  } else if (retval == CV_RTFUNC_FAIL) {
1249  cvProcessError(cv_mem, CV_RTFUNC_FAIL, "CVODE", "cvRcheck2", MSGCV_RTFUNC_FAILED, tlo);
1250  return(CV_RTFUNC_FAIL);
1251  } else if (retval == RTFOUND) {

1493 /*
1494  * CVodeGetDky
1495  *
1496  * This routine computes the k-th derivative of the interpolating
1497  * polynomial at the time t and stores the result in the vector dky.
1498  * The formula is:
1499  * q
1500  * dky = SUM c(j,k) * (t - tn)^(j-k) * h^(-j) * zn[j] ,
1501  * j=k
1502  * where c(j,k) = j*(j-1)*...*(j-k+1), q is the current order, and
1503  * zn[j] is the j-th column of the Nordsieck history array.
1504  *
1505  * This function is called by CVode with k = 0 and t = tout, but
1506  * may also be called directly by the user.
1507  */
1508 
1509 int CVodeGetDky(void *cvode_mem, realtype t, int k, N_Vector dky)
1510 {
1511  realtype s, c, r;
1512  realtype tfuzz, tp, tn1;
1513  int i, j;
1514  CVodeMem cv_mem;
1515 
1516  /* Check all inputs for legality */
1517 
1518  if (cvode_mem == NULL) {
1519  cvProcessError(NULL, CV_MEM_NULL, "CVODE", "CVodeGetDky", MSGCV_NO_MEM);
1520  return(CV_MEM_NULL);
1521  }
1522  cv_mem = (CVodeMem) cvode_mem;
1523 
1524  if (dky == NULL) {
1525  cvProcessError(cv_mem, CV_BAD_DKY, "CVODE", "CVodeGetDky", MSGCV_NULL_DKY);
1526  return(CV_BAD_DKY);
1527  }
1528 
1529  if ((k < 0) || (k > q)) {
1530  cvProcessError(cv_mem, CV_BAD_K, "CVODE", "CVodeGetDky", MSGCV_BAD_K);
1531  return(CV_BAD_K);
1532  }
1533 
1534  /* Allow for some slack */
1535  tfuzz = FUZZ_FACTOR * uround * (SUNRabs(tn) + SUNRabs(hu));
1536  if (hu < ZERO) tfuzz = -tfuzz;
1537  tp = tn - hu - tfuzz;
1538  tn1 = tn + tfuzz;
1539  if ((t-tp)*(t-tn1) > ZERO) {
1540  cvProcessError(cv_mem, CV_BAD_T, "CVODE", "CVodeGetDky", MSGCV_BAD_T, t, tn-hu, tn);
1541  return(CV_BAD_T);
1542  }
1543 
1544  /* Sum the differentiated interpolating polynomial */
1545 
1546  s = (t - tn) / h;
1547  for (j=q; j >= k; j--) {
1548  c = ONE;
1549  for (i=j; i >= j-k+1; i--) c *= i;
1550  if (j == q) {
1551  N_VScale(c, zn[q], dky);
1552  } else {
1553  N_VLinearSum(c, zn[j], s, dky, dky);
1554  }
1555  }
1556  if (k == 0) return(CV_SUCCESS);
1557  r = SUNRpowerI(h,-k);
1558  N_VScale(r, dky, dky);
1559  return(CV_SUCCESS);
1560 }

1778 /*
1779  * cvHin
1780  *
1781  * This routine computes a tentative initial step size h0.
1782  * If tout is too close to tn (= t0), then cvHin returns CV_TOO_CLOSE
1783  * and h remains uninitialized. Note that here tout is either the value
1784  * passed to CVode at the first call or the value of tstop (if tstop is
1785  * enabled and it is closer to t0=tn than tout).
1786  * If the RHS function fails unrecoverably, cvHin returns CV_RHSFUNC_FAIL.
1787  * If the RHS function fails recoverably too many times and recovery is
1788  * not possible, cvHin returns CV_REPTD_RHSFUNC_ERR.
1789  * Otherwise, cvHin sets h to the chosen value h0 and returns CV_SUCCESS.
1790  *
1791  * The algorithm used seeks to find h0 as a solution of
1792  * (WRMS norm of (h0^2 ydd / 2)) = 1,
1793  * where ydd = estimated second derivative of y.
1794  *
1795  * We start with an initial estimate equal to the geometric mean of the
1796  * lower and upper bounds on the step size.
1797  *
1798  * Loop up to MAX_ITERS times to find h0.
1799  * Stop if new and previous values differ by a factor < 2.
1800  * Stop if hnew/hg > 2 after one iteration, as this probably means
1801  * that the ydd value is bad because of cancellation error.
1802  *
1803  * For each new proposed hg, we allow MAX_ITERS attempts to
1804  * resolve a possible recoverable failure from f() by reducing
1805  * the proposed stepsize by a factor of 0.2. If a legal stepsize
1806  * still cannot be found, fall back on a previous value if possible,
1807  * or else return CV_REPTD_RHSFUNC_ERR.
1808  *
1809  * Finally, we apply a bias (0.5) and verify that h0 is within bounds.
1810  */
1811 
1812 static int cvHin(CVodeMem cv_mem, realtype tout)
1813 {
1814  int retval, sign, count1, count2;
1815  realtype tdiff, tdist, tround, hlb, hub;
1816  realtype hg, hgs, hs, hnew, hrat, h0, yddnrm;
1817  booleantype hgOK, hnewOK;
1818 
1819  /* If tout is too close to tn, give up */
1820 
1821  if ((tdiff = tout-tn) == ZERO) return(CV_TOO_CLOSE);
1822 
1823  sign = (tdiff > ZERO) ? 1 : -1;
1824  tdist = SUNRabs(tdiff);
1825  tround = uround * SUNMAX(SUNRabs(tn), SUNRabs(tout));
1826 
1827  if (tdist < TWO*tround) return(CV_TOO_CLOSE);
1828 
1829  /*
1830  Set lower and upper bounds on h0, and take geometric mean
1831  as first trial value.
1832  Exit with this value if the bounds cross each other.
1833  */
1834 
1835  hlb = HLB_FACTOR * tround;
1836  hub = cvUpperBoundH0(cv_mem, tdist);
1837 
1838  hg = SUNRsqrt(hlb*hub);
1839 
1840  if (hub < hlb) {
1841  if (sign == -1) h = -hg;
1842  else h = hg;
1843  return(CV_SUCCESS);
1844  }
1845 
1846  /* Outer loop */
1847 
1848  hnewOK = FALSE;
1849  hs = hg; /* safeguard against 'uninitialized variable' warning */
1850 
1851  for(count1 = 1; count1 <= MAX_ITERS; count1++) {
1852 
1853  /* Attempts to estimate ydd */
1854 
1855  hgOK = FALSE;
1856 
1857  for (count2 = 1; count2 <= MAX_ITERS; count2++) {
1858  hgs = hg*sign;
1859  retval = cvYddNorm(cv_mem, hgs, &yddnrm);
1860  /* If f() failed unrecoverably, give up */
1861  if (retval < 0) return(CV_RHSFUNC_FAIL);
1862  /* If successful, we can use ydd */
1863  if (retval == CV_SUCCESS) {hgOK = TRUE; break;}
