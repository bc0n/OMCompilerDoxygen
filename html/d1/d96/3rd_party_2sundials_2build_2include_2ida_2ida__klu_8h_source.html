32 /*
33  * -----------------------------------------------------------------
34  * Function : IDAKLU
35  * -----------------------------------------------------------------
36  * A call to the IDAKLU function links the main integrator
37  * with the IDAKLU linear solver module.
38  *
39  * ida_mem is the pointer to integrator memory returned by
40  * IDACreate.
41  *
42  *
43  * IDAKLU returns:
44  * IDASLU_SUCCESS = 0 if successful
45  * IDASLU_LMEM_FAIL = -1 if there was a memory allocation failure
46  * IDASLU_ILL_INPUT = -2 if NVECTOR found incompatible
47  *
48  * NOTE: The KLU linear solver assumes a serial implementation
49  * of the NVECTOR package. Therefore, IDAKLU will first
50  * test for a compatible N_Vector internal representation
51  * by checking that the functions N_VGetArrayPointer and
52  * N_VSetArrayPointer exist.
53  * -----------------------------------------------------------------
54  */

58 /*
59  * -----------------------------------------------------------------
60  * IDAKLUReInit
61  * -----------------------------------------------------------------
62  * This routine reinitializes memory and flags for a new factorization
63  * (symbolic and numeric) to be conducted at the next solver setup
64  * call. This routine is useful in the cases where the number of nonzeroes
65  * has changed or if the structure of the linear system has changed
66  * which would require a new symbolic (and numeric factorization).
67  *
68  * The reinit_type argumenmt governs the level of reinitialization:
69  *
70  * reinit_type = 1: The Jacobian matrix will be destroyed and
71  * a new one will be allocated based on the nnz
72  * value passed to this call. New symbolic and
73  * numeric factorizations will be completed at the next
74  * solver setup.
75  *
76  * reinit_type = 2: Only symbolic and numeric factorizations will be
77  * completed. It is assumed that the Jacobian size
78  * has not exceeded the size of nnz given in the prior
79  * call to IDAKLU.
80  *
81  * This routine assumes no other changes to solver use are necessary.
82  *
83  * The return value is IDASLS_SUCCESS = 0, IDASLS_MEM_NULL = -1,
84  * IDASLS_LMEM_NULL = -2, IDASLS_ILL_INPUT = -3, or IDASLS_MEM_FAIL = -4.
85  *
86  * -----------------------------------------------------------------
87  */
88 
89 SUNDIALS_EXPORT int IDAKLUReInit(void *ida_mem_v, int n, int nnz,
90  int reinit_type);
