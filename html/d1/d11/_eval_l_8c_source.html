202 static inline void num_hessian0(double * v, const double * const lambda,
203  const double objFactor , OptData *optData, const int i, const int j){
204 
205  const modelica_boolean la = optData->s.lagrange;
206  const modelica_boolean upCost = la && objFactor != 0;
207  DATA * data = optData->data;
208  threadData_t *threadData = optData->threadData;
209 
210  const int nv = optData->dim.nv;
211  const int nx = optData->dim.nx;
212  const int nJ = optData->dim.nJ;
213  const modelica_real * const vmax = optData->bounds.vmax;
214  const modelica_real * const vnom = optData->bounds.vnom;
215 
216  int ii,jj, l;
217  long double v_save, h;
218  modelica_real * realV[3];
219 
220 
221  for(l = 1; l<3; ++l){
222  realV[l] = data->localData[l]->realVars;
223  data->localData[l]->realVars = optData->v[i][j];
224  data->localData[l]->timeValue = (modelica_real) optData->time.t[i][j];
225  }
226  data->localData[0]->timeValue = (modelica_real) optData->time.t[i][j];
227 
228  for(ii = 0; ii < nv; ++ii){
229  /********************/
230  v_save = (long double) v[ii];
231  h = (long double)DF_STEP(v_save);
232  v[ii] += h;
233  if( v[ii] >= vmax[ii]){
234  h *= -1.0;
235  v[ii] = v_save + h;
236  }
237  /********************/
238  for(l = 0; l < nx; ++l)
239  data->localData[0]->realVars[l] = v[l]*vnom[l];
240  for(; l <nv; ++l)
241  data->simulationInfo->inputVars[l-nx] = (modelica_real) v[l]*vnom[l];
242  data->callback->input_function(data, threadData);
243  /*data->callback->functionDAE(data);*/
244  updateDiscreteSystem(data, threadData);
245  /********************/
246  diffSynColoredOptimizerSystem(optData, optData->tmpJ, i,j,2);
247  /********************/
248  v[ii] = (double)v_save;
249  /********************/
250  for(jj = 0; jj <ii+1; ++jj){
251  if(optData->s.H0[ii][jj]){
252  for(l = 0; l < nJ; ++l){
253  if(optData->s.Hg[l][ii][jj] && lambda[l] != 0)
254  optData->H[l][ii][jj] = (long double)(optData->tmpJ[l][jj] - optData->J[i][j][l][jj])*lambda[l]/h;
255  }
256  }
257  }
258  /********************/
259  if(upCost){
260  h = objFactor/h;
261  for(jj = 0; jj <ii+1; ++jj){
262  if(optData->s.Hl[ii][jj]){
263  optData->Hl[ii][jj] = (long double)(optData->tmpJ[nJ][jj] - optData->J[i][j][nJ][jj])*h;
264  }else{
265  optData->Hl[ii][jj] = 0.0;
266  }
267  }
268  }
269  /********************/
270  }
271 
272  for(l = 1; l<3; ++l){
273  data->localData[l]->realVars = realV[l];
274  }
275 
276 }

284 static inline void num_hessian1(double * v, const double * const lambda,
285  const double objFactor, OptData *optData, const int i, const int j){
286 
287  const modelica_boolean la = optData->s.lagrange;
288  const modelica_boolean ma = optData->s.mayer;
289  const modelica_boolean upCost = la && objFactor != 0;
290 
291  const int nv = optData->dim.nv;
292  const int nx = optData->dim.nx;
293  const int np = optData->dim.np;
294  const int nJ = optData->dim.nJ;
295  const int nJ1 = optData->dim.nJ + 1;
296  const int nsi = optData->dim.nsi;
297  const int ncf = optData->dim.ncf;
298  const modelica_real * const vmax = optData->bounds.vmax;
299  const modelica_real * const vnom = optData->bounds.vnom;
300  const modelica_boolean upFinalCon = np == j + 1 && nsi == i +1;
301  const modelica_boolean upCost2 = upFinalCon && ma && objFactor != 0;
302  const short indexJ = (upCost2) ? 3 : 2;
303  int ii,jj, l,k;
304  long double v_save, h;
305  DATA * data = optData->data;
306  threadData_t *threadData = optData->threadData;
307 
308  modelica_real * realV[3];
309 
310  data->localData[0]->timeValue = (modelica_real) optData->time.t[i][j];
311  for(l = 1; l<3; ++l){
312  realV[l] = data->localData[l]->realVars;
313  data->localData[l]->realVars = optData->v[i][j];
314  data->localData[l]->timeValue = (modelica_real) optData->time.t[i][j];
315  }
316 
317  for(ii = 0; ii < nv; ++ii){
318  /********************/
319  v_save = (long double) v[ii];
320  h = (long double)DF_STEP(v_save);
321  v[ii] += h;
322  if(v[ii] > vmax[ii]){
323  h *= -1.0;
324  v[ii] = v_save + h;
325  }
326  /********************/
327  for(l = 0; l < nx; ++l)
328  data->localData[0]->realVars[l] = v[l]*vnom[l];
329  for(; l <nv; ++l)
330  data->simulationInfo->inputVars[l-nx] = (modelica_real) v[l]*vnom[l];
331 
332  data->callback->input_function(data, threadData);
333  /*data->callback->functionDAE(data);*/
334  updateDiscreteSystem(data, threadData);
335  /********************/
336  diffSynColoredOptimizerSystem(optData, optData->tmpJ, i,j,indexJ);
337  /********************/
338  v[ii] = (double)v_save;
339  /********************/
340  for(jj = 0; jj <ii+1; ++jj){
341  if(optData->s.H0[ii][jj]){
342  for(l = 0; l < nJ; ++l){
343  if(optData->s.Hg[l][ii][jj])
344  optData->H[l][ii][jj] = (long double)(optData->tmpJ[l][jj] - optData->J[i][j][l][jj])*lambda[l]/h;
345  }
346  }
347  }
348  /********************/
349  if(upCost){
350  long double hh;
351  hh = objFactor/h;
352  for(jj = 0; jj <ii+1; ++jj){
353  if(optData->s.Hl[ii][jj]){
354  optData->Hl[ii][jj] = (long double)(optData->tmpJ[nJ][jj] - optData->J[i][j][nJ][jj])*hh;
355  }
356  }
357  }
358  /********************/
359  if(upCost2){
360  long double hh;
361  hh = objFactor/h;
362  for(jj = 0; jj <ii+1; ++jj){
363  if(optData->s.Hm[ii][jj]){
364  optData->Hm[ii][jj] = (long double)(optData->tmpJ[nJ1][jj] - optData->J[i][j][nJ1][jj])*hh;
365  }
366  }
367  }
368  /********************/
369  if(upFinalCon && ncf > 0){
370  diffSynColoredOptimizerSystemF(optData, optData->tmpJf);
371  for(jj = 0; jj <ii+1; ++jj){
372  if(optData->s.H0[ii][jj]){
373  for(l = 0; l < ncf; ++l){
374  if(optData->s.Hcf[l][ii][jj]){
375  optData->Hcf[l][ii][jj] = (long double)(optData->tmpJf[l][jj] - optData->Jf[l][jj])*lambda[nJ+l]/h;
376  }
377  }
378  }
379  }
380  }
381  /********************/
382  }
383 
384  for(l = 1; l<3; ++l){
385  data->localData[l]->realVars = realV[l];
386  }
387 
388 }
