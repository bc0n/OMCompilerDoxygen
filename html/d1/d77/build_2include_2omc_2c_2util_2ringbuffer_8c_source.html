75 void *getRingData(RINGBUFFER *rb, int i)
76 {
77  assertStreamPrint(NULL, rb->nElements > 0, "empty RingBuffer");
78  assertStreamPrint(NULL, i < rb->nElements, "index [%d] out of range [%d:%d]", i, -rb->nElements+1, rb->nElements-1);
79  assertStreamPrint(NULL, -rb->nElements < i, "index [%d] out of range [%d:%d]", i, -rb->nElements+1, rb->nElements-1);
80  return ((char*)rb->buffer)+(((rb->firstElement+i)%rb->bufferSize)*rb->itemSize);
81 }

83 void expandRingBuffer(RINGBUFFER *rb)
84 {
85  int i;
86 
87  void *tmp = calloc(2*rb->bufferSize, rb->itemSize);
88  assertStreamPrint(NULL, 0!=tmp, "out of memory");
89 
90  for(i=0; i<rb->nElements; i++) {
91  memcpy(((char*)tmp)+(i*rb->itemSize), getRingData(rb, i), rb->itemSize);
92  }
93 
94  free(rb->buffer);
95  rb->buffer = tmp;
96  rb->bufferSize *= 2;
97  rb->firstElement = 0;
98 }
