393  void NetStreamStorage::writeByEndianess(const unsigned char * b, unsigned int sz)
394  {
395  const unsigned char * e = &(b[sz]);
396  if (bigEndian_)
397  store.insert(store.end(), b, e);
398  else
399  store.insert(store.end(), std::reverse_iterator<const unsigned char *>(e), std::reverse_iterator<const unsigned char *>(b));
400  iter_ = store.begin();

405  void NetStreamStorage::readByEndianess(unsigned char * array, int sz)
