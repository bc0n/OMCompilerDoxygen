50 /*
51  Some of the functions never return to the caller. In order to compile
52  external Modelica C-code in most compilers, noreturn attributes need to
53  be present to avoid warnings or errors.
54 
55  The following macros handle noreturn attributes according to the latest
56  C11/C++11 standard with fallback to GNU, Clang or MSVC extensions if using
57  an older compiler.
58 */
59

62 #define MODELICA_NORETURNATTR
63 #elif defined(__cplusplus) && __cplusplus >= 201103L
64 #if (defined(__GNUC__) && __GNUC__ >= 5) || \
65  (defined(__GNUC__) && defined(__GNUC_MINOR__) && __GNUC__ == 4 && __GNUC_MINOR__ >= 8)
66 #define MODELICA_NORETURN [[noreturn]]
67 #define MODELICA_NORETURNATTR
68 #elif (defined(__GNUC__) && __GNUC__ >= 3) || \
69  (defined(__GNUC__) && defined(__GNUC_MINOR__) && __GNUC__ == 2 && __GNUC_MINOR__ >= 8)
70 #define MODELICA_NORETURN
71 #define MODELICA_NORETURNATTR __attribute__((noreturn))
72 #elif defined(__GNUC__)
73 #define MODELICA_NORETURN
74 #define MODELICA_NORETURNATTR

143 char* ModelicaAllocateString(size_t len);
144 /*
145 Allocate memory for a Modelica string which is used as return
146 argument of an external Modelica function. Note, that the storage
147 for string arrays (= pointer to string array) is still provided by the
148 calling program, as for any other array. If an error occurs, this
149 function does not return, but calls "ModelicaError".
150 */
151

153 char* ModelicaAllocateStringWithErrorReturn(size_t len);
154 /*
155 Same as ModelicaAllocateString, except that in case of error, the
156 function returns 0. This allows the external function to close files
157 and free other open resources in case of error. After cleaning up
158 resources use ModelicaError or ModelicaFormatError to signal
159 the error.
160 */
