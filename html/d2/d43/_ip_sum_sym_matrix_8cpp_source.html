57  void SumSymMatrix::MultVectorImpl(Number alpha, const Vector &x,
58  Number beta, Vector &y) const
59  {
60  // A few sanity checks
61  DBG_ASSERT(Dim()==x.Dim());
62  DBG_ASSERT(Dim()==y.Dim());
63 
64  // Take care of the y part of the addition
65  if ( beta!=0.0 ) {
66  y.Scal(beta);
67  }
68  else {
69  y.Set(0.0); // In case y hasn't been initialized yet
70  }
71 
72  for (Index iterm=0; iterm<NTerms(); iterm++) {
73  DBG_ASSERT(IsValid(matrices_[iterm]));
74  matrices_[iterm]->MultVector(alpha*factors_[iterm], x,
75  1.0, y);
76  }
77  }
