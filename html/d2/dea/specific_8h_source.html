1 /*
2  * This is a reimplementation of a subset of the pthread_getspecific/setspecific
3  * interface. This appears to outperform the standard linuxthreads one
4  * by a significant margin.
5  * The major restriction is that each thread may only make a single
6  * pthread_setspecific call on a single key. (The current data structure
7  * doesn't really require that. The restriction should be easily removable.)
8  * We don't currently support the destruction functions, though that
9  * could be done.
10  * We also currently assume that only one pthread_setspecific call
11  * can be executed at a time, though that assumption would be easy to remove
12  * by adding a lock.

46 /* All such accessible structures preserve the invariant that if either */
47 /* thread is a valid pthread id or qtid is a valid "quick thread id" */

58 /* We represent each thread-specific datum as two tables. The first is */
59 /* a cache, indexed by a "quick thread identifier". The "quick" thread */
60 /* identifier is an easy to compute value, which is guaranteed to */
61 /* determine the thread, though a thread may correspond to more than */
62 /* one value. We typically use the address of a page in the stack. */
63 /* The second is a hash table, indexed by pthread_self(). It is used */
64 /* only as a backup. */
65 
66 /* Return the "quick thread id". Default version. Assumes page size, */
67 /* or at least thread stack separation, is at least 4K. */
68 /* Must be defined so that it never returns 0. (Page 0 can't really be */
69 /* part of any stack, since that would make 0 a valid stack pointer.) */
70 #define quick_thread_id() (((word)GC_approx_sp()) >> 12)

101 GC_INLINE void * GC_getspecific(tsd * key)
102 {
103  word qtid = quick_thread_id();
104  tse * volatile * entry_ptr = &key->cache[CACHE_HASH(qtid)];
105  tse * entry = *entry_ptr; /* Must be loaded only once. */
106 
107  GC_ASSERT(qtid != INVALID_QTID);
108  if (EXPECT(entry -> qtid == qtid, TRUE)) {
109  GC_ASSERT(entry -> thread == pthread_self());
110  return TS_REVEAL_PTR(entry -> value);
111  }
112  return GC_slow_getspecific(key, qtid, entry_ptr);
