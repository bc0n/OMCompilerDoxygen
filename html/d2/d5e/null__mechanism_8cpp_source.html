67 int zmq::null_mechanism_t::next_handshake_command (msg_t *msg_)
68 {
69  if (ready_command_sent || error_command_sent) {
70  errno = EAGAIN;
71  return -1;
72  }
73  if (zap_connected && !zap_reply_received) {
74  if (zap_request_sent) {
75  errno = EAGAIN;
76  return -1;
77  }
78  int rc = send_zap_request ();
79  if (rc != 0)
80  return -1;
81  zap_request_sent = true;
82  rc = receive_and_process_zap_reply ();
83  if (rc != 0)
84  return -1;
85  zap_reply_received = true;
86  }
87 
88  if (zap_reply_received
89  && strncmp (status_code, "200", sizeof status_code) != 0) {
90  const int rc = msg_->init_size (6 + 1 + sizeof status_code);
91  zmq_assert (rc == 0);
92  unsigned char *msg_data =
93  static_cast <unsigned char *> (msg_->data ());
94  memcpy (msg_data, "\5ERROR", 6);
95  msg_data [6] = sizeof status_code;
96  memcpy (msg_data + 7, status_code, sizeof status_code);
97  error_command_sent = true;
98  return 0;
99  }
100 
101  unsigned char *const command_buffer = (unsigned char *) malloc (512);
102  alloc_assert (command_buffer);
103 
104  unsigned char *ptr = command_buffer;
105 
106  // Add mechanism string
107  memcpy (ptr, "\5READY", 6);
108  ptr += 6;
109 
110  // Add socket type property
111  const char *socket_type = socket_type_string (options.type);
112  ptr += add_property (ptr, "Socket-Type", socket_type, strlen (socket_type));
113 
114  // Add identity property
115  if (options.type == ZMQ_REQ
116  || options.type == ZMQ_DEALER
117  || options.type == ZMQ_ROUTER)
118  ptr += add_property (ptr, "Identity", options.identity, options.identity_size);
119 
120  const size_t command_size = ptr - command_buffer;
121  const int rc = msg_->init_size (command_size);
122  errno_assert (rc == 0);
123  memcpy (msg_->data (), command_buffer, command_size);
124  free (command_buffer);
125 
126  ready_command_sent = true;
127 
128  return 0;
129 }

131 int zmq::null_mechanism_t::process_handshake_command (msg_t *msg_)
132 {
133  if (ready_command_received || error_command_received) {
134  // Temporary support for security debugging
135  puts ("NULL I: client sent invalid NULL handshake (duplicate READY)");
136  errno = EPROTO;
137  return -1;
138  }
139 
140  const unsigned char *cmd_data =
141  static_cast <unsigned char *> (msg_->data ());
142  const size_t data_size = msg_->size ();
143 
144  int rc = 0;
145  if (data_size >= 6 && !memcmp (cmd_data, "\5READY", 6))
146  rc = process_ready_command (cmd_data, data_size);
147  else
148  if (data_size >= 6 && !memcmp (cmd_data, "\5ERROR", 6))
149  rc = process_error_command (cmd_data, data_size);
150  else {
151  // Temporary support for security debugging
152  puts ("NULL I: client sent invalid NULL handshake (not READY)");
153  errno = EPROTO;
154  rc = -1;
155  }
156 
157  if (rc == 0) {
158  rc = msg_->close ();
159  errno_assert (rc == 0);
160  rc = msg_->init ();
161  errno_assert (rc == 0);
162  }
163  return rc;
164 }

175 {

217 int zmq::null_mechanism_t::send_zap_request ()
218 {
219  int rc;
220  msg_t msg;
221 
222  // Address delimiter frame
223  rc = msg.init ();
224  errno_assert (rc == 0);
225  msg.set_flags (msg_t::more);
226  rc = session->write_zap_msg (&msg);
227  if (rc != 0)
228  return close_and_return (&msg, -1);
229 
230  // Version frame
231  rc = msg.init_size (3);
232  errno_assert (rc == 0);
233  memcpy (msg.data (), "1.0", 3);
234  msg.set_flags (msg_t::more);
235  rc = session->write_zap_msg (&msg);
236  if (rc != 0)
237  return close_and_return (&msg, -1);
238 
239  // Request id frame
240  rc = msg.init_size (1);
241  errno_assert (rc == 0);
242  memcpy (msg.data (), "1", 1);
243  msg.set_flags (msg_t::more);
244  rc = session->write_zap_msg (&msg);
245  if (rc != 0)
246  return close_and_return (&msg, -1);
247 
248  // Domain frame
249  rc = msg.init_size (options.zap_domain.length ());
250  errno_assert (rc == 0);
251  memcpy (msg.data (), options.zap_domain.c_str (), options.zap_domain.length ());
252  msg.set_flags (msg_t::more);
253  rc = session->write_zap_msg (&msg);
254  if (rc != 0)
255  return close_and_return (&msg, -1);
256 
257  // Address frame
258  rc = msg.init_size (peer_address.length ());
259  errno_assert (rc == 0);
260  memcpy (msg.data (), peer_address.c_str (), peer_address.length ());
261  msg.set_flags (msg_t::more);
262  rc = session->write_zap_msg (&msg);
263  if (rc != 0)
264  return close_and_return (&msg, -1);
265 
266  // Identity frame
267  rc = msg.init_size (options.identity_size);
268  errno_assert (rc == 0);
269  memcpy (msg.data (), options.identity, options.identity_size);
270  msg.set_flags (msg_t::more);
271  rc = session->write_zap_msg (&msg);
272  if (rc != 0)
273  return close_and_return (&msg, -1);
274 
275  // Mechanism frame
276  rc = msg.init_size (4);
277  errno_assert (rc == 0);
278  memcpy (msg.data (), "NULL", 4);
279  rc = session->write_zap_msg (&msg);
280  if (rc != 0)
281  return close_and_return (&msg, -1);
282 
283  return 0;
284 }

286 int zmq::null_mechanism_t::receive_and_process_zap_reply ()
287 {
288  int rc = 0;
289  msg_t msg [7]; // ZAP reply consists of 7 frames
290 
291  // Initialize all reply frames
292  for (int i = 0; i < 7; i++) {
293  rc = msg [i].init ();
294  errno_assert (rc == 0);
295  }
296 
297  for (int i = 0; i < 7; i++) {
298  rc = session->read_zap_msg (&msg [i]);
299  if (rc == -1)
300  return close_and_return (msg, -1);
301  if ((msg [i].flags () & msg_t::more) == (i < 6? 0: msg_t::more)) {
302  // Temporary support for security debugging
303  puts ("NULL I: ZAP handler sent incomplete reply message");
304  errno = EPROTO;
305  return close_and_return (msg, -1);
306  }
307  }
308 
309  // Address delimiter frame
310  if (msg [0].size () > 0) {
311  // Temporary support for security debugging
312  puts ("NULL I: ZAP handler sent malformed reply message");
313  errno = EPROTO;
314  return close_and_return (msg, -1);
315  }
316 
317  // Version frame
318  if (msg [1].size () != 3 || memcmp (msg [1].data (), "1.0", 3)) {
319  // Temporary support for security debugging
320  puts ("NULL I: ZAP handler sent bad version number");
321  errno = EPROTO;
322  return close_and_return (msg, -1);
323  }
324 
325  // Request id frame
326  if (msg [2].size () != 1 || memcmp (msg [2].data (), "1", 1)) {
327  // Temporary support for security debugging
328  puts ("NULL I: ZAP handler sent bad request ID");
329  errno = EPROTO;
330  return close_and_return (msg, -1);
331  }
332 
333  // Status code frame
334  if (msg [3].size () != 3) {
335  // Temporary support for security debugging
336  puts ("NULL I: ZAP handler rejected client authentication");
337  errno = EPROTO;
338  return close_and_return (msg, -1);
339  }
340 
341  // Save status code
342  memcpy (status_code, msg [3].data (), sizeof status_code);
343 
344  // Save user id
345  set_user_id (msg [5].data (), msg [5].size ());
346 
347  // Process metadata frame
348  rc = parse_metadata (static_cast <const unsigned char*> (msg [6].data ()),
349  msg [6].size (), true);
350 
351  if (rc != 0)
352  return close_and_return (msg, -1);
353 
354  // Close all reply frames
355  for (int i = 0; i < 7; i++) {
356  const int rc2 = msg [i].close ();
357  errno_assert (rc2 == 0);
358  }
359 
360  return 0;
361 }
