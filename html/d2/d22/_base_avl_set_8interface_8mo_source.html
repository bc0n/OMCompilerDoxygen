85 replaceable function add
86  "Inserts a new node in the tree."
87  input Tree inTree;
88  input Key inKey;
89  output Tree tree=inTree;
90 algorithm
91  tree := match tree
92  local
93  Key key;
94  Integer key_comp;
95  Tree outTree;
96 
97  // Empty tree.
98  case EMPTY()
99  then LEAF(inKey);
100 
101  case NODE(key = key)
102  algorithm
103  key_comp := keyCompare(inKey, key);
104 
105  if key_comp == -1 then
106  // Replace left branch.
107  tree.left := add(tree.left, inKey);
108  elseif key_comp == 1 then
109  // Replace right branch.
110  tree.right := add(tree.right, inKey);
111  end if;
112  then
113  if key_comp == 0 then tree else balance(tree);
114 
115  case LEAF(key = key)
116  algorithm
117  key_comp := keyCompare(inKey, key);
118 
119  if key_comp == -1 then
120  // Replace left branch.
121  outTree := NODE(tree.key, 2, LEAF(inKey), EMPTY());
122  elseif key_comp == 1 then
123  // Replace right branch.
124  outTree := NODE(tree.key, 2, EMPTY(), LEAF(inKey));
125  else
126  outTree := tree;
127  end if;
128  then outTree; // No need to balance addition in a leaf
129 
130  end match;
131 end add;

265 function intersection
266  "Takes two sets and returns the intersection as well as the remainder
267  of both sets after removing the duplicates in both sets."
268  input Tree tree1, tree2;
269  output Tree intersect=Tree.EMPTY(), rest1=Tree.EMPTY(), rest2=Tree.EMPTY();
270 protected
271  list<Key> keylist1, keylist2;
272  Key k1, k2;
273  Integer key_comp;
274 algorithm
275  if isEmpty(tree1) then
276  rest2 := tree2;
277  return;
278  end if;
279  if isEmpty(tree2) then
280  rest1 := tree1;
281  return;
282  end if;
283 
284  // we operate on sorted lists from the trees!
285  k1::keylist1 := listKeys(tree1);
286  k2::keylist2 := listKeys(tree2);
287  while true loop
288  key_comp := keyCompare(k1, k2);
289  if key_comp > 0 then
290  if isPresent(rest2) then rest2 := add(rest2, k2); end if;
291  if listEmpty(keylist2) then break; end if;
292  k2::keylist2 := keylist2;
293  elseif key_comp < 0 then
294  if isPresent(rest1) then rest1 := add(rest1, k1); end if;
295  if listEmpty(keylist1) then break; end if;
296  k1::keylist1 := keylist1;
297  else // equal keys: advance both lists
298  intersect := add(intersect, k1);
299  if listEmpty(keylist1) or listEmpty(keylist2) then break; end if;
300  k1::keylist1 := keylist1;
301  k2::keylist2 := keylist2;
302  end if;
303  end while;
304  if isPresent(rest1) and not listEmpty(keylist1) then
305  for key in keylist1 loop
306  rest1 := add(rest1, key);
307  end for;
308  end if;
309  if isPresent(rest2) and not listEmpty(keylist2) then
310  for key in keylist2 loop
311  rest2 := add(rest2, key);
312  end for;
313  end if;
314 end intersection;

328 function balance
329  "Balances a Tree"
330  input Tree inTree;
331  output Tree outTree = inTree;
332 algorithm
333  outTree := match outTree
334  local
335  Integer lh, rh, diff;
336  Tree child, balanced_tree;
337 
338  case LEAF() then inTree;
339 
340  case NODE()
341  algorithm
342  lh := height(outTree.left);
343  rh := height(outTree.right);
344  diff := lh - rh;
345 
346  if diff < -1 then
347  balanced_tree := if calculateBalance(outTree.right) > 0
348  then rotateLeft(setTreeLeftRight(outTree, left=outTree.left, right=rotateRight(outTree.right)))
349  else rotateLeft(outTree);
350  elseif diff > 1 then
351  balanced_tree := if calculateBalance(outTree.left) < 0
352  then rotateRight(setTreeLeftRight(outTree, left=rotateLeft(outTree.left), right=outTree.right))
353  else rotateRight(outTree);
354  elseif outTree.height <> max(lh, rh) + 1 then
355  outTree.height := max(lh, rh) + 1;
356  balanced_tree := outTree;
357  else
358  balanced_tree := outTree;
359  end if;
360  then
361  balanced_tree;
362 
363  end match;
364 end balance;
