259 /*
260  * -----------------------------------------------------------------
261  * int (*ida_linit)(IDAMem IDA_mem);
262  * -----------------------------------------------------------------
263  * The purpose of ida_linit is to allocate memory for the
264  * solver-specific fields in the structure *(idamem->ida_lmem) and
265  * perform any needed initializations of solver-specific memory,
266  * such as counters/statistics. An (*ida_linit) should return
267  * 0 if it has successfully initialized the IDA linear solver and
268  * a non-zero value otherwise. If an error does occur, an appropriate
269  * message should be sent to the error handler function.
270  * ----------------------------------------------------------------
271  */
272 
273 /*
274  * -----------------------------------------------------------------
275  * int (*ida_lsetup)(IDAMem IDA_mem, N_Vector yyp, N_Vector ypp,
276  * N_Vector resp,
277  * N_Vector tempv1, N_Vector tempv2, N_Vector tempv3);
278  * -----------------------------------------------------------------
279  * The job of ida_lsetup is to prepare the linear solver for
280  * subsequent calls to ida_lsolve. Its parameters are as follows:
281  *
282  * idamem - problem memory pointer of type IDAMem. See the big
283  * typedef earlier in this file.
284  *
285  *
286  * yyp - the predicted y vector for the current IDA internal
287  * step.
288  *
289  * ypp - the predicted y' vector for the current IDA internal
290  * step.
291  *
292  * resp - F(tn, yyp, ypp).
293  *
294  * tempv1, tempv2, tempv3 - temporary N_Vectors provided for use
295  * by ida_lsetup.
296  *
297  * The ida_lsetup routine should return 0 if successful,
298  * a positive value for a recoverable error, and a negative value
299  * for an unrecoverable error.
300  * -----------------------------------------------------------------
301  */
302 
303 /*
304  * -----------------------------------------------------------------
305  * int (*ida_lsolve)(IDAMem IDA_mem, N_Vector b, N_Vector weight,
306  * N_Vector ycur, N_Vector ypcur, N_Vector rescur);
307  * -----------------------------------------------------------------
308  * ida_lsolve must solve the linear equation P x = b, where
309  * P is some approximation to the system Jacobian
310  * J = (dF/dy) + cj (dF/dy')
311  * evaluated at (tn,ycur,ypcur) and the RHS vector b is input.
312  * The N-vector ycur contains the solver's current approximation
313  * to y(tn), ypcur contains that for y'(tn), and the vector rescur
314  * contains the N-vector residual F(tn,ycur,ypcur).
315  * The solution is to be returned in the vector b.
316  *
317  * The ida_lsolve routine should return 0 if successful,
318  * a positive value for a recoverable error, and a negative value
319  * for an unrecoverable error.
320  * -----------------------------------------------------------------
321  */
322 
323 /*
324  * -----------------------------------------------------------------
325  * int (*ida_lperf)(IDAMem IDA_mem, int perftask);
326  * -----------------------------------------------------------------
327  * ida_lperf is called two places in IDA where linear solver
328  * performance data is required by IDA. For perftask = 0, an
329  * initialization of performance variables is performed, while for
330  * perftask = 1, the performance is evaluated.
331  * -----------------------------------------------------------------
332  */
333 
334 /*
335  * -----------------------------------------------------------------
336  * int (*ida_lfree)(IDAMem IDA_mem);
337  * -----------------------------------------------------------------
338  * ida_lfree should free up any memory allocated by the linear
339  * solver. This routine is called once a problem has been
340  * completed and the linear solver is no longer needed.
341  * -----------------------------------------------------------------
342  */
