44 omc_mmap_read_unix omc_mmap_open_read_unix(const char *fileName)
45 {
46  struct stat s;
47  omc_mmap_read_unix res = {0};
48  int fd = open(fileName, O_RDONLY);
49  if (fd < 0) {
50  throwStreamPrint(NULL, "Failed to open file %s for reading: %s\n", fileName, strerror(errno));
51  }
52  if (fstat(fd, &s) < 0) {
53  close(fd);
54  throwStreamPrint(NULL, "fstat %s failed: %s\n", fileName, strerror(errno));
55  }
56  res.size = s.st_size;
57  res.data = (const char*) mmap(0, res.size, PROT_READ, MAP_SHARED, fd, 0);
58  close(fd);
59  if (res.data == MAP_FAILED) {
60  throwStreamPrint(NULL, "mmap(file=\"%s\",fd=%d,size=%ld kB) failed: %s\n", fileName, fd, (long) s.st_size, strerror(errno));
61  }
62  return res;
63 }

65 omc_mmap_write_unix omc_mmap_open_write_unix(const char *fileName, size_t size)
66 {
67  omc_mmap_write_unix res = {0};
68  int fd = open(fileName, O_RDWR | O_CREAT, S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH);
69  if (fd < 0) {
70  throwStreamPrint(NULL, "Failed to open file %s for reading: %s\n", fileName, strerror(errno));
71  }
72  if (size == 0) {
73  struct stat s;
74  if (fstat(fd, &s) < 0) {
75  close(fd);
76  throwStreamPrint(NULL, "fstat %s failed: %s\n", fileName, strerror(errno));
77  }
78  res.size = s.st_size;
79  } else {
80  res.size = size;
81  lseek(fd, size, SEEK_SET);
82  }
83  res.data = res.size == 0 ? NULL : (char*) mmap(0, res.size, PROT_WRITE, MAP_SHARED, fd, 0);
84  close(fd);
85  if (res.data == MAP_FAILED) {
86  throwStreamPrint(NULL, "mmap(file=\"%s\",fd=%d,size=%ld kB) failed: %s\n", fileName, fd, (long) res.size, strerror(errno));
87  }
88  return res;
89 }

103 static FILE* omc_mmap_common(const char *fileName, const char *mode, size_t *size, char **data)
104 {
105  FILE *file = fopen(fileName, mode);
106  size_t fileSize;
107  if (!file) {
108  throwStreamPrint(NULL, "Failed to open file %s for reading: %s\n", fileName, strerror(errno));
109  }
110  fseek(file, 0, SEEK_END);
111  fileSize = ftell(file);
112  rewind(file);
113  if (*size == 0) {
114  *size = fileSize;
115  }
116  if (*size > fileSize) {
117  *data = (char*) calloc(*size,1);
118  } else {
119  *data = (char*) malloc(*size);
120  }
121  if (1 != fread(*data, (*size > fileSize ? fileSize : *size), 1, file)) {
122  throwStreamPrint(NULL, "Failed to read file data: %s\n", fileName);
123  }
124  return file;
125 }
