25  Number beta, Vector &y) const
26  {
27  DBG_ASSERT(IsValid(matrix_));
28 
29  // Take care of the y part of the addition
30  if ( beta!=0.0 ) {
31  y.Scal(beta);
32  }
33  else {
34  y.Set(0.0); // In case y hasn't been initialized yet
35  }
36
