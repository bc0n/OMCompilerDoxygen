83 void zmq::vmci_listener_t::in_event ()
84 {
85  fd_t fd = accept ();
86 
87  // If connection was reset by the peer in the meantime, just ignore it.
88  if (fd == retired_fd) {
89  socket->event_accept_failed (endpoint, zmq_errno());
90  return;
91  }
92 
93  tune_vmci_buffer_size (this->get_ctx (), fd, options.vmci_buffer_size, options.vmci_buffer_min_size, options.vmci_buffer_max_size);
94 
95  if (options.vmci_connect_timeout > 0)
96  {
97 #if defined ZMQ_HAVE_WINDOWS
98  tune_vmci_connect_timeout (this->get_ctx (), fd, options.vmci_connect_timeout);
99 #else
100  struct timeval timeout = {0, options.vmci_connect_timeout * 1000};
101  tune_vmci_connect_timeout (this->get_ctx (), fd, timeout);
102 #endif
103  }
104 
105  // Create the engine object for this connection.
106  stream_engine_t *engine = new (std::nothrow)
107  stream_engine_t (fd, options, endpoint);
108  alloc_assert (engine);
109 
110  // Choose I/O thread to run connecter in. Given that we are already
111  // running in an I/O thread, there must be at least one available.
112  io_thread_t *io_thread = choose_io_thread (options.affinity);
113  zmq_assert (io_thread);
114 
115  // Create and launch a session object.
116  session_base_t *session = session_base_t::create (io_thread, false, socket,
117  options, NULL);
118  errno_assert (session);
119  session->inc_seqnum ();
120  launch_child (session);
121  send_attach (session, engine, false);
122  socket->event_accepted (endpoint, fd);
123 }

143 int zmq::vmci_listener_t::set_address (const char *addr_)
144 {
145  // Create addr on stack for auto-cleanup
146  std::string addr (addr_);
147 
148  // Initialise the address structure.
149  vmci_address_t address(this->get_ctx ());
150  int rc = address.resolve (addr.c_str());
151  if (rc != 0)
152  return -1;
153 
154  // Create a listening socket.
155  s = open_socket (this->get_ctx ()->get_vmci_socket_family (), SOCK_STREAM, 0);
156 #ifdef ZMQ_HAVE_WINDOWS
157  if (s == INVALID_SOCKET) {
158  errno = wsa_error_to_errno(WSAGetLastError());
159  return -1;
160  }
161 #if !defined _WIN32_WCE
162  // On Windows, preventing sockets to be inherited by child processes.
163  BOOL brc = SetHandleInformation((HANDLE)s, HANDLE_FLAG_INHERIT, 0);
164  win_assert(brc);
165 #endif
166 #else
167  if (s == -1)
168  return -1;
169 #endif
170 
171  address.to_string (endpoint);
172 
173  // Bind the socket.
174  rc = bind (s, address.addr (), address.addrlen ());
175 #ifdef ZMQ_HAVE_WINDOWS
176  if (rc == SOCKET_ERROR) {
177  errno = wsa_error_to_errno(WSAGetLastError());
178  goto error;
179  }
180 #else
181  if (rc != 0)
182  goto error;
183 #endif
184 
185  // Listen for incoming connections.
186  rc = listen (s, options.backlog);
187 #ifdef ZMQ_HAVE_WINDOWS
188  if (rc == SOCKET_ERROR) {
189  errno = wsa_error_to_errno(WSAGetLastError());
190  goto error;
191  }
192 #else
193  if (rc != 0)
194  goto error;
195 #endif
196 
197  socket->event_listening (endpoint, s);
198  return 0;
199 
200  error:
201  int err = errno;
202  close ();
203  errno = err;
204  return -1;
205 }

221 zmq::fd_t zmq::vmci_listener_t::accept ()
222 {
223  // Accept one connection and deal with different failure modes.
224  // The situation where connection cannot be accepted due to insufficient
225  // resources is considered valid and treated by ignoring the connection.
226  zmq_assert (s != retired_fd);
227  fd_t sock = ::accept (s, NULL, NULL);
228 
229 #ifdef ZMQ_HAVE_WINDOWS
230  if (sock == INVALID_SOCKET) {
231  wsa_assert(WSAGetLastError() == WSAEWOULDBLOCK ||
232  WSAGetLastError() == WSAECONNRESET ||
233  WSAGetLastError() == WSAEMFILE ||
234  WSAGetLastError() == WSAENOBUFS);
235  return retired_fd;
236  }
237 #if !defined _WIN32_WCE
238  // On Windows, preventing sockets to be inherited by child processes.
239  BOOL brc = SetHandleInformation((HANDLE)sock, HANDLE_FLAG_INHERIT, 0);
240  win_assert(brc);
241 #endif
242 #else
243  if (sock == -1) {
244  errno_assert(errno == EAGAIN || errno == EWOULDBLOCK ||
245  errno == EINTR || errno == ECONNABORTED || errno == EPROTO ||
246  errno == ENOBUFS || errno == ENOMEM || errno == EMFILE ||
247  errno == ENFILE);
248  return retired_fd;
249  }
250 #endif
251 
252  // Race condition can cause socket not to be closed (if fork happens
253  // between accept and this point).
254 #ifdef FD_CLOEXEC
255  int rc = fcntl (sock, F_SETFD, FD_CLOEXEC);
256  errno_assert (rc != -1);
257 #endif
258 
259  return sock;
260 }
