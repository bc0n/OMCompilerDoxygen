109 void zmq::vmci_connecter_t::out_event ()
110 {
111  fd_t fd = connect ();
112  rm_fd (handle);
113  handle_valid = false;
114 
115  // Handle the error condition by attempt to reconnect.
116  if (fd == retired_fd) {
117  close ();
118  add_reconnect_timer();
119  return;
120  }
121 
122  tune_vmci_buffer_size (this->get_ctx (), fd, options.vmci_buffer_size,
123  options.vmci_buffer_min_size, options.vmci_buffer_max_size);
124 
125  if (options.vmci_connect_timeout > 0)
126  {
127 #if defined ZMQ_HAVE_WINDOWS
128  tune_vmci_connect_timeout (this->get_ctx (), fd, options.vmci_connect_timeout);
129 #else
130  struct timeval timeout = {0, options.vmci_connect_timeout * 1000};
131  tune_vmci_connect_timeout (this->get_ctx (), fd, timeout);
132 #endif
133  }
134 
135  // Create the engine object for this connection.
136  stream_engine_t *engine = new (std::nothrow)
137  stream_engine_t (fd, options, endpoint);
138  alloc_assert (engine);
139 
140  // Attach the engine to the corresponding session object.
141  send_attach (session, engine);
142 
143  // Shut the connecter down.
144  terminate ();
145 
146  socket->event_connected (endpoint, fd);
147 }

156 void zmq::vmci_connecter_t::start_connecting ()
157 {
158  // Open the connecting socket.
159  int rc = open ();
160 
161  // Connect may succeed in synchronous manner.
162  if (rc == 0) {
163  handle = add_fd (s);
164  handle_valid = true;
165  out_event ();
166  }
167 
168  // Handle any other error condition by eventual reconnect.
169  else {
170  if (s != retired_fd)
171  close ();
172  add_reconnect_timer ();
173  }
174 }

184 int zmq::vmci_connecter_t::get_new_reconnect_ivl ()
185 {
186  // The new interval is the current interval + random value.
187  int this_interval = current_reconnect_ivl +
188  (generate_random () % options.reconnect_ivl);
189 
190  // Only change the current reconnect interval if the maximum reconnect
191  // interval was set and if it's larger than the reconnect interval.
192  if (options.reconnect_ivl_max > 0 &&
193  options.reconnect_ivl_max > options.reconnect_ivl) {
194 
195  // Calculate the next interval
196  current_reconnect_ivl = current_reconnect_ivl * 2;
197  if(current_reconnect_ivl >= options.reconnect_ivl_max) {
198  current_reconnect_ivl = options.reconnect_ivl_max;
199  }
200  }
201  return this_interval;
202 }

204 int zmq::vmci_connecter_t::open ()
205 {
206  zmq_assert (s == retired_fd);
207 
208  int family = this->get_ctx ()->get_vmci_socket_family ();
209  if (family == -1)
210  return -1;
211 
212  // Create the socket.
213  s = open_socket (family, SOCK_STREAM, 0);
214 #ifdef ZMQ_HAVE_WINDOWS
215  if (s == INVALID_SOCKET) {
216  errno = wsa_error_to_errno(WSAGetLastError());
217  return -1;
218  }
219 #else
220  if (s == -1)
221  return -1;
222 #endif
223 
224  // Connect to the remote peer.
225  int rc = ::connect (
226  s, addr->resolved.vmci_addr->addr (),
227  addr->resolved.vmci_addr->addrlen ());
228 
229  // Connect was successful immediately.
230  if (rc == 0)
231  return 0;
232 
233  // Forward the error.
234  return -1;
235 }

251 zmq::fd_t zmq::vmci_connecter_t::connect ()
252 {
253  // Following code should handle both Berkeley-derived socket
254  // implementations and Solaris.
255  int err = 0;
256 #if defined ZMQ_HAVE_HPUX
257  int len = sizeof (err);
258 #else
259  socklen_t len = sizeof (err);
260 #endif
261  int rc = getsockopt (s, SOL_SOCKET, SO_ERROR, (char*) &err, &len);
262 
263  // Assert if the error was caused by 0MQ bug.
264  // Networking problems are OK. No need to assert.
265 #ifdef ZMQ_HAVE_WINDOWS
266  zmq_assert(rc == 0);
267  if (err != 0) {
268  if (err != WSAECONNREFUSED
269  && err != WSAETIMEDOUT
270  && err != WSAECONNABORTED
271  && err != WSAEHOSTUNREACH
272  && err != WSAENETUNREACH
273  && err != WSAENETDOWN
274  && err != WSAEACCES
275  && err != WSAEINVAL
276  && err != WSAEADDRINUSE
277  && err != WSAECONNRESET)
278  {
279  wsa_assert_no(err);
280  }
281  return retired_fd;
282  }
283 #else
284  // Following code should handle both Berkeley-derived socket
285  // implementations and Solaris.
286  if (rc == -1)
287  err = errno;
288  if (err != 0) {
289  errno = err;
290  errno_assert(
291  errno == ECONNREFUSED ||
292  errno == ECONNRESET ||
293  errno == ETIMEDOUT ||
294  errno == EHOSTUNREACH ||
295  errno == ENETUNREACH ||
296  errno == ENETDOWN ||
297  errno == EINVAL);
298  return retired_fd;
299  }
300 #endif
301 
302  fd_t result = s;
303  s = retired_fd;
304  return result;
305 }
