42 int CMP_CALLMODEL compareImprovementVar(const pricerec *current, const pricerec *candidate)
43 {
44  register int result = COMP_PREFERNONE;
45  register lprec *lp = current->lp;
46  register REAL testvalue, margin = PREC_IMPROVEGAP;
47  int currentcolno, currentvarno = current->varno,
48  candidatecolno, candidatevarno = candidate->varno;
49  MYBOOL isdual = candidate->isdual;
50 
51  if(isdual) {
52  candidatevarno = lp->var_basic[candidatevarno];
53  currentvarno = lp->var_basic[currentvarno];
54  }
55  candidatecolno = candidatevarno - lp->rows;
56  currentcolno = currentvarno - lp->rows;
57 
58  /* Do pivot-based selection unless Bland's (first index) rule is active */
59  if(lp->_piv_rule_ != PRICER_FIRSTINDEX) {
60 
61  MYBOOL candbetter;
62 
63  /* Find the largest value - normalize in case of the dual, since
64  constraint violation is expressed as a negative number. */
65  /* Use absolute test for "small numbers", relative otherwise */
66  testvalue = candidate->pivot;
67  if(fabs(testvalue) < LIMIT_ABS_REL)
68  testvalue -= current->pivot;
69  else
70  testvalue = my_reldiff(testvalue, current->pivot);
71  if(isdual)
72  testvalue = -testvalue;
73 
74  candbetter = (MYBOOL) (testvalue > 0);
75  if(candbetter) {
76  if(testvalue > margin)
77  result = COMP_PREFERCANDIDATE;
78  }
79 #if 0 /* Give more opportunity to optimize on non-primary criteria */
80  else if (testvalue < -margin)
81 #else /* Give reduced opportunity to optimize on non-primary criteria */
82  else if (testvalue < -lp->epsvalue)
83 #endif
84  result = COMP_PREFERINCUMBENT;
85 
86 #ifdef UseSortOnBound
87  /* Extra selection criterion based on the variable's range;
88  variable with - DUAL: small bound out; PRIMAL: large bound in */
89  if(result == COMP_PREFERNONE) {
90  testvalue = lp->upbo[candidatevarno] - lp->upbo[currentvarno];
91  if(testvalue < -margin)
92  result = COMP_PREFERINCUMBENT;
93  else if(testvalue > margin)
94  result = COMP_PREFERCANDIDATE;
95  result = my_chsign(isdual, result);
96  }
97 #endif
98 
99 #ifdef UseSortOnColumnLength
100  /* Prevent long columns from entering the basis */
101  if(result == COMP_PREFERNONE) {
102  if(candidatecolno > 0)
103  testvalue = mat_collength(lp->matA, candidatecolno) +
104  (is_obj_in_basis(lp) && (lp->obj[candidatecolno] != 0) ? 1 : 0);
105  else
106  testvalue = 1;
107  if(currentcolno > 0)
108  testvalue -= mat_collength(lp->matA, currentcolno) +
109  (is_obj_in_basis(lp) && (lp->obj[currentcolno] != 0) ? 1 : 0);
110  else
111  testvalue -= 1;
112  if(testvalue > 0)
113  result = COMP_PREFERINCUMBENT;
114  else if(testvalue < 0)
115  result = COMP_PREFERCANDIDATE;
116  result = my_chsign(isdual, result);
117  }
118 #endif
119 
120  /* Select absolute best if the non-primary criteria failed to separate */
121  if((result == COMP_PREFERNONE) && candbetter) {
122  result = COMP_PREFERCANDIDATE;
123  goto Finish;
124  }
125  }
126 
127  /* Final tie-breakers */
128  if(result == COMP_PREFERNONE) {
129 
130  /* Add randomization tie-braker */
131  if(lp->piv_strategy & PRICE_RANDOMIZE) {
132  result = my_sign(PRICER_RANDFACT - rand_uniform(lp, 1.0));
133  if(candidatevarno < currentvarno)
134  result = -result;
135  }
136 
137  /* Resolve ties via index ordinal */
138  if(result == COMP_PREFERNONE) {
139  if(candidatevarno < currentvarno)
140  result = COMP_PREFERCANDIDATE;
141  else /* if(candidatevarno > currentvarno) */
142  result = COMP_PREFERINCUMBENT;
143  if(lp->_piv_left_)
144  result = -result;
145  }
146  }
147 
148 Finish:
149  return( result );

411 /* Function to add a valid pivot candidate into the specified list */
412 STATIC int addCandidateVar(pricerec *candidate, multirec *multi, findCompare_func findCompare, MYBOOL allowSortedExpand)
413 {
414  int insertpos, delta = 1;
415  pricerec *targetrec;
416 
417  /* Find the insertion point (if any) */
418  if((multi->freeList[0] == 0) ||
419  (multi->sorted && allowSortedExpand) ||
420  (candidate->isdual && (multi->used == 1) && ((multi->step_last >= multi->epszero) ||
421  multi_truncatingvar(multi, ((pricerec *) (multi->sortedList[0].pvoidreal.ptr))->varno)))
422  ) {
423  UNIONTYPE QSORTrec searchTarget;
424 
425  /* Make sure that the list is sorted before the search for an insertion point */
426  if((multi->freeList[0] == 0) && !multi->sorted) {
427  multi->sorted = QS_execute(multi->sortedList, multi->used, findCompare, &insertpos);
428  multi->dirty = (MYBOOL) (insertpos > 0);
429  }
430 
431  /* Perform the search */
432  searchTarget.pvoidint2.ptr = (void *) candidate;
433  insertpos = sizeof(searchTarget);
434  insertpos = findIndexEx(&searchTarget, multi->sortedList-delta, multi->used, delta, insertpos, findCompare, TRUE);
435  if(insertpos > 0)
436  return( -1 );
437  insertpos = -insertpos - delta;
438 
439  /* Check if the candidate is worse than the worst of the list */
440  if(((insertpos >= multi->size) && (multi->freeList[0] == 0)) ||
441  ((insertpos == multi->used) && (!allowSortedExpand ||
442  (multi->step_last >= multi->epszero))))
443  return( -1 );
444 
445 #ifdef Paranoia
446  /* Do validation */
447  if((insertpos < 0) || (insertpos > multi->used))
448  return( -1 );
449 #endif
450 
451  /* Define the target for storing the candidate;
452  Case 1: List is full and we must discard the previously worst candidate
453  Case 2: List is not full and we simply use the next free position */
454  if(multi->freeList[0] == 0)
455  targetrec = (pricerec *) multi->sortedList[multi->used-1].pvoidreal.ptr;
456  else {
457  delta = multi->freeList[0]--;
458  delta = multi->freeList[delta];
459  targetrec = &(multi->items[delta]);
460  }
461  }
462  else {
463  delta = multi->freeList[0]--;
464  delta = multi->freeList[delta];
465  targetrec = &(multi->items[delta]);
466  insertpos = multi->used;
467  }
468 
469  /* Insert the new candidate record in the data store */
470  MEMCOPY(targetrec, candidate, 1);
471 
472  /* Store the pointer data and handle tree cases:
473  Case 1: The list is unsorted and not full; simply append pointer to list,
474  Case 2: The list is sorted and full; insert the pointer by discarding previous last,
475  Case 3: The list is sorted and not full; shift the inferior items down, and increment count */
476  if((multi->used < multi->size) && (insertpos >= multi->used)) {
477  QS_append(multi->sortedList, insertpos, targetrec);
478  multi->used++;
479  }
480  else {
481  if(multi->used == multi->size)
482  QS_insert(multi->sortedList, insertpos, targetrec, multi->size-1); /* Discard previous last */
483  else {
484  QS_insert(multi->sortedList, insertpos, targetrec, multi->used); /* Keep previous last */
485  multi->used++;
486  }
487  }
488  multi->active = insertpos;
489 
490 #ifdef Paranoia
491  if((insertpos >= multi->size) || (insertpos >= multi->used))
492  report(multi->lp, SEVERE, "addCandidateVar: Insertion point beyond limit!\n");
493 #endif
494 
495  return( insertpos );
496 }

532 /* Bound flip variable accumulation routine */
533 STATIC MYBOOL collectMinorVar(pricerec *candidate, multirec *longsteps, MYBOOL isphase2, MYBOOL isbatch)
534 {
535  int inspos;
536 
537  /* 1. Check for ratio and pivot validity (to have the extra flexibility that all
538  bound-flip candidates are also possible as basis-entering variables */
539  if(!validSubstitutionVar(candidate))
540  return( FALSE );
541 
542  /* 2. If the free-list is empty we need to see if we have a better candidate,
543  and for this the candidate list has to be sorted by merit */
544  if(!isbatch &&
545  !longsteps->sorted && (longsteps->used > 1) &&
546  ((longsteps->freeList[0] == 0) ||
547  multi_truncatingvar(longsteps, candidate->varno) ||
548  (longsteps->step_last >= longsteps->epszero) )) {
549  longsteps->sorted = QS_execute(longsteps->sortedList, longsteps->used,
550  (findCompare_func *) compareSubstitutionQS, &inspos);
551  longsteps->dirty = (MYBOOL) (inspos > 0);
552  if(longsteps->dirty)
553  multi_recompute(longsteps, 0, isphase2, TRUE);
554  }
555 
556  /* 3. Now handle three cases...
557  - Add to the list when the list is not full and there is opportunity for improvement,
558  - Check if we should replace an incumbent when the list is full,
559  - Check if we should replace an incumbent when the list is not full, there is no room
560  for improvement, but the current candidate is better than an incumbent. */
561  inspos = addCandidateVar(candidate, longsteps, (findCompare_func *) compareSubstitutionQS, TRUE);
562 
563  /* 4. Recompute steps and objective, and (if relevant) determine if we
564  may be suboptimal in relation to an incumbent MILP solution. */
565  return( (MYBOOL) (inspos >= 0) &&
566  ((isbatch == TRUE) || multi_recompute(longsteps, inspos, isphase2, TRUE)) );
567 }

660 /* Routine to verify accuracy of the current basis factorization */
661 STATIC MYBOOL serious_facterror(lprec *lp, REAL *bvector, int maxcols, REAL tolerance)
662 {
663  int i, j, ib, ie, nz, nc;
664  REAL sum, tsum = 0, err = 0;
665  MATrec *mat = lp->matA;
666 
667  if(bvector == 0)
668  bvector = lp->bsolveVal;
669  nc =0;
670  nz = 0;
671  for(i = 1; (i <= lp->rows) && (nc <= maxcols); i++) {
672 
673  /* Do we have a non-slack variable? (we choose to skip slacks,
674  since they have "natural" good accuracy properties) */
675  j = lp->var_basic[i] - lp->rows;
676  if(j <= 0)
677  continue;
678  nc++;
679 
680  /* Compute cross product for basic, non-slack column */
681  ib = mat->col_end[j-1];
682  ie = mat->col_end[j];
683  nz += ie - ib;
684  sum = get_OF_active(lp, j+lp->rows, bvector[0]);
685  for(; ib < ie; ib++)
686  sum += COL_MAT_VALUE(ib)*bvector[COL_MAT_ROWNR(ib)];
687 
688  /* Catch high precision early, so we don't to uneccessary work */
689  tsum += sum;
690  SETMAX(err, fabs(sum));
691  if((tsum / nc > tolerance / 100) && (err < tolerance / 100))
692  break;
693  }
694  err /= mat->infnorm;
695  return( (MYBOOL) (err >= tolerance) );
696 }

759 /* Primal: Prevent acceptance of an entering variable when the magnitude of
760  other candidates is also very small.
761  Dual: Prevent acceptance of a leaving variable when the magnitude of
762  other candidates is also very small.
763 
764  Both of these cases are associated with numerical stalling, which we could
765  argue should be detected and handled by the stalling monitor routine. */
766 STATIC MYBOOL verify_stability(lprec *lp, MYBOOL isprimal, REAL xfeas, REAL sfeas, int nfeas)
767 {
768  MYBOOL testOK = TRUE;
769  return( testOK );
770 
771 #if 1
772  /* Try to make dual feasibility as tight as possible */
773  if(!isprimal)
774 /* if(lp->P1extraVal == 0) */
775  {
776  xfeas /= (1+lp->rhsmax);
777  sfeas /= (1+lp->rhsmax);
778  }
779 #endif
780  xfeas = fabs(xfeas); /* Maximum (positive) infeasibility */
781 /* if(xfeas < lp->epspivot) { */
782  if(xfeas < lp->epssolution) {
783  REAL f;
784  sfeas = fabs(sfeas); /* Make sum of infeasibilities positive */
785  xfeas = (sfeas-xfeas)/nfeas; /* Average "residual" feasibility */
786  f = 1 + log10((REAL) nfeas); /* Some numerical complexity scalar */
787  /* Numerical errors can interact to cause non-convergence, and the
788  idea is to relax the tolerance to account for this and only
789  marginally weakening the (user-specified) tolerance. */
790  if((sfeas-xfeas) < f*lp->epsprimal)
791  testOK = FALSE;
792  }
793  return( testOK );
794 }

833 STATIC int colprim(lprec *lp, REAL *drow, int *nzdrow, MYBOOL skipupdate, int partialloop, int *candidatecount, MYBOOL updateinfeas, REAL *xviol)
834 {

837  pricerec current, candidate;
838  MYBOOL collectMP = FALSE;

840 
841  /* Identify pivot column according to pricing strategy; set
842  entering variable initial threshold reduced cost value to "0" */
843  current.pivot = lp->epsprimal; /* Minimum acceptable improvement */
844  current.varno = 0;
845  current.lp = lp;
846  current.isdual = FALSE;
847  candidate.lp = lp;
848  candidate.isdual = FALSE;
849  *candidatecount = 0;
850 
851  /* Update local value of pivot setting and determine active multiple pricing set */
852  lp->_piv_rule_ = get_piv_rule(lp);
853 doLoop:
854  nloop++;
855  if((lp->multivars != NULL) && ((lp->simplex_mode & SIMPLEX_PRIMAL_PRIMAL) != 0)) {
856  collectMP = multi_mustupdate(lp->multivars);
857  if(collectMP) {
858  multi_restart(lp->multivars);
859  coltarget = NULL;
860  }
861  else
862  coltarget = multi_indexSet(lp->multivars, FALSE);
863  }
864

866  (i.e. the previous iteration was not a "minor" iteration/bound flip) */
867  if(!skipupdate) {
868 #ifdef UsePrimalReducedCostUpdate
869  /* Recompute from scratch only at the beginning, otherwise update */
870  if((lp->current_iter > 0) && (refactRecent(lp) == AUTOMATIC))
871 #endif
872  compute_reducedcosts(lp, FALSE, 0, coltarget, (MYBOOL) ((nloop <= 1) || (partialloop > 1)),
873  NULL, NULL,
874  drow, nzdrow,
875  MAT_ROUNDDEFAULT);
876  }

879  have only been updated for columns in the active partial range. */
880  ix = 1;
881  iy = nzdrow[0];
882  ninfeas = 0;
883  xinfeas = 0;
884  sinfeas = 0;
885  makePriceLoop(lp, &ix, &iy, &iz);
886  iy *= iz;
887  for(; ix*iz <= iy; ix += iz) {
888  i = nzdrow[ix];
889 #if 0 /* Not necessary since we masked them out in compute_reducedcosts() */
890  if(i > lp->sum-abs(lp->P1extraDim))
891  continue;
892 #endif

894  /* Check if the pivot candidate is on the block-list */
895  if(lp->rejectpivot[0] > 0) {
896  int kk;
897  for(kk = 1; (kk <= lp->rejectpivot[0]) && (i != lp->rejectpivot[kk]); kk++);
898  if(kk <= lp->rejectpivot[0])
899  continue;
900  }
901 
902  /* Retrieve the applicable reduced cost - threshold should not be smaller than 0 */
903  f = my_chsign(lp->is_lower[i], drow[i]);
904  if(f <= epsvalue)
905  continue;
906 
907  /* Find entering variable according to strategy (largest positive f) */
908  ninfeas++;
909  SETMAX(xinfeas, f);
910  sinfeas += f;
911  candidate.pivot = normalizeEdge(lp, i, f, FALSE);
912  candidate.varno = i;
913  if(findImprovementVar(&current, &candidate, collectMP, candidatecount))
914  break;
915  }
916

918  if(lp->multivars != NULL) {
919  if(collectMP) {
920  if(!lp->multivars->sorted)
921  lp->multivars->sorted = QS_execute(lp->multivars->sortedList, lp->multivars->used,
922  (findCompare_func *) compareImprovementQS, NULL);
923  coltarget = multi_indexSet(lp->multivars, TRUE);
924  }
925  else if((current.varno == 0) && (lp->multivars->retries == 0)) {
926  ix = partial_blockStart(lp, FALSE);
927  iy = partial_blockEnd(lp, FALSE);
928  lp->multivars->used = 0;
929  lp->multivars->retries++;
930  goto doLoop;
931  }
932  /* Shrink the candidate list */
933  lp->multivars->retries = 0;
934  if(current.varno != 0)
935  multi_removevar(lp->multivars, current.varno);
936  }
937 
938  /* Check for optimality */
939  if(xviol != NULL)
940  *xviol = xinfeas;
941  if(updateinfeas)
942  lp->suminfeas = fabs(sinfeas);
943  if((lp->multivars == NULL) && (current.varno > 0) &&
944  !verify_stability(lp, TRUE, xinfeas, sinfeas, ninfeas))
945  current.varno = 0;
946 
947  /* Produce statistics */
948  if(lp->spx_trace) {
949  if(current.varno > 0)
950  report(lp, DETAILED, "colprim: Column %d reduced cost = " RESULTVALUEMASK "\n",
951  current.varno, current.pivot);
952  else
953  report(lp, DETAILED, "colprim: No positive reduced costs found, optimality!\n");
954  }
955 
956  return( current.varno );
957 } /* colprim */
