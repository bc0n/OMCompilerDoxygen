134 /*
135  * We use signals to stop threads during GC.
136  *
137  * Suspended threads wait in signal handler for SIG_THR_RESTART.
138  * That's more portable than semaphores or condition variables.
139  * (We do use sem_post from a signal handler, but that should be portable.)
140  *
141  * The thread suspension signal SIG_SUSPEND is now defined in gc_priv.h.
142  * Note that we can't just stop a thread; we need it to save its stack
143  * pointer(s) and acknowledge.

285 STATIC void GC_suspend_handler_inner(ptr_t dummy GC_ATTR_UNUSED,
286  void * context GC_ATTR_UNUSED)
287 {
288  pthread_t self = pthread_self();
289  GC_thread me;
290  IF_CANCEL(int cancel_state;)
291  AO_t my_stop_count = AO_load_acquire(&GC_stop_count);
292  /* After the barrier, this thread should see */
293  /* the actual content of GC_threads. */
294 
295  DISABLE_CANCEL(cancel_state);
296  /* pthread_setcancelstate is not defined to be async-signal-safe. */
297  /* But the glibc version appears to be in the absence of */
298  /* asynchronous cancellation. And since this signal handler */
299  /* to block on sigsuspend, which is both async-signal-safe */
300  /* and a cancellation point, there seems to be no obvious way */
301  /* out of it. In fact, it looks to me like an async-signal-safe */
302  /* cancellation point is inherently a problem, unless there is */
303  /* some way to disable cancellation in the handler. */
304 # ifdef DEBUG_THREADS
305  GC_log_printf("Suspending %p\n", (void *)self);
306 # endif
307  GC_ASSERT(((word)my_stop_count & 1) == 0);
308 
309  me = GC_lookup_thread_async(self);
310 
311 # ifdef GC_ENABLE_SUSPEND_THREAD
312  if (AO_load(&me->suspended_ext)) {
313  GC_store_stack_ptr(me);
314  sem_post(&GC_suspend_ack_sem);
315  suspend_self_inner(me);
316 # ifdef DEBUG_THREADS
317  GC_log_printf("Continuing %p on GC_resume_thread\n", (void *)self);
318 # endif
319  RESTORE_CANCEL(cancel_state);
320  return;
321  }
322 # endif
323 
324  if (((word)me->stop_info.last_stop_count & ~(word)0x1)
325  == (word)my_stop_count) {
326  /* Duplicate signal. OK if we are retrying. */
327  if (!GC_retry_signals) {
328  WARN("Duplicate suspend signal in thread %p\n", self);
329  }
330  RESTORE_CANCEL(cancel_state);
331  return;
332  }
333  GC_store_stack_ptr(me);
334 
335 # ifdef THREAD_SANITIZER
336  /* TSan disables signals around signal handlers. Without */
337  /* a pthread_sigmask call, sigsuspend may block forever. */
338  {
339  sigset_t set;
340  sigemptyset(&set);
341  GC_ASSERT(GC_sig_suspend != SIGNAL_UNSET);
342  GC_ASSERT(GC_sig_thr_restart != SIGNAL_UNSET);
343  sigaddset(&set, GC_sig_suspend);
344  sigaddset(&set, GC_sig_thr_restart);
345  if (pthread_sigmask(SIG_UNBLOCK, &set, NULL) != 0)
346  ABORT("pthread_sigmask failed in suspend handler");
347  }
348 # endif
349  /* Tell the thread that wants to stop the world that this */
350  /* thread has been stopped. Note that sem_post() is */
351  /* the only async-signal-safe primitive in LinuxThreads. */
352  sem_post(&GC_suspend_ack_sem);
353  AO_store_release(&me->stop_info.last_stop_count, my_stop_count);
354 
355  /* Wait until that thread tells us to restart by sending */
356  /* this thread a GC_sig_thr_restart signal (should be masked */
357  /* at this point thus there is no race). */
358  /* We do not continue until we receive that signal, */
359  /* but we do not take that as authoritative. (We may be */
360  /* accidentally restarted by one of the user signals we */
361  /* don't block.) After we receive the signal, we use a */
362  /* primitive and expensive mechanism to wait until it's */
363  /* really safe to proceed. Under normal circumstances, */
364  /* this code should not be executed. */
365  do {
366  sigsuspend (&suspend_handler_mask);
367  } while (AO_load_acquire(&GC_world_is_stopped)
368  && AO_load(&GC_stop_count) == my_stop_count);
369 
370 # ifdef DEBUG_THREADS
371  GC_log_printf("Continuing %p\n", (void *)self);
372 # endif
373 # ifndef GC_NETBSD_THREADS_WORKAROUND
374  if (GC_retry_signals)
375 # endif
376  {
377  /* If the RESTART signal loss is possible (though it should be */
378  /* less likely than losing the SUSPEND signal as we do not do */
379  /* much between the first sem_post and sigsuspend calls), more */
380  /* handshaking is provided to work around it. */
381  sem_post(&GC_suspend_ack_sem);
382 # ifdef GC_NETBSD_THREADS_WORKAROUND
383  if (GC_retry_signals)
384 # endif
385  {
386  /* Set the flag (the lowest bit of last_stop_count) that the */
387  /* thread has been restarted. */
388  AO_store_release(&me->stop_info.last_stop_count,
389  (AO_t)((word)my_stop_count | 1));
390  }
391  }
392  RESTORE_CANCEL(cancel_state);
393 }

414 static int resend_lost_signals(int n_live_threads,
415  int (*suspend_restart_all)(void))
416 {
417 # define WAIT_UNIT 3000
418 # define RETRY_INTERVAL 100000
419 
420  if (n_live_threads > 0) {
421  unsigned long wait_usecs = 0; /* Total wait since retry. */
422  for (;;) {
423  int ack_count;
424 
425  sem_getvalue(&GC_suspend_ack_sem, &ack_count);
426  if (ack_count == n_live_threads)
427  break;
428  if (wait_usecs > RETRY_INTERVAL) {
429  int newly_sent = suspend_restart_all();
430 
431  GC_COND_LOG_PRINTF("Resent %d signals after timeout\n", newly_sent);
432  sem_getvalue(&GC_suspend_ack_sem, &ack_count);
433  if (newly_sent < n_live_threads - ack_count) {
434  WARN("Lost some threads while stopping or starting world?!\n", 0);
435  n_live_threads = ack_count + newly_sent;
436  }
437  wait_usecs = 0;
438  }
439 
440 # ifdef LINT2
441  /* Workaround "waiting while holding a lock" warning. */
442 # undef WAIT_UNIT
443 # define WAIT_UNIT 1
444  sched_yield();
445 # elif defined(CPPCHECK) /* || _POSIX_C_SOURCE >= 199309L */
446  {
447  struct timespec ts;
448 
449  ts.tv_sec = 0;
450  ts.tv_nsec = WAIT_UNIT * 1000;
451  (void)nanosleep(&ts, NULL);
452  }
453 # else
454  usleep(WAIT_UNIT);
455 # endif
456  wait_usecs += WAIT_UNIT;
457  }
458  }
459  return n_live_threads;
460 }

535  GC_API void GC_CALL GC_suspend_thread(GC_SUSPEND_THREAD_ID thread) {
536  GC_thread t;
537  IF_CANCEL(int cancel_state;)
538  DCL_LOCK_STATE;
539 
540  LOCK();
541  t = GC_lookup_thread((pthread_t)thread);
542  if (t == NULL || t -> suspended_ext) {
543  UNLOCK();
544  return;
545  }
546 
547  /* Set the flag making the change visible to the signal handler. */
548  AO_store_release(&t->suspended_ext, TRUE);
549 
550  if (THREAD_EQUAL((pthread_t)thread, pthread_self())) {
551  UNLOCK();
552  /* It is safe as "t" cannot become invalid here (no race with */
553  /* GC_unregister_my_thread). */
554  (void)GC_do_blocking(suspend_self_inner, t);
555  return;
556  }
557  if ((t -> flags & FINISHED) != 0) {
558  /* Terminated but not joined yet. */
559  UNLOCK();
560  return;
561  }
562 
563  DISABLE_CANCEL(cancel_state);
564  /* GC_suspend_thread is not a cancellation point. */
565 # ifdef PARALLEL_MARK
566  /* Ensure we do not suspend a thread while it is rebuilding */
567  /* a free list, otherwise such a dead-lock is possible: */
568  /* thread 1 is blocked in GC_wait_for_reclaim holding */
569  /* the allocation lock, thread 2 is suspended in */
570  /* GC_reclaim_generic invoked from GC_generic_malloc_many */
571  /* (with GC_fl_builder_count > 0), and thread 3 is blocked */
572  /* acquiring the allocation lock in GC_resume_thread. */
573  if (GC_parallel)
574  GC_wait_for_reclaim();
575 # endif
576 
577  if (GC_manual_vdb) {
578  /* See the relevant comment in GC_stop_world. */
579  GC_acquire_dirty_lock();
580  }
581  /* Else do not acquire the lock as the write fault handler might */
582  /* be trying to acquire this lock too, and the suspend handler */
583  /* execution is deferred until the write fault handler completes. */
584 
585  /* TODO: Support GC_retry_signals (not needed for TSan) */
586  switch (RAISE_SIGNAL(t, GC_sig_suspend)) {
587  /* ESRCH cannot happen as terminated threads are handled above. */
588  case 0:
589  break;
590  default:
591  ABORT("pthread_kill failed");
592  }
593 
594  /* Wait for the thread to complete threads table lookup and */
595  /* stack_ptr assignment. */
596  GC_ASSERT(GC_thr_initialized);
597  while (sem_wait(&GC_suspend_ack_sem) != 0) {
598  if (errno != EINTR)
599  ABORT("sem_wait for handler failed (suspend_self)");
600  }
601  if (GC_manual_vdb)
602  GC_release_dirty_lock();
603  RESTORE_CANCEL(cancel_state);
604  UNLOCK();
605  }

639 /* We hold allocation lock. Should do exactly the right thing if the */
640 /* world is stopped. Should not fail if it isn't. */
641 GC_INNER void GC_push_all_stacks(void)
642 {
643  GC_bool found_me = FALSE;
644  size_t nthreads = 0;
645  int i;
646  GC_thread p;
647  ptr_t lo, hi;
648  /* On IA64, we also need to scan the register backing store. */
649  IF_IA64(ptr_t bs_lo; ptr_t bs_hi;)
650  struct GC_traced_stack_sect_s *traced_stack_sect;
651  pthread_t self = pthread_self();
652  word total_size = 0;
653 
654  if (!EXPECT(GC_thr_initialized, TRUE))
655  GC_thr_init();
656 # ifdef DEBUG_THREADS
657  GC_log_printf("Pushing stacks from thread %p\n", (void *)self);
658 # endif
659  for (i = 0; i < THREAD_TABLE_SZ; i++) {
660  for (p = GC_threads[i]; p != 0; p = p -> next) {
661  if (p -> flags & FINISHED) continue;
662  ++nthreads;
663  traced_stack_sect = p -> traced_stack_sect;
664  if (THREAD_EQUAL(p -> id, self)) {
665  GC_ASSERT(!p->thread_blocked);
666 # ifdef SPARC
667  lo = (ptr_t)GC_save_regs_in_stack();
668 # else
669  lo = GC_approx_sp();
670 # endif
671  found_me = TRUE;
672  IF_IA64(bs_hi = (ptr_t)GC_save_regs_in_stack();)
673  } else {
674  lo = (ptr_t)AO_load((volatile AO_t *)&p->stop_info.stack_ptr);
675  IF_IA64(bs_hi = p -> backing_store_ptr;)
676  if (traced_stack_sect != NULL
677  && traced_stack_sect->saved_stack_ptr == lo) {
678  /* If the thread has never been stopped since the recent */
679  /* GC_call_with_gc_active invocation then skip the top */
680  /* "stack section" as stack_ptr already points to. */
681  traced_stack_sect = traced_stack_sect->prev;
682  }
683  }
684  if ((p -> flags & MAIN_THREAD) == 0) {
685  hi = p -> stack_end;
686  IF_IA64(bs_lo = p -> backing_store_end);
687  } else {
688  /* The original stack. */
689  hi = GC_stackbottom;
690  IF_IA64(bs_lo = BACKING_STORE_BASE;)
691  }
692 # ifdef DEBUG_THREADS
693  GC_log_printf("Stack for thread %p = [%p,%p)\n",
694  (void *)p->id, (void *)lo, (void *)hi);
695 # endif
696  if (0 == lo) ABORT("GC_push_all_stacks: sp not set!");
697  if (p->altstack != NULL && (word)p->altstack <= (word)lo
698  && (word)lo <= (word)p->altstack + p->altstack_size) {
699  hi = p->altstack + p->altstack_size;
700  /* FIXME: Need to scan the normal stack too, but how ? */
701  /* FIXME: Assume stack grows down */
702  }
703  GC_push_all_stack_sections(lo, hi, traced_stack_sect);
704 # ifdef STACK_GROWS_UP
705  total_size += lo - hi;
706 # else
707  total_size += hi - lo; /* lo <= hi */
708 # endif
709 # ifdef NACL
710  /* Push reg_storage as roots, this will cover the reg context. */
711  GC_push_all_stack((ptr_t)p -> stop_info.reg_storage,
712  (ptr_t)(p -> stop_info.reg_storage + NACL_GC_REG_STORAGE_SIZE));
713  total_size += NACL_GC_REG_STORAGE_SIZE * sizeof(ptr_t);
714 # endif
715 # ifdef IA64
716 # ifdef DEBUG_THREADS
717  GC_log_printf("Reg stack for thread %p = [%p,%p)\n",
718  (void *)p->id, (void *)bs_lo, (void *)bs_hi);
719 # endif
720  /* FIXME: This (if p->id==self) may add an unbounded number of */
721  /* entries, and hence overflow the mark stack, which is bad. */
722  GC_push_all_register_sections(bs_lo, bs_hi,
723  THREAD_EQUAL(p -> id, self),
724  traced_stack_sect);
725  total_size += bs_hi - bs_lo; /* bs_lo <= bs_hi */
726 # endif
727  }
728  }
729  GC_VERBOSE_LOG_PRINTF("Pushed %d thread stacks\n", (int)nthreads);
730  if (!found_me && !GC_in_thread_creation)
731  ABORT("Collecting from unknown thread");
732  GC_total_stacksize = total_size;
733 }

742 /* We hold the allocation lock. Suspend all threads that might */
743 /* still be running. Return the number of suspend signals that */
744 /* were sent. */
745 STATIC int GC_suspend_all(void)
746 {
747  int n_live_threads = 0;
748  int i;
749 # ifndef NACL
750  GC_thread p;
751 # ifndef GC_OPENBSD_UTHREADS
752  int result;
753 # endif
754  pthread_t self = pthread_self();
755 
756  for (i = 0; i < THREAD_TABLE_SZ; i++) {
757  for (p = GC_threads[i]; p != 0; p = p -> next) {
758  if (!THREAD_EQUAL(p -> id, self)) {
759  if ((p -> flags & FINISHED) != 0) continue;
760  if (p -> thread_blocked) /* Will wait */ continue;
761 # ifndef GC_OPENBSD_UTHREADS
762 # ifdef GC_ENABLE_SUSPEND_THREAD
763  if (p -> suspended_ext) continue;
764 # endif
765  if (AO_load(&p->stop_info.last_stop_count) == GC_stop_count)
766  continue; /* matters only if GC_retry_signals */
767  n_live_threads++;
768 # endif
769 # ifdef DEBUG_THREADS
770  GC_log_printf("Sending suspend signal to %p\n", (void *)p->id);
771 # endif
772 
773 # ifdef GC_OPENBSD_UTHREADS
774  {
775  stack_t stack;
776 
777  GC_acquire_dirty_lock();
778  if (pthread_suspend_np(p -> id) != 0)
779  ABORT("pthread_suspend_np failed");
780  GC_release_dirty_lock();
781  if (pthread_stackseg_np(p->id, &stack))
782  ABORT("pthread_stackseg_np failed");
783  p -> stop_info.stack_ptr = (ptr_t)stack.ss_sp - stack.ss_size;
784  if (GC_on_thread_event)
785  GC_on_thread_event(GC_EVENT_THREAD_SUSPENDED,
786  (void *)p->id);
787  }
788 # else
789  /* The synchronization between GC_dirty (based on */
790  /* test-and-set) and the signal-based thread suspension */
791  /* is performed in GC_stop_world because */
792  /* GC_release_dirty_lock cannot be called before */
793  /* acknowledging the thread is really suspended. */
794  result = RAISE_SIGNAL(p, GC_sig_suspend);
795  switch(result) {
796  case ESRCH:
797  /* Not really there anymore. Possible? */
798  n_live_threads--;
799  break;
800  case 0:
801  if (GC_on_thread_event)
802  GC_on_thread_event(GC_EVENT_THREAD_SUSPENDED,
803  (void *)(word)THREAD_SYSTEM_ID(p));
804  /* Note: thread id might be truncated. */
805  break;
806  default:
807  ABORT_ARG1("pthread_kill failed at suspend",
808  ": errcode= %d", result);
809  }
810 # endif
811  }
812  }
813  }
814 
815 # else /* NACL */
816 # ifndef NACL_PARK_WAIT_NANOSECONDS
817 # define NACL_PARK_WAIT_NANOSECONDS (100 * 1000)
818 # endif
819 # define NANOS_PER_SECOND (1000UL * 1000 * 1000)
820  unsigned long num_sleeps = 0;
821 
822 # ifdef DEBUG_THREADS
823  GC_log_printf("pthread_stop_world: num_threads=%d\n",
824  GC_nacl_num_gc_threads - 1);
825 # endif
826  GC_nacl_thread_parker = pthread_self();
827  GC_nacl_park_threads_now = 1;
828 
829  if (GC_manual_vdb)
830  GC_acquire_dirty_lock();
831  while (1) {
832  int num_threads_parked = 0;
833  struct timespec ts;
834  int num_used = 0;
835 
836  /* Check the 'parked' flag for each thread the GC knows about. */
837  for (i = 0; i < MAX_NACL_GC_THREADS
838  && num_used < GC_nacl_num_gc_threads; i++) {
839  if (GC_nacl_thread_used[i] == 1) {
840  num_used++;
841  if (GC_nacl_thread_parked[i] == 1) {
842  num_threads_parked++;
843  if (GC_on_thread_event)
844  GC_on_thread_event(GC_EVENT_THREAD_SUSPENDED, (void *)(word)i);
845  }
846  }
847  }
848  /* -1 for the current thread. */
849  if (num_threads_parked >= GC_nacl_num_gc_threads - 1)
850  break;
851  ts.tv_sec = 0;
852  ts.tv_nsec = NACL_PARK_WAIT_NANOSECONDS;
853 # ifdef DEBUG_THREADS
854  GC_log_printf("Sleep waiting for %d threads to park...\n",
855  GC_nacl_num_gc_threads - num_threads_parked - 1);
856 # endif
857  /* This requires _POSIX_TIMERS feature. */
858  nanosleep(&ts, 0);
859  if (++num_sleeps > NANOS_PER_SECOND / NACL_PARK_WAIT_NANOSECONDS) {
860  WARN("GC appears stalled waiting for %" WARN_PRIdPTR
861  " threads to park...\n",
862  GC_nacl_num_gc_threads - num_threads_parked - 1);
863  num_sleeps = 0;
864  }
865  }
866  if (GC_manual_vdb)
867  GC_release_dirty_lock();
868 # endif /* NACL */
869  return n_live_threads;
870 }

985  {

990  if (GC_nacl_thread_parker == self)
