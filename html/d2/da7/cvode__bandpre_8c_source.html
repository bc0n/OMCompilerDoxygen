80 {

84  long int mup, mlp, storagemu;

86 
87  if (cvode_mem == NULL) {
88  cvProcessError(NULL, CVSPILS_MEM_NULL, "CVBANDPRE", "CVBandPrecInit", MSGBP_MEM_NULL);
89  return(CVSPILS_MEM_NULL);
90  }
91  cv_mem = (CVodeMem) cvode_mem;
92 
93  /* Test if one of the SPILS linear solvers has been attached */
94  if (cv_mem->cv_lmem == NULL) {
95  cvProcessError(cv_mem, CVSPILS_LMEM_NULL, "CVBANDPRE", "CVBandPrecInit", MSGBP_LMEM_NULL);
96  return(CVSPILS_LMEM_NULL);
97  }
98  cvspils_mem = (CVSpilsMem) cv_mem->cv_lmem;
99 
100  /* Test if the NVECTOR package is compatible with the BAND preconditioner */
101  if(vec_tmpl->ops->nvgetarraypointer == NULL) {
102  cvProcessError(cv_mem, CVSPILS_ILL_INPUT, "CVBANDPRE", "CVBandPrecInit", MSGBP_BAD_NVECTOR);
103  return(CVSPILS_ILL_INPUT);
104  }
105 
106  pdata = NULL;
107  pdata = (CVBandPrecData) malloc(sizeof *pdata); /* Allocate data memory */
108  if (pdata == NULL) {
109  cvProcessError(cv_mem, CVSPILS_MEM_FAIL, "CVBANDPRE", "CVBandPrecInit", MSGBP_MEM_FAIL);
110  return(CVSPILS_MEM_FAIL);
111  }
112 
113  /* Load pointers and bandwidths into pdata block. */
114  pdata->cvode_mem = cvode_mem;
115  pdata->N = N;
116  pdata->mu = mup = SUNMIN(N-1, SUNMAX(0,mu));
117  pdata->ml = mlp = SUNMIN(N-1, SUNMAX(0,ml));
118 
119  /* Initialize nfeBP counter */
120  pdata->nfeBP = 0;
121 
122  /* Allocate memory for saved banded Jacobian approximation. */
123  pdata->savedJ = NULL;
124  pdata->savedJ = NewBandMat(N, mup, mlp, mup);

126  free(pdata); pdata = NULL;

128  return(CVSPILS_MEM_FAIL);
129  }
130

132  storagemu = SUNMIN(N-1, mup+mlp);
133  pdata->savedP = NULL;

152

241 /*
242  * -----------------------------------------------------------------
243  * CVBandPrecSetup
244  * -----------------------------------------------------------------
245  * Together CVBandPrecSetup and CVBandPrecSolve use a banded
246  * difference quotient Jacobian to create a preconditioner.
247  * CVBandPrecSetup calculates a new J, if necessary, then
248  * calculates P = I - gamma*J, and does an LU factorization of P.
249  *
250  * The parameters of CVBandPrecSetup are as follows:
251  *
252  * t is the current value of the independent variable.
253  *
254  * y is the current value of the dependent variable vector,
255  * namely the predicted value of y(t).
256  *
257  * fy is the vector f(t,y).
258  *
259  * jok is an input flag indicating whether Jacobian-related
260  * data needs to be recomputed, as follows:
261  * jok == FALSE means recompute Jacobian-related data
262  * from scratch.
263  * jok == TRUE means that Jacobian data from the
264  * previous PrecSetup call will be reused
265  * (with the current value of gamma).
266  * A CVBandPrecSetup call with jok == TRUE should only
267  * occur after a call with jok == FALSE.
268  *
269  * *jcurPtr is a pointer to an output integer flag which is
270  * set by CVBandPrecond as follows:
271  * *jcurPtr = TRUE if Jacobian data was recomputed.
272  * *jcurPtr = FALSE if Jacobian data was not recomputed,
273  * but saved data was reused.
274  *
275  * gamma is the scalar appearing in the Newton matrix.
276  *
277  * bp_data is a pointer to preconditoner data (set by CVBandPrecInit)
278  *
279  * tmp1, tmp2, and tmp3 are pointers to memory allocated
280  * for vectors of length N for work space. This
281  * routine uses only tmp1 and tmp2.
282  *
283  * The value to be returned by the CVBandPrecSetup function is
284  * 0 if successful, or
285  * 1 if the band factorization failed.
286  * -----------------------------------------------------------------
287  */
288 
289 static int CVBandPrecSetup(realtype t, N_Vector y, N_Vector fy,
290  booleantype jok, booleantype *jcurPtr,
291  realtype gamma, void *bp_data,
292  N_Vector tmp1, N_Vector tmp2, N_Vector tmp3)
293 {
294  CVBandPrecData pdata;
295  CVodeMem cv_mem;
296  int retval;
297  long int ier;
298 
299  /* Assume matrix and lpivots have already been allocated. */
300  pdata = (CVBandPrecData) bp_data;
301 
302  cv_mem = (CVodeMem) pdata->cvode_mem;
303 
304  if (jok) {
305 
306  /* If jok = TRUE, use saved copy of J. */
307  *jcurPtr = FALSE;
308  BandCopy(savedJ, savedP, mu, ml);
309 
310  } else {
311 
312  /* If jok = FALSE, call CVBandPDQJac for new J value. */
313  *jcurPtr = TRUE;
314  SetToZero(savedJ);
315 
316  retval = CVBandPDQJac(pdata, t, y, fy, tmp1, tmp2);
317  if (retval < 0) {
318  cvProcessError(cv_mem, -1, "CVBANDPRE", "CVBandPrecSetup", MSGBP_RHSFUNC_FAILED);
319  return(-1);
320  }
321  if (retval > 0) {
322  return(1);
323  }
324 
325  BandCopy(savedJ, savedP, mu, ml);
326 
327  }
328 
329  /* Scale and add I to get savedP = I - gamma*J. */
330  BandScale(-gamma, savedP);
331  AddIdentity(savedP);
332 
333  /* Do LU factorization of matrix. */
334  ier = BandGBTRF(savedP, lpivots);
335 
336  /* Return 0 if the LU was complete; otherwise return 1. */
337  if (ier > 0) return(1);
338  return(0);
339 }

341 /*
342  * -----------------------------------------------------------------
343  * CVBandPrecSolve
344  * -----------------------------------------------------------------
345  * CVBandPrecSolve solves a linear system P z = r, where P is the
346  * matrix computed by CVBandPrecond.
347  *
348  * The parameters of CVBandPrecSolve used here are as follows:
349  *
350  * r is the right-hand side vector of the linear system.
351  *
352  * bp_data is a pointer to preconditoner data (set by CVBandPrecInit)
353  *
354  * z is the output vector computed by CVBandPrecSolve.
355  *
356  * The value returned by the CVBandPrecSolve function is always 0,
357  * indicating success.
358  * -----------------------------------------------------------------
359  */
360 
361 static int CVBandPrecSolve(realtype t, N_Vector y, N_Vector fy,
362  N_Vector r, N_Vector z,
363  realtype gamma, realtype delta,
364  int lr, void *bp_data, N_Vector tmp)
365 {
366  CVBandPrecData pdata;
367  realtype *zd;
368 
369  /* Assume matrix and lpivots have already been allocated. */
370  pdata = (CVBandPrecData) bp_data;
371 
372  /* Copy r to z. */
373  N_VScale(ONE, r, z);
374 
375  /* Do band backsolve on the vector z. */
376  zd = N_VGetArrayPointer(z);
377

409 /*
410  * -----------------------------------------------------------------
411  * CVBandPDQJac
412  * -----------------------------------------------------------------
413  * This routine generates a banded difference quotient approximation to
414  * the Jacobian of f(t,y). It assumes that a band matrix of type
415  * DlsMat is stored column-wise, and that elements within each column
416  * are contiguous. This makes it possible to get the address of a column
417  * of J via the macro BAND_COL and to write a simple for loop to set
418  * each of the elements of a column in succession.
419  * -----------------------------------------------------------------
420  */
421 
422 static int CVBandPDQJac(CVBandPrecData pdata,
423  realtype t, N_Vector y, N_Vector fy,
424  N_Vector ftemp, N_Vector ytemp)
425 {
426  CVodeMem cv_mem;
427  realtype fnorm, minInc, inc, inc_inv, srur;
428  long int group, i, j, width, ngroups, i1, i2;
429  realtype *col_j, *ewt_data, *fy_data, *ftemp_data, *y_data, *ytemp_data;
430  int retval;
431 
432  cv_mem = (CVodeMem) pdata->cvode_mem;
433 
434  /* Obtain pointers to the data for ewt, fy, ftemp, y, ytemp. */
435  ewt_data = N_VGetArrayPointer(ewt);
436  fy_data = N_VGetArrayPointer(fy);
437  ftemp_data = N_VGetArrayPointer(ftemp);
438  y_data = N_VGetArrayPointer(y);
439  ytemp_data = N_VGetArrayPointer(ytemp);
440 
441  /* Load ytemp with y = predicted y vector. */
442  N_VScale(ONE, y, ytemp);
443 
444  /* Set minimum increment based on uround and norm of f. */
445  srur = SUNRsqrt(uround);
446  fnorm = N_VWrmsNorm(fy, ewt);
447  minInc = (fnorm != ZERO) ?
448  (MIN_INC_MULT * SUNRabs(h) * uround * N * fnorm) : ONE;
449 
450  /* Set bandwidth and number of column groups for band differencing. */
451  width = ml + mu + 1;
452  ngroups = SUNMIN(width, N);
453 
454  for (group = 1; group <= ngroups; group++) {
455 
456  /* Increment all y_j in group. */
457  for(j = group-1; j < N; j += width) {
458  inc = SUNMAX(srur*SUNRabs(y_data[j]), minInc/ewt_data[j]);
459  ytemp_data[j] += inc;
460  }
461 
462  /* Evaluate f with incremented y. */
463 
464  retval = f(t, ytemp, ftemp, user_data);
465  nfeBP++;
466  if (retval != 0) return(retval);
467 
468  /* Restore ytemp, then form and load difference quotients. */
469  for (j = group-1; j < N; j += width) {
470  ytemp_data[j] = y_data[j];
471  col_j = BAND_COL(savedJ,j);
472  inc = SUNMAX(srur*SUNRabs(y_data[j]), minInc/ewt_data[j]);
473  inc_inv = ONE/inc;
474  i1 = SUNMAX(0, j-mu);
475  i2 = SUNMIN(j+ml, N-1);
476  for (i=i1; i <= i2; i++)
477  BAND_COL_ELEM(col_j,i,j) =
478  inc_inv * (ftemp_data[i] - fy_data[i]);
479  }
480  }
481 
482  return(0);
483 }
