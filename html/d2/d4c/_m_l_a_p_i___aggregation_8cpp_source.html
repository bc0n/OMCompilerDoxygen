224 int GetGlobalAggregates(Epetra_RowMatrix& A, Teuchos::ParameterList& List,
225  double* thisns, Epetra_IntVector& aggrinfo)
226 {
227  int naggregates = GetAggregates(A,List,thisns,aggrinfo);
228  const Epetra_Comm& comm = A.Comm();
229  std::vector<int> local(comm.NumProc());
230  std::vector<int> global(comm.NumProc());
231  for (int i=0; i<comm.NumProc(); ++i) local[i] = 0;
232  local[comm.MyPID()] = naggregates;
233  comm.SumAll(&local[0],&global[0],comm.NumProc());
234  int offset = 0;
235  for (int i=0; i<comm.MyPID(); ++i) offset += global[i];
236  for (int i=0; i<aggrinfo.MyLength(); ++i)
237  if (aggrinfo[i]<naggregates) aggrinfo[i] += offset;
238  else aggrinfo[i] = -1;
239  return naggregates;
240 }

243 void GetPtent(const Epetra_RowMatrix& A, Teuchos::ParameterList& List,
244  double* thisns, Teuchos::RCP<Epetra_CrsMatrix>& Ptent,
245  Teuchos::RCP<Epetra_MultiVector>& NextNS, const int domainoffset)
246 {
247  const int nsdim = List.get<int>("null space: dimension",-1);
248  if (nsdim<=0) ML_THROW("null space dimension not given",-1);
249  const Epetra_Map& rowmap = A.RowMatrixRowMap();
250  const int mylength = rowmap.NumMyElements();
251 
252  // wrap nullspace into something more handy
253  Epetra_MultiVector ns(View,rowmap,thisns,mylength,nsdim);
254 
255  // vector to hold aggregation information
256  Epetra_IntVector aggs(rowmap,false);
257 
258  // aggregation with global aggregate numbering
259  int naggregates = GetGlobalAggregates(const_cast<Epetra_RowMatrix&>(A),List,thisns,aggs);
260 
261  // build a domain map for Ptent
262  // find first aggregate on proc
263  int firstagg = -1;
264  int offset = -1;
265  for (int i=0; i<mylength; ++i)
266  if (aggs[i]>=0)
267  {
268  offset = firstagg = aggs[i];
269  break;
270  }
271  offset *= nsdim;
272  if (offset<0) ML_THROW("could not find any aggregate on proc",-2);
273  std::vector<int> coarsegids(naggregates*nsdim);
274  for (int i=0; i<naggregates; ++i)
275  for (int j=0; j<nsdim; ++j)
276  {
277  coarsegids[i*nsdim+j] = offset + domainoffset;
278  ++offset;
279  }
280  Epetra_Map pdomainmap(-1,naggregates*nsdim,&coarsegids[0],0,A.Comm());
281 
282  // loop aggregates and build ids for dofs
283  std::map<int,std::vector<int> > aggdofs;
284  std::map<int,std::vector<int> >::iterator fool;
285  for (int i=0; i<naggregates; ++i)
286  {
287  std::vector<int> gids(0);
288  aggdofs.insert(std::pair<int,std::vector<int> >(firstagg+i,gids));
289  }
290  for (int i=0; i<mylength; ++i)
291  {
292  if (aggs[i]<0) continue;
293  std::vector<int>& gids = aggdofs[aggs[i]];
294  gids.push_back(aggs.Map().GID(i));
295  }
296 
297 #if 0 // debugging output
298  for (int proc=0; proc<A.Comm().NumProc(); ++proc)
299  {
300  if (proc==A.Comm().MyPID())
301  {
302  for (fool=aggdofs.begin(); fool!=aggdofs.end(); ++fool)
303  {
304  std::cout << "Proc " << proc << " Aggregate " << fool->first << " Dofs ";
305  std::vector<int>& gids = fool->second;
306  for (int i=0; i<(int)gids.size(); ++i) std::cout << gids[i] << " ";
307  std::cout << std::endl;
308  }
309  }
310  fflush(stdout);
311  A.Comm().Barrier();
312  }
313 #endif
314 
315  // coarse level nullspace to be filled
316  NextNS = Teuchos::rcp(new Epetra_MultiVector(pdomainmap,nsdim,true));
317  Epetra_MultiVector& nextns = *NextNS;
318 
319  // matrix Ptent
320  Ptent = Teuchos::rcp(new Epetra_CrsMatrix(Copy,rowmap,nsdim));
321 
322  // loop aggregates and extract the appropriate slices of the nullspace.
323  // do QR and assemble Q and R to Ptent and NextNS
324  for (fool=aggdofs.begin(); fool!=aggdofs.end(); ++fool)
325  {
326  // extract aggregate-local junk of nullspace
327  const int aggsize = (int)fool->second.size();
328  Epetra_SerialDenseMatrix Bagg(aggsize,nsdim);
329  for (int i=0; i<aggsize; ++i)
330  for (int j=0; j<nsdim; ++j)
331  Bagg(i,j) = (*ns(j))[ns.Map().LID(fool->second[i])];
332 
333  // Bagg = Q*R
334  int m = Bagg.M();
335  int n = Bagg.N();
336  int lwork = n*10;
337  int info = 0;
338  int k = std::min(m,n);
339  if (k!=n) ML_THROW("Aggregate too small, fatal!",-1);
340  std::vector<double> work(lwork);
341  std::vector<double> tau(k);
342  Epetra_LAPACK lapack;
343  lapack.GEQRF(m,n,Bagg.A(),m,&tau[0],&work[0],lwork,&info);
344  if (info) ML_THROW("Lapack dgeqrf returned nonzero",info);
345  if (work[0]>lwork)
346  {
347  lwork = (int)work[0];
348  work.resize(lwork);
349  }
350 
351  // get R (stored on Bagg) and assemble it into nextns
352  int agg_cgid = fool->first*nsdim;
353  if (!nextns.Map().MyGID(agg_cgid+domainoffset))
354  ML_THROW("Missing coarse column id on this proc",-1);
355  for (int i=0; i<n; ++i)
356  for (int j=i; j<n; ++j)
357  (*nextns(j))[nextns.Map().LID(domainoffset+agg_cgid+i)] = Bagg(i,j);
358 
359  // get Q and assemble it into Ptent
360  lapack.ORGQR(m,n,k,Bagg.A(),m,&tau[0],&work[0],lwork,&info);
361  if (info) ML_THROW("Lapack dorgqr returned nonzero",info);
362  for (int i=0; i<aggsize; ++i)
363  {
364  const int actgrow = fool->second[i];
365  for (int j=0; j<nsdim; ++j)
366  {
367  int actgcol = fool->first*nsdim+j+domainoffset;
368  int errone = Ptent->SumIntoGlobalValues(actgrow,1,&Bagg(i,j),&actgcol);
369  if (errone>0)
370  {
371  int errtwo = Ptent->InsertGlobalValues(actgrow,1,&Bagg(i,j),&actgcol);
372  if (errtwo<0) ML_THROW("Epetra_CrsMatrix::InsertGlobalValues returned negative nonzero",errtwo);
373  }
374  else if (errone) ML_THROW("Epetra_CrsMatrix::SumIntoGlobalValues returned negative nonzero",errone);
375  }
376  } // for (int i=0; i<aggsize; ++i)
377  } // for (fool=aggdofs.begin(); fool!=aggdofs.end(); ++fool)
378  int err = Ptent->FillComplete(pdomainmap,rowmap);
379  if (err) ML_THROW("Epetra_CrsMatrix::FillComplete returned nonzero",err);
380  err = Ptent->OptimizeStorage();
381  if (err) ML_THROW("Epetra_CrsMatrix::OptimizeStorage returned nonzero",err);
382 
383  return;
384 }
