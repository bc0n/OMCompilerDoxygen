85 int ML_GridAGX_Get_Element(ML_GridAGX *grid,int index,
86  ML_ElementAGX *elmnt)
87 {
88  int i, *nlist, ind, ncnt, nd;
89  double x, y, z;
90 
91  if ( grid->ML_id != ML_ID_GRIDAGX )
92  {
93  printf("ML_GridAGX_Get_Element : wrong object. \n");
94  exit(1);
95  }
96  if (index < 0 || index >= grid->Nelements)
97  {
98  printf("ML_GridAGX_Get_Element : access error.\n");
99  exit(-1);
100  }
101 
102  ML_ElementAGX_Reuse( elmnt );
103  ML_memory_alloc( (void **) &nlist, 30 * sizeof(int), "GD1");
104  ML_IntList_Get_Sublist( grid->ele_nodes, index, &ncnt, nlist );
105  if ( ncnt > 30 )
106  {
107  printf("Warning : Int_lists - sublist length > 30.\n");
108  exit(0);
109  }
110  nd = ML_GridAGX_Get_Dimension( grid );
111  for ( i = 0; i < ncnt; i++)
112  {
113  ind = nlist[i];
114  x = grid->x[ind];
115  y = grid->y[ind];
116  if ( nd > 2 ) z = grid->z[ind]; else z = 0.0;
117  ML_ElementAGX_Load_VertCoordinate( elmnt, ind, x, y, z );
118  }
119  ML_memory_free( (void **) &nlist);
120  return 0;
121 }
