32 /*
33  * -----------------------------------------------------------------
34  * Type : SlsMat
35  * -----------------------------------------------------------------
36  * The type SlsMat is defined to be a pointer to a structure
37  * with various sizes, a data field, and arrays for the row and
38  * column information for the sparse matrix entries.
39  * The M and N fields indicates the number
40  * of rows and columns, respectively. The data field is a one
41  * dimensional array used for component storage. The NNZ field indicates
42  * the number of nonzero entries in the matrix. The integer array, asub,
43  * holds the row index for each of the matrix entries. The integer
44  * array, xa, holds the index entry for the starting value of each column.
45  * -----------------------------------------------------------------
46  * The relevant fields in DlsMat are:
47  * M - number of rows
48  * N - number of columns
49  * NNZ - the number of nonzero entries in the matrix
50  * data - pointer to a contiguous block of realtype variables
51  * rowvals - row indices of each nonzero entry
52  * colptrs - starting index of the first entry in data in each column
53  *
54  * The nonzero entries of the matrix are stored in
55  * compressed column format. Row indices of entries in
56  * column j are stored in rowvals[colptrs[j]] through rowvals[colptrs[j+i]-1]
57  * and corresponding numerical values of the matrix are stored
58  * in the same entries of data.
59  * -----------------------------------------------------------------
60  */
61 
62 typedef struct _SlsMat {
63  int M;
64  int N;
65  int NNZ;
66  realtype *data;
67  int *rowvals;
68  int *colptrs;
69 } *SlsMat;
