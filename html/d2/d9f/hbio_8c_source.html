223 int readHB_info(const char* filename, int* M, int* N, int* nz, char** Type,
224  int* Nrhs)
225 {
226 /****************************************************************************/
227 /* The readHB_info function opens and reads the header information from */
228 /* the specified Harwell-Boeing file, and reports back the number of rows */
229 /* and columns in the stored matrix (M and N), the number of nonzeros in */
230 /* the matrix (nz), and the number of right-hand-sides stored along with */
231 /* the matrix (Nrhs). */
232 /* */
233 /* For a description of the Harwell Boeing standard, see: */
234 /* Duff, et al., ACM TOMS Vol.15, No.1, March 1989 */
235 /* */
236 /* ---------- */
237 /* **CAVEAT** */
238 /* ---------- */
239 /* ** If the input file does not adhere to the H/B format, the ** */
240 /* ** results will be unpredictable. ** */
241 /* */
242 /****************************************************************************/
243  FILE *in_file;
244  int Ptrcrd, Indcrd, Valcrd, Rhscrd;
245  int Nrow, Ncol, Nnzero, Nrhsix;
246  char *mat_type;
247  char Title[73], Key[9], Rhstype[4];
248  char Ptrfmt[17], Indfmt[17], Valfmt[21], Rhsfmt[21];
249 
250  mat_type = (char *) malloc(4);
251  if ( mat_type == NULL ) IOHBTerminate("Insufficient memory for mat_typen");
252 
253  if ( (in_file = fopen( filename, "r")) == NULL ) {
254  fprintf(stderr,"Error: Cannot open file: %s\n",filename);
255  return 0;
256  }
257 
258  readHB_header(in_file, Title, Key, mat_type, &Nrow, &Ncol, &Nnzero,
259  Nrhs, &Nrhsix,
260  Ptrfmt, Indfmt, Valfmt, Rhsfmt,
261  &Ptrcrd, &Indcrd, &Valcrd, &Rhscrd, Rhstype);
262  fclose(in_file);
263  *Type = mat_type;
264  *(*Type+3) = (char) NULL;
265  *M = Nrow;
266  *N = Ncol;
267  *nz = Nnzero;
268  if (Rhscrd == 0) {*Nrhs = 0;}
269 
270 /* In verbose mode, print some of the header information: */
271 /*
272  if (verbose == 1)
273  {
274  printf("Reading from Harwell-Boeing file %s (verbose on)...\n",filename);
275  printf(" Title: %s\n",Title);
276  printf(" Key: %s\n",Key);
277  printf(" The stored matrix is %i by %i with %i nonzeros.\n",
278  *M, *N, *nz );
279  printf(" %i right-hand--side(s) stored.\n",*Nrhs);
280  }
281 */
282 
283  return 1;
284 
285 }

289 int readHB_header(FILE* in_file, char* Title, char* Key, char* Type,
290  int* Nrow, int* Ncol, int* Nnzero, int* Nrhs, int* Nrhsix,
291  char* Ptrfmt, char* Indfmt, char* Valfmt, char* Rhsfmt,
292  int* Ptrcrd, int* Indcrd, int* Valcrd, int* Rhscrd,
293  char *Rhstype)
294 {
295 /*************************************************************************/
296 /* Read header information from the named H/B file... */
297 /*************************************************************************/
298  int Totcrd,Neltvl;
299  char line[BUFSIZ];
300 
301 /* First line: */
302  fgets(line, BUFSIZ, in_file);
303  if ( sscanf(line,"%*s") < 0 )
304  IOHBTerminate("iohb.c: Null (or blank) first line of HB file.\n");
305  (void) sscanf(line, "%72c%8[^\n]", Title, Key);
306  *(Key+8) = (char) NULL;
307  *(Title+72) = (char) NULL;
308 
309 /* Second line: */
310  fgets(line, BUFSIZ, in_file);
311  if ( sscanf(line,"%*s") < 0 )
312  IOHBTerminate("iohb.c: Null (or blank) second line of HB file.\n");
313  if ( sscanf(line,"%i",&Totcrd) != 1) Totcrd = 0;
314  if ( sscanf(line,"%*i%i",Ptrcrd) != 1) *Ptrcrd = 0;
315  if ( sscanf(line,"%*i%*i%i",Indcrd) != 1) *Indcrd = 0;
316  if ( sscanf(line,"%*i%*i%*i%i",Valcrd) != 1) *Valcrd = 0;
317  if ( sscanf(line,"%*i%*i%*i%*i%i",Rhscrd) != 1) *Rhscrd = 0;
318 
319 /* Third line: */
320  fgets(line, BUFSIZ, in_file);
321  if ( sscanf(line,"%*s") < 0 )
322  IOHBTerminate("iohb.c: Null (or blank) third line of HB file.\n");
323  if ( sscanf(line, "%3c", Type) != 1)
324  IOHBTerminate("iohb.c: Invalid Type info, line 3 of Harwell-Boeing file.\n");
325  upcase(Type);
326  if ( sscanf(line,"%*3c%i",Nrow) != 1) *Nrow = 0 ;
327  if ( sscanf(line,"%*3c%*i%i",Ncol) != 1) *Ncol = 0 ;
328  if ( sscanf(line,"%*3c%*i%*i%i",Nnzero) != 1) *Nnzero = 0 ;
329  if ( sscanf(line,"%*3c%*i%*i%*i%i",&Neltvl) != 1) Neltvl = 0 ;
330 
331 /* Fourth line: */
332  fgets(line, BUFSIZ, in_file);
333  if ( sscanf(line,"%*s") < 0 )
334  IOHBTerminate("iohb.c: Null (or blank) fourth line of HB file.\n");
335  if ( sscanf(line, "%16c",Ptrfmt) != 1)
336  IOHBTerminate("iohb.c: Invalid format info, line 4 of Harwell-Boeing file.\n");
337  if ( sscanf(line, "%*16c%16c",Indfmt) != 1)
338  IOHBTerminate("iohb.c: Invalid format info, line 4 of Harwell-Boeing file.\n");
339  if ( sscanf(line, "%*16c%*16c%20c",Valfmt) != 1)
340  IOHBTerminate("iohb.c: Invalid format info, line 4 of Harwell-Boeing file.\n");
341  sscanf(line, "%*16c%*16c%*20c%20c",Rhsfmt);
342  *(Ptrfmt+16) = (char) NULL;
343  *(Indfmt+16) = (char) NULL;
344  *(Valfmt+20) = (char) NULL;
345  *(Rhsfmt+20) = (char) NULL;
346 
347 /* (Optional) Fifth line: */
348  if (*Rhscrd != 0 )
349  {
350  fgets(line, BUFSIZ, in_file);
351  if ( sscanf(line,"%*s") < 0 )
352  IOHBTerminate("iohb.c: Null (or blank) fifth line of HB file.\n");
353  if ( sscanf(line, "%3c", Rhstype) != 1)
354  IOHBTerminate("iohb.c: Invalid RHS type information, line 5 of Harwell-Boeing file.\n");
355  if ( sscanf(line, "%*3c%i", Nrhs) != 1) *Nrhs = 0;
356  if ( sscanf(line, "%*3c%*i%i", Nrhsix) != 1) *Nrhsix = 0;
357  }
358  return 1;
359 }

362 int readHB_mat_double(const char* filename, int colptr[], int rowind[],
363  double val[])
364 {
365 /****************************************************************************/
366 /* This function opens and reads the specified file, interpreting its */
367 /* contents as a sparse matrix stored in the Harwell/Boeing standard */
368 /* format and creating compressed column storage scheme vectors to hold */
369 /* the index and nonzero value information. */
370 /* */
371 /* ---------- */
372 /* **CAVEAT** */
373 /* ---------- */
374 /* Parsing real formats from Fortran is tricky, and this file reader */
375 /* does not claim to be foolproof. It has been tested for cases when */
376 /* the real values are printed consistently and evenly spaced on each */
377 /* line, with Fixed (F), and Exponential (E or D) formats. */
378 /* */
379 /* ** If the input file does not adhere to the H/B format, the ** */
380 /* ** results will be unpredictable. ** */
381 /* */
382 /****************************************************************************/
383  FILE *in_file;
384  int i,j,ind,col,offset,count,last,Nrhs,Nrhsix;
385  int Ptrcrd, Indcrd, Valcrd, Rhscrd;
386  int Nrow, Ncol, Nnzero, Nentries;
387  int Ptrperline, Ptrwidth, Indperline, Indwidth;
388  int Valperline, Valwidth, Valprec;
389  int Valflag; /* Indicates 'E','D', or 'F' float format */
390  char* ThisElement;
391  char Title[73], Key[9], Type[4], Rhstype[4];
392  char Ptrfmt[17], Indfmt[17], Valfmt[21], Rhsfmt[21];
393  char line[BUFSIZ];
394 
395  if ( (in_file = fopen( filename, "r")) == NULL ) {
396  fprintf(stderr,"Error: Cannot open file: %s\n",filename);
397  return 0;
398  }
399 
400  readHB_header(in_file, Title, Key, Type, &Nrow, &Ncol, &Nnzero,
401  &Nrhs, &Nrhsix,
402  Ptrfmt, Indfmt, Valfmt, Rhsfmt,
403  &Ptrcrd, &Indcrd, &Valcrd, &Rhscrd, Rhstype);
404 
405 /* Parse the array input formats from Line 3 of HB file */
406  ParseIfmt(Ptrfmt,&Ptrperline,&Ptrwidth);
407  ParseIfmt(Indfmt,&Indperline,&Indwidth);
408  if ( Type[0] != 'P' ) { /* Skip if pattern only */
409  ParseRfmt(Valfmt,&Valperline,&Valwidth,&Valprec,&Valflag);
410  }
411 
412 /* Read column pointer array: */
413 
414  offset = 1-_SP_base; /* if base 0 storage is declared (via macro definition), */
415  /* then storage entries are offset by 1 */
416 
417  ThisElement = (char *) malloc(Ptrwidth+1);
418  if ( ThisElement == NULL ) IOHBTerminate("Insufficient memory for ThisElement.");
419  *(ThisElement+Ptrwidth) = (char) NULL;
420  count=0;
421  for (i=0;i<Ptrcrd;i++)
422  {
423  fgets(line, BUFSIZ, in_file);
424  if ( sscanf(line,"%*s") < 0 )
425  IOHBTerminate("iohb.c: Null (or blank) line in pointer data region of HB file.\n");
426  col = 0;
427  for (ind = 0;ind<Ptrperline;ind++)
428  {
429  if (count > Ncol) break;
430  strncpy(ThisElement,line+col,Ptrwidth);
431  /* ThisElement = substr(line,col,Ptrwidth); */
432  colptr[count] = atoi(ThisElement)-offset;
433  count++; col += Ptrwidth;
434  }
435  }
436  free(ThisElement);
437 
438 /* Read row index array: */
439 
440  ThisElement = (char *) malloc(Indwidth+1);
441  if ( ThisElement == NULL ) IOHBTerminate("Insufficient memory for ThisElement.");
442  *(ThisElement+Indwidth) = (char) NULL;
443  count = 0;
444  for (i=0;i<Indcrd;i++)
445  {
446  fgets(line, BUFSIZ, in_file);
447  if ( sscanf(line,"%*s") < 0 )
448  IOHBTerminate("iohb.c: Null (or blank) line in index data region of HB file.\n");
449  col = 0;
450  for (ind = 0;ind<Indperline;ind++)
451  {
452  if (count == Nnzero) break;
453  strncpy(ThisElement,line+col,Indwidth);
454 /* ThisElement = substr(line,col,Indwidth); */
455  rowind[count] = atoi(ThisElement)-offset;
456  count++; col += Indwidth;
457  }
458  }
459  free(ThisElement);
460 
461 /* Read array of values: */
462 
463  if ( Type[0] != 'P' ) { /* Skip if pattern only */
464 
465  if ( Type[0] == 'C' ) Nentries = 2*Nnzero;
466  else Nentries = Nnzero;
467 
468  ThisElement = (char *) malloc(Valwidth+1);
469  if ( ThisElement == NULL )
470  IOHBTerminate("Insufficient memory for ThisElement.");
471  *(ThisElement+Valwidth) = (char) NULL;
472  count = 0;
473  for (i=0;i<Valcrd;i++)
474  {
475  fgets(line, BUFSIZ, in_file);
476  if ( sscanf(line,"%*s") < 0 )
477  IOHBTerminate("iohb.c: Null (or blank) line in value data region of HB file.\n");
478  if (Valflag == 'D') {
479  while( strchr(line,'D') ) *strchr(line,'D') = 'E';
480 /* *strchr(Valfmt,'D') = 'E'; */
481  }
482  col = 0;
483  for (ind = 0;ind<Valperline;ind++)
484  {
485  if (count == Nentries) break;
486  strncpy(ThisElement,line+col,Valwidth);
487  /*ThisElement = substr(line,col,Valwidth);*/
488  if ( Valflag != 'F' && strchr(ThisElement,'E') == NULL ) {
489  /* insert a char prefix for exp */
490  last = strlen(ThisElement);
491  for (j=last+1;j>=0;j--) {
492  ThisElement[j] = ThisElement[j-1];
493  if ( ThisElement[j] == '+' || ThisElement[j] == '-' ) {
494  ThisElement[j-1] = Valflag;
495  break;
496  }
497  }
498  }
499  val[count] = atof(ThisElement);
500  count++; col += Valwidth;
501  }
502  }
503  free(ThisElement);
504  }
505 
506  fclose(in_file);
507  return 1;
508 }

534  if ( *val == NULL ) IOHBTerminate("Insufficient memory for val.\n");
535  }
536  } /* No val[] space needed if pattern only */
