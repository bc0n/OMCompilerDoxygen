236 protected function transposeRow "author: PA
237  Helper function to transposeMatrix2.
238  Input: BackendDAE.AdjacencyMatrix (eqn => var)
239  Input: row number (variable)
240  Input: iterator (start with one)
241  inputs: (int list list, int /* row */,int /* iter */)
242  outputs: int list"
243  input list<Integer> row;
244  input output BackendDAE.AdjacencyMatrixT mt;
245  input output Integer indx;
246 algorithm
247  (mt, indx) := match (row)
248  local
249  Integer i, indx1, iabs;
250  list<Integer> res, col;
251 
252  case {}
253  then (mt, indx+1);
254 
255  case i::res equation
256  iabs = intAbs(i);
257  mt = Array.expand(iabs - arrayLength(mt), mt, {});
258  col = mt[iabs];
259  indx1 = if intLt(i, 0) then -indx else indx;
260  arrayUpdate(mt, iabs, indx1::col);
261  then transposeRow(res, mt, indx);
262  end match;
263 end transposeRow;

265 public function absAdjacencyMatrix "author: PA
266  Applies absolute value to all entries in the incidence matrix.
267  This can be used when e.g. der(x) and x are considered the same variable."
268  input BackendDAE.AdjacencyMatrix m;
269  output BackendDAE.AdjacencyMatrix res;
270 protected
271  list<list<Integer>> lst, lst_1;
272  Integer i = 1;
273  Integer minn;
274 algorithm
275  res := Dangerous.arrayCreateNoInit(arrayLength(m),{});
276  for v in m loop
277  minn := List.fold(v,intMin,0);
278  if minn < 0 then
279  Dangerous.arrayUpdateNoBoundsChecking(res, i, List.map(v,intAbs));
280  else
281  Dangerous.arrayUpdateNoBoundsChecking(res, i, v);
282  end if;
283  i := i+1;
284  end for;
285 end absAdjacencyMatrix;
