29  ElementByElement_SingleElement(Epetra_Comm& Comm,
30  const int NumMyFEs,
31  const int NumVerticesPerFE,
32  int* MyFEs,
33  const int NumPDEEqns,
34  Epetra_SerialDenseMatrix* FEMatrix,
35  const int NumMyBoundaryRows,
36  int* MyBoundaryRows,
37  const double* MyBoundaryValues,
38  const Epetra_Map& GraphMap,
39  const int MaxEntriesPerGraphRow = 0) :
40  NumMyFEs_(NumMyFEs),
41  NumVerticesPerFE_(NumVerticesPerFE),
42  MyFEs_(MyFEs),
43  NumPDEEqns_(NumPDEEqns),
44  FEMatrix_(FEMatrix),
45  NumMyBoundaryRows_(NumMyBoundaryRows),
46  MyBoundaryRows_(MyBoundaryRows),
47  MyBoundaryValues_(MyBoundaryValues),
48  Comm_(Comm),
49  Graph_(0)
50  {
51  // build the graph using GraphMap
52 
53  Graph_ = new Epetra_FECrsGraph(Copy, GraphMap, MaxEntriesPerGraphRow);
54 
55  for (int ie = 0; ie < NumMyFEs; ++ie)
56  {
57  const int* ptr = &(MyFEs_[ie * NumVerticesPerFE]);
58  Graph_->InsertGlobalIndices(NumVerticesPerFE, ptr,
59  NumVerticesPerFE, ptr);
60  }
61 
62  Graph_->GlobalAssemble();
63 
64  // convert MyFEs into local column map ordering
65 
66  for (int ie = 0; ie < NumMyFEs; ++ie)
67  {
68  int* ptr = &(MyFEs_[ie * NumVerticesPerFE]);
69  for (int i = 0; i < NumVerticesPerFE; ++i)
70  {
71  ptr[i] = Graph_->ColMap().LID(ptr[i]);
72  assert (ptr[i] != -1);
73  }
74  }
75 
76  // build the map for the operator, which is the "extended"
77  // version of GraphMap
78 
79  std::vector<int> MyGlobalElements2(Graph_->ColMap().NumMyElements() * NumPDEEqns);
80  int* MyGlobalElements = Graph_->RowMap().MyGlobalElements();
81  int NumMyElements = Graph_->RowMap().NumMyElements();
82 
83  for (int i = 0; i < NumMyElements; ++i)
84  for (int j = 0; j < NumPDEEqns; ++j)
85  MyGlobalElements2[i * NumPDEEqns + j] = MyGlobalElements[i] * NumPDEEqns + j;
86  OperatorMap_ = new Epetra_Map(-1, NumMyElements * NumPDEEqns,
87  &MyGlobalElements2[0], 0, Comm_);
88 
89  // expand the column map as well
90 
91  MyGlobalElements = Graph_->ColMap().MyGlobalElements();
92  NumMyElements = Graph_->ColMap().NumMyElements();
93 
94  for (int i = 0; i < NumMyElements; ++i)
95  for (int j = 0; j < NumPDEEqns; ++j)
96  MyGlobalElements2[i * NumPDEEqns + j] = MyGlobalElements[i] * NumPDEEqns + j;
97  OperatorColMap_ = new Epetra_Map(-1, NumMyElements * NumPDEEqns,
98  &MyGlobalElements2[0], 0, Comm_);
99 
100  ColImporter_ = new Epetra_Import(*OperatorColMap_, *OperatorMap_);
101 
102  // memory allocation for element-by-element multiplication
103 
104  DenseX.Reshape(NumPDEEqns * NumVerticesPerFE, NumMyFEs_);
105  DenseY.Reshape(NumPDEEqns * NumVerticesPerFE, NumMyFEs_);
106 
107  // now arrange the boundary conditions. I make the assumption that
108  // each BC row has been specified on a different processor.
109 
110  MyGlobalElements2.resize(NumMyBoundaryRows_);
111  for (int i = 0; i < NumMyBoundaryRows_; ++i)
112  MyGlobalElements2[i] = MyBoundaryRows_[i];
113 
114  Epetra_Map BoundaryMap(-1, NumMyBoundaryRows_, &MyGlobalElements2[0], 0, Comm_);
115 
116  Epetra_Vector BoundaryVector(BoundaryMap);
117 
118  for (int i = 0; i < NumMyBoundaryRows_; ++i)
119  BoundaryVector[i] = MyBoundaryValues_[i];
120 
121  // now build a map containing the boundaries for ghost nodes only
122  int count = 0;
123  for (int i = 0; i < NumMyBoundaryRows_; ++i)
124  {
125  if (GraphMap.LID(MyBoundaryRows_[i]) == -1)
126  MyGlobalElements2[++count] = MyBoundaryRows_[i];
127  }
128 
129  ColBoundaryMap_ = new Epetra_Map(-1, count, &MyGlobalElements2[0], 0, Comm_);
130 
131  ColBoundaryVector_ = new Epetra_Vector(*ColBoundaryMap_);
132 
133  Epetra_Import Importer(*ColBoundaryMap_, BoundaryMap);
134  ColBoundaryVector_->Import(BoundaryVector, Importer, Insert);
135 
136  ColBoundaryImporter_ = new Epetra_Import(*OperatorColMap_, *ColBoundaryMap_);
137 
138  for (int i = 0; i < NumMyBoundaryRows_; ++i)
139  MyBoundaryRows_[i] = OperatorColMap_->LID(MyBoundaryRows_[i]);
140  }

165  int Apply(const Epetra_MultiVector& X, Epetra_MultiVector& Y) const
166  {
167  assert (X.NumVectors() == 1); // FIXME
168 
169  // import ghost nodes
170  Epetra_MultiVector ColX(*OperatorColMap_, X.NumVectors());
171  ML_CHK_ERR(ColX.Import(X, *ColImporter_, Insert));
172  // fix boundary conditions
173  ML_CHK_ERR(SetMyBoundaryRows(ColX));
174  ML_CHK_ERR(ColX.Import(*ColBoundaryVector_, *ColBoundaryImporter_, Insert));
175 
176  // now redistribute the vector into Dense
177  for (int ie = 0; ie < NumMyFEs_; ++ie)
178  {
179  const int* ptr = &(MyFEs_[ie * NumVerticesPerFE_]);
180  for (int i = 0; i < NumVerticesPerFE_; ++i)
181  {
182  for (int j = 0; j < NumPDEEqns_; ++j)
183  DenseX(i * NumPDEEqns_ + j, ie) = ColX[0][ptr[i] * NumPDEEqns_ + j];
184  }
185  }
186 
187  DenseY.Multiply('N', 'N', 1.0, *FEMatrix_, DenseX, 0.0);
188 
189  ColX.PutScalar(0.0);
190 
191  // put values back into Y
192  for (int ie = 0; ie < NumMyFEs_; ++ie)
193  {
194  const int* ptr = &(MyFEs_[ie * NumVerticesPerFE_]);
195  for (int i = 0; i < NumVerticesPerFE_; ++i)
196  {
197  for (int j = 0; j < NumPDEEqns_; ++j)
198  ColX[0][ptr[i] * NumPDEEqns_ + j] += DenseY(i * NumPDEEqns_ + j, ie);
199  }
200  }
201 
202  Y.PutScalar(0.0);
203  ML_CHK_ERR(Y.Export(ColX, *ColImporter_, Add));
204  ML_CHK_ERR(ResetMyBoundaryRows(Y));
205  return(0);
206  }
