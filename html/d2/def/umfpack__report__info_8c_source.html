27 #define RATIO(a,b,c) (((b) == 0) ? (c) : (((double) a)/((double) b)))

33 PRIVATE void print_ratio
34 (
35  char *what,
36  char *format,
37  double estimate,
38  double actual
39 )
40 {
41  if (estimate < 0 && actual < 0) /* double relop, but ignore Nan case */
42  {
43  return ;
44  }
45  PRINTF ((" %-27s", what)) ;
46  if (estimate >= 0) /* double relop, but ignore Nan case */
47  {
48  PRINTF ((format, estimate)) ;
49  }
50  else
51  {
52  PRINTF ((" -")) ;
53  }
54  if (actual >= 0) /* double relop, but ignore Nan case */
55  {
56  PRINTF ((format, actual)) ;
57  }
58  else
59  {
60  PRINTF ((" -")) ;
61  }
62  if (estimate >= 0 && actual >= 0) /* double relop, but ignore Nan case */
63  {
64  PRINTF ((" %5.0f%%\n", 100 * RATIO (actual, estimate, 1))) ;
65  }
66  else
67  {
68  PRINTF ((" -\n")) ;
69  }
70 }
