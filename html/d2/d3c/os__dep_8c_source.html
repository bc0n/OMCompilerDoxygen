117 /* Repeatedly perform a read call until the buffer is filled or */
118 /* we encounter EOF. */
119 STATIC ssize_t GC_repeat_read(int fd, char *buf, size_t count)
120 {
121 # define READ read
122  size_t num_read = 0;
123 
124  ASSERT_CANCEL_DISABLED();
125  while (num_read < count) {
126  ssize_t result = READ(fd, buf + num_read, count - num_read);
127 
128  if (result < 0) return result;
129  if (result == 0) break;
130  num_read += result;
131  }
132 # undef READ
133  return num_read;
134 }

166 /* Copy the contents of /proc/self/maps to a buffer in our address */
167 /* space. Return the address of the buffer, or zero on failure. */
168 /* This code could be simplified if we could determine its size ahead */
169 /* of time. */
170 GC_INNER char * GC_get_maps(void)
171 {
172  ssize_t result;
173  static char *maps_buf = NULL;
174  static size_t maps_buf_sz = 1;
175  size_t maps_size;
176 # ifdef THREADS
177  size_t old_maps_size = 0;
178 # endif
179 
180  /* The buffer is essentially static, so there must be a single client. */
181  GC_ASSERT(I_HOLD_LOCK());
182 
183  /* Note that in the presence of threads, the maps file can */
184  /* essentially shrink asynchronously and unexpectedly as */
185  /* threads that we already think of as dead release their */
186  /* stacks. And there is no easy way to read the entire */
187  /* file atomically. This is arguably a misfeature of the */
188  /* /proc/.../maps interface. */
189  /* Since we expect the file can grow asynchronously in rare */
190  /* cases, it should suffice to first determine */
191  /* the size (using lseek or read), and then to reread the */
192  /* file. If the size is inconsistent we have to retry. */
193  /* This only matters with threads enabled, and if we use */
194  /* this to locate roots (not the default). */
195 
196 # ifdef THREADS
197  /* Determine the initial size of /proc/self/maps. */
198  /* Note that lseek doesn't work, at least as of 2.6.15. */
199  maps_size = GC_get_maps_len();
200  if (0 == maps_size) return 0;
201 # else
202  maps_size = 4000; /* Guess */
203 # endif
204 
205  /* Read /proc/self/maps, growing maps_buf as necessary. */
206  /* Note that we may not allocate conventionally, and */
207  /* thus can't use stdio. */
208  do {
209  int f;
210 
211  while (maps_size >= maps_buf_sz) {
212  GC_scratch_recycle_no_gww(maps_buf, maps_buf_sz);
213  /* Grow only by powers of 2, since we leak "too small" buffers.*/
214  while (maps_size >= maps_buf_sz) maps_buf_sz *= 2;
215  maps_buf = GC_scratch_alloc(maps_buf_sz);
216 # ifdef THREADS
217  /* Recompute initial length, since we allocated. */
218  /* This can only happen a few times per program */
219  /* execution. */
220  maps_size = GC_get_maps_len();
221  if (0 == maps_size) return 0;
222 # endif
223  if (maps_buf == 0) return 0;
224  }
225  GC_ASSERT(maps_buf_sz >= maps_size + 1);
226  f = open("/proc/self/maps", O_RDONLY);
227  if (-1 == f) return 0;
228 # ifdef THREADS
229  old_maps_size = maps_size;
230 # endif
231  maps_size = 0;
232  do {
233  result = GC_repeat_read(f, maps_buf, maps_buf_sz-1);
234  if (result <= 0)
235  break;
236  maps_size += result;
237  } while ((size_t)result == maps_buf_sz-1);
238  close(f);
239  if (result <= 0)
240  return 0;
241 # ifdef THREADS
242  if (maps_size > old_maps_size) {
243  /* This might be caused by e.g. thread creation. */
244  WARN("Unexpected asynchronous /proc/self/maps growth"
245  " (to %" WARN_PRIdPTR " bytes)\n", maps_size);
246  }
247 # endif
248  } while (maps_size >= maps_buf_sz
249 # ifdef THREADS
250  || maps_size < old_maps_size
251 # endif
252  );
253  maps_buf[maps_size] = '\0';
254  return maps_buf;
255 }

257 /*
258  * GC_parse_map_entry parses an entry from /proc/self/maps so we can
259  * locate all writable data segments that belong to shared libraries.
260  * The format of one of these entries and the fields we care about
261  * is as follows:
262  * XXXXXXXX-XXXXXXXX r-xp 00000000 30:05 260537 name of mapping...\n
263  * ^^^^^^^^ ^^^^^^^^ ^^^^ ^^
264  * start end prot maj_dev
265  *
266  * Note that since about august 2003 kernels, the columns no longer have
267  * fixed offsets on 64-bit kernels. Hence we no longer rely on fixed offsets
268  * anywhere, which is safer anyway.
269  */
270 
271 /* Assign various fields of the first line in buf_ptr to (*start), */
272 /* (*end), (*prot), (*maj_dev) and (*mapping_name). mapping_name may */
273 /* be NULL. (*prot) and (*mapping_name) are assigned pointers into the */
274 /* original buffer. */
275 #if (defined(DYNAMIC_LOADING) && defined(USE_PROC_FOR_LIBRARIES)) \
276  || defined(IA64) || defined(INCLUDE_LINUX_THREAD_DESCR) \
277  || defined(REDIRECT_MALLOC)
278  GC_INNER char *GC_parse_map_entry(char *buf_ptr, ptr_t *start, ptr_t *end,
279  char **prot, unsigned int *maj_dev,
280  char **mapping_name)
281  {
282  unsigned char *start_start, *end_start, *maj_dev_start;
283  unsigned char *p; /* unsigned for isspace, isxdigit */
284 
285  if (buf_ptr == NULL || *buf_ptr == '\0') {
286  return NULL;
287  }
288 
289  p = (unsigned char *)buf_ptr;
290  while (isspace(*p)) ++p;
291  start_start = p;
292  GC_ASSERT(isxdigit(*start_start));
293  *start = (ptr_t)strtoul((char *)start_start, (char **)&p, 16);
294  GC_ASSERT(*p=='-');
295 
296  ++p;
297  end_start = p;
298  GC_ASSERT(isxdigit(*end_start));
299  *end = (ptr_t)strtoul((char *)end_start, (char **)&p, 16);
300  GC_ASSERT(isspace(*p));
301 
302  while (isspace(*p)) ++p;
303  GC_ASSERT(*p == 'r' || *p == '-');
304  *prot = (char *)p;
305  /* Skip past protection field to offset field */
306  while (!isspace(*p)) ++p; while (isspace(*p)) ++p;
307  GC_ASSERT(isxdigit(*p));
308  /* Skip past offset field, which we ignore */
309  while (!isspace(*p)) ++p; while (isspace(*p)) ++p;
310  maj_dev_start = p;
311  GC_ASSERT(isxdigit(*maj_dev_start));
312  *maj_dev = strtoul((char *)maj_dev_start, NULL, 16);
313 
314  if (mapping_name == 0) {
315  while (*p && *p++ != '\n');
316  } else {
317  while (*p && *p != '\n' && *p != '/' && *p != '[') p++;
318  *mapping_name = (char *)p;
319  while (*p && *p++ != '\n');
320  }
321  return (char *)p;
322  }
323 #endif /* REDIRECT_MALLOC || DYNAMIC_LOADING || IA64 || ... */

326  /* Try to read the backing store base from /proc/self/maps. */
327  /* Return the bounds of the writable mapping with a 0 major device, */
328  /* which includes the address passed as data. */
329  /* Return FALSE if there is no such mapping. */
330  GC_INNER GC_bool GC_enclosing_mapping(ptr_t addr, ptr_t *startp,
331  ptr_t *endp)
332  {
333  char *prot;
334  ptr_t my_start, my_end;
335  unsigned int maj_dev;
336  char *maps = GC_get_maps();
337  char *buf_ptr = maps;
338 
339  if (0 == maps) return(FALSE);
340  for (;;) {
341  buf_ptr = GC_parse_map_entry(buf_ptr, &my_start, &my_end,
342  &prot, &maj_dev, 0);
343 
344  if (buf_ptr == NULL) return FALSE;
345  if (prot[1] == 'w' && maj_dev == 0) {
346  if ((word)my_end > (word)addr && (word)my_start <= (word)addr) {
347  *startp = my_start;
348  *endp = my_end;
349  return TRUE;
350  }
351  }
352  }
353  return FALSE;
354  }

550  /* Return the first non-addressable location > p or bound. */
551  /* Requires the allocation lock. */
552  STATIC ptr_t GC_find_limit_openbsd(ptr_t p, ptr_t bound)
553  {
554  static volatile ptr_t result;
555  /* Safer if static, since otherwise it may not be */
556  /* preserved across the longjmp. Can safely be */
557  /* static since it's only called with the */
558  /* allocation lock held. */
559 
560  struct sigaction act;
561  word pgsz = (word)sysconf(_SC_PAGESIZE);
562 
563  GC_ASSERT((word)bound >= pgsz);
564  GC_ASSERT(I_HOLD_LOCK());
565 
566  act.sa_handler = GC_fault_handler_openbsd;
567  sigemptyset(&act.sa_mask);
568  act.sa_flags = SA_NODEFER | SA_RESTART;
569  /* act.sa_restorer is deprecated and should not be initialized. */
570  sigaction(SIGSEGV, &act, &old_segv_act);
571 
572  if (SETJMP(GC_jmp_buf_openbsd) == 0) {
573  result = (ptr_t)((word)p & ~(pgsz-1));
574  for (;;) {
575  if ((word)result >= (word)bound - pgsz) {
576  result = bound;
577  break;
578  }
579  result += pgsz; /* no overflow expected */
580  GC_noop1((word)(*result));
581  }
582  }
583 
584 # ifdef THREADS
585  /* Due to the siglongjump we need to manually unmask SIGPROF. */
586  __syscall(SYS_sigprocmask, SIG_UNBLOCK, sigmask(SIGPROF));
587 # endif
588 
589  sigaction(SIGSEGV, &old_segv_act, 0);
590  return(result);
591  }

594  /* Requires the allocation lock. */
595  STATIC ptr_t GC_skip_hole_openbsd(ptr_t p, ptr_t bound)
596  {

598  static volatile int firstpass;
599 
600  struct sigaction act;
601  word pgsz = (word)sysconf(_SC_PAGESIZE);
602 
603  GC_ASSERT((word)bound >= pgsz);
604  GC_ASSERT(I_HOLD_LOCK());
605 
606  act.sa_handler = GC_fault_handler_openbsd;
607  sigemptyset(&act.sa_mask);
608  act.sa_flags = SA_NODEFER | SA_RESTART;
609  /* act.sa_restorer is deprecated and should not be initialized. */
610  sigaction(SIGSEGV, &act, &old_segv_act);

617  result = bound;
618  } else {

620  GC_noop1((word)(*result));
621  }
622  }

982  /* Return the first non-addressable location > p (up) or */
983  /* the smallest location q s.t. [q,p) is addressable (!up). */
984  /* We assume that p (up) or p-1 (!up) is addressable. */
985  /* Requires allocation lock. */
986  STATIC ptr_t GC_find_limit_with_bound(ptr_t p, GC_bool up, ptr_t bound)
987  {
988  static volatile ptr_t result;
989  /* Safer if static, since otherwise it may not be */
990  /* preserved across the longjmp. Can safely be */
991  /* static since it's only called with the */
992  /* allocation lock held. */
993 
994  GC_ASSERT(up ? (word)bound >= MIN_PAGE_SIZE
995  : (word)bound <= ~(word)MIN_PAGE_SIZE);
996  GC_ASSERT(I_HOLD_LOCK());
997  GC_setup_temporary_fault_handler();
998  if (SETJMP(GC_jmp_buf) == 0) {
999  result = (ptr_t)(((word)(p))
1000  & ~(MIN_PAGE_SIZE-1));
1001  for (;;) {
1002  if (up) {
1003  if ((word)result >= (word)bound - MIN_PAGE_SIZE) {
1004  result = bound;
1005  break;
1006  }
1007  result += MIN_PAGE_SIZE; /* no overflow expected */
1008  } else {
1009  if ((word)result <= (word)bound + MIN_PAGE_SIZE) {
1010  result = bound - MIN_PAGE_SIZE;
1011  /* This is to compensate */
1012  /* further result increment (we */
1013  /* do not modify "up" variable */
1014  /* since it might be clobbered */
1015  /* by setjmp otherwise). */
1016  break;
1017  }
1018  result -= MIN_PAGE_SIZE; /* no underflow expected */
1019  }
1020  GC_noop1((word)(*result));
1021  }
1022  }
1023  GC_reset_fault_handler();
1024  if (!up) {
1025  result += MIN_PAGE_SIZE;
1026  }
1027  return(result);
1028  }

1104  STATIC ptr_t GC_linux_main_stack_base(void)
1105  {
1106  /* We read the stack base value from /proc/self/stat. We do this */
1107  /* using direct I/O system calls in order to avoid calling malloc */
1108  /* in case REDIRECT_MALLOC is defined. */
1109 # ifndef STAT_READ
1110  /* Also defined in pthread_support.c. */
1111 # define STAT_BUF_SIZE 4096
1112 # define STAT_READ read
1113 # endif
1114  /* Should probably call the real read, if read is wrapped. */
1115  char stat_buf[STAT_BUF_SIZE];
1116  int f;
1117  word result;
1118  int i, buf_offset = 0, len;
1119 
1120  /* First try the easy way. This should work for glibc 2.2 */
1121  /* This fails in a prelinked ("prelink" command) executable */
1122  /* since the correct value of __libc_stack_end never */
1123  /* becomes visible to us. The second test works around */
1124  /* this. */
1125 # ifdef USE_LIBC_PRIVATES
1126  if (0 != &__libc_stack_end && 0 != __libc_stack_end ) {
1127 # if defined(IA64)
1128  /* Some versions of glibc set the address 16 bytes too */
1129  /* low while the initialization code is running. */
1130  if (((word)__libc_stack_end & 0xfff) + 0x10 < 0x1000) {
1131  return __libc_stack_end + 0x10;
1132  } /* Otherwise it's not safe to add 16 bytes and we fall */
1133  /* back to using /proc. */
1134 # elif defined(SPARC)
1135  /* Older versions of glibc for 64-bit SPARC do not set this */
1136  /* variable correctly, it gets set to either zero or one. */
1137  if (__libc_stack_end != (ptr_t) (unsigned long)0x1)
1138  return __libc_stack_end;
1139 # else
1140  return __libc_stack_end;
1141 # endif
1142  }
1143 # endif
1144  f = open("/proc/self/stat", O_RDONLY);
1145  if (f < 0)
1146  ABORT("Couldn't read /proc/self/stat");
1147  len = STAT_READ(f, stat_buf, STAT_BUF_SIZE);
1148  close(f);
1149 
1150  /* Skip the required number of fields. This number is hopefully */
1151  /* constant across all Linux implementations. */
1152  for (i = 0; i < STAT_SKIP; ++i) {
1153  while (buf_offset < len && isspace(stat_buf[buf_offset++])) {
1154  /* empty */
1155  }
1156  while (buf_offset < len && !isspace(stat_buf[buf_offset++])) {
1157  /* empty */
1158  }
1159  }
1160  /* Skip spaces. */
1161  while (buf_offset < len && isspace(stat_buf[buf_offset])) {
1162  buf_offset++;
1163  }
1164  /* Find the end of the number and cut the buffer there. */
1165  for (i = 0; buf_offset + i < len; i++) {
1166  if (!isdigit(stat_buf[buf_offset + i])) break;
1167  }
1168  if (buf_offset + i >= len) ABORT("Could not parse /proc/self/stat");
1169  stat_buf[buf_offset + i] = '\0';
1170 
1171  result = (word)STRTOULL(&stat_buf[buf_offset], NULL, 10);
1172  if (result < 0x100000 || (result & (sizeof(word) - 1)) != 0)
1173  ABORT("Absurd stack bottom value");

1175  }

2801  * Routines for accessing dirty bits on virtual pages.
2802  * There are six ways to maintain this information:
2803  * DEFAULT_VDB: A simple dummy implementation that treats every page
2804  * as possibly dirty. This makes incremental collection
2805  * useless, but the implementation is still correct.
2806  * Manual VDB: Stacks and static data are always considered dirty.
2807  * Heap pages are considered dirty if GC_dirty(p) has been
2808  * called on some pointer p pointing to somewhere inside
2809  * an object on that page. A GC_dirty() call on a large
2810  * object directly dirties only a single page, but for the
2811  * manual VDB we are careful to treat an object with a dirty
2812  * page as completely dirty.
2813  * In order to avoid races, an object must be marked dirty
2814  * after it is written, and a reference to the object
2815  * must be kept on a stack or in a register in the interim.
2816  * With threads enabled, an object directly reachable from the
2817  * stack at the time of a collection is treated as dirty.
2818  * In single-threaded mode, it suffices to ensure that no
2819  * collection can take place between the pointer assignment
2820  * and the GC_dirty() call.
2821  * PCR_VDB: Use PPCRs virtual dirty bit facility.
2822  * PROC_VDB: Use the /proc facility for reading dirty bits. Only
2823  * works under some SVR4 variants. Even then, it may be
2824  * too slow to be entirely satisfactory. Requires reading
2825  * dirty bits for entire address space. Implementations tend
2826  * to assume that the client is a (slow) debugger.
2827  * MPROTECT_VDB:Protect pages and then catch the faults to keep track of
2828  * dirtied pages. The implementation (and implementability)
2829  * is highly system dependent. This usually fails when system
2830  * calls write to a protected page. We prevent the read system
2831  * call from doing so. It is the clients responsibility to
2832  * make sure that other system calls are similarly protected
2833  * or write only to the stack.
2834  * GWW_VDB: Use the Win32 GetWriteWatch functions, if available, to
2835  * read dirty bits. In case it is not available (because we
2836  * are running on Windows 95, Windows 2000 or earlier),
2837  * MPROTECT_VDB may be defined as a fallback strategy.
