57 STATIC MYBOOL stallMonitor_creepingObj(lprec *lp)
58 {
59  OBJmonrec *monitor = lp->monitor;
60 
61  if(monitor->countstep > 1) {
62  REAL deltaOF = (monitor->objstep[monitor->currentstep] -
63  monitor->objstep[monitor->startstep]) / monitor->countstep;
64  deltaOF /= MAX(1, (monitor->idxstep[monitor->currentstep] -
65  monitor->idxstep[monitor->startstep]));
66  deltaOF = my_chsign(monitor->isdual, deltaOF);
67  return( (MYBOOL) (deltaOF < monitor->epsvalue) );

69  else

141 STATIC MYBOOL stallMonitor_check(lprec *lp, int rownr, int colnr, int lastnr,
142  MYBOOL minit, MYBOOL approved, MYBOOL *forceoutEQ)
143 {
144  OBJmonrec *monitor = lp->monitor;
145  MYBOOL isStalled, isCreeping, acceptance = TRUE;
146  int altrule,
147 #ifdef Paranoia
148  msglevel = NORMAL;
149 #else
150  msglevel = DETAILED;
151 #endif
152  REAL deltaobj = lp->suminfeas;
153 
154  /* Accept unconditionally if this is the first or second call */
155  monitor->active = FALSE;
156  if(monitor->Icount <= 1) {
157  if(monitor->Icount == 1) {
158  monitor->prevobj = lp->rhs[0];
159  monitor->previnfeas = deltaobj;
160  }
161  monitor->Icount++;
162  return( acceptance );
163  }
164 
165  /* Define progress as primal objective less sum of (primal/dual) infeasibilities */
166  monitor->thisobj = lp->rhs[0];
167  monitor->thisinfeas = deltaobj;
168  if(lp->spx_trace &&
169  (lastnr > 0))
170  report(lp, NORMAL, "%s: Objective at iter %10.0f is " RESULTVALUEMASK " (%4d: %4d %s- %4d)\n",
171  monitor->spxfunc,
172  (double) get_total_iter(lp), monitor->thisobj, rownr, lastnr,
173  my_if(minit == ITERATE_MAJORMAJOR, "<","|"), colnr);
174  monitor->pivrule = get_piv_rule(lp);
175 
176  /* Check if we have a stationary solution at selected tolerance level;
177  allow some difference in case we just refactorized the basis. */
178  deltaobj = my_reldiff(monitor->thisobj, monitor->prevobj);
179  deltaobj = fabs(deltaobj); /* Pre v5.2 version */
180  isStalled = (MYBOOL) (deltaobj < monitor->epsvalue);
181 
182  /* Also require that we have a measure of infeasibility-stalling */
183  if(isStalled) {
184  REAL testvalue, refvalue = monitor->epsvalue;
185 #if 1
186  if(monitor->isdual)
187  refvalue *= 1000*log10(9.0+lp->rows);
188  else
189  refvalue *= 1000*log10(9.0+lp->columns);
190 #else
191  refvalue *= 1000*log10(9.0+lp->sum);
192 #endif
193  testvalue = my_reldiff(monitor->thisinfeas, monitor->previnfeas);
194  isStalled &= (fabs(testvalue) < refvalue);
195 
196  /* Check if we should force "major" pivoting, i.e. no bound flips;
197  this is activated when we see the feasibility deteriorate */
198 /* if(!isStalled && (testvalue > 0) && (TRUE || is_action(lp->anti_degen, ANTIDEGEN_BOUNDFLIP))) */
199 #if !defined _PRICE_NOBOUNDFLIP
200  if(!isStalled && (testvalue > 0) && is_action(lp->anti_degen, ANTIDEGEN_BOUNDFLIP))
201  acceptance = AUTOMATIC;
202  }
203 #else
204  if(!isStalled && (testvalue > 0) && !ISMASKSET(lp->piv_strategy, PRICE_NOBOUNDFLIP)) {
205  SETMASK(lp->piv_strategy, PRICE_NOBOUNDFLIP);
206  acceptance = AUTOMATIC;
207  }
208  }
209  else
210  CLEARMASK(lp->piv_strategy, PRICE_NOBOUNDFLIP);
211 #endif
212 
213 #if 1
214  isCreeping = FALSE;
215 #else
216  isCreeping |= stallMonitor_creepingObj(lp);
217 /* isCreeping |= stallMonitor_shortSteps(lp); */
218 #endif
219  if(isStalled || isCreeping) {
220 
221  /* Update counters along with specific tolerance for bound flips */
222 #if 1
223  if(minit != ITERATE_MAJORMAJOR) {
224  if(++monitor->Mcycle > 2) {
225  monitor->Mcycle = 0;
226  monitor->Ncycle++;
227  }
228  }
229  else
230 #endif
231  monitor->Ncycle++;
232 
233  /* Start to monitor for variable cycling if this is the initial stationarity */
234  if(monitor->Ncycle <= 1) {
235  monitor->Ccycle = colnr;
236  monitor->Rcycle = rownr;
237  }
238 
239  /* Check if we should change pivoting strategy */
240  else if(isCreeping || /* We have OF creep */
241  (monitor->Ncycle > monitor->limitstall[monitor->isdual]) || /* KE empirical value */
242  ((monitor->Ccycle == rownr) && (monitor->Rcycle == colnr))) { /* Obvious cycling */
243 
244  monitor->active = TRUE;
245 
246  /* Try to force out equality slacks to combat degeneracy */
247  if((lp->fixedvars > 0) && (*forceoutEQ != TRUE)) {
248  *forceoutEQ = TRUE;
249  goto Proceed;
250  }
251 
252  /* Our options are now to select an alternative rule or to do bound perturbation;
253  check if these options are available to us or if we must signal failure and break out. */
254  approved &= monitor->pivdynamic && (monitor->ruleswitches < monitor->limitruleswitches);
255  if(!approved && !is_anti_degen(lp, ANTIDEGEN_STALLING)) {
256  lp->spx_status = DEGENERATE;
257  report(lp, msglevel, "%s: Stalling at iter %10.0f; no alternative strategy left.\n",
258  monitor->spxfunc, (double) get_total_iter(lp));
259  acceptance = FALSE;
260  return( acceptance );
261  }
262 
263  /* See if we can do the appropriate alternative rule. */
264  switch (monitor->oldpivrule) {
265  case PRICER_FIRSTINDEX: altrule = PRICER_DEVEX;
266  break;
267  case PRICER_DANTZIG: altrule = PRICER_DEVEX;
268  break;
269  case PRICER_DEVEX: altrule = PRICER_STEEPESTEDGE;
270  break;
271  case PRICER_STEEPESTEDGE: altrule = PRICER_DEVEX;
272  break;
273  default: altrule = PRICER_FIRSTINDEX;
274  }
275  if(approved &&
276  (monitor->pivrule != altrule) && (monitor->pivrule == monitor->oldpivrule)) {
277 
278  /* Switch rule to combat degeneracy. */
279  monitor->ruleswitches++;
280  lp->piv_strategy = altrule;
281  monitor->Ccycle = 0;
282  monitor->Rcycle = 0;
283  monitor->Ncycle = 0;
284  monitor->Mcycle = 0;
285  report(lp, msglevel, "%s: Stalling at iter %10.0f; changed to '%s' rule.\n",
286  monitor->spxfunc, (double) get_total_iter(lp),
287  get_str_piv_rule(get_piv_rule(lp)));
288  if((altrule == PRICER_DEVEX) || (altrule == PRICER_STEEPESTEDGE))
289  restartPricer(lp, AUTOMATIC);
290  }
291 
292  /* If not, code for bound relaxation/perturbation */
293  else {
294  report(lp, msglevel, "%s: Stalling at iter %10.0f; proceed to bound relaxation.\n",
295  monitor->spxfunc, (double) get_total_iter(lp));
296  acceptance = FALSE;
297  lp->spx_status = DEGENERATE;
298  return( acceptance );
299  }
300  }
301  }
302 
303  /* Otherwise change back to original selection strategy as soon as possible */
304  else {
305  if(monitor->pivrule != monitor->oldpivrule) {
306  lp->piv_strategy = monitor->oldpivstrategy;
307  altrule = monitor->oldpivrule;
308  if((altrule == PRICER_DEVEX) || (altrule == PRICER_STEEPESTEDGE))
309  restartPricer(lp, AUTOMATIC);
310  report(lp, msglevel, "...returned to original pivot selection rule at iter %.0f.\n",
311  (double) get_total_iter(lp));
312  }
313  stallMonitor_update(lp, monitor->thisobj);
314  monitor->Ccycle = 0;
315  monitor->Rcycle = 0;
316  monitor->Ncycle = 0;
317  monitor->Mcycle = 0;
318  }
319 
320  /* Update objective progress tracker */
321 Proceed:
322  monitor->Icount++;
323  if(deltaobj >= monitor->epsvalue)
324  monitor->prevobj = monitor->thisobj;
325  monitor->previnfeas = monitor->thisinfeas;
326 
327  return( acceptance );
328 }

346  objective function values to populate primal phase 1. */

376  break;
377  }
378  }
379  }
380

382 #if 0

385  bvar = 0;

425  add_columnex(lp, 2, avalue, rownr);
426 
427  /* Free the temporary sparse array storage */
428  if(idxarray == NULL)
429  FREE(rownr);
430  if(nzarray == NULL)
431  FREE(avalue);
432 
433  /* Now set the artificial variable to be basic */
434  set_basisvar(lp, bvar, lp->sum);
435  lp->P1extraDim++;
436  }
437  else {
438  report(lp, CRITICAL, "add_artificial: Could not find replacement basis variable for row %d\n",
439  forrownr);
440  lp->basis_valid = FALSE;
441  }
442 
443  }
444 
445  return(add);

564 {

566  primalphase1, pricerCanChange, minit, stallaccept, pendingunbounded;
567  int i, j, k, colnr = 0, rownr = 0, lastnr = 0,
568  candidatecount = 0, minitcount = 0, ok = TRUE;
569  LREAL theta = 0.0;
570  REAL epsvalue, xviolated, cviolated = 0.0,
571  *prow = NULL, *pcol = NULL,
572  *drow = lp->drow;
573  int *workINT = NULL,
574  *nzdrow = lp->nzdrow;
575 
576  if(lp->spx_trace)
577  report(lp, DETAILED, "Entered primal simplex algorithm with feasibility %s\n",
578  my_boolstr(primalfeasible));
579 
580  /* Add sufficent number of artificial variables to make the problem feasible
581  through the first phase; delete when primal feasibility has been achieved */
582  lp->P1extraDim = 0;
583  if(!primalfeasible) {
584  lp->simplex_mode = SIMPLEX_Phase1_PRIMAL;
585 #ifdef Paranoia
586  if(!verify_basis(lp))
587  report(lp, SEVERE, "primloop: No valid basis for artificial variables\n");
588 #endif
589 #if 0
590  /* First check if we can get away with a single artificial variable */
591  if(lp->equalities == 0) {
592  i = (int) feasibilityOffset(lp, !primal);
593  add_artificial(lp, i, prow, (int *) pcol);
594  }
595  else
596 #endif
597  /* Otherwise add as many artificial variables as is necessary
598  to force primal feasibility. */
599  for(i = 1; i <= lp->rows; i++) {
600  add_artificial(lp, i, NULL, NULL);
601  }
602 
603  /* Make sure we update the working objective */
604  if(lp->P1extraDim > 0) {
605 #if 1 /* v5.1 code: Not really necessary since we do not price the artificial
606  variables (stored at the end of the column list, they are initially
607  basic and are never allowed to enter the basis, once they exit) */
608  ok = allocREAL(lp, &(lp->drow), lp->sum+1, AUTOMATIC) &&
609  allocINT(lp, &(lp->nzdrow), lp->sum+1, AUTOMATIC);
610  if(!ok)
611  goto Finish;
612  lp->nzdrow[0] = 0;
613  drow = lp->drow;
614  nzdrow = lp->nzdrow;
615 #endif
616  mat_validate(lp->matA);
617  set_OF_p1extra(lp, 0.0);
618  }
619  if(lp->spx_trace)
620  report(lp, DETAILED, "P1extraDim count = %d\n", lp->P1extraDim);
621 
622  simplexPricer(lp, (MYBOOL)!primal);
623  invert(lp, INITSOL_USEZERO, TRUE);
624  }
625  else {
626  lp->simplex_mode = SIMPLEX_Phase2_PRIMAL;
627  restartPricer(lp, (MYBOOL)!primal);
628  }
629 
630  /* Create work arrays and optionally the multiple pricing structure */
631  ok = allocREAL(lp, &(lp->bsolveVal), lp->rows + 1, FALSE) &&
632  allocREAL(lp, &prow, lp->sum + 1, TRUE) &&
633  allocREAL(lp, &pcol, lp->rows + 1, TRUE);
634  if(is_piv_mode(lp, PRICE_MULTIPLE) && (lp->multiblockdiv > 1)) {
635  lp->multivars = multi_create(lp, FALSE);
636  ok &= (lp->multivars != NULL) &&
637  multi_resize(lp->multivars, lp->sum / lp->multiblockdiv, 2, FALSE, TRUE);
638  }
639  if(!ok)
640  goto Finish;
641 
642  /* Initialize regular primal simplex algorithm variables */
643  lp->spx_status = RUNNING;
644  minit = ITERATE_MAJORMAJOR;
645  epsvalue = lp->epspivot;
646  pendingunbounded = FALSE;
647 
648  ok = stallMonitor_create(lp, FALSE, "primloop");
649  if(!ok)
650  goto Finish;
651 
652  lp->rejectpivot[0] = 0;
653 
654  /* Iterate while we are successful; exit when the model is infeasible/unbounded,
655  or we must terminate due to numeric instability or user-determined reasons */
656  while((lp->spx_status == RUNNING) && !userabort(lp, -1)) {
657 
658  primalphase1 = (MYBOOL) (lp->P1extraDim > 0);
659  clear_action(&lp->spx_action, ACTION_REINVERT | ACTION_ITERATE);
660 
661  /* Check if we have stalling (from numerics or degenerate cycling) */
662  pricerCanChange = !primalphase1;
663  stallaccept = stallMonitor_check(lp, rownr, colnr, lastnr, minit, pricerCanChange, &forceoutEQ);
664  if(!stallaccept)
665  break;
666 
667  /* Find best column to enter the basis */
668 RetryCol:
669 #if 0
670  if(verify_solution(lp, FALSE, "spx_loop") > 0)
671  i = 1; /* This is just a debug trap */
672 #endif
673  if(!changedphase) {
674  i = 0;
675  do {
676  i++;
677  colnr = colprim(lp, drow, nzdrow, (MYBOOL) (minit == ITERATE_MINORRETRY), i, &candidatecount, TRUE, &xviolated);
678  } while ((colnr == 0) && (i < partial_countBlocks(lp, (MYBOOL) !primal)) &&
679  partial_blockStep(lp, (MYBOOL) !primal));
680 
681  /* Handle direct outcomes */
682  if(colnr == 0)
683  lp->spx_status = OPTIMAL;
684  if(lp->rejectpivot[0] > 0)
685  minit = ITERATE_MAJORMAJOR;
686 
687  /* See if accuracy check during compute_reducedcosts flagged refactorization */
688  if(is_action(lp->spx_action, ACTION_REINVERT))
689  bfpfinal = TRUE;
690 
691  }
692 
693  /* Make sure that we do not erroneously conclude that an unbounded model is optimal */
694 #ifdef primal_UseRejectionList
695  if((colnr == 0) && (lp->rejectpivot[0] > 0)) {
696  lp->spx_status = UNBOUNDED;
697  if((lp->spx_trace && (lp->bb_totalnodes == 0)) ||
698  (lp->bb_trace && (lp->bb_totalnodes > 0)))
699  report(lp, DETAILED, "The model is primal unbounded.\n");
700  colnr = lp->rejectpivot[1];
701  rownr = 0;
702  lp->rejectpivot[0] = 0;
703  ok = FALSE;
704  break;
705  }
706 #endif
707 
708  /* Check if we found an entering variable (indicating that we are still dual infeasible) */
709  if(colnr > 0) {
710  changedphase = FALSE;
711  fsolve(lp, colnr, pcol, NULL, lp->epsmachine, 1.0, TRUE); /* Solve entering column for Pi */
712 
713  /* Do special anti-degeneracy column selection, if specified */
714  if(is_anti_degen(lp, ANTIDEGEN_COLUMNCHECK) && !check_degeneracy(lp, pcol, NULL)) {
715  if(lp->rejectpivot[0] < DEF_MAXPIVOTRETRY/3) {
716  i = ++lp->rejectpivot[0];
717  lp->rejectpivot[i] = colnr;
718  report(lp, DETAILED, "Entering column %d found to be non-improving due to degeneracy.\n",
719  colnr);
720  minit = ITERATE_MINORRETRY;
721  goto RetryCol;
722  }
723  else {
724  lp->rejectpivot[0] = 0;
725  report(lp, DETAILED, "Gave up trying to find a strictly improving entering column.\n");
726  }
727  }
728 
729  /* Find the leaving variable that gives the most stringent bound on the entering variable */
730  theta = drow[colnr];
731  rownr = rowprim(lp, colnr, &theta, pcol, workINT, forceoutEQ, &cviolated);
732 
733 #ifdef AcceptMarginalAccuracy
734  /* Check for marginal accuracy */
735  if((rownr > 0) && (xviolated+cviolated < lp->epspivot)) {
736  if(lp->bb_trace || (lp->bb_totalnodes == 0))
737  report(lp, DETAILED, "primloop: Assuming convergence with reduced accuracy %g.\n",
738  MAX(xviolated, cviolated));
739  rownr = 0;
740  colnr = 0;
741  goto Optimality;
742  }
743  else
744 #endif
745 
746  /* See if we can do a straight artificial<->slack replacement (when "colnr" is a slack) */
747  if((lp->P1extraDim != 0) && (rownr == 0) && (colnr <= lp->rows))
748  rownr = findAnti_artificial(lp, colnr);
749 
750  if(rownr > 0) {
751  pendingunbounded = FALSE;
752  lp->rejectpivot[0] = 0;
753  set_action(&lp->spx_action, ACTION_ITERATE);
754  if(!lp->obj_in_basis) /* We must manually copy the reduced cost for RHS update */
755  pcol[0] = my_chsign(!lp->is_lower[colnr], drow[colnr]);
756  lp->bfp_prepareupdate(lp, rownr, colnr, pcol);
757  }
758 
759  /* We may be unbounded... */
760  else {
761  /* First make sure that we are not suffering from precision loss */
762 #ifdef primal_UseRejectionList
763  if(lp->rejectpivot[0] < DEF_MAXPIVOTRETRY) {
764  lp->spx_status = RUNNING;
765  lp->rejectpivot[0]++;
766  lp->rejectpivot[lp->rejectpivot[0]] = colnr;
767  report(lp, DETAILED, "...trying to recover via another pivot column.\n");
768  minit = ITERATE_MINORRETRY;
769  goto RetryCol;
770  }
771  else
772 #endif
773  /* Check that we are not having numerical problems */
774  if(!refactRecent(lp) && !pendingunbounded) {
775  bfpfinal = TRUE;
776  pendingunbounded = TRUE;
777  set_action(&lp->spx_action, ACTION_REINVERT);
778  }
779 
780  /* Conclude that the model is unbounded */
781  else {
782  lp->spx_status = UNBOUNDED;
783  report(lp, DETAILED, "The model is primal unbounded.\n");
784  break;
785  }
786  }
787  }
788 
789  /* We handle optimality and phase 1 infeasibility ... */
790  else {
791 
792 Optimality:
793  /* Handle possible transition from phase 1 to phase 2 */
794  if(!primalfeasible || isP1extra(lp)) {
795 
796  if(feasiblePhase1(lp, epsvalue)) {
797  lp->spx_status = RUNNING;
798  if(lp->bb_totalnodes == 0) {
799  report(lp, NORMAL, "Found feasibility by primal simplex after %10.0f iter.\n",
800  (double) get_total_iter(lp));
801  if((lp->usermessage != NULL) && (lp->msgmask & MSG_LPFEASIBLE))
802  lp->usermessage(lp, lp->msghandle, MSG_LPFEASIBLE);
803  }
804  changedphase = FALSE;
805  primalfeasible = TRUE;
806  lp->simplex_mode = SIMPLEX_Phase2_PRIMAL;
807  set_OF_p1extra(lp, 0.0);
808 
809  /* We can do two things now;
810  1) delete the rows belonging to those variables, since they are redundant, OR
811  2) drive out the existing artificial variables via pivoting. */
812  if(lp->P1extraDim > 0) {
813 
814 #ifdef Phase1EliminateRedundant
815  /* If it is not a MIP model we can try to delete redundant rows */
816  if((lp->bb_totalnodes == 0) && (MIP_count(lp) == 0)) {
817  while(lp->P1extraDim > 0) {
818  i = lp->rows;
819  while((i > 0) && (lp->var_basic[i] <= lp->sum-lp->P1extraDim))
820  i--;
821 #ifdef Paranoia
822  if(i <= 0) {
823  report(lp, SEVERE, "primloop: Could not find redundant artificial.\n");
824  break;
825  }
826 #endif
827  /* Obtain column and row indeces */
828  j = lp->var_basic[i]-lp->rows;
829  k = get_artificialRow(lp, j);
830 
831  /* Delete row before column due to basis "compensation logic" */
832  if(lp->is_basic[k]) {
833  lp->is_basic[lp->rows+j] = FALSE;
834  del_constraint(lp, k);
835  }
836  else
837  set_basisvar(lp, i, k);
838  del_column(lp, j);
839  lp->P1extraDim--;
840  }
841  lp->basis_valid = TRUE;
842  }
843  /* Otherwise we drive out the artificials by elimination pivoting */
844  else
845  eliminate_artificials(lp, prow);
846 
847 #else
848  /* Indicate phase 2 with artificial variables by negating P1extraDim */
849  lp->P1extraDim = my_flipsign(lp->P1extraDim);
850 #endif
851  }
852 
853  /* We must refactorize since the OF changes from phase 1 to phase 2 */
854  set_action(&lp->spx_action, ACTION_REINVERT);
855  bfpfinal = TRUE;
856  }
857 
858  /* We are infeasible in phase 1 */
859  else {
860  lp->spx_status = INFEASIBLE;
861  minit = ITERATE_MAJORMAJOR;
862  if(lp->spx_trace)
863  report(lp, NORMAL, "Model infeasible by primal simplex at iter %10.0f.\n",
864  (double) get_total_iter(lp));
865  }
866  }
867 
868  /* Handle phase 1 optimality */
869  else {
870  /* (Do nothing special) */
871  }
872 
873  /* Check if we are still primal feasible; the default assumes that this check
874  is not necessary after the relaxed problem has been solved satisfactorily. */
875  if((lp->bb_level <= 1) || (lp->improve & IMPROVE_BBSIMPLEX) /* || (lp->bb_rule & NODE_RCOSTFIXING) */) { /* NODE_RCOSTFIXING fix */
876  set_action(&lp->piv_strategy, PRICE_FORCEFULL);
877  i = rowdual(lp, lp->rhs, FALSE, FALSE, NULL);
878  clear_action(&lp->piv_strategy, PRICE_FORCEFULL);
879  if(i > 0) {
880  lp->spx_status = LOSTFEAS;
881  if(lp->total_iter == 0)
882  report(lp, DETAILED, "primloop: Lost primal feasibility at iter %10.0f: will try to recover.\n",
883  (double) get_total_iter(lp));
884  }
885  }
886  }
887 
888  /* Pivot row/col and update the inverse */
889  if(is_action(lp->spx_action, ACTION_ITERATE)) {
890  lastnr = lp->var_basic[rownr];
891 
892  if(refactRecent(lp) == AUTOMATIC)
893  minitcount = 0;
894  else if(minitcount > MAX_MINITUPDATES) {
895  recompute_solution(lp, INITSOL_USEZERO);
896  minitcount = 0;
897  }
898  minit = performiteration(lp, rownr, colnr, theta, primal,
899  (MYBOOL) (/*(candidatecount > 1) && */
900  (stallaccept != AUTOMATIC)),
901  NULL, NULL,
902  pcol, NULL, NULL);
903  if(minit != ITERATE_MAJORMAJOR)
904  minitcount++;
905 
906  if((lp->spx_status == USERABORT) || (lp->spx_status == TIMEOUT))
907  break;
908  else if(minit == ITERATE_MINORMAJOR)
909  continue;
910 #ifdef UsePrimalReducedCostUpdate
911  /* Do a fast update of the reduced costs in preparation for the next iteration */
912  if(minit == ITERATE_MAJORMAJOR)
913  update_reducedcosts(lp, primal, lastnr, colnr, pcol, drow);
914 #endif
915 
916  /* Detect if an auxiliary variable has left the basis and delete it; if
917  the non-basic variable only changed bound (a "minor iteration"), the
918  basic artificial variable did not leave and there is nothing to do */
919  if((minit == ITERATE_MAJORMAJOR) && (lastnr > lp->sum - abs(lp->P1extraDim))) {
920 #ifdef Paranoia
921  if(lp->is_basic[lastnr] || !lp->is_basic[colnr])
922  report(lp, SEVERE, "primloop: Invalid basis indicator for variable %d at iter %10.0f.\n",
923  lastnr, (double) get_total_iter(lp));
924 #endif
925  del_column(lp, lastnr-lp->rows);
926  if(lp->P1extraDim > 0)
927  lp->P1extraDim--;
928  else
929  lp->P1extraDim++;
930  if(lp->P1extraDim == 0) {
931  colnr = 0;
932  changedphase = TRUE;
933  stallMonitor_reset(lp);
934  }
935  }
936  }
937 
938  if(lp->spx_status == SWITCH_TO_DUAL)
939  ;
940  else if(!changedphase && lp->bfp_mustrefactorize(lp)) {
941 #ifdef ResetMinitOnReinvert
942  minit = ITERATE_MAJORMAJOR;
943 #endif
944  if(!invert(lp, INITSOL_USEZERO, bfpfinal))
945  lp->spx_status = SINGULAR_BASIS;
946  bfpfinal = FALSE;
947  }
948  }
949 
950  /* Remove any remaining artificial variables (feasible or infeasible model) */
951  lp->P1extraDim = abs(lp->P1extraDim);
952 /* if((lp->P1extraDim > 0) && (lp->spx_status != DEGENERATE)) { */
953  if(lp->P1extraDim > 0) {
954  clear_artificials(lp);
955  if(lp->spx_status != OPTIMAL)
956  restore_basis(lp);
957  i = invert(lp, INITSOL_USEZERO, TRUE);
958  }
959 #ifdef Paranoia
960  if(!verify_basis(lp))
961  report(lp, SEVERE, "primloop: Invalid basis detected due to internal error\n");
962 #endif
963 
964  /* Switch to dual phase 1 simplex for MIP models during
965  B&B phases, since this is typically far more efficient */
966 #ifdef ForceDualSimplexInBB
967  if((lp->bb_totalnodes == 0) && (MIP_count(lp) > 0) &&
968  ((lp->simplex_strategy & SIMPLEX_Phase1_DUAL) == 0)) {
969  lp->simplex_strategy &= ~SIMPLEX_Phase1_PRIMAL;
970  lp->simplex_strategy += SIMPLEX_Phase1_DUAL;
971  }
972 #endif
973 
974 Finish:
975  stallMonitor_finish(lp);
976  multi_free(&(lp->multivars));
977  FREE(prow);
978  FREE(pcol);
979  FREE(lp->bsolveVal);
980 
981  return(ok);
982 } /* primloop */

984 STATIC int dualloop(lprec *lp, MYBOOL dualfeasible, int dualinfeasibles[], REAL dualoffset)
985 {
986  MYBOOL primal = FALSE, inP1extra, dualphase1 = FALSE, changedphase = TRUE,
987  pricerCanChange, minit, stallaccept, longsteps,
988  forceoutEQ = FALSE, bfpfinal = FALSE;
989  int i, colnr = 0, rownr = 0, lastnr = 0,
990  candidatecount = 0, minitcount = 0,
991 #ifdef FixInaccurateDualMinit
992  minitcolnr = 0,
993 #endif
994  ok = TRUE;
995  int *boundswaps = NULL;
996  LREAL theta = 0.0;
997  REAL epsvalue, xviolated, cviolated,
998  *prow = NULL, *pcol = NULL,
999  *drow = lp->drow;
1000  int *nzprow = NULL, *workINT = NULL,
1001  *nzdrow = lp->nzdrow;
1002 
1003  if(lp->spx_trace)
1004  report(lp, DETAILED, "Entered dual simplex algorithm with feasibility %s.\n",
1005  my_boolstr(dualfeasible));
1006 
1007  /* Allocate work arrays */
1008  ok = allocREAL(lp, &prow, lp->sum + 1, TRUE) &&
1009  allocINT (lp, &nzprow, lp->sum + 1, FALSE) &&
1010  allocREAL(lp, &pcol, lp->rows + 1, TRUE);
1011  if(!ok)
1012  goto Finish;
1013 
1014  /* Set non-zero P1extraVal value to force dual feasibility when the dual
1015  simplex is used as a phase 1 algorithm for the primal simplex.
1016  The value will be reset when primal feasibility has been achieved, or
1017  a dual non-feasibility has been encountered (no candidate for a first
1018  leaving variable) */
1019  inP1extra = (MYBOOL) (dualoffset != 0);
1020  if(inP1extra) {
1021  set_OF_p1extra(lp, dualoffset);
1022  simplexPricer(lp, (MYBOOL)!primal);
1023  invert(lp, INITSOL_USEZERO, TRUE);
1024  }
1025  else
1026  restartPricer(lp, (MYBOOL)!primal);

1029 #if 0
1030  longsteps = TRUE;
1031 #elif 0
1032  longsteps = (MYBOOL) ((MIP_count(lp) > 0) && (lp->bb_level > 1));
1033 #elif 0
1034  longsteps = (MYBOOL) ((MIP_count(lp) > 0) && (lp->solutioncount >= 1));
1035 #else
1036  longsteps = FALSE;
1037 #endif
1038 #ifdef UseLongStepDualPhase1
1039  longsteps = !dualfeasible && (MYBOOL) (dualinfeasibles != NULL);
1040 #endif
1041 
1042  if(longsteps) {
1043  lp->longsteps = multi_create(lp, TRUE);
1044  ok = (lp->longsteps != NULL) &&
1045  multi_resize(lp->longsteps, MIN(lp->boundedvars+2, 11), 1, TRUE, TRUE);
1046  if(!ok)
1047  goto Finish;
1048 #ifdef UseLongStepPruning
1049  lp->longsteps->objcheck = TRUE;
1050 #endif
1051  boundswaps = multi_indexSet(lp->longsteps, FALSE);
1052  }
1053 
1054  /* Do regular dual simplex variable initializations */
1055  lp->spx_status = RUNNING;
1056  minit = ITERATE_MAJORMAJOR;
1057  epsvalue = lp->epspivot;
1058 
1059  ok = stallMonitor_create(lp, TRUE, "dualloop");
1060  if(!ok)
1061  goto Finish;
1062 
1063  lp->rejectpivot[0] = 0;
1064  if(dualfeasible)
1065  lp->simplex_mode = SIMPLEX_Phase2_DUAL;
1066  else
1067  lp->simplex_mode = SIMPLEX_Phase1_DUAL;
1068 
1069  /* Check if we have equality slacks in the basis and we should try to
1070  drive them out in order to reduce chance of degeneracy in Phase 1.
1071  forceoutEQ = FALSE : Only eliminate assured "good" violated
1072  equality constraint slacks
1073  AUTOMATIC: Seek more elimination of equality constraint
1074  slacks (but not as aggressive as the rule
1075  used in lp_solve v4.0 and earlier)
1076  TRUE: Force remaining equality slacks out of the
1077  basis */
1078  if(dualphase1 || inP1extra ||
1079  ((lp->fixedvars > 0) && is_anti_degen(lp, ANTIDEGEN_FIXEDVARS))) {
1080  forceoutEQ = AUTOMATIC;
1081  }
1082 #if 1
1083  if(is_anti_degen(lp, ANTIDEGEN_DYNAMIC) && (bin_count(lp, TRUE)*2 > lp->columns)) {
1084  switch (forceoutEQ) {
1085  case FALSE: forceoutEQ = AUTOMATIC;
1086  break;
1087  /* case AUTOMATIC: forceoutEQ = TRUE;
1088  break;
1089  default: forceoutEQ = TRUE; */
1090  }
1091  }
1092 #endif
1093 
1094  while((lp->spx_status == RUNNING) && !userabort(lp, -1)) {
1095 
1096  /* Check if we have stalling (from numerics or degenerate cycling) */
1097  pricerCanChange = !dualphase1 && !inP1extra;
1098  stallaccept = stallMonitor_check(lp, rownr, colnr, lastnr, minit, pricerCanChange, &forceoutEQ);
1099  if(!stallaccept)
1100  break;
1101 
1102  /* Store current LP index for reference at next iteration */
1103  changedphase = FALSE;
1104 
1105  /* Compute (pure) dual phase1 offsets / reduced costs if appropriate */
1106  dualphase1 &= (MYBOOL) (lp->simplex_mode == SIMPLEX_Phase1_DUAL);
1107  if(longsteps && dualphase1 && !inP1extra) {
1108  obtain_column(lp, dualinfeasibles[1], pcol, NULL, NULL);
1109  i = 2;
1110  for(i = 2; i <= dualinfeasibles[0]; i++)
1111  mat_multadd(lp->matA, pcol, dualinfeasibles[i], 1.0);
1112  /* Solve (note that solved pcol will be used instead of lp->rhs) */
1113  ftran(lp, pcol, NULL, lp->epsmachine);
1114  }
1115 
1116  /* Do minor iterations (non-basic variable bound flips) for as
1117  long as possible since this is a cheap way of iterating */
1118 #if (defined dual_Phase1PriceEqualities) || (defined dual_UseRejectionList)
1119 RetryRow:
1120 #endif
1121  if(minit != ITERATE_MINORRETRY) {
1122  i = 0;
1123  do {
1124  i++;
1125  rownr = rowdual(lp, my_if(dualphase1, pcol, NULL), forceoutEQ, TRUE, &xviolated);
1126  } while ((rownr == 0) && (i < partial_countBlocks(lp, (MYBOOL) !primal)) &&
1127  partial_blockStep(lp, (MYBOOL) !primal));
1128  }
1129 
1130  /* Make sure that we do not erroneously conclude that an infeasible model is optimal */
1131 #ifdef dual_UseRejectionList
1132  if((rownr == 0) && (lp->rejectpivot[0] > 0)) {
1133  lp->spx_status = INFEASIBLE;
1134  if((lp->spx_trace && (lp->bb_totalnodes == 0)) ||
1135  (lp->bb_trace && (lp->bb_totalnodes > 0)))
1136  report(lp, DETAILED, "The model is primal infeasible.\n");
1137  rownr = lp->rejectpivot[1];
1138  colnr = 0;
1139  lp->rejectpivot[0] = 0;
1140  ok = FALSE;
1141  break;
1142  }
1143 #endif
1144 
1145  /* If we found a leaving variable, find a matching entering one */
1146  clear_action(&lp->spx_action, ACTION_ITERATE);
1147  if(rownr > 0) {
1148  colnr = coldual(lp, rownr, prow, nzprow, drow, nzdrow,
1149  (MYBOOL) (dualphase1 && !inP1extra),
1150  (MYBOOL) (minit == ITERATE_MINORRETRY), &candidatecount, &cviolated);
1151  if(colnr < 0) {
1152  minit = ITERATE_MAJORMAJOR;
1153  continue;
1154  }
1155 #ifdef AcceptMarginalAccuracy
1156  else if(xviolated+cviolated < lp->epspivot) {
1157  if(lp->bb_trace || (lp->bb_totalnodes == 0))
1158  report(lp, DETAILED, "dualloop: Assuming convergence with reduced accuracy %g.\n",
1159  MAX(xviolated, cviolated));
1160  rownr = 0;
1161  colnr = 0;
1162  }
1163 #endif
1164  /* Check if the long-dual found reason to prune the B&B tree */
1165  if(lp->spx_status == FATHOMED)
1166  break;
1167  }
1168  else
1169  colnr = 0;
1170 
1171  /* Process primal-infeasible row */
1172  if(rownr > 0) {
1173 
1174  if(colnr > 0) {
1175 #ifdef Paranoia
1176  if((rownr > lp->rows) || (colnr > lp->sum)) {
1177  report(lp, SEVERE, "dualloop: Invalid row %d(%d) and column %d(%d) pair selected at iteration %.0f\n",
1178  rownr, lp->rows, colnr-lp->columns, lp->columns, (double) get_total_iter(lp));
1179  lp->spx_status = UNKNOWNERROR;
1180  break;
1181  }
1182 #endif
1183  fsolve(lp, colnr, pcol, workINT, lp->epsmachine, 1.0, TRUE);
1184 
1185 #ifdef FixInaccurateDualMinit
1186  /* Prevent bound flip-flops during minor iterations; used to detect
1187  infeasibility after triggering of minor iteration accuracy management */
1188  if(colnr != minitcolnr)
1189  minitcolnr = 0;
1190 #endif
1191 
1192  /* Getting division by zero here; catch it and try to recover */
1193  if(pcol[rownr] == 0) {
1194  if(lp->spx_trace)
1195  report(lp, DETAILED, "dualloop: Attempt to divide by zero (pcol[%d])\n", rownr);

1534  lp->bb_status = lp->spx_status;
1535  lp->P1extraDim = 0;
1536  set_OF_p1extra(lp, 0);

1538  lost_feas_count = 0;
1539  lost_feas_state = FALSE;
1540  lp->simplex_mode = SIMPLEX_DYNAMIC;
1541 
1542  /* Compute the number of fixed basic and bounded variables (used in long duals) */
1543  lp->fixedvars = 0;
1544  lp->boundedvars = 0;
1545  for(i = 1; i <= lp->rows; i++) {
1546  j = lp->var_basic[i];
1547  if((j <= lp->rows) && is_fixedvar(lp, j))
1548  lp->fixedvars++;
1549  if((lp->upbo[i] < lp->infinite) && (lp->upbo[i] > lp->epsprimal))
1550  lp->boundedvars++;
1551  }
1552  for(; i <= lp->sum; i++){
1553  if((lp->upbo[i] < lp->infinite) && (lp->upbo[i] > lp->epsprimal))
1554  lp->boundedvars++;
1555  }
1556 #ifdef UseLongStepDualPhase1
1557  allocINT(lp, &infeasibles, lp->columns + 1, FALSE);
1558  infeasibles[0] = 0;
1559 #endif
1560 
1561  /* Reinvert for initialization, if necessary */
1562  isbb = (MYBOOL) ((MIP_count(lp) > 0) && (lp->bb_level > 1));
1563  if(is_action(lp->spx_action, ACTION_REINVERT)) {
1564  if(isbb && (lp->bb_bounds->nodessolved == 0))
1565 /* if(isbb && (lp->bb_basis->pivots == 0)) */
1566  recompute_solution(lp, INITSOL_SHIFTZERO);
1567  else {
1568  i = my_if(is_action(lp->spx_action, ACTION_REBASE), INITSOL_SHIFTZERO, INITSOL_USEZERO);
1569  invert(lp, (MYBOOL) i, TRUE);
1570  }
1571  }
1572  else if(is_action(lp->spx_action, ACTION_REBASE))
1573  recompute_solution(lp, INITSOL_SHIFTZERO);
1574 
1575  /* Optionally try to do bound flips to obtain dual feasibility */
1576  if(is_action(lp->improve, IMPROVE_DUALFEAS) || (lp->rows == 0))
1577  boundflip_count = &i;
1578  else
1579  boundflip_count = NULL;
1580 
1581  /* Loop for as long as is needed */
1582  while(lp->spx_status == RUNNING) {
1583 
1584  /* Check for dual and primal feasibility */
1585  dualfeasible = isbb ||
1586  isDualFeasible(lp, lp->epsprimal, boundflip_count, infeasibles, &dualoffset);
1587 
1588  /* Recompute if the dual feasibility check included bound flips */
1589  if(is_action(lp->spx_action, ACTION_RECOMPUTE))
1590  recompute_solution(lp, INITSOL_USEZERO);
1591  primalfeasible = isPrimalFeasible(lp, lp->epsprimal, NULL, &primaloffset);
1592 
1593  if(userabort(lp, -1))
1594  break;
1595 
1596  if(lp->spx_trace) {
1597  if(primalfeasible)
1598  report(lp, NORMAL, "Start at primal feasible basis\n");
1599  else if(dualfeasible)
1600  report(lp, NORMAL, "Start at dual feasible basis\n");
1601  else if(lost_feas_count > 0)
1602  report(lp, NORMAL, "Continuing at infeasible basis\n");
1603  else
1604  report(lp, NORMAL, "Start at infeasible basis\n");
1605  }
1606 
1607  /* Now do the simplex magic */
1608  if(((lp->simplex_strategy & SIMPLEX_Phase1_DUAL) == 0) ||
1609  ((MIP_count(lp) > 0) && (lp->total_iter == 0) &&
1610  is_presolve(lp, PRESOLVE_REDUCEMIP))) {
1611  if(!lost_feas_state && primalfeasible && ((lp->simplex_strategy & SIMPLEX_Phase2_DUAL) > 0))
1612  lp->spx_status = SWITCH_TO_DUAL;
1613  else
1614  primloop(lp, primalfeasible, 0.0);
1615  if(lp->spx_status == SWITCH_TO_DUAL)
1616  dualloop(lp, TRUE, NULL, 0.0);
1617  }
1618  else {
1619  if(!lost_feas_state && primalfeasible && ((lp->simplex_strategy & SIMPLEX_Phase2_PRIMAL) > 0))
1620  lp->spx_status = SWITCH_TO_PRIMAL;
1621  else
1622  dualloop(lp, dualfeasible, infeasibles, dualoffset);
1623  if(lp->spx_status == SWITCH_TO_PRIMAL)
1624  primloop(lp, TRUE, 0.0);
1625  }
1626 
1627  /* Check for simplex outcomes that always involve breaking out of the loop;
1628  this includes optimality, unboundedness, pure infeasibility (i.e. not
1629  loss of feasibility), numerical failure and perturbation-based degeneracy
1630  handling */
1631  i = lp->spx_status;
1632  primalfeasible = (MYBOOL) (i == OPTIMAL);
1633  if(primalfeasible || (i == UNBOUNDED))
1634  break;
1635  else if(((i == INFEASIBLE) && is_anti_degen(lp, ANTIDEGEN_INFEASIBLE)) ||
1636  ((i == LOSTFEAS) && is_anti_degen(lp, ANTIDEGEN_LOSTFEAS)) ||
1637  ((i == NUMFAILURE) && is_anti_degen(lp, ANTIDEGEN_NUMFAILURE)) ||
1638  ((i == DEGENERATE) && is_anti_degen(lp, ANTIDEGEN_STALLING))) {
1639  /* Check if we should not loop here, but do perturbations */
1640  if((lp->bb_level <= 1) || is_anti_degen(lp, ANTIDEGEN_DURINGBB))
1641  break;
1642

1644 #ifdef AssumeHighAccuracyInBB
1645  if((lp->bb_level > 1) && (i == INFEASIBLE))
1646  break;
1647 #endif
1648  }
1649 
1650  /* Check for outcomes that may involve trying another simplex loop */
1651  if(lp->spx_status == SINGULAR_BASIS) {
1652  lost_feas_state = FALSE;
1653  singular_count++;
1654  if(singular_count >= DEF_MAXSINGULARITIES) {
1655  report(lp, IMPORTANT, "spx_run: Failure due to too many singular bases.\n");
1656  lp->spx_status = NUMFAILURE;
1657  break;
1658  }
1659  if(lp->spx_trace || (lp->verbose > DETAILED))
1660  report(lp, NORMAL, "spx_run: Singular basis; attempting to recover.\n");
1661  lp->spx_status = RUNNING;
1662  /* Singular pivots are simply skipped by the inversion, leaving a row's
1663  slack variable in the basis instead of the singular user variable. */
1664  }
1665  else {
1666  lost_feas_state = (MYBOOL) (lp->spx_status == LOSTFEAS);
1667 #if 0
1668  /* Optionally handle loss of numerical accuracy as loss of feasibility,
1669  but only attempt a single loop to try to recover from this. */
1670  lost_feas_state |= (MYBOOL) ((lp->spx_status == NUMFAILURE) && (lost_feas_count < 1));
1671 #endif
1672  if(lost_feas_state) {
1673  lost_feas_count++;
1674  if(lost_feas_count < DEF_MAXSINGULARITIES) {
1675  report(lp, DETAILED, "spx_run: Recover lost feasibility at iter %10.0f.\n",
1676  (double) get_total_iter(lp));
1677  lp->spx_status = RUNNING;
1678  }
1679  else {
1680  report(lp, IMPORTANT, "spx_run: Lost feasibility %d times - iter %10.0f and %9.0f nodes.\n",
1681  lost_feas_count, (double) get_total_iter(lp), (double) lp->bb_totalnodes);
1682  lp->spx_status = NUMFAILURE;
1683  }
1684  }
1685  }
1686  }
1687 
1688  /* Update iteration tallies before returning */
1689  lp->total_iter += lp->current_iter;
1690  lp->current_iter = 0;
1691  lp->total_bswap += lp->current_bswap;
1692  lp->current_bswap = 0;
1693  FREE(infeasibles);
1694 
1695  return(lp->spx_status);
1696 } /* spx_run */
