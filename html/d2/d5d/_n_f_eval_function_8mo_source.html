101 function evaluateNormal
102  input Function fn;
103  input list<Expression> args;
104  output Expression result;
105 protected
106  list<Statement> fn_body;
107  list<Binding> bindings;
108  ReplTree.Tree repl;
109  Integer call_count, limit;
110  Pointer<Integer> call_counter = fn.callCounter;
111  FlowControl ctrl;
112 algorithm
113  // Functions contain a mutable call counter that's increased by one at the
114  // start of each evaluation, and decreased by one when the evalution is
115  // finished. This is used to limit the number of recursive functions calls.
116  call_count := Pointer.access(call_counter) + 1;
117  limit := Flags.getConfigInt(Flags.EVAL_RECURSION_LIMIT);
118 
119  if call_count > limit then
120  Pointer.update(call_counter, 0);
121  Error.addSourceMessage(Error.EVAL_RECURSION_LIMIT_REACHED,
122  {String(limit), Absyn.pathString(Function.name(fn))}, InstNode.info(fn.node));
123  fail();
124  end if;
125 
126  Pointer.update(call_counter, call_count);
127 
128  try
129  fn_body := Function.getBody(fn);
130  repl := createReplacements(fn, args);
131  // TODO: Also apply replacements to the replacements themselves, i.e. the
132  // bindings of the function parameters. But they probably need to be
133  // sorted by dependencies first.
134  fn_body := applyReplacements(repl, fn_body);
135  fn_body := optimizeBody(fn_body);
136  ctrl := evaluateStatements(fn_body);
137 
138  if ctrl <> FlowControl.ASSERTION then
139  result := createResult(repl, fn.outputs);
140  else
141  fail();
142  end if;
143  else
144  // Make sure we always decrease the call counter even if the evaluation fails.
145  Pointer.update(call_counter, call_count - 1);
146  fail();
147  end try;
148 
149  Pointer.update(call_counter, call_count - 1);
150 end evaluateNormal;

154  input list<Expression> args;
155  output Expression result;
156 protected
157  String name, lang;
158  ComponentRef output_ref;
159  Option<SCode.Annotation> ann;

162  Sections.EXTERNAL(name = name, args = ext_args, outputRef = output_ref, language = lang, ann = ann) :=
163  Class.getSections(InstNode.getClass(fn.node));

168  elseif isKnownExternalFunc(name, ann) then

188 function createReplacements
189  input Function fn;
190  input list<Expression> args;
191  output ReplTree.Tree repl;
192 protected
193  Expression arg;
194  list<Expression> rest_args = args;
195 algorithm
196  repl := ReplTree.new();
197 
198  // Add inputs to the replacement tree. Since they can't be assigned to the
199  // replacements don't need to be mutable.
200  for i in fn.inputs loop
201  arg :: rest_args := rest_args;
202  repl := addInputReplacement(i, "", arg, repl);
203  end for;
204 
205  // Add outputs and local variables to the replacement tree. These do need to
206  // be mutable to allow assigning to them.
207  repl := List.fold(fn.outputs, function addMutableReplacement(prefix = ""), repl);
208  repl := List.fold(fn.locals, function addMutableReplacement(prefix = ""), repl);
209 
210  // Apply the replacements to the replacements themselves. This is done after
211  // building the tree to make sure all the replacements are available.
212  repl := ReplTree.map(repl, function applyBindingReplacement(repl = repl));
213 end createReplacements;

345 function applyReplacementCref
346  input ReplTree.Tree repl;
347  input ComponentRef cref;
348  input Expression exp;
349  output Expression outExp;
350 protected
351  list<ComponentRef> cref_parts;
352  Option<Expression> repl_exp;
353  InstNode parent, node;
354 algorithm
355  // Explode the cref into a list of parts in reverse order.
356  cref_parts := ComponentRef.toListReverse(cref);
357 
358  // If the list is empty it's probably an iterator or _, which shouldn't be replaced.
359  if listEmpty(cref_parts) then
360  outExp := exp;
361  else
362  // Look up the replacement for the first part in the replacement tree.
363  parent := ComponentRef.node(listHead(cref_parts));
364  repl_exp := ReplTree.getOpt(repl, InstNode.name(parent));
365 
366  if isSome(repl_exp) then
367  SOME(outExp) := repl_exp;
368  else
369  outExp := exp;
370  return;
371  end if;
372 
373  outExp := Expression.applySubscripts(ComponentRef.getSubscripts(listHead(cref_parts)), outExp);
374  cref_parts := listRest(cref_parts);
375 
376  if not listEmpty(cref_parts) then
377  try
378  // If the cref consists of more than one identifier we need to look up
379  // the corresponding record field in the expression.
380  for cr in cref_parts loop
381  node := ComponentRef.node(cr);
382  outExp := Expression.makeImmutable(outExp);
383  outExp := Expression.lookupRecordField(InstNode.name(node), outExp);
384  outExp := Expression.applySubscripts(ComponentRef.getSubscripts(cr), outExp);
385  end for;
386  else
387  Error.assertion(false, getInstanceName() + " could not find replacement for " +
388  ComponentRef.toString(cref), sourceInfo());
389  end try;
390  end if;
391  end if;
392 end applyReplacementCref;

400 function optimizeStatement
401  input output Statement stmt;
402 algorithm
403  () := match stmt
404  local
405  Expression iter_exp;
406 
407  // Replace iterators in for loops with mutable expressions, so we don't need
408  // to do it each time we enter a for loop during evaluation.
409  case Statement.FOR()
410  algorithm
411  // Make a mutable expression with a placeholder value.
412  iter_exp := Expression.makeMutable(Expression.EMPTY(Type.UNKNOWN()));
413  // Replace the iterator with the expression in the body of the for loop.
414  stmt.body := list(
415  Statement.mapExp(s, function Expression.replaceIterator(
416  iterator = stmt.iterator, iteratorValue = iter_exp))
417  for s in stmt.body);
418  // Replace the iterator node with the mutable expression too.
419  stmt.iterator := InstNode.EXP_NODE(iter_exp);
420  then
421  ();
422 
423  else ();
424  end match;
425 end optimizeStatement;

427 function createResult
428  input ReplTree.Tree repl;
429  input list<InstNode> outputs;
430  output Expression exp;
431 protected
432  list<Expression> expl;
433  list<Type> types;
434  Expression e;
435 algorithm
436  if listLength(outputs) == 1 then
437  exp := Ceval.evalExp(ReplTree.get(repl, InstNode.name(listHead(outputs))));
438  assertAssignedOutput(listHead(outputs), exp);
439  else
440  expl := {};
441  types := {};
442 
443  for o in outputs loop
444  e := Ceval.evalExp(ReplTree.get(repl, InstNode.name(o)));
445  assertAssignedOutput(o, e);
446  expl := e :: expl;
447  end for;
448 
449  expl := listReverseInPlace(expl);
450  types := list(Expression.typeOf(e) for e in expl);
451  exp := Expression.TUPLE(Type.TUPLE(types, NONE()), expl);
452  end if;
453 end createResult;

528  input Expression value;

530  () := match (variable, value)
531  local
532  Expression var, val;
533  list<Expression> vals;
534  Mutable<Expression> var_ptr;
535

537  algorithm

549  ();

579 function assignArrayElement
580  input Expression arrayExp;
581  input list<Subscript> subscripts;
582  input Expression value;
583  output Expression result;
584 protected
585  Expression sub, val;
586  list<Subscript> rest_subs;
587  Integer idx;
588  list<Expression> subs, vals;
589 algorithm
590  result := match (arrayExp, subscripts)
591  case (Expression.ARRAY(), Subscript.INDEX(sub) :: rest_subs) guard Expression.isScalarLiteral(sub)
592  algorithm
593  idx := Expression.toInteger(sub);
594 
595  if listEmpty(rest_subs) then
596  arrayExp.elements := List.set(arrayExp.elements, idx, value);
597  else
598  arrayExp.elements := List.set(arrayExp.elements, idx,
599  assignArrayElement(listGet(arrayExp.elements, idx), rest_subs, value));
600  end if;
601  then
602  arrayExp;
603 
604  case (Expression.ARRAY(), Subscript.SLICE(sub) :: rest_subs)
605  algorithm
606  subs := Expression.arrayElements(sub);
607  vals := Expression.arrayElements(value);
608 
609  if listEmpty(rest_subs) then
610  for s in subs loop
611  val :: vals := vals;
612  idx := Expression.toInteger(s);
613  arrayExp.elements := List.set(arrayExp.elements, idx, val);
614  end for;
615  else
616  for s in subs loop
617  val :: vals := vals;
618  idx := Expression.toInteger(s);
619  arrayExp.elements := List.set(arrayExp.elements, idx,
620  assignArrayElement(listGet(arrayExp.elements, idx), rest_subs, val));
621  end for;
622  end if;
623  then
624  arrayExp;
625 
626  case (Expression.ARRAY(), Subscript.WHOLE() :: rest_subs)
627  algorithm
628  if not listEmpty(rest_subs) then
629  arrayExp.elements := list(assignArrayElement(e, rest_subs, v) threaded for
630  e in arrayExp.elements, v in Expression.arrayElements(value));
631  end if;
632  then
633  arrayExp;
634 
635  else
636  algorithm
637  Error.assertion(false, getInstanceName() + ": unimplemented case for " +
638  Expression.toString(arrayExp) +
639  Subscript.toStringList(subscripts) + " = " +
640  Expression.toString(value), sourceInfo());
641  then
642  fail();
643 
644  end match;
645 end assignArrayElement;

662 function assignRecord

664  input Expression rhs;

707 function evaluateFor
708  input InstNode iterator;
709  input Option<Expression> range;
710  input list<Statement> forBody;
711  input DAE.ElementSource source;
712  output FlowControl ctrl;
713 protected
714  RangeIterator range_iter;
715  Mutable<Expression> iter_exp;
716  Expression range_exp, value;
717  list<Statement> body = forBody;
718  Integer i = 0, limit = Flags.getConfigInt(Flags.EVAL_LOOP_LIMIT);
719 algorithm
720  range_exp := Ceval.evalExp(Util.getOption(range), EvalTarget.STATEMENT(source));
721  range_iter := RangeIterator.fromExp(range_exp);
722 
723  if RangeIterator.hasNext(range_iter) then
724  InstNode.EXP_NODE(exp = Expression.MUTABLE(exp = iter_exp)) := iterator;
725 
726  // Loop through each value in the iteration range.
727  while RangeIterator.hasNext(range_iter) loop
728  (range_iter, value) := RangeIterator.next(range_iter);
729  // Update the mutable expression with the iteration value and evaluate the statement.
730  Mutable.update(iter_exp, value);
731  ctrl := evaluateStatements(body);
732 
733  if ctrl <> FlowControl.NEXT then
734  if ctrl == FlowControl.BREAK then
735  ctrl := FlowControl.NEXT;
736  end if;
737 
738  break;
739  end if;
740 
741  i := i + 1;
742  if i > limit then
743  Error.addSourceMessage(Error.EVAL_LOOP_LIMIT_REACHED, {String(limit)},
744  ElementSource.getInfo(source));
745  fail();
746  end if;
747  end while;
748  end if;
749 end evaluateFor;
