387  * Parameters

389  * update_index == On input, ordering of update and external

789 void Generate_mesh(int N_elements, ML_GridAGX **meshp, int *N_update,
790  int *update[], int proc_config[])
791 {
792  int i, j, k, nprocs, nprocs_1d, mypid, mypid_y, mypid_z;
793  int icnt, *intlist, offset, nelmnt_1d, nelmnt_part_yz, estart_y;
794  ml_big_int *big_intlist;
795  int estart_z, nelmnt_local, nstart_y, nstart_z, index, eend_y, eend_z;
796  int nelmnt_2d, nnode_1d, nnode_local_y, nnode_local_z, nnode_local;
797  int nnode_2d, nnode_expand, nnode_x, nnode_y, nnode_z, ***cube;
798  int y_begin, z_begin;
799  double *coord, h;
800  ML_GridAGX *mesh;
801 
802  /* --------------------------------------------------------------- */
803  /* box in the processors */
804  /* --------------------------------------------------------------- */
805 
806  mypid = proc_config[AZ_node];
807  nprocs = proc_config[AZ_N_procs];
808 
809  nprocs_1d = (int) pow( (double) nprocs, 0.50001 );
810  if ( nprocs_1d * nprocs_1d != nprocs ) {
811  printf("Error: nprocs should be a square (%d).\n",nprocs_1d);
812  exit(1);
813  }
814  mypid_y = mypid % nprocs_1d;
815  mypid_z = mypid / nprocs_1d;
816 
817  /* --------------------------------------------------------------- */
818  /* process the grid */
819  /* --------------------------------------------------------------- */
820 
821  nelmnt_1d = (int) pow( (double) N_elements, 0.333334 );
822 
823  /* partition the elements along the yz-axis */
824 
825  nelmnt_part_yz = nelmnt_1d / nprocs_1d;
826  if (nelmnt_part_yz * nprocs_1d != nelmnt_1d) {
827  printf("Error: nelmnt_part not good. %d %d\n",nelmnt_part_yz,nelmnt_1d);
828  exit(-1);
829  }
830  nelmnt_local = nelmnt_part_yz * nelmnt_part_yz * nelmnt_1d;
831 
832  /* initialize the mesh */
833 
834  ML_GridAGX_Create( meshp );
835  mesh = (*meshp);
836  h = 1.0 / (double) nelmnt_1d;
837  ML_GridAGX_Set_Dimension(mesh, 3);
838  ML_GridAGX_Set_NElmnts(mesh, nelmnt_local, 8);
839 
840  /* generate the global element numbers of the local elements */
841 
842  ML_memory_alloc((void**) &big_intlist, nelmnt_local * sizeof(ml_big_int), "AP1");
843  estart_y = nelmnt_part_yz * mypid_y;
844  estart_z = nelmnt_part_yz * mypid_z;
845  eend_y = estart_y + nelmnt_part_yz - 1;
846  eend_z = estart_z + nelmnt_part_yz - 1;
847  icnt = 0;
848  nelmnt_2d = nelmnt_1d * nelmnt_1d;
849  for (k = estart_z; k <= eend_z; k++)
850  for (j = estart_y; j <= eend_y; j++)
851  for (i = 0; i < nelmnt_1d; i++)
852  big_intlist[icnt++] = i + j * nelmnt_1d + k * nelmnt_2d;
853  ML_GridAGX_Load_ElmntGlobalNum(mesh, nelmnt_local, big_intlist);
854  ML_memory_free( (void **) &big_intlist);
855 
856  /* generate the local computational grid */
857 
858  nnode_x = nelmnt_1d + 1;
859  nnode_y = nelmnt_part_yz + 1;
860  nnode_z = nelmnt_part_yz + 1;
861  ML_memory_alloc((void**) &cube, nnode_x * sizeof(int**), "AP2");
862  for ( i = 0; i < nnode_x; i++ )
863  ML_memory_alloc((void**) &(cube[i]), nnode_y * sizeof(int*), "AP3");
864  for ( i = 0; i < nnode_x; i++ )
865  for ( j = 0; j < nnode_y; j++ )
866  ML_memory_alloc((void**) &(cube[i][j]),nnode_z*sizeof(int),"AP4");
867  icnt = 0;
868  y_begin = 1;
869  z_begin = 1;
870  if ( estart_y == 0 ) y_begin--;
871  if ( estart_z == 0 ) z_begin--;
872  for ( k = z_begin; k < nnode_z; k++ )
873  for ( j = y_begin; j < nnode_y; j++ )
874  for ( i = 0; i < nnode_x; i++ ) cube[i][j][k] = icnt++;
875  if ( y_begin == 1 || z_begin == 1 ) {
876  for ( i = 0; i < nnode_x; i++ ) cube[i][0][0] = icnt++;
877  }
878  if ( y_begin == 1 ) {
879  for ( k = 1; k < nnode_z; k++ )
880  for ( i = 0; i < nnode_x; i++ )
881  cube[i][0][k] = icnt++;
882  }
883  if ( z_begin == 1 ) {
884  for ( j = 1; j < nnode_y; j++ )
885  for ( i = 0; i < nnode_x; i++ )
886  cube[i][j][0] = icnt++;
887  }
888 
889  /* generate the global node IDs */
890 
891  nnode_local_y = nnode_y - 1;
892  nnode_local_z = nnode_z - 1;
893  if (mypid_y == 0 ) nnode_local_y++;
894  if (mypid_z == 0 ) nnode_local_z++;
895  nnode_1d = nelmnt_1d + 1;
896  nnode_local = nnode_local_y * nnode_local_z * nnode_1d;
897  (*N_update) = nnode_local;
898  nnode_expand = nnode_y * nnode_z * nnode_x;
899  (*update) = (int*) malloc(nnode_expand * sizeof(int));
900 
901  nstart_y = mypid_y * nnode_local_y;
902  nstart_z = mypid_z * nnode_local_z;
903  nnode_2d = nnode_1d * nnode_1d;
904  for (k = 0; k < nnode_z; k++)
905  for (j = 0; j < nnode_y; j++)
906  for (i = 0; i < nnode_x; i++) {
907  index = cube[i][j][k];
908  offset = i + (j+nstart_y)*nnode_1d + (k+nstart_z)*nnode_2d;
909  (*update)[index] = offset;
910  }
911 
912  ML_GridAGX_Set_NVert(mesh, *N_update);
913  ML_GridAGX_Load_VertGlobalNum(mesh, nnode_expand, *update);
914 
915  /* generate coordinates of local nodes */
916 
917  ML_memory_alloc((void**) &coord, 3*nnode_expand*sizeof(double),"AP6");
918  for (k = 0; k < nnode_z; k++)
919  for (j = 0; j < nnode_y; j++)
920  for (i = 0; i < nnode_x; i++) {
921  index = cube[i][j][k];
922  offset = i + (j+nstart_y)*nnode_1d + (k+nstart_z)*nnode_2d;
923  coord[index*3] = (double) (h * i);
924  coord[index*3+1] = (double) (h * (j + nstart_y));
925  coord[index*3+2] = (double) (h * (k + nstart_z));
926  }
927  ML_GridAGX_Load_AllVertCoordinates(mesh, nnode_expand, coord);
928  ML_memory_free( (void **) &coord);
929 
930  /* finally construct the element-to-node lists */
931 
932  ML_memory_alloc((void**) &intlist, 8 * sizeof(int), "AP7");
933 
934  k = (eend_z - estart_z + 1) * (eend_y - estart_y + 1) * nelmnt_1d;
935  for (k = estart_z; k <= eend_z; k++)
936  for (j = estart_y; j <= eend_y; j++)
937  for (i = 0; i < nelmnt_1d; i++) {
938  intlist[0] = cube[i][j-estart_y][k-estart_z];
939  intlist[1] = cube[i+1][j-estart_y][k-estart_z];
940  intlist[2] = cube[i][j-estart_y+1][k-estart_z];
941  intlist[3] = cube[i+1][j-estart_y+1][k-estart_z];
942  intlist[4] = cube[i][j-estart_y][k-estart_z+1];
943  intlist[5] = cube[i+1][j-estart_y][k-estart_z+1];
944  intlist[6] = cube[i][j-estart_y+1][k-estart_z+1];
945  intlist[7] = cube[i+1][j-estart_y+1][k-estart_z+1];
946  ML_IntList_Load_Sublist(mesh->ele_nodes, 8, intlist);
947  }
948  ML_memory_free( (void **) &intlist);
949  for ( i = 0; i < nnode_x; i++ )
950  for ( j = 0; j < nnode_y; j++ )
951  ML_memory_free( (void**) &(cube[i][j]) );
952  for ( i = 0; i < nnode_x; i++ )
953  ML_memory_free( (void**) &(cube[i]));
954  ML_memory_free( (void**) &cube);
955 }
