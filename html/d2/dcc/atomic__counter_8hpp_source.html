94  // Atomic addition. Returns the old value.
95  inline integer_t add (integer_t increment_)
96  {
97  integer_t old_value;
98 
99 #if defined ZMQ_ATOMIC_COUNTER_WINDOWS
100  old_value = InterlockedExchangeAdd ((LONG*) &value, increment_);
101 #elif defined ZMQ_ATOMIC_COUNTER_INTRINSIC
102  old_value = __atomic_fetch_add(&value, increment_, __ATOMIC_ACQ_REL);
103 #elif defined ZMQ_ATOMIC_COUNTER_CXX11
104  old_value = value.fetch_add(increment_, std::memory_order_acq_rel);
105 #elif defined ZMQ_ATOMIC_COUNTER_ATOMIC_H
106  integer_t new_value = atomic_add_32_nv (&value, increment_);
107  old_value = new_value - increment_;
108 #elif defined ZMQ_ATOMIC_COUNTER_TILE
109  old_value = arch_atomic_add (&value, increment_);
110 #elif defined ZMQ_ATOMIC_COUNTER_X86
111  __asm__ volatile (
112  "lock; xadd %0, %1 \n\t"
113  : "=r" (old_value), "=m" (value)
114  : "0" (increment_), "m" (value)
115  : "cc", "memory");
116 #elif defined ZMQ_ATOMIC_COUNTER_ARM
117  integer_t flag, tmp;
118  __asm__ volatile (
119  " dmb sy\n\t"
120  "1: ldrex %0, [%5]\n\t"
121  " add %2, %0, %4\n\t"
122  " strex %1, %2, [%5]\n\t"
123  " teq %1, #0\n\t"
124  " bne 1b\n\t"
125  " dmb sy\n\t"
126  : "=&r"(old_value), "=&r"(flag), "=&r"(tmp), "+Qo"(value)
127  : "Ir"(increment_), "r"(&value)
128  : "cc");
129 #elif defined ZMQ_ATOMIC_COUNTER_MUTEX
130  sync.lock ();
131  old_value = value;
132  value += increment_;
133  sync.unlock ();
134 #else
135 #error atomic_counter is not implemented for this platform
136 #endif
137  return old_value;
138  }

140  // Atomic subtraction. Returns false if the counter drops to zero.
141  inline bool sub (integer_t decrement)
142  {
143 #if defined ZMQ_ATOMIC_COUNTER_WINDOWS
144  LONG delta = - ((LONG) decrement);
145  integer_t old = InterlockedExchangeAdd ((LONG*) &value, delta);
146  return old - decrement != 0;
147 #elif defined ZMQ_ATOMIC_COUNTER_INTRINSIC
148  integer_t nv = __atomic_sub_fetch(&value, decrement, __ATOMIC_ACQ_REL);
149  return nv != 0;
150 #elif defined ZMQ_ATOMIC_COUNTER_CXX11
151  integer_t old = value.fetch_sub(decrement, std::memory_order_acq_rel);
152  return old - decrement != 0;
153 #elif defined ZMQ_ATOMIC_COUNTER_ATOMIC_H
154  int32_t delta = - ((int32_t) decrement);
155  integer_t nv = atomic_add_32_nv (&value, delta);
156  return nv != 0;
157 #elif defined ZMQ_ATOMIC_COUNTER_TILE
158  int32_t delta = - ((int32_t) decrement);
159  integer_t nv = arch_atomic_add (&value, delta);
160  return nv != 0;
161 #elif defined ZMQ_ATOMIC_COUNTER_X86
162  integer_t oldval = -decrement;
163  volatile integer_t *val = &value;
164  __asm__ volatile ("lock; xaddl %0,%1"
165  : "=r" (oldval), "=m" (*val)
166  : "0" (oldval), "m" (*val)
167  : "cc", "memory");
168  return oldval != decrement;
169 #elif defined ZMQ_ATOMIC_COUNTER_ARM
170  integer_t old_value, flag, tmp;
171  __asm__ volatile (
172  " dmb sy\n\t"
173  "1: ldrex %0, [%5]\n\t"
174  " sub %2, %0, %4\n\t"
175  " strex %1, %2, [%5]\n\t"
176  " teq %1, #0\n\t"
177  " bne 1b\n\t"
178  " dmb sy\n\t"
179  : "=&r"(old_value), "=&r"(flag), "=&r"(tmp), "+Qo"(value)
180  : "Ir"(decrement), "r"(&value)
181  : "cc");
182  return old_value - decrement != 0;
183 #elif defined ZMQ_ATOMIC_COUNTER_MUTEX
184  sync.lock ();
185  value -= decrement;
186  bool result = value ? true : false;
187  sync.unlock ();
188  return result;
189 #else
190 #error atomic_counter is not implemented for this platform
191 #endif
192  }
