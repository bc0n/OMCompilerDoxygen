76  * Auxiliary data is assumed to be communicated by common blocks.

88 
89  /* Initialize all pointers to NULL */
90  uu_data = fval_data = jacdata = v1_data = v2_data = NULL;
91 
92  /* NOTE: The user-supplied routine should set ier to an
93  appropriate value, but we preset the value to zero
94  (meaning SUCCESS) so the user need only reset the

96  ier = 0;

104  eband = (J->s_mu) + mlower + 1;

109  uu_data, fval_data,

111  v1_data, v2_data, &ier);

113  return(ier);
