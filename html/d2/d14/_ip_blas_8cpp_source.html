66  Number IpBlasDdot(Index size, const Number *x, Index incX, const Number *y,
67  Index incY)
68  {
69  if (incX > 0 && incY > 0)
70  {
71  ipfint n=size, INCX=incX, INCY=incY;
72 
73  return F77_FUNC(ddot,DDOT)(&n, x, &INCX, y, &INCY);
74  }
75  else
76  {
77  Number s = 0.0;
78 
79  for (; size; --size, x += incX, y += incY)
80  s += *x * *y;
81 
82  return s;
83  }
84  }

111  void IpBlasDcopy(Index size, const Number *x, Index incX, Number *y, Index incY)
112  {
113  if (incX > 0)

115  ipfint N=size, INCX=incX, INCY=incY;
116 
117  F77_FUNC(dcopy,DCOPY)(&N, x, &INCX, y, &INCY);
118  }
119  else if (incY == 1)
120  {
121  for (; size; --size, ++y)
122  *y = *x;
123  }
124  else
125  {
