33  void
34  PDPerturbationHandler::RegisterOptions(SmartPtr<RegisteredOptions> roptions)
35  {
36  roptions->AddLowerBoundedNumberOption(
37  "max_hessian_perturbation",
38  "Maximum value of regularization parameter for handling negative curvature.",
39  0, true,
40  1e20,
41  "In order to guarantee that the search directions are indeed proper "
42  "descent directions, Ipopt requires that the inertia of the "
43  "(augmented) linear system for the step computation has the "
44  "correct number of negative and positive eigenvalues. The idea "
45  "is that this guides the algorithm away from maximizers and makes "
46  "Ipopt more likely converge to first order optimal points that "
47  "are minimizers. If the inertia is not correct, a multiple of the "
48  "identity matrix is added to the Hessian of the Lagrangian in the "
49  "augmented system. This parameter gives the maximum value of the "
50  "regularization parameter. If a regularization of that size is "
51  "not enough, the algorithm skips this iteration and goes to the "
52  "restoration phase. (This is delta_w^max in the implementation paper.)");
53  roptions->AddLowerBoundedNumberOption(
54  "min_hessian_perturbation",
55  "Smallest perturbation of the Hessian block.",
56  0., false, 1e-20,
57  "The size of the perturbation of the Hessian block is never selected "
58  "smaller than this value, unless no perturbation is necessary. (This "
59  "is delta_w^min in implementation paper.)");
60  roptions->AddLowerBoundedNumberOption(
61  "perturb_inc_fact_first",
62  "Increase factor for x-s perturbation for very first perturbation.",
63  1., true, 100.,
64  "The factor by which the perturbation is increased when a trial value "
65  "was not sufficient - this value is used for the computation of the "
66  "very first perturbation and allows a different value for for the first "
67  "perturbation than that used for the remaining perturbations. "
68  "(This is bar_kappa_w^+ in the implementation paper.)");
69  roptions->AddLowerBoundedNumberOption(
70  "perturb_inc_fact",
71  "Increase factor for x-s perturbation.",
72  1., true, 8.,
73  "The factor by which the perturbation is increased when a trial value "
74  "was not sufficient - this value is used for the computation of "
75  "all perturbations except for the first. "
76  "(This is kappa_w^+ in the implementation paper.)");
77  roptions->AddBoundedNumberOption(
78  "perturb_dec_fact",
79  "Decrease factor for x-s perturbation.",
80  0., true, 1., true, 1./3.,
81  "The factor by which the perturbation is decreased when a trial value "
82  "is deduced from the size of the most recent successful perturbation. "
83  "(This is kappa_w^- in the implementation paper.)");
84  roptions->AddLowerBoundedNumberOption(
85  "first_hessian_perturbation",
86  "Size of first x-s perturbation tried.",
87  0., true, 1e-4,
88  "The first value tried for the x-s perturbation in the inertia "
89  "correction scheme."
90  "(This is delta_0 in the implementation paper.)");
91  roptions->AddLowerBoundedNumberOption(
92  "jacobian_regularization_value",
93  "Size of the regularization for rank-deficient constraint Jacobians.",
94  0., false, 1e-8,
95  "(This is bar delta_c in the implementation paper.)");
96  roptions->AddLowerBoundedNumberOption(
97  "jacobian_regularization_exponent",
98  "Exponent for mu in the regularization for rank-deficient constraint Jacobians.",
99  0., false, 0.25,
100  "(This is kappa_c in the implementation paper.)");
101  roptions->AddStringOption2(
102  "perturb_always_cd",
103  "Active permanent perturbation of constraint linearization.",
104  "no",
105  "no", "perturbation only used when required",
106  "yes", "always use perturbation",
107  "This options makes the delta_c and delta_d perturbation be used for "
108  "the computation of every search direction. Usually, it is only used "
109  "when the iteration matrix is singular.");

148  bool
149  PDPerturbationHandler::ConsiderNewSystem(Number& delta_x, Number& delta_s,
150  Number& delta_c, Number& delta_d)
151  {
152  DBG_START_METH("PDPerturbationHandler::ConsiderNewSystem",dbg_verbosity);
153 
154  // Check if we can conclude that some components of the system are
155  // structurally degenerate
156  finalize_test();
157 
158  // Store the perturbation from the previous matrix
159  if (reset_last_) {
160  delta_x_last_ = delta_x_curr_;
161  delta_s_last_ = delta_s_curr_;
162  delta_c_last_ = delta_c_curr_;
163  delta_d_last_ = delta_d_curr_;
164  }
165  else {
166  if (delta_x_curr_ > 0.) {
167  delta_x_last_ = delta_x_curr_;
168  }
169  if (delta_s_curr_ > 0.) {
170  delta_s_last_ = delta_s_curr_;
171  }
172  if (delta_c_curr_ > 0.) {
173  delta_c_last_ = delta_c_curr_;
174  }
175  if (delta_d_curr_ > 0.) {
176  delta_d_last_ = delta_d_curr_;
177  }
178  }
179 
180  DBG_ASSERT((hess_degenerate_ != NOT_YET_DETERMINED ||
181  jac_degenerate_ != DEGENERATE) &&
182  (jac_degenerate_ != NOT_YET_DETERMINED ||
183  hess_degenerate_ != DEGENERATE));
184 
185  if (hess_degenerate_ == NOT_YET_DETERMINED ||
186  jac_degenerate_ == NOT_YET_DETERMINED) {
187  if (!perturb_always_cd_) {
188  test_status_ = TEST_DELTA_C_EQ_0_DELTA_X_EQ_0;
189  }
190  else {
191  test_status_ = TEST_DELTA_C_GT_0_DELTA_X_EQ_0;
192  }
193  }
194  else {
195  test_status_ = NO_TEST;
196  }
197 
198  if (jac_degenerate_ == DEGENERATE) {
199  delta_c = delta_c_curr_ = delta_cd();
200  IpData().Append_info_string("l");
201  }
202  else if (perturb_always_cd_) {
203  delta_c = delta_c_curr_ = delta_cd();
204  }
205  else {
206  delta_c = delta_c_curr_ = 0.;
207  }
208  delta_d = delta_d_curr_ = delta_c;
209 
210  if (hess_degenerate_ == DEGENERATE) {
211  delta_x_curr_ = 0.;
212  delta_s_curr_ = 0.;
213  bool retval = get_deltas_for_wrong_inertia(delta_x, delta_s,
214  delta_c, delta_d);
215  if (!retval) {
216  return false;
217  }
218  }
219  else {
220  delta_x = 0.;
221  delta_s = delta_x;
222  }
223 
224  delta_x_curr_ = delta_x;
225  delta_s_curr_ = delta_s;
226  delta_c_curr_ = delta_c;
227  delta_d_curr_ = delta_d;
228 
229  IpData().Set_info_regu_x(delta_x);
230 
231  get_deltas_for_wrong_inertia_called_ = false;
232 
233  return true;
234  }

236  bool
237  PDPerturbationHandler::PerturbForSingularity(
238  Number& delta_x, Number& delta_s,
239  Number& delta_c, Number& delta_d)
240  {
241  DBG_START_METH("PDPerturbationHandler::PerturbForSingularity",
242  dbg_verbosity);
243 
244  bool retval;
245 
246  // Check for structural degeneracy
247  if (hess_degenerate_ == NOT_YET_DETERMINED ||
248  jac_degenerate_ == NOT_YET_DETERMINED) {
249  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
250  "Degeneracy test for hess_degenerate_ = %d and jac_degenerate_ = %d\n test_status_ = %d\n",
251  hess_degenerate_, jac_degenerate_, test_status_);
252  switch (test_status_) {
253  case TEST_DELTA_C_EQ_0_DELTA_X_EQ_0:
254  DBG_ASSERT(delta_x_curr_ == 0. && delta_c_curr_ == 0.);
255  // in this case we haven't tried anything for this matrix yet
256  if (jac_degenerate_ == NOT_YET_DETERMINED) {
257  delta_d_curr_ = delta_c_curr_ = delta_cd();
258  test_status_ = TEST_DELTA_C_GT_0_DELTA_X_EQ_0;
259  }
260  else {
261  DBG_ASSERT(hess_degenerate_ == NOT_YET_DETERMINED);
262  retval = get_deltas_for_wrong_inertia(delta_x, delta_s,
263  delta_c, delta_d);
264  if (!retval) {
265  return false;
266  }
267  DBG_ASSERT(delta_c == 0. && delta_d == 0.);
268  test_status_ = TEST_DELTA_C_EQ_0_DELTA_X_GT_0;
269  }
270  break;
271  case TEST_DELTA_C_GT_0_DELTA_X_EQ_0:
272  DBG_ASSERT(delta_x_curr_ == 0. && delta_c_curr_ > 0.);
273  DBG_ASSERT(jac_degenerate_ == NOT_YET_DETERMINED);
274  if (!perturb_always_cd_) {
275  delta_d_curr_ = delta_c_curr_ = 0.;
276  retval = get_deltas_for_wrong_inertia(delta_x, delta_s,
277  delta_c, delta_d);
278  if (!retval) {
279  return false;
280  }
281  DBG_ASSERT(delta_c == 0. && delta_d == 0.);
282  test_status_ = TEST_DELTA_C_EQ_0_DELTA_X_GT_0;
283  }
284  else {
285  retval = get_deltas_for_wrong_inertia(delta_x, delta_s,
286  delta_c, delta_d);
287  if (!retval) {
288  return false;
289  }
290  DBG_ASSERT(delta_c > 0. && delta_d > 0.);
291  test_status_ = TEST_DELTA_C_GT_0_DELTA_X_GT_0;
292  }
293  break;
294  case TEST_DELTA_C_EQ_0_DELTA_X_GT_0:
295  DBG_ASSERT(delta_x_curr_ > 0. && delta_c_curr_ == 0.);
296  delta_d_curr_ = delta_c_curr_ = delta_cd();
297  retval = get_deltas_for_wrong_inertia(delta_x, delta_s,
298  delta_c, delta_d);
299  if (!retval) {
300  return false;
301  }
302  test_status_ = TEST_DELTA_C_GT_0_DELTA_X_GT_0;
303  break;
304  case TEST_DELTA_C_GT_0_DELTA_X_GT_0:
305  retval = get_deltas_for_wrong_inertia(delta_x, delta_s,
306  delta_c, delta_d);
307  if (!retval) {
308  return false;
309  }
310  break;
311  case NO_TEST:
312  DBG_ASSERT(false && "we should not get here.");
313  }
314  }
315  else {
316  if (delta_c_curr_ > 0.) { // || get_deltas_for_wrong_inertia_called_) {
317  // If we already used a perturbation for the constraints, we do
318  // the same thing as if we were encountering negative curvature
319  retval = get_deltas_for_wrong_inertia(delta_x, delta_s,
320  delta_c, delta_d);
321  if (!retval) {
322  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
323  "Can't get_deltas_for_wrong_inertia for delta_x_curr_ = %e and delta_c_curr_ = %e\n",
324  delta_x_curr_, delta_c_curr_);
325  return false;
326  }
327  }
328  else {
329  // Otherwise we now perturb the lower right corner
330  delta_d_curr_ = delta_c_curr_ = delta_cd();
331 
332  // ToDo - also perturb Hessian?
333  IpData().Append_info_string("L");
334  }
335  }
336 
337  delta_x = delta_x_curr_;
338  delta_s = delta_s_curr_;
339  delta_c = delta_c_curr_;
340  delta_d = delta_d_curr_;
341 
342  IpData().Set_info_regu_x(delta_x);
343 
344  return true;
345  }

347  bool
348  PDPerturbationHandler::get_deltas_for_wrong_inertia(
349  Number& delta_x, Number& delta_s,
350  Number& delta_c, Number& delta_d)
351  {
352  if (delta_x_curr_ == 0.) {
353  if (delta_x_last_ == 0.) {
354  delta_x_curr_ = delta_xs_init_;
355  }
356  else {
357  delta_x_curr_ = Max(delta_xs_min_,
358  delta_x_last_*delta_xs_dec_fact_);
359  }
360  }
361  else {
362  if (delta_x_last_ == 0. || 1e5*delta_x_last_<delta_x_curr_) {
363  delta_x_curr_ = delta_xs_first_inc_fact_*delta_x_curr_;
364  }
365  else {
366  delta_x_curr_ = delta_xs_inc_fact_*delta_x_curr_;
367  }
368  }
369  if (delta_x_curr_ > delta_xs_max_) {
370  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA,
371  "delta_x perturbation is becoming too large: %e\n",
372  delta_x_curr_);
373  delta_x_last_ = 0.;
374  delta_s_last_ = 0.;
375  IpData().Append_info_string("dx");
376  return false;
377  }
378 
379  delta_s_curr_ = delta_x_curr_;
380 
381  delta_x = delta_x_curr_;
382  delta_s = delta_s_curr_;
383  delta_c = delta_c_curr_;
384  delta_d = delta_d_curr_;
385 
386  IpData().Set_info_regu_x(delta_x);
387 
388  get_deltas_for_wrong_inertia_called_ = true;
389 
390  return true;
391  }

393  bool
394  PDPerturbationHandler::PerturbForWrongInertia(
395  Number& delta_x, Number& delta_s,
396  Number& delta_c, Number& delta_d)
397  {
398  DBG_START_METH("PDPerturbationHandler::PerturbForWrongInertia",
399  dbg_verbosity);
400 
401  // Check if we can conclude that components of the system are
402  // structurally degenerate (we only get here if the most recent
403  // perturbation for a test did not result in a singular system)
404  finalize_test();
405 
406  bool retval = get_deltas_for_wrong_inertia(delta_x, delta_s, delta_c, delta_d);
407  if (!retval && delta_c==0.) {
408  DBG_ASSERT(delta_d == 0.);
409  delta_c_curr_ = delta_cd();
410  delta_d_curr_ = delta_c_curr_;
411  delta_x_curr_ = 0.;
412  delta_s_curr_ = 0.;
413  test_status_ = NO_TEST;
414  if (hess_degenerate_ == DEGENERATE) {
415  hess_degenerate_ = NOT_YET_DETERMINED;
416  }
417  retval = get_deltas_for_wrong_inertia(delta_x, delta_s, delta_c, delta_d);
418  }
419  return retval;
420  }

439  void
440  PDPerturbationHandler::finalize_test()
441  {
442  switch (test_status_) {
443  case NO_TEST:
444  return;
445  case TEST_DELTA_C_EQ_0_DELTA_X_EQ_0:
446  if (hess_degenerate_ == NOT_YET_DETERMINED &&
447  jac_degenerate_ == NOT_YET_DETERMINED) {
448  hess_degenerate_ = NOT_DEGENERATE;
449  jac_degenerate_ = NOT_DEGENERATE;
450  IpData().Append_info_string("Nhj ");
451  }
452  else if (hess_degenerate_ == NOT_YET_DETERMINED) {
453  hess_degenerate_ = NOT_DEGENERATE;
454  IpData().Append_info_string("Nh ");
455  }
456  else if (jac_degenerate_ == NOT_YET_DETERMINED) {
457  jac_degenerate_ = NOT_DEGENERATE;
458  IpData().Append_info_string("Nj ");
459  }
460  break;
461  case TEST_DELTA_C_GT_0_DELTA_X_EQ_0:
462  if (hess_degenerate_ == NOT_YET_DETERMINED) {
463  hess_degenerate_ = NOT_DEGENERATE;
464  IpData().Append_info_string("Nh ");
465  }
466  if (jac_degenerate_ == NOT_YET_DETERMINED) {
467  degen_iters_++;
468  if (degen_iters_ >= degen_iters_max_) {
469  jac_degenerate_ = DEGENERATE;
470  IpData().Append_info_string("Dj ");
471  }
472  IpData().Append_info_string("L");
473  }
474  break;
475  case TEST_DELTA_C_EQ_0_DELTA_X_GT_0:
476  if (jac_degenerate_ == NOT_YET_DETERMINED) {
477  jac_degenerate_ = NOT_DEGENERATE;
478  IpData().Append_info_string("Nj ");
479  }
480  if (hess_degenerate_ == NOT_YET_DETERMINED) {
481  degen_iters_++;
482  if (degen_iters_ >= degen_iters_max_) {
483  hess_degenerate_ = DEGENERATE;
484  IpData().Append_info_string("Dh ");
485  }
486  }
487  break;
488  case TEST_DELTA_C_GT_0_DELTA_X_GT_0:
489  degen_iters_++;
490  if (degen_iters_ >= degen_iters_max_) {
491  hess_degenerate_ = DEGENERATE;
492  jac_degenerate_ = DEGENERATE;
493  IpData().Append_info_string("Dhj ");
494  }
495  IpData().Append_info_string("L");
496  break;
497  }
498  }
