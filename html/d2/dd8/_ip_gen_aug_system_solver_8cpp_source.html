71  ESymSolverStatus GenAugSystemSolver::MultiSolve(
72  const SymMatrix* W,
73  double W_factor,
74  const Vector* D_x,
75  double delta_x,
76  const Vector* D_s,
77  double delta_s,
78  const Matrix* J_c,
79  const Vector* D_c,
80  double delta_c,
81  const Matrix* J_d,
82  const Vector* D_d,
83  double delta_d,
84  std::vector<SmartPtr<const Vector> >& rhs_xV,
85  std::vector<SmartPtr<const Vector> >& rhs_sV,
86  std::vector<SmartPtr<const Vector> >& rhs_cV,
87  std::vector<SmartPtr<const Vector> >& rhs_dV,
88  std::vector<SmartPtr<Vector> >& sol_xV,
89  std::vector<SmartPtr<Vector> >& sol_sV,
90  std::vector<SmartPtr<Vector> >& sol_cV,
91  std::vector<SmartPtr<Vector> >& sol_dV,
92  bool check_NegEVals,
93  Index numberOfNegEVals)
94  {
95  DBG_START_METH("GenAugSystemSolver::MultiSolve",dbg_verbosity);
96  DBG_ASSERT(J_c && J_d && "Currently, you MUST specify J_c and J_d in the augmented system");
97 
98  DBG_ASSERT(W_factor == 0.0 || W_factor == 1.0);
99 
100  const Index nrhs = (Index)rhs_xV.size();
101  DBG_ASSERT(nrhs>0);
102  DBG_ASSERT(nrhs==(Index)rhs_sV.size());
103  DBG_ASSERT(nrhs==(Index)rhs_cV.size());
104  DBG_ASSERT(nrhs==(Index)rhs_dV.size());
105  DBG_ASSERT(nrhs==(Index)sol_xV.size());
106  DBG_ASSERT(nrhs==(Index)sol_sV.size());
107  DBG_ASSERT(nrhs==(Index)sol_cV.size());
108  DBG_ASSERT(nrhs==(Index)sol_dV.size());
109 
110  // Check if the input data has changed:
111  bool new_matrix =
112  AugmentedSystemChanged(W, W_factor, D_x, delta_x, D_s, delta_s,
113  *J_c, D_c, delta_c, *J_d, D_d, delta_d);
114 
115  // Get the individual arrays to be given to the
116  // GenKKTSolverInterface
117  const Index n_x = rhs_xV[0]->Dim();
118  const Index n_c = rhs_cV[0]->Dim();
119  const Index n_d = rhs_dV[0]->Dim();
120  const Number* dx_vals=NULL;
121  if (D_x) {
122  const DenseVector* dD_x = dynamic_cast<const DenseVector*> (D_x);
123  if (dD_x && !dD_x->IsHomogeneous()) {
124  dx_vals = dD_x->Values();
125  }
126  else if (D_x->GetTag() != d_x_tag_) {
127  delete [] dx_vals_copy_;
128  dx_vals_copy_ = new Number[n_x];
129  TripletHelper::FillValuesFromVector(n_x, *D_x, dx_vals_copy_);
130  dx_vals = dx_vals_copy_;
131  }
132  }
133  const Number* ds_vals=NULL;
134  if (D_s) {
135  const DenseVector* dD_s = dynamic_cast<const DenseVector*> (D_s);
136  if (dD_s && !dD_s->IsHomogeneous()) {
137  ds_vals = dD_s->Values();
138  }
139  else if (D_s->GetTag() != d_s_tag_) {
140  delete [] ds_vals_copy_;
141  ds_vals_copy_ = new Number[n_d];
142  TripletHelper::FillValuesFromVector(n_d, *D_s, ds_vals_copy_);
143  ds_vals = ds_vals_copy_;
144  }
145  }
146  const Number* dc_vals=NULL;
147  if (D_c) {
148  const DenseVector* dD_c = dynamic_cast<const DenseVector*> (D_c);
149  if (dD_c && !dD_c->IsHomogeneous()) {
150  dc_vals = dD_c->Values();
151  }
152  else if (D_c->GetTag() != d_c_tag_) {
153  delete [] dc_vals_copy_;
154  dc_vals_copy_ = new Number[n_c];
155  TripletHelper::FillValuesFromVector(n_c, *D_c, dc_vals_copy_);
156  dc_vals = dc_vals_copy_;
157  }
158  }
159  const Number* dd_vals=NULL;
160  if (D_d) {
161  const DenseVector* dD_d = dynamic_cast<const DenseVector*> (D_d);
162  if (dD_d && !dD_d->IsHomogeneous()) {
163  dd_vals = dD_d->Values();
164  }
165  else if (D_d->GetTag() != d_d_tag_) {
166  delete [] dd_vals_copy_;
167  dd_vals_copy_ = new Number[n_d];
168  TripletHelper::FillValuesFromVector(n_d, *D_d, dd_vals_copy_);
169  dd_vals = dd_vals_copy_;
170  }
171  }
172 
173  const Index dim = n_x+n_d+n_c+n_d;
174  Number* rhssol = new Number[nrhs*dim];
175  for (Index irhs=0; irhs<nrhs; irhs++) {
176  // TODO: make order an option
177  TripletHelper::FillValuesFromVector(n_x, *rhs_xV[irhs],
178  &rhssol[irhs*dim]);
179  TripletHelper::FillValuesFromVector(n_c, *rhs_cV[irhs],
180  &rhssol[irhs*dim+n_x]);
181  TripletHelper::FillValuesFromVector(n_d, *rhs_dV[irhs],
182  &rhssol[irhs*dim+n_x+n_c]);
183  TripletHelper::FillValuesFromVector(n_d, *rhs_sV[irhs],
184  &rhssol[irhs*dim+n_x+n_c+n_d]);
185  }
186 
187  bool done = false;
188  ESymSolverStatus retval = SYMSOLVER_FATAL_ERROR;
189  const SymMatrix* Wgive = NULL;
190  if (W && W_factor==1.0) {
191  Wgive = W;
192  }
193  while (!done) {
194  retval = solver_interface_->MultiSolve(new_matrix, n_x, n_c, n_d,
195  Wgive, J_c, J_d,
196  dx_vals, ds_vals,
197  dc_vals, dd_vals,
198  delta_x, delta_s,
199  delta_c, delta_d,
200  nrhs, rhssol,
201  check_NegEVals, numberOfNegEVals);
202  if (retval==SYMSOLVER_CALL_AGAIN) {
203  DBG_PRINT((1, "Solver interface asks to be called again. Don't really se why...?\n"));
204  }
205  else {
206  done = true;
207  }
208  }
209 
210  // Copy the values back into the vectors
211  if (retval==SYMSOLVER_SUCCESS) {
212  for (Index irhs=0; irhs<nrhs; irhs++) {
213  TripletHelper::PutValuesInVector(n_x, &rhssol[irhs*dim],
214  *sol_xV[irhs]);
215  TripletHelper::PutValuesInVector(n_c, &rhssol[irhs*dim+n_x],
216  *sol_cV[irhs]);
217  TripletHelper::PutValuesInVector(n_d, &rhssol[irhs*dim+n_x+n_c],
218  *sol_dV[irhs]);
219  TripletHelper::PutValuesInVector(n_d, &rhssol[irhs*dim+n_x+n_c+n_d],
220  *sol_sV[irhs]);
221  }
222  }
223  else if (retval==SYMSOLVER_FATAL_ERROR) {
224  delete [] rhssol;
225  THROW_EXCEPTION(FATAL_ERROR_IN_LINEAR_SOLVER,"A fatal error occured in the linear solver.");
226  }
227  else {
228  Jnlst().Printf(J_DETAILED, J_LINEAR_ALGEBRA, "Factorization failed with retval = %d\n", retval);
229  }
230 
231  delete [] rhssol;
232 
233  return retval;
234  }
