91 static inline void *mmc_mk_box(mmc_sint_t _slots, mmc_uint_t ctor, ...)
92 {
93  mmc_sint_t i;
94  va_list argp;
95 
96  struct mmc_struct *p = (struct mmc_struct *) mmc_alloc_words(_slots+1);
97  p->header = MMC_STRUCTHDR(_slots, ctor);
98  va_start(argp, ctor);
99  for (i = 0; i < _slots; i++) {
100  p->data[i] = va_arg(argp, void*);
101  }
102  va_end(argp);
103 #ifdef MMC_MK_DEBUG
104  fprintf(stderr, "BOXNN slots:%d ctor: %u\n", _slots, ctor); fflush(NULL);
105 #endif
106  return MMC_TAGPTR(p);
107 }

187 /*
188  * Generated (Meta)Records should access a static, constant value of
189  * the record_description structure. This means the additional cost
190  * of including the description is 1 word of memory and O(1) time.
191  * When sending structures between the compiler and generated files,
192  * the descriptions will be duplicated, and cost additional memory
193  * for each and every Values.Value copied. /sjoelund 2009-05-20
194  */
