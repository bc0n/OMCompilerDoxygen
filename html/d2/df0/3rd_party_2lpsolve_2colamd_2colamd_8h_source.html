170 /* ========================================================================== */
171 /* === Colamd recommended memory size ======================================= */
172 /* ========================================================================== */
173 
174 /*
175  The recommended length Alen of the array A passed to colamd is given by
176  the COLAMD_RECOMMENDED (nnz, n_row, n_col) macro. It returns -1 if any
177  argument is negative. 2*nnz space is required for the row and column
178  indices of the matrix. COLAMD_C (n_col) + COLAMD_R (n_row) space is
179  required for the Col and Row arrays, respectively, which are internal to
180  colamd. An additional n_col space is the minimal amount of "elbow room",
181  and nnz/5 more space is recommended for run time efficiency.
182 
183  This macro is not needed when using symamd.
184 */
185 
186 #define COLAMD_C(n_col) (((n_col) + 1) * sizeof (Colamd_Col) / sizeof (int))
187 #define COLAMD_R(n_row) (((n_row) + 1) * sizeof (Colamd_Row) / sizeof (int))
188 
189 #define COLAMD_RECOMMENDED(nnz, n_row, n_col) \
190 ( \
191 ((nnz) < 0 || (n_row) < 0 || (n_col) < 0) \
192 ? \
193  (-1) \
194 : \
195  (int) (2 * (nnz) + COLAMD_C (n_col) + COLAMD_R (n_row) + (n_col) + ((nnz) / 5)) \
196 )
