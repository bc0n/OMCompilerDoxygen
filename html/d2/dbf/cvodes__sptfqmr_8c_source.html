104 /*
105  * -----------------------------------------------------------------
106  * Function : CVSptfqmr
107  * -----------------------------------------------------------------
108  * This routine initializes the memory record and sets various function
109  * fields specific to the Sptfqmr linear solver module. CVSptfqmr first
110  * calls the existing lfree routine if this is not NULL. It then sets
111  * the cv_linit, cv_lsetup, cv_lsolve, cv_lfree fields in (*cvode_mem)
112  * to be CVSptfqmrInit, CVSptfqmrSetup, CVSptfqmrSolve, and CVSptfqmrFree,
113  * respectively. It allocates memory for a structure of type
114  * CVSpilsMemRec and sets the cv_lmem field in (*cvode_mem) to the
115  * address of this structure. It sets setupNonNull in (*cvode_mem),
116  * and sets various fields in the CVSpilsMemRec structure.
117  * Finally, CVSptfqmr allocates memory for ytemp and x, and calls
118  * SptfqmrMalloc to allocate memory for the Sptfqmr solver.
119  * -----------------------------------------------------------------
120  */
121 
122 int CVSptfqmr(void *cvode_mem, int pretype, int maxl)
123 {
124  CVodeMem cv_mem;
125  CVSpilsMem cvspils_mem;
126  SptfqmrMem sptfqmr_mem;
127  int mxl;
128 
129  /* Return immediately if cvode_mem is NULL */
130  if (cvode_mem == NULL) {
131  cvProcessError(NULL, CVSPILS_MEM_NULL, "CVSPTFQMR", "CVSptfqmr", MSGS_CVMEM_NULL);
132  return(CVSPILS_MEM_NULL);
133  }
134  cv_mem = (CVodeMem) cvode_mem;
135 
136  /* Check if N_VDotProd is present */
137  if (vec_tmpl->ops->nvdotprod == NULL) {
138  cvProcessError(cv_mem, CVSPILS_ILL_INPUT, "CVSPTFQMR", "CVSptfqmr", MSGS_BAD_NVECTOR);
139  return(CVSPILS_ILL_INPUT);
140  }
141 
142  if (lfree != NULL) lfree(cv_mem);
143 
144  /* Set four main function fields in cv_mem */
145  linit = CVSptfqmrInit;
146  lsetup = CVSptfqmrSetup;
147  lsolve = CVSptfqmrSolve;
148  lfree = CVSptfqmrFree;

151  cvspils_mem = NULL;
152  cvspils_mem = (CVSpilsMem) malloc(sizeof(struct CVSpilsMemRec));
153  if (cvspils_mem == NULL) {
154  cvProcessError(cv_mem, CVSPILS_MEM_FAIL, "CVSPTFQMR", "CVSptfqmr", MSGS_MEM_FAIL);
155  return(CVSPILS_MEM_FAIL);
156  }
157

159  cvspils_mem->s_type = SPILS_SPTFQMR;
160 
161  /* Set Sptfqmr parameters that have been passed in call sequence */
162  cvspils_mem->s_pretype = pretype;
163  mxl = cvspils_mem->s_maxl = (maxl <= 0) ? CVSPILS_MAXL : maxl;
164 
165  /* Set defaults for Jacobian-related fileds */
166  jtimesDQ = TRUE;
167  jtimes = NULL;
168  j_data = NULL;
169 
170  /* Set defaults for preconditioner-related fields */
171  cvspils_mem->s_pset = NULL;
172  cvspils_mem->s_psolve = NULL;
173  cvspils_mem->s_pfree = NULL;
174  cvspils_mem->s_P_data = cv_mem->cv_user_data;
175 
176  /* Set default values for the rest of the Sptfqmr parameters */
177  cvspils_mem->s_eplifac = CVSPILS_EPLIN;
178 
179  cvspils_mem->s_last_flag = CVSPILS_SUCCESS;
180 
181  setupNonNull = FALSE;
182 
183  /* Check for legal pretype */
184  if ((pretype != PREC_NONE) && (pretype != PREC_LEFT) &&
185  (pretype != PREC_RIGHT) && (pretype != PREC_BOTH)) {
186  cvProcessError(cv_mem, CVSPILS_ILL_INPUT, "CVSPTFQMR", "CVSptfqmr", MSGS_BAD_PRETYPE);
187  free(cvspils_mem); cvspils_mem = NULL;
188  return(CVSPILS_ILL_INPUT);
189  }
190 
191  /* Allocate memory for ytemp and x */
192  ytemp = N_VClone(vec_tmpl);
193  if (ytemp == NULL) {
194  cvProcessError(cv_mem, CVSPILS_MEM_FAIL, "CVSPTFQMR", "CVSptfqmr", MSGS_MEM_FAIL);
195  free(cvspils_mem); cvspils_mem = NULL;
196  return(CVSPILS_MEM_FAIL);
197  }
198  x = N_VClone(vec_tmpl);
199  if (x == NULL) {
200  cvProcessError(cv_mem, CVSPILS_MEM_FAIL, "CVSPTFQMR", "CVSptfqmr", MSGS_MEM_FAIL);
201  N_VDestroy(ytemp);
202  free(cvspils_mem); cvspils_mem = NULL;
203  return(CVSPILS_MEM_FAIL);
204  }
205 
206  /* Compute sqrtN from a dot product */
207  N_VConst(ONE, ytemp);
208  sqrtN = SUNRsqrt(N_VDotProd(ytemp, ytemp));
209 
210  /* Call SptfqmrMalloc to allocate workspace for Sptfqmr */
211  sptfqmr_mem = NULL;
212  sptfqmr_mem = SptfqmrMalloc(mxl, vec_tmpl);
213  if (sptfqmr_mem == NULL) {
214  cvProcessError(cv_mem, CVSPILS_MEM_FAIL, "CVSPTFQMR", "CVSptfqmr", MSGS_MEM_FAIL);

217  free(cvspils_mem); cvspils_mem = NULL;
218  return(CVSPILS_MEM_FAIL);
219  }
220 
221  /* Attach SPTFQMR memory to spils memory structure */
222  spils_mem = (void *) sptfqmr_mem;
223 
224  /* Attach linear solver memory to integrator memory */
225  lmem = cvspils_mem;
226 
227  return(CVSPILS_SUCCESS);
228 }

287 /*
288  * -----------------------------------------------------------------
289  * Function : CVSptfqmrSetup
290  * -----------------------------------------------------------------
291  * This routine does the setup operations for the Sptfqmr linear solver.
292  * It makes a decision as to whether or not to signal for reevaluation
293  * of Jacobian data in the pset routine, based on various state
294  * variables, then it calls pset. If we signal for reevaluation,
295  * then we reset jcur = *jcurPtr to TRUE, regardless of the pset output.
296  * In any case, if jcur == TRUE, we increment npe and save nst in nstlpre.
297  * -----------------------------------------------------------------
298  */
299 
300 static int CVSptfqmrSetup(CVodeMem cv_mem, int convfail, N_Vector ypred,
301  N_Vector fpred, booleantype *jcurPtr, N_Vector vtemp1,
302  N_Vector vtemp2, N_Vector vtemp3)
303 {
304  booleantype jbad, jok;
305  realtype dgamma;
306  int retval;
307  CVSpilsMem cvspils_mem;
308 
309  cvspils_mem = (CVSpilsMem) lmem;
310 
311  /* Use nst, gamma/gammap, and convfail to set J eval. flag jok */
312  dgamma = SUNRabs((gamma/gammap) - ONE);
313  jbad = (nst == 0) || (nst > nstlpre + CVSPILS_MSBPRE) ||
314  ((convfail == CV_FAIL_BAD_J) && (dgamma < CVSPILS_DGMAX)) ||
315  (convfail == CV_FAIL_OTHER);
316  *jcurPtr = jbad;
317  jok = !jbad;
318 
319  /* Call pset routine and possibly reset jcur */
320  retval = pset(tn, ypred, fpred, jok, jcurPtr, gamma, P_data,
321  vtemp1, vtemp2, vtemp3);
322  if (retval < 0) {
323  cvProcessError(cv_mem, SPTFQMR_PSET_FAIL_UNREC, "CVSPTFQMR", "CVSptfqmrSetup", MSGS_PSET_FAILED);
324  last_flag = SPTFQMR_PSET_FAIL_UNREC;
325  }
326  if (retval > 0) {
327  last_flag = SPTFQMR_PSET_FAIL_REC;
328  }
329 
330  if (jbad) *jcurPtr = TRUE;
331 
332  /* If jcur = TRUE, increment npe and save nst value */
333  if (*jcurPtr) {
334  npe++;
335  nstlpre = nst;
336  }
337 
338  last_flag = SPTFQMR_SUCCESS;
339 
340  /* Return the same value that pset returned */
341  return(retval);
342 }

344 /*
345  * -----------------------------------------------------------------
346  * Function : CVSptfqmrSolve
347  * -----------------------------------------------------------------
348  * This routine handles the call to the generic solver SptfqmrSolve
349  * for the solution of the linear system Ax = b with the SPTFQMR method.
350  * The solution x is returned in the vector b.
351  *
352  * If the WRMS norm of b is small, we return x = b (if this is the first
353  * Newton iteration) or x = 0 (if a later Newton iteration).
354  *
355  * Otherwise, we set the tolerance parameter and initial guess (x = 0),
356  * call SptfqmrSolve, and copy the solution x into b. The x-scaling and
357  * b-scaling arrays are both equal to weight.
358  *
359  * The counters nli, nps, and ncfl are incremented, and the return value
360  * is set according to the success of SptfqmrSolve. The success flag is
361  * returned if SptfqmrSolve converged, or if this is the first Newton
362  * iteration and the residual norm was reduced below its initial value.
363  * -----------------------------------------------------------------
364  */
365 
366 static int CVSptfqmrSolve(CVodeMem cv_mem, N_Vector b, N_Vector weight,
367  N_Vector ynow, N_Vector fnow)
368 {
369  realtype bnorm, res_norm;
370  CVSpilsMem cvspils_mem;
371  SptfqmrMem sptfqmr_mem;
372  int nli_inc, nps_inc, retval;
373 
374  cvspils_mem = (CVSpilsMem) lmem;
375 
376  sptfqmr_mem = (SptfqmrMem) spils_mem;
377 
378  /* Test norm(b); if small, return x = 0 or x = b */
379  deltar = eplifac * tq[4];
380 
381  bnorm = N_VWrmsNorm(b, weight);
382  if (bnorm <= deltar) {
383  if (mnewt > 0) N_VConst(ZERO, b);
384  return(0);
385  }
386 
387  /* Set vectors ycur and fcur for use by the Atimes and Psolve routines */
388  ycur = ynow;
389  fcur = fnow;
390 
391  /* Set inputs delta and initial guess x = 0 to SptfqmrSolve */
392  delta = deltar * sqrtN;
393  N_VConst(ZERO, x);
394 
395  /* Call SptfqmrSolve and copy x to b */
396  retval = SptfqmrSolve(sptfqmr_mem, cv_mem, x, b, pretype, delta,
397  cv_mem, weight, weight, CVSpilsAtimes, CVSpilsPSolve,
398  &res_norm, &nli_inc, &nps_inc);
399 
400  N_VScale(ONE, x, b);
401 
402  /* Increment counters nli, nps, and ncfl */
403  nli += nli_inc;
404  nps += nps_inc;
405  if (retval != SPTFQMR_SUCCESS) ncfl++;
406 
407  /* Interpret return value from SpgmrSolve */
408 
409  last_flag = retval;
410 
411  switch(retval) {
412 
413  case SPTFQMR_SUCCESS:
414  return(0);
415  break;
416  case SPTFQMR_RES_REDUCED:
417  if (mnewt == 0) return(0);
418  else return(1);
419  break;
420  case SPTFQMR_CONV_FAIL:
421  return(1);
422  break;
423  case SPTFQMR_PSOLVE_FAIL_REC:
424  return(1);
425  break;
426  case SPTFQMR_ATIMES_FAIL_REC:
427  return(1);
428  break;
429  case SPTFQMR_MEM_NULL:
430  return(-1);
431  break;
432  case SPTFQMR_ATIMES_FAIL_UNREC:
433  cvProcessError(cv_mem, SPTFQMR_ATIMES_FAIL_UNREC, "CVSPTFQMR", "CVSptfqmrSolve", MSGS_JTIMES_FAILED);
434  return(-1);
435  break;
436  case SPTFQMR_PSOLVE_FAIL_UNREC:
437  cvProcessError(cv_mem, SPTFQMR_PSOLVE_FAIL_UNREC, "CVSPTFQMR", "CVSptfqmrSolve", MSGS_PSOLVE_FAILED);
438  return(-1);
439  break;
440  }
441 
442  return(0);
443 
444 }
