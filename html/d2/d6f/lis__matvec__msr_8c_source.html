56  n = A->n;

58  {
59  n = A->n;
60  #ifdef _OPENMP

64  {
65  t = A->D->value[i] * x[i];
66  js = A->L->index[i];

103 void lis_matvect_msr(LIS_MATRIX A, LIS_SCALAR x[], LIS_SCALAR y[])
104 {
105  LIS_INT i,j,js,je,jj;
106  LIS_INT n,np;
107  LIS_SCALAR t;
108  #ifdef _OPENMP
109  LIS_INT k,is,ie,nprocs;
110  LIS_SCALAR *w;
111  #endif
112 
113  n = A->n;
114  np = A->np;
115 
116  if( A->is_splited )
117  {
118  for(i=0; i<n; i++)
119  {
120  y[i] = A->D->value[i] * x[i];
121  }
122  for(i=0; i<n; i++)
123  {
124  t = x[i];
125  js = A->L->index[i];
126  je = A->L->index[i+1];
127  for(j=js;j<je;j++)
128  {
129  jj = A->L->index[j];
130  y[jj] += A->L->value[j] * t;
131  }
132  js = A->U->index[i];
133  je = A->U->index[i+1];
134  for(j=js;j<je;j++)
135  {
136  jj = A->U->index[j];
137  y[jj] += A->U->value[j] * t;
138  }
139  }
140  }
141  else
142  {
143  #ifdef _OPENMP
144  nprocs = omp_get_max_threads();
145  w = (LIS_SCALAR *)lis_malloc( nprocs*np*sizeof(LIS_SCALAR),"lis_matvect_msr::w" );
146  #pragma omp parallel private(i,j,js,je,t,jj,k)
147  {
148  k = omp_get_thread_num();
149  #pragma omp for
150  for(j=0;j<nprocs;j++)
151  {
152  memset( &w[j*np], 0, np*sizeof(LIS_SCALAR) );
153  }
154  #pragma omp for
155  for(i=0; i<n; i++)
156  {
157  js = A->index[i];
158  je = A->index[i+1];
159  t = x[i];
160  for(j=js;j<je;j++)
161  {
162  jj = k*np+A->index[j];
163  w[jj] += A->value[j] * t;
164  }
165  w[k*np+i] += A->value[i] * x[i];
166  }
167  #pragma omp for
168  for(i=0;i<np;i++)
169  {
170  t = 0.0;
171  for(j=0;j<nprocs;j++)
172  {
173  t += w[j*np+i];
174  }
175  y[i] = t;
176  }
177  }
178  lis_free(w);
179  #else
180  for(i=0; i<n; i++)
181  {
182  y[i] = A->value[i] * x[i];
183  }
184  for(i=0; i<n; i++)
185  {
186  t = x[i];
187  js = A->index[i];
188  je = A->index[i+1];
189  for(j=js;j<je;j++)
190  {
191  jj = A->index[j];
192  y[jj] += A->value[j] * t;
193  }
194  }
195  #endif
196  }
197 }
