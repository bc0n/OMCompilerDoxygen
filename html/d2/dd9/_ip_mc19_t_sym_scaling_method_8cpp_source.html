64  DBG_PRINT((2, "%5d A[%5d,%5d] = %23.15e\n", i, airn[i], ajcn[i], a[i]));

142  // Get the symmetric scaling factors as mean of the general ones
143  // If some of the entries in A2 are too large, the scaling factors
144  // are NaN. Here, we check for this and return no scaling factors

164  i, scaling_factors[i]));
165  }
166  }
167

173  delete[] AJCN2;
174

176  }
