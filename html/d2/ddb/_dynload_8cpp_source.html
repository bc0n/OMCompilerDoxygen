59 static int value_to_type_desc(void *value, type_description *desc)
60 {
61  init_type_description(desc);
62 
63  switch (MMC_HDRCTOR(MMC_GETHDR(value))) {
64  case Values__INTEGER_3dBOX1: {
65  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET];
66  desc->type = TYPE_DESC_INT;
67  desc->data.integer = MMC_UNTAGFIXNUM(data);
68  }; break;
69  case Values__ENUM_5fLITERAL_3dBOX2: {
70  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET+1];
71  desc->type = TYPE_DESC_INT;
72  desc->data.integer = MMC_UNTAGFIXNUM(data);
73  }; break;
74  case Values__REAL_3dBOX1: {
75  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET];
76  desc->type = TYPE_DESC_REAL;
77  desc->data.real = mmc_prim_get_real(data);
78  }; break;
79  case Values__BOOL_3dBOX1: {
80  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET];
81  desc->type = TYPE_DESC_BOOL;
82  desc->data.boolean = (data == MMC_TRUE);
83  }; break;
84  case Values__STRING_3dBOX1: {
85  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET];
86  desc->type = TYPE_DESC_STRING;
87  desc->data.string = data;
88  }; break;
89  case Values__ARRAY_3dBOX2: {
90  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET];
91  void *dimLst = MMC_STRUCTDATA(value)[UNBOX_OFFSET+1];
92  if (parse_array(desc, data, dimLst)) {
93  printf("Parsing of array failed\n");
94  return -1;
95  }
96  }; break;
97  case Values__RECORD_3dBOX4: {
98  void *path = MMC_STRUCTDATA(value)[UNBOX_OFFSET];
99  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET+1];
100  void *names = MMC_STRUCTDATA(value)[UNBOX_OFFSET+2];
101  void *index = MMC_STRUCTDATA(value)[UNBOX_OFFSET+3];
102  if (-1 != ((mmc_sint_t)index >> 1)) {
103  desc->type = TYPE_DESC_MMC;
104  desc->data.mmc = value_to_mmc(value);
105  break;
106  }
107  desc->type = TYPE_DESC_RECORD;
108  //fprintf(stderr, "makepath: %s\n", path_to_name(path));
109  desc->data.record.record_name = path_to_name(path, '.');
110  while ((MMC_GETHDR(names) != MMC_NILHDR) &&
111  (MMC_GETHDR(data) != MMC_NILHDR)) {
112  type_description *elem;
113  void *nptr;
114 
115  nptr = MMC_CAR(names);
116  elem = add_modelica_record_member(desc, MMC_STRINGDATA(nptr),
117  MMC_HDRSTRLEN(MMC_GETHDR(nptr)));
118 
119  if (value_to_type_desc(MMC_CAR(data), elem)) {
120  return -1;
121  }
122  data = MMC_CDR(data);
123  names = MMC_CDR(names);
124  }
125  }; break;
126 
127  case Values__TUPLE_3dBOX1: {
128  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET];
129  desc->type = TYPE_DESC_TUPLE;
130  while (MMC_GETHDR(data) != MMC_NILHDR) {
131  type_description *elem;
132 
133  elem = add_tuple_member(desc);
134 
135  if (value_to_type_desc(MMC_CAR(data), elem)) {
136  return -1;
137  }
138  data = MMC_CDR(data);
139  }
140  }; break;
141  case Values__META_5fTUPLE_3dBOX1:
142  case Values__LIST_3dBOX1:
143  case Values__OPTION_3dBOX1:
144  desc->type = TYPE_DESC_MMC;
145  desc->data.mmc = value_to_mmc(value);
146  if (desc->data.mmc == 0) {
147  c_add_message(NULL,-1, ErrorType_runtime, ErrorLevel_error, "systemimpl.c:value_to_type_desc failed\n", NULL, 0);
148  return -1;
149  }
150  break;
151  /* unsupported */
152  case Values__META_5fARRAY_3dBOX1:
153  c_add_message(NULL,-1, ErrorType_runtime, ErrorLevel_error, "systemimpl.c:value_to_type_desc failed: Values.META_ARRAY\n", NULL, 0);
154  return -1;
155  case Values__CODE_3dBOX1:
156  c_add_message(NULL,-1, ErrorType_runtime, ErrorLevel_error, "systemimpl.c:value_to_type_desc failed: Values.CODE\n", NULL, 0);
157  return -1;
158  default:
159  c_add_message(NULL,-1, ErrorType_runtime, ErrorLevel_error, "systemimpl.c:value_to_type_desc failed\n", NULL, 0);
160  return -1;
161  }
162 
163  return 0;
164 }

166 static int execute_function(threadData_t* threadData,
167  void *in_arg, void **out_arg,
168  int (* func)(threadData_t*,
169  type_description *,
170  type_description *), int printDebug)
171 {
172  type_description arglst[MMC_NUM_ARGS + 1], crashbuf[50], *arg = NULL;
173  type_description crashbufretarg, retarg;
174  void *v = NULL;
175  int retval = 0;
176 
177  if (printDebug) { fprintf(stderr, "[dynload]: input parameters:\n"); fflush(stderr); }
178 
179  v = in_arg;
180  arg = arglst;
181 
182  while (!listEmpty(v)) {
183  void *val = MMC_CAR(v);
184  if (value_to_type_desc(val, arg)) {
185  if (printDebug)
186  {
187  puttype(arg);
188  fprintf(stderr, "[dynload]: returning from execute function due to value_to_type_desc failure!\n"); fflush(stderr);
189  }
190  return -1;
191  }
192  if (printDebug) puttype(arg);
193  ++arg;
194  v = MMC_CDR(v);
195  }
196 
197  init_type_description(arg);
198  init_type_description(&crashbufretarg);
199  init_type_description(&retarg);
200  init_type_description(&crashbuf[5]);
201 
202  retarg.retval = 1;
203 
204  if (printDebug) { fprintf(stderr, "[dynload]: calling the function\n"); fflush(stderr); }
205 
206  fflush(stdout);
207  /* call our function pointer! */
208  try { /* Don't let external C functions throwing C++ exceptions kill OMC! */
209  retval = func(threadData, arglst, &retarg);
210  } catch (...) {
211  retval = 1;
212  }
213  /* Flush all buffers for deterministic behaviour; in particular for the testsuite */
214  fflush(NULL);
215 
216  /* free the type description for the input parameters! */
217  arg = arglst;
218  while (arg->type != TYPE_DESC_NONE) {
219  free_type_description(arg);
220  ++arg;
221  }
222 
223  if (retval) {
224  *out_arg = Values__META_5fFAIL;
225  return 0;
226  } else {
227  if (printDebug) { fprintf(stderr, "[dynload]: output results:\n"); fflush(stderr); puttype(&retarg); }
228 
229  (*out_arg) = type_desc_to_value(&retarg);
230  /* out_arg doesn't seem to get freed, something we can do anything about?
231  * adrpo: 2009-09. it shouldn't be freed!
232  */
233 
234  free_type_description(&retarg);
235 
236  if ((*out_arg) == NULL) {
237  printf("[dynload]: Unable to parse returned values.\n");
238  return -1;
239  }
240 
241  return 0;
242  }

245 static void *generate_array(enum type_desc_e type, int curdim, int ndims,
246  _index_t *dim_size, void **data)
247 {
248  void *lst = (void *) mmc_mk_nil();
249  void *dimLst = (void*) mmc_mk_nil();
250  int i, cur_dim_size = dim_size[curdim - 1];
251 
252  if (curdim == ndims) {
253  type_description tmp;
254  tmp.type = type;
255 
256  switch (type) {
257  case TYPE_DESC_REAL: {
258  modelica_real *ptr = *((modelica_real**)data);
259  for (i = 0; i < cur_dim_size; ++i, --ptr) {
260  tmp.data.real = *ptr;
261  lst = (void *) mmc_mk_cons(type_desc_to_value(&tmp), lst);
262  }
263  *data = ptr;
264  }; break;
265  case TYPE_DESC_INT: {
266  modelica_integer *ptr = *((modelica_integer**)data);
267  for (i = 0; i < cur_dim_size; ++i, --ptr) {
268  tmp.data.integer = *ptr;
269  lst = (void *) mmc_mk_cons(type_desc_to_value(&tmp), lst);
270  }
271  *data = ptr;
272  }; break;
273  case TYPE_DESC_BOOL: {
274  modelica_boolean *ptr = *((modelica_boolean**)data);
275  for (i = 0; i < cur_dim_size; ++i, --ptr) {
276  tmp.data.boolean = *ptr;
277  lst = (void *) mmc_mk_cons(type_desc_to_value(&tmp), lst);
278  }
279  *data = ptr;
280  }; break;
281  case TYPE_DESC_STRING: {
282  modelica_string *ptr = *((modelica_string**)data);
283  for (i = 0; i < cur_dim_size; ++i, --ptr) {
284  tmp.data.string = *ptr;
285  lst = (void *) mmc_mk_cons(type_desc_to_value(&tmp), lst);
286  }
287  *data = ptr;
288  }; break;
289  default:
290  assert(0);
291  return NULL;
292  }
293  } else {
294  for (i = 0; i < cur_dim_size; ++i) {
295  lst = (void *) mmc_mk_cons(generate_array(type, curdim + 1, ndims, dim_size,
296  data), lst);
297  }
298  }
299  for (i = ndims; i >= curdim; i--) {
300  dimLst = (void *) mmc_mk_cons(mmc_mk_icon(dim_size[i-1]), dimLst);
301  }
302 
303  return Values__ARRAY(lst, dimLst);
304 }

307 static void *name_to_path(const char *name)
308 {
309  const char *last = name, *pos = NULL;
310  char *tmp;
311  void *ident = NULL;
312  int need_replace = 0;
313  while ((pos = strchr(last, '_')) != NULL) {
314  if (pos[1] == '_') {
315  last = pos + 2;
316  need_replace = 1;
317  continue;
318  } else
319  break;
320  }
321 
322  if (pos == NULL) {
323  if (need_replace) {
324  tmp = _replace(name, "__", "_");
325  ident = mmc_mk_scon(tmp);
326  GC_free(tmp);
327  } else {
328  /* memcpy(&tmp, &name, sizeof(char *)); */ /* don't try this at home */
329  ident = mmc_mk_scon((char*)name);
330  }
331  return Absyn__IDENT(ident);
332  } else {
333  size_t len = pos - name;
334  tmp = (char*)GC_malloc_atomic(len + 1);
335  memcpy(tmp, name, len);
336  tmp[len] = '\0';
337  if (need_replace) {
338  char *tmp2 = _replace(tmp, "__", "_");
339  ident = mmc_mk_scon(tmp2);
340  GC_free(tmp2);
341  } else {
342  ident = mmc_mk_scon(tmp);
343  }
344  GC_free(tmp);
345  return Absyn__QUALIFIED(ident, name_to_path(pos + 1));
346  }
347 }

349 static int mmc_to_value(void* mmc, void** res)
350 {
351  mmc_uint_t hdr;
352  int numslots;
353  unsigned ctor;
354  int i;
355  void* t;
356  if (MMC_IS_INTEGER(mmc)) {
357  *res = Values__INTEGER(mmc_mk_icon(MMC_UNTAGFIXNUM(mmc)));
358  return 0;
359  }
360  hdr = MMC_GETHDR(mmc);
361  if (hdr == MMC_REALHDR) {
362  *res = Values__REAL(mmc_mk_rcon(mmc_unbox_real(mmc)));
363  return 0;
364  }
365  if (MMC_HDRISSTRING(hdr)) {
366  MMC_CHECK_STRING(mmc);
367  // We need to duplicate the string because literals may not be accessible anymore... Bleh
368  *res = Values__STRING(mmc_mk_scon(MMC_STRINGDATA(mmc)));
369  return 0;
370  }
371  if (hdr == MMC_NILHDR) {
372  *res = Values__LIST(mmc_mk_nil());
373  return 0;
374  }
375 
376  numslots = MMC_HDRSLOTS(hdr);
377  ctor = 255 & (hdr >> 2);
378 
379 
380  if (numslots>0 && ctor == MMC_ARRAY_TAG) {
381  void *varlst = (void *) mmc_mk_nil();
382  for (i=numslots; i>0; i--) {
383  assert(0 == mmc_to_value(MMC_FETCH(MMC_OFFSET(MMC_UNTAGPTR(mmc),i)),&t));
384  varlst = mmc_mk_cons(t, varlst);
385  }
386  *res = (void *) Values__META_5fARRAY(varlst);
387  return 0;
388  }
389 
390  if (numslots>0 && ctor > 1) { /* RECORD */
391  void *namelst = (void *) mmc_mk_nil();
392  void *varlst = (void *) mmc_mk_nil();
393  struct record_description* desc = (struct record_description*) MMC_FETCH(MMC_OFFSET(MMC_UNTAGPTR(mmc),1));
394  assert(desc != NULL);
395  for (i=numslots; i>1; i--) {
396  assert(0 == mmc_to_value(MMC_FETCH(MMC_OFFSET(MMC_UNTAGPTR(mmc),i)),&t));
397  varlst = mmc_mk_cons(t, varlst);
398  t = (void*) desc->fieldNames[i-2];
399  namelst = mmc_mk_cons(mmc_mk_scon(t != NULL ? (const char*) t : "(null)"), namelst);
400  }
401  *res = (void *) Values__RECORD(name_to_path(desc->path),
402  varlst, namelst, mmc_mk_icon(((mmc_sint_t)ctor)-3));
403  return 0;
404  }
405 
406  if (numslots>0 && ctor == 0) { /* TUPLE */
407  void *varlst = (void *) mmc_mk_nil();
408  for (i=numslots; i>0; i--) {
409  assert(0 == mmc_to_value(MMC_FETCH(MMC_OFFSET(MMC_UNTAGPTR(mmc),i)),&t));
410  varlst = mmc_mk_cons(t, varlst);
411  }
412  *res = (void *) Values__META_5fTUPLE(varlst);
413  return 0;
414  }
415 
416  if (numslots==0 && ctor==1) /* NONE() */ {
417  *res = Values__OPTION(mmc_mk_none());
418  return 0;
419  }
420 
421  if (numslots==1 && ctor==1) /* SOME(x) */ {
422  assert(0 == mmc_to_value(MMC_FETCH(MMC_OFFSET(MMC_UNTAGPTR(mmc),1)),&t));
423  *res = Values__OPTION(mmc_mk_some(t));
424  return 0;
425  }
426 
427  if (numslots==2 && ctor==1) { /* CONS-PAIR */
428  /* Transform list by first reversing it to preserve the order */
429  void *varlst;
430  mmc = listReverse(mmc);
431  varlst = (void *) mmc_mk_nil();
432  while (!MMC_NILTEST(mmc)) {
433  assert(0 == mmc_to_value(MMC_CAR(mmc),&t));
434  varlst = mmc_mk_cons(t, varlst);
435  mmc = MMC_CDR(mmc);
436  }
437  *res = (void *) Values__LIST(varlst);
438  return 0;
439  }
440 
441  fprintf(stderr, "mmc_to_value: %d slots; ctor %d - FAILED to detect the type\n", numslots, ctor);
442  return 1;
443 }

445 static void *value_to_mmc(void* value)
446 {
447  switch (MMC_HDRCTOR(MMC_GETHDR(value))) {
448  case Values__INTEGER_3dBOX1:
449  case Values__BOOL_3dBOX1: {
450  return mmc_mk_icon(MMC_UNTAGFIXNUM(MMC_STRUCTDATA(value)[UNBOX_OFFSET+0]));
451  };
452  case Values__STRING_3dBOX1: {
453  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET+0];
454  return mmc_mk_scon(MMC_STRINGDATA(data));
455  };
456  case Values__REAL_3dBOX1: {
457  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET+0];
458  return mmc_mk_rcon(mmc_prim_get_real(data));
459  }
460  case Values__ARRAY_3dBOX2:
461  printf("[dynload]: Parsing of array inside uniontype failed\n");
462  return 0;
463  case Values__RECORD_3dBOX4: {
464  void *path = MMC_STRUCTDATA(value)[UNBOX_OFFSET+0];
465  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET+1];
466  void *names = MMC_STRUCTDATA(value)[UNBOX_OFFSET+2];
467  void *index = MMC_STRUCTDATA(value)[UNBOX_OFFSET+3];
468  int index_int = ((mmc_sint_t)index >> 1);
469  int i=0;
470  void *tmp = names;
471  void **data_mmc;
472  struct record_description* desc = (struct record_description*) malloc(sizeof(struct record_description));
473  if (desc == NULL) {
474  fprintf(stderr, "[dynload]: value_to_mmc: malloc failed\n");
475  return 0;
476  }
477  while (MMC_GETHDR(tmp) != MMC_NILHDR) {
478  i++; tmp = MMC_CDR(tmp);
479  }
480  /* duplicate string? will this give problems with GC? */
481  desc->path = path_to_name(path, '_');
482  desc->name = path_to_name(path, '.');
483  desc->fieldNames = (const char**) malloc(i*sizeof(char*));
484  data_mmc = (void**) malloc((i+1)*sizeof(void*));
485  i=0;
486  data_mmc[0] = desc;
487  while (MMC_GETHDR(names) != MMC_NILHDR) {
488  desc->fieldNames[i] = MMC_STRINGDATA(MMC_CAR(names));
489  names = MMC_CDR(names);
490  data_mmc[i+1] = value_to_mmc(MMC_CAR(data));
491  i++;
492  data = MMC_CDR(data);
493  }
494  return mmc_mk_box_arr(i+1, index_int+3, (void**) data_mmc);
495  }; break;
496  case Values__OPTION_3dBOX1: {
497  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET+0];
498  int numslots = MMC_HDRSLOTS(MMC_GETHDR(data));
499  if (numslots == 0) return mmc_mk_none();
500  return mmc_mk_some(value_to_mmc(MMC_STRUCTDATA(data)[0]));
501  }; break;
502  case Values__LIST_3dBOX1: {
503  void* data = MMC_STRUCTDATA(value)[UNBOX_OFFSET+0];
504  void* tmp = mmc_mk_nil();
505  while (MMC_GETHDR(data) != MMC_NILHDR) {
506  tmp = mmc_mk_cons(value_to_mmc(MMC_CAR(data)), tmp);
507  data = MMC_CDR(data);
508  }
509  /* Transform list by first reversing it to preserve the order */
510  return listReverse(tmp);
511  };
512  case Values__META_5fARRAY_3dBOX1: {
513  void* data = MMC_STRUCTDATA(value)[UNBOX_OFFSET+0];
514  void* tmp = mmc_mk_nil();
515  while (MMC_GETHDR(data) != MMC_NILHDR) {
516  tmp = mmc_mk_cons(value_to_mmc(MMC_CAR(data)), tmp);
517  data = MMC_CDR(data);
518  }
519  return listArray(listReverse(tmp));
520  };
521  case Values__META_5fTUPLE_3dBOX1: {
522  void *data = MMC_STRUCTDATA(value)[UNBOX_OFFSET+0];
523  int len=0;
524  void *tmp = data;
525  void **data_mmc;
526  void *res;
527 
528  while (MMC_GETHDR(tmp) != MMC_NILHDR) {
529  len++; tmp = MMC_CDR(tmp);
530  }
531  res = mmc_mk_box_no_assign(len, 0, 0);
532  len = 0;
533  tmp = data;
534  while (MMC_GETHDR(tmp) != MMC_NILHDR) {
535  threadData_t *threadData = NULL;
536  arrayUpdate(res, ++len, value_to_mmc(MMC_CAR(tmp)));
537  tmp = MMC_CDR(tmp);
538  }
539  return res;
540  };
541  case Values__ENUM_5fLITERAL_3dBOX2:
542  case Values__CODE_3dBOX1:
543  /* unsupported */
544  fprintf(stderr, "%s:%d: enum,code unsupported in MetaModelica data\n", __FILE__, __LINE__);
545  return 0;
546  default:
547  /* unsupported */
548  fprintf(stderr, "%s:%d: Error, unknown type", __FILE__, __LINE__);
549  printAny(value);
550  fprintf(stderr, "\n");
551  return 0;
552  }
553 }

555 void *type_desc_to_value(type_description *desc)
556 {
557  switch (desc->type) {
558  case TYPE_DESC_NONE:
559  return NULL;
560  case TYPE_DESC_NORETCALL:
561  return (void *) Values__NORETCALL;
562  case TYPE_DESC_REAL:
563  return (void *) Values__REAL(mmc_mk_rcon(desc->data.real));
564  case TYPE_DESC_INT:
565  return (void *) Values__INTEGER(mmc_mk_icon(desc->data.integer));
566  case TYPE_DESC_BOOL:
567  if(getMyBool(desc))
568  return (void *) Values__BOOL(MMC_TRUE);
569  return (void *) Values__BOOL(MMC_FALSE);
570  case TYPE_DESC_STRING:
571  /* Duplicate since we unload the object */
572  return (void *) Values__STRING(mmc_mk_scon(MMC_STRINGDATA(desc->data.string)));
573  case TYPE_DESC_TUPLE: {
574  type_description *e = desc->data.tuple.element + desc->data.tuple.elements;
575  void *lst = (void *) mmc_mk_nil();
576  while (e > desc->data.tuple.element) {
577  void *t = type_desc_to_value(--e);
578  if (t == NULL)
579  return NULL;
580  lst = mmc_mk_cons(t, lst);
581  }
582  return (void *) Values__TUPLE(lst);
583  };
584  case TYPE_DESC_RECORD: {

693 static int get_array_data(int curdim, int dims, const _index_t *dim_size,
694  void *arrdata, enum type_desc_e type, void **data)
695 {
696  void *ptr = arrdata;
697  assert(curdim > 0 && curdim <= dims);
698  if (curdim == dims) {
699  while (MMC_GETHDR(ptr) != MMC_NILHDR) {
700  void *item = MMC_CAR(ptr);
701 
702  switch (type) {
703  case TYPE_DESC_REAL: {
704  modelica_real *ptr = *((modelica_real**)data);
705  if (MMC_HDRCTOR(MMC_GETHDR(item)) != Values__REAL_3dBOX1)
706  return -1;
707  *ptr = mmc_prim_get_real(MMC_STRUCTDATA(item)[UNBOX_OFFSET+0]);
708  *data = ++ptr;
709  }; break;
710  case TYPE_DESC_INT: {
711  modelica_integer *ptr = *((modelica_integer**)data);
712  if (MMC_HDRCTOR(MMC_GETHDR(item)) != Values__INTEGER_3dBOX1)
713  return -1;
714  *ptr = MMC_UNTAGFIXNUM(MMC_STRUCTDATA(item)[UNBOX_OFFSET+0]);
715  *data = ++ptr;
716  }; break;
717  case TYPE_DESC_BOOL: {
718  modelica_boolean *ptr = *((modelica_boolean**)data);
719  if (MMC_HDRCTOR(MMC_GETHDR(item)) != Values__BOOL_3dBOX1)
720  return -1;
721  *ptr = (MMC_STRUCTDATA(item)[UNBOX_OFFSET+0] == MMC_TRUE);
722  *data = ++ptr;
723  }; break;
724  case TYPE_DESC_STRING: {
725  modelica_string *ptr = *((modelica_string**)data);
726  void *str;
727  if (MMC_HDRCTOR(MMC_GETHDR(item)) != Values__STRING_3dBOX1)
728  return -1;
729  str = MMC_STRUCTDATA(item)[UNBOX_OFFSET+0];
730  *ptr = str;
731  *data = ++ptr;
732  }; break;
733  default:
734  assert(0);
735  return -1;
736  }
737 
738  ptr = MMC_CDR(ptr);
739  }
740  } else {
741  while (MMC_GETHDR(ptr) != MMC_NILHDR) {
742  void *item = MMC_CAR(ptr);
743  if (MMC_HDRCTOR(MMC_GETHDR(item)) != Values__ARRAY_3dBOX2)
744  return -1;
745 
746  if (get_array_data(curdim + 1, dims, dim_size, MMC_STRUCTDATA(item)[UNBOX_OFFSET+0],
747  type, data))
748  return -1;
749 
750  ptr = MMC_CDR(ptr);
751  }
752  }
753 
754  return 0;
