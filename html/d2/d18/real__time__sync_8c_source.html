38 void omc_real_time_sync_init(threadData_t *threadData, DATA *data)
39 {
40  data->real_time_sync.maxLate = INT64_MIN;
41 
42  omc_real_time_sync_update(data, data->real_time_sync.scaling);
43 
44  if (data->real_time_sync.enabled == 0) {
45  return;
46  }
47 
48 #if defined(__linux__)
49  if (mlockall(MCL_CURRENT | MCL_FUTURE) == -1) {
50  warningStreamPrint(LOG_RT, 0, __FILE__ ": mlockall failed (recommended to run as root to lock memory into RAM while doing real-time simulation): %s\n", strerror(errno));
51  }
52  struct sched_param param = {.sched_priority = 49 /* 50=interrupt handler */ };
53  if(sched_setscheduler(0, SCHED_FIFO, &param) == -1) {
54  warningStreamPrint(LOG_RT, 0, __FILE__ ": sched_setscheduler failed: %s\n", strerror(errno));
55  }
56 #endif
57 }

59 void omc_real_time_sync_update(DATA *data, double scaling)
60 {
61  data->real_time_sync.scaling = scaling;
62  if (scaling == 0) {
63  data->real_time_sync.enabled = 0;
64  return;
65  }
66  data->real_time_sync.enabled = 1;
67  data->real_time_sync.time = data->localData[0]->timeValue;
68  rt_ext_tp_tick_realtime(&data->real_time_sync.clock);
